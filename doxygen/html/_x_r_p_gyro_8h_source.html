18/**
19 * Use a rate gyro to return the robots heading relative to a starting position.
20 *
21 * This class is for the XRP onboard gyro, and will only work in
22 * simulation/XRP mode. Only one instance of a XRPGyro is supported.
23 */
24class XRPGyro {
25 public:
26 /**
27 * Constructs an XRPGyro.
28 *
29 * <p>Only one instance of a XRPGyro is supported.
30 */
31 XRPGyro();
32
33 /**
34 * Return the actual angle in degrees that the robot is currently facing.
35 *
36 * The angle is based on integration of the returned rate form the gyro.
37 * The angle is continuous, that is, it will continue from 360->361 degrees.
38 * This allows algorithms that wouldn't want to see a discontinuity in the
39 * gyro output as it sweeps from 360 to 0 on the second time around.
40 *
41 * @return the current heading of the robot in degrees.
42 */
43 double GetAngle() const;
44
45 /**
46 * Gets the angle the robot is facing.
47 *
48 * @return A Rotation2d with the current heading.
49 */
50 frc::Rotation2d GetRotation2d() const;
51
52 /**
53 * Return the rate of rotation of the gyro
54 *
55 * The rate is based on the most recent reading of the gyro.
56 *
57 * @return the current rate in degrees per second
58 */
59 double GetRate() const;
60
61 /**
62 * Gets the rate of turn in degrees-per-second around the X-axis
63 */
64 double GetRateX() const;
65
66 /**
67 * Gets the rate of turn in degrees-per-second around the Y-axis
68 */
69 double GetRateY() const;
70
71 /**
72 * Gets the rate of turn in degrees-per-second around the Z-axis
73 */
74 double GetRateZ() const;
75
76 /**
77 * Gets the currently reported angle around the X-axis
78 */
79 double GetAngleX() const;
80
81 /**
82 * Gets the currently reported angle around the X-axis
83 */
84 double GetAngleY() const;
85
86 /**
87 * Gets the currently reported angle around the X-axis
88 */
89 double GetAngleZ() const;
90
91 /**
92 * Resets the gyro
93 */
94 void Reset();
95
96 private:
97 hal::SimDevice m_simDevice;
98 hal::SimDouble m_simRateX;
99 hal::SimDouble m_simRateY;
100 hal::SimDouble m_simRateZ;
101 hal::SimDouble m_simAngleX;
102 hal::SimDouble m_simAngleY;
103 hal::SimDouble m_simAngleZ;
104
105 double m_angleXOffset = 0;
106 double m_angleYOffset = 0;
107 double m_angleZOffset = 0;
108};
