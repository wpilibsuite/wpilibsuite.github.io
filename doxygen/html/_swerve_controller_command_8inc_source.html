81template <size_t NumModules>
82void SwerveControllerCommand<NumModules>::Initialize() {
83 if (m_desiredRotation == nullptr) {
84 m_desiredRotation = [&] {
85 return m_trajectory.States().back().pose.Rotation();
86 };
87 }
88 m_timer.Restart();
89}
90
91template <size_t NumModules>
92void SwerveControllerCommand<NumModules>::Execute() {
93 auto curTime = m_timer.Get();
94 auto m_desiredState = m_trajectory.Sample(curTime);
95
96 auto targetChassisSpeeds =
97 m_controller.Calculate(m_pose(), m_desiredState, m_desiredRotation());
98 auto targetModuleStates =
99 m_kinematics.ToSwerveModuleStates(targetChassisSpeeds);
100
101 m_outputStates(targetModuleStates);
102}
103
104template <size_t NumModules>
105void SwerveControllerCommand<NumModules>::End(bool interrupted) {
106 m_timer.Stop();
107}
108
109template <size_t NumModules>
110bool SwerveControllerCommand<NumModules>::IsFinished() {
111 return m_timer.HasElapsed(m_trajectory.TotalTime());
112}
