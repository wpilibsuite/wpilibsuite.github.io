21template <int States, int Inputs, int Outputs>
22ExtendedKalmanFilter<States, Inputs, Outputs>::ExtendedKalmanFilter(
23 std::function<StateVector(const StateVector&, const InputVector&)> f,
24 std::function<OutputVector(const StateVector&, const InputVector&)> h,
25 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
26 units::second_t dt)
27 : m_f(std::move(f)), m_h(std::move(h)) {
28 m_contQ = MakeCovMatrix(stateStdDevs);
29 m_contR = MakeCovMatrix(measurementStdDevs);
30 m_residualFuncY = [](const OutputVector& a,
31 const OutputVector& b) -> OutputVector { return a - b; };
32 m_addFuncX = [](const StateVector& a, const StateVector& b) -> StateVector {
33 return a + b;
34 };
35 m_dt = dt;
36
37 StateMatrix contA = NumericalJacobianX<States, States, Inputs>(
38 m_f, m_xHat, InputVector::Zero());
39 Matrixd<Outputs, States> C = NumericalJacobianX<Outputs, States, Inputs>(
40 m_h, m_xHat, InputVector::Zero());
41
42 StateMatrix discA;
43 StateMatrix discQ;
44 DiscretizeAQ<States>(contA, m_contQ, dt, &discA, &discQ);
45
46 Matrixd<Outputs, Outputs> discR = DiscretizeR<Outputs>(m_contR, dt);
47
48 if (IsDetectable<States, Outputs>(discA, C) && Outputs <= States) {
49 m_initP =
50 DARE<States, Outputs>(discA.transpose(), C.transpose(), discQ, discR);
51 } else {
52 m_initP = StateMatrix::Zero();
53 }
54 m_P = m_initP;
55}

94template <int States, int Inputs, int Outputs>
95void ExtendedKalmanFilter<States, Inputs, Outputs>::Predict(
96 const InputVector& u, units::second_t dt) {
97 // Find continuous A
98 StateMatrix contA =
99 NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
100
101 // Find discrete A and Q
102 StateMatrix discA;
103 StateMatrix discQ;
104 DiscretizeAQ<States>(contA, m_contQ, dt, &discA, &discQ);
105
106 m_xHat = RK4(m_f, m_xHat, u, dt);
107
108 // Pₖ₊₁⁻ = APₖ⁻Aᵀ + Q
109 m_P = discA * m_P * discA.transpose() + discQ;
110
111 m_dt = dt;
112}

130template <int States, int Inputs, int Outputs>
131template <int Rows>
132void ExtendedKalmanFilter<States, Inputs, Outputs>::Correct(
133 const InputVector& u, const Vectord<Rows>& y,
134 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
135 const Matrixd<Rows, Rows>& R,
136 std::function<Vectord<Rows>(const Vectord<Rows>&, const Vectord<Rows>&)>
137 residualFuncY,
138 std::function<StateVector(const StateVector&, const StateVector&)>
139 addFuncX) {
140 const Matrixd<Rows, States> C =
141 NumericalJacobianX<Rows, States, Inputs>(h, m_xHat, u);
142 const Matrixd<Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
143
144 Matrixd<Rows, Rows> S = C * m_P * C.transpose() + discR;
145
146 // We want to put K = PCᵀS⁻¹ into Ax = b form so we can solve it more
147 // efficiently.
148 //
149 // K = PCᵀS⁻¹
150 // KS = PCᵀ
151 // (KS)ᵀ = (PCᵀ)ᵀ
152 // SᵀKᵀ = CPᵀ
153 //
154 // The solution of Ax = b can be found via x = A.solve(b).
155 //
156 // Kᵀ = Sᵀ.solve(CPᵀ)
157 // K = (Sᵀ.solve(CPᵀ))ᵀ
158 Matrixd<States, Rows> K =
159 S.transpose().ldlt().solve(C * m_P.transpose()).transpose();
160
161 // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + Kₖ₊₁(y − h(x̂ₖ₊₁⁻, uₖ₊₁))
162 m_xHat = addFuncX(m_xHat, K * residualFuncY(y, h(m_xHat, u)));
163
164 // Pₖ₊₁⁺ = (I−Kₖ₊₁C)Pₖ₊₁⁻(I−Kₖ₊₁C)ᵀ + Kₖ₊₁RKₖ₊₁ᵀ
165 // Use Joseph form for numerical stability
166 m_P = (StateMatrix::Identity() - K * C) * m_P *
167 (StateMatrix::Identity() - K * C).transpose() +
168 K * discR * K.transpose();
169}
