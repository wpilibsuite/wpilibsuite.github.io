16/**
17 * Use a rate gyro to return the robots heading relative to a starting position.
18 *
19 * This class is for the Romi onboard gyro, and will only work in
20 * simulation/Romi mode. Only one instance of a RomiGyro is supported.
21 */
22class RomiGyro {
23 public:
24 RomiGyro();
25
26 /**
27 * Return the actual angle in degrees that the robot is currently facing.
28 *
29 * The angle is based on integration of the returned rate form the gyro.
30 * The angle is continuous, that is, it will continue from 360->361 degrees.
31 * This allows algorithms that wouldn't want to see a discontinuity in the
32 * gyro output as it sweeps from 360 to 0 on the second time around.
33 *
34 * @return the current heading of the robot in degrees.
35 */
36 double GetAngle() const;
37
38 /**
39 * Return the rate of rotation of the gyro
40 *
41 * The rate is based on the most recent reading of the gyro.
42 *
43 * @return the current rate in degrees per second
44 */
45 double GetRate() const;
46
47 /**
48 * Gets the rate of turn in degrees-per-second around the X-axis
49 */
50 double GetRateX() const;
51
52 /**
53 * Gets the rate of turn in degrees-per-second around the Y-axis
54 */
55 double GetRateY() const;
56
57 /**
58 * Gets the rate of turn in degrees-per-second around the Z-axis
59 */
60 double GetRateZ() const;
61
62 /**

64 */
65 double GetAngleX() const;
66
67 /**
68 * Gets the currently reported angle around the X-axis
69 */
70 double GetAngleY() const;
71
72 /**
73 * Gets the currently reported angle around the X-axis
74 */
75 double GetAngleZ() const;
76
77 /**
78 * Resets the gyro

80 void Reset();
