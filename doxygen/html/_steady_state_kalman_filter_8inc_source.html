22template <int States, int Inputs, int Outputs>
23SteadyStateKalmanFilter<States, Inputs, Outputs>::SteadyStateKalmanFilter(
24 LinearSystem<States, Inputs, Outputs>& plant,
25 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
26 units::second_t dt) {
27 m_plant = &plant;
28
29 auto contQ = MakeCovMatrix(stateStdDevs);
30 auto contR = MakeCovMatrix(measurementStdDevs);
31
32 Matrixd<States, States> discA;
33 Matrixd<States, States> discQ;
34 DiscretizeAQ<States>(plant.A(), contQ, dt, &discA, &discQ);
35
36 auto discR = DiscretizeR<Outputs>(contR, dt);
37
38 const auto& C = plant.C();
39
40 if (!IsDetectable<States, Outputs>(discA, C)) {
41 std::string msg = fmt::format(
42 "The system passed to the Kalman filter is undetectable!\n\n"
43 "A =\n{}\nC =\n{}\n",
44 discA, C);
45
46 wpi::math::MathSharedStore::ReportError(msg);
47 throw std::invalid_argument(msg);
48 }
49
50 Matrixd<States, States> P =
51 DARE<States, Outputs>(discA.transpose(), C.transpose(), discQ, discR);
52
53 // S = CPCᵀ + R
54 Matrixd<Outputs, Outputs> S = C * P * C.transpose() + discR;
55
56 // We want to put K = PCᵀS⁻¹ into Ax = b form so we can solve it more
57 // efficiently.
58 //
59 // K = PCᵀS⁻¹
60 // KS = PCᵀ
61 // (KS)ᵀ = (PCᵀ)ᵀ
62 // SᵀKᵀ = CPᵀ
63 //
64 // The solution of Ax = b can be found via x = A.solve(b).
65 //
66 // Kᵀ = Sᵀ.solve(CPᵀ)
67 // K = (Sᵀ.solve(CPᵀ))ᵀ
68 m_K = S.transpose().ldlt().solve(C * P.transpose()).transpose();
69
70 Reset();
71}

79template <int States, int Inputs, int Outputs>
80void SteadyStateKalmanFilter<States, Inputs, Outputs>::Correct(
81 const InputVector& u, const OutputVector& y) {
82 const auto& C = m_plant->C();
83 const auto& D = m_plant->D();
84
85 // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − (Cx̂ₖ₊₁⁻ + Duₖ₊₁))
86 m_xHat += m_K * (y - (C * m_xHat + D * u));
87}
