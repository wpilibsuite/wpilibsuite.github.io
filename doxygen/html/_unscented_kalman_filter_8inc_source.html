21template <int States, int Inputs, int Outputs>
22UnscentedKalmanFilter<States, Inputs, Outputs>::UnscentedKalmanFilter(
23 std::function<StateVector(const StateVector&, const InputVector&)> f,
24 std::function<OutputVector(const StateVector&, const InputVector&)> h,
25 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
26 units::second_t dt)
27 : m_f(std::move(f)), m_h(std::move(h)) {
28 m_contQ = MakeCovMatrix(stateStdDevs);
29 m_contR = MakeCovMatrix(measurementStdDevs);
30 m_meanFuncX = [](const Matrixd<States, 2 * States + 1>& sigmas,
31 const Vectord<2 * States + 1>& Wm) -> StateVector {
32 return sigmas * Wm;
33 };
34 m_meanFuncY = [](const Matrixd<Outputs, 2 * States + 1>& sigmas,
35 const Vectord<2 * States + 1>& Wc) -> OutputVector {
36 return sigmas * Wc;
37 };
38 m_residualFuncX = [](const StateVector& a,
39 const StateVector& b) -> StateVector { return a - b; };
40 m_residualFuncY = [](const OutputVector& a,
41 const OutputVector& b) -> OutputVector { return a - b; };
42 m_addFuncX = [](const StateVector& a, const StateVector& b) -> StateVector {
43 return a + b;
44 };
45 m_dt = dt;
46
47 Reset();
48}

81template <int States, int Inputs, int Outputs>
82void UnscentedKalmanFilter<States, Inputs, Outputs>::Predict(
83 const InputVector& u, units::second_t dt) {
84 m_dt = dt;
85
86 // Discretize Q before projecting mean and covariance forward
87 StateMatrix contA =
88 NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
89 StateMatrix discA;
90 StateMatrix discQ;
91 DiscretizeAQ<States>(contA, m_contQ, m_dt, &discA, &discQ);
92 Eigen::internal::llt_inplace<double, Eigen::Lower>::blocked(discQ);
93
94 Matrixd<States, 2 * States + 1> sigmas =
95 m_pts.SquareRootSigmaPoints(m_xHat, m_S);
96
97 for (int i = 0; i < m_pts.NumSigmas(); ++i) {
98 StateVector x = sigmas.template block<States, 1>(0, i);
99 m_sigmasF.template block<States, 1>(0, i) = RK4(m_f, x, u, dt);
100 }
101
102 auto [xHat, S] = SquareRootUnscentedTransform<States, States>(
103 m_sigmasF, m_pts.Wm(), m_pts.Wc(), m_meanFuncX, m_residualFuncX,
104 discQ.template triangularView<Eigen::Lower>());
105 m_xHat = xHat;
106 m_S = S;
107}

109template <int States, int Inputs, int Outputs>
110template <int Rows>
111void UnscentedKalmanFilter<States, Inputs, Outputs>::Correct(
112 const InputVector& u, const Vectord<Rows>& y,
113 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
114 const Matrixd<Rows, Rows>& R) {
115 auto meanFuncY = [](const Matrixd<Outputs, 2 * States + 1>& sigmas,
116 const Vectord<2 * States + 1>& Wc) -> Vectord<Rows> {
117 return sigmas * Wc;
118 };
119 auto residualFuncX = [](const StateVector& a,
120 const StateVector& b) -> StateVector {
121 return a - b;
122 };
123 auto residualFuncY = [](const Vectord<Rows>& a,
124 const Vectord<Rows>& b) -> Vectord<Rows> {
125 return a - b;
126 };
127 auto addFuncX = [](const StateVector& a,
128 const StateVector& b) -> StateVector { return a + b; };
129 Correct<Rows>(u, y, std::move(h), R, std::move(meanFuncY),
130 std::move(residualFuncY), std::move(residualFuncX),
131 std::move(addFuncX));
132}

134template <int States, int Inputs, int Outputs>
135template <int Rows>
136void UnscentedKalmanFilter<States, Inputs, Outputs>::Correct(
137 const InputVector& u, const Vectord<Rows>& y,
138 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
139 const Matrixd<Rows, Rows>& R,
140 std::function<Vectord<Rows>(const Matrixd<Rows, 2 * States + 1>&,
141 const Vectord<2 * States + 1>&)>
142 meanFuncY,
143 std::function<Vectord<Rows>(const Vectord<Rows>&, const Vectord<Rows>&)>
144 residualFuncY,
145 std::function<StateVector(const StateVector&, const StateVector&)>
146 residualFuncX,
147 std::function<StateVector(const StateVector&, const StateVector&)>
148 addFuncX) {
149 Matrixd<Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
150 Eigen::internal::llt_inplace<double, Eigen::Lower>::blocked(discR);
151
152 // Transform sigma points into measurement space
153 Matrixd<Rows, 2 * States + 1> sigmasH;
154 Matrixd<States, 2 * States + 1> sigmas =
155 m_pts.SquareRootSigmaPoints(m_xHat, m_S);
156 for (int i = 0; i < m_pts.NumSigmas(); ++i) {
157 sigmasH.template block<Rows, 1>(0, i) =
158 h(sigmas.template block<States, 1>(0, i), u);
159 }
160
161 // Mean and covariance of prediction passed through UT
162 auto [yHat, Sy] = SquareRootUnscentedTransform<Rows, States>(
163 sigmasH, m_pts.Wm(), m_pts.Wc(), meanFuncY, residualFuncY,
164 discR.template triangularView<Eigen::Lower>());
165
166 // Compute cross covariance of the state and the measurements
167 Matrixd<States, Rows> Pxy;
168 Pxy.setZero();
169 for (int i = 0; i < m_pts.NumSigmas(); ++i) {
170 // Pxy += (sigmas_f[:, i] - x̂)(sigmas_h[:, i] - ŷ)ᵀ W_c[i]
171 Pxy += m_pts.Wc(i) *
172 (residualFuncX(m_sigmasF.template block<States, 1>(0, i), m_xHat)) *
173 (residualFuncY(sigmasH.template block<Rows, 1>(0, i), yHat))
174 .transpose();
175 }
176
177 // K = (P_{xy} / S_yᵀ) / S_y
178 // K = (S_y \ P_{xy}ᵀ)ᵀ / S_y
179 // K = (S_yᵀ \ (S_y \ P_{xy}ᵀ))ᵀ
180 Matrixd<States, Rows> K =
181 Sy.transpose()
182 .fullPivHouseholderQr()
183 .solve(Sy.fullPivHouseholderQr().solve(Pxy.transpose()))
184 .transpose();
185
186 // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − ŷ)
187 m_xHat = addFuncX(m_xHat, K * residualFuncY(y, yHat));
188
189 Matrixd<States, Rows> U = K * Sy;
190 for (int i = 0; i < Rows; i++) {
191 Eigen::internal::llt_inplace<double, Eigen::Upper>::rankUpdate(
192 m_S, U.template block<States, 1>(0, i), -1);
193 }
194}
