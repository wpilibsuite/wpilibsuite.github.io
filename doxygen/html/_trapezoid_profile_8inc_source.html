17template <class Distance>
18typename TrapezoidProfile<Distance>::State
19TrapezoidProfile<Distance>::Calculate(units::second_t t, State current,
20 State goal) {
21 m_direction = ShouldFlipAcceleration(current, goal) ? -1 : 1;
22 m_current = Direct(current);
23 goal = Direct(goal);
24 if (m_current.velocity > m_constraints.maxVelocity) {
25 m_current.velocity = m_constraints.maxVelocity;
26 }
27
28 // Deal with a possibly truncated motion profile (with nonzero initial or
29 // final velocity) by calculating the parameters as if the profile began and
30 // ended at zero velocity
31 units::second_t cutoffBegin =
32 m_current.velocity / m_constraints.maxAcceleration;
33 Distance_t cutoffDistBegin =
34 cutoffBegin * cutoffBegin * m_constraints.maxAcceleration / 2.0;
35
36 units::second_t cutoffEnd = goal.velocity / m_constraints.maxAcceleration;
37 Distance_t cutoffDistEnd =
38 cutoffEnd * cutoffEnd * m_constraints.maxAcceleration / 2.0;
39
40 // Now we can calculate the parameters as if it was a full trapezoid instead
41 // of a truncated one
42
43 Distance_t fullTrapezoidDist =
44 cutoffDistBegin + (goal.position - m_current.position) + cutoffDistEnd;
45 units::second_t accelerationTime =
46 m_constraints.maxVelocity / m_constraints.maxAcceleration;
47
48 Distance_t fullSpeedDist =
49 fullTrapezoidDist -
50 accelerationTime * accelerationTime * m_constraints.maxAcceleration;
51
52 // Handle the case where the profile never reaches full speed
53 if (fullSpeedDist < Distance_t{0}) {
54 accelerationTime =
55 units::math::sqrt(fullTrapezoidDist / m_constraints.maxAcceleration);
56 fullSpeedDist = Distance_t{0};
57 }
58
59 m_endAccel = accelerationTime - cutoffBegin;
60 m_endFullSpeed = m_endAccel + fullSpeedDist / m_constraints.maxVelocity;
61 m_endDecel = m_endFullSpeed + accelerationTime - cutoffEnd;
62 State result = m_current;
63
64 if (t < m_endAccel) {
65 result.velocity += t * m_constraints.maxAcceleration;
66 result.position +=
67 (m_current.velocity + t * m_constraints.maxAcceleration / 2.0) * t;
68 } else if (t < m_endFullSpeed) {
69 result.velocity = m_constraints.maxVelocity;
70 result.position += (m_current.velocity +
71 m_endAccel * m_constraints.maxAcceleration / 2.0) *
72 m_endAccel +
73 m_constraints.maxVelocity * (t - m_endAccel);
74 } else if (t <= m_endDecel) {
75 result.velocity =
76 goal.velocity + (m_endDecel - t) * m_constraints.maxAcceleration;
77 units::second_t timeLeft = m_endDecel - t;
78 result.position =
79 goal.position -
80 (goal.velocity + timeLeft * m_constraints.maxAcceleration / 2.0) *
81 timeLeft;
82 } else {
83 result = goal;
84 }
85
86 return Direct(result);
87}

89template <class Distance>
90units::second_t TrapezoidProfile<Distance>::TimeLeftUntil(
91 Distance_t target) const {
92 Distance_t position = m_current.position * m_direction;
93 Velocity_t velocity = m_current.velocity * m_direction;
94
95 units::second_t endAccel = m_endAccel * m_direction;
96 units::second_t endFullSpeed = m_endFullSpeed * m_direction - endAccel;
97
98 if (target < position) {
99 endAccel *= -1.0;
100 endFullSpeed *= -1.0;
101 velocity *= -1.0;
102 }
103
104 endAccel = units::math::max(endAccel, 0_s);
105 endFullSpeed = units::math::max(endFullSpeed, 0_s);
106
107 const Acceleration_t acceleration = m_constraints.maxAcceleration;
108 const Acceleration_t deceleration = -m_constraints.maxAcceleration;
109
110 Distance_t distToTarget = units::math::abs(target - position);
111
112 if (distToTarget < Distance_t{1e-6}) {
113 return 0_s;
114 }
115
116 Distance_t accelDist =
117 velocity * endAccel + 0.5 * acceleration * endAccel * endAccel;
118
119 Velocity_t decelVelocity;
120 if (endAccel > 0_s) {
121 decelVelocity = units::math::sqrt(
122 units::math::abs(velocity * velocity + 2 * acceleration * accelDist));
123 } else {
124 decelVelocity = velocity;
125 }
126
127 Distance_t fullSpeedDist = m_constraints.maxVelocity * endFullSpeed;
128 Distance_t decelDist;
129
130 if (accelDist > distToTarget) {
131 accelDist = distToTarget;
132 fullSpeedDist = Distance_t{0};
133 decelDist = Distance_t{0};
134 } else if (accelDist + fullSpeedDist > distToTarget) {
135 fullSpeedDist = distToTarget - accelDist;
136 decelDist = Distance_t{0};
137 } else {
138 decelDist = distToTarget - fullSpeedDist - accelDist;
139 }
140
141 units::second_t accelTime =
142 (-velocity + units::math::sqrt(units::math::abs(
143 velocity * velocity + 2 * acceleration * accelDist))) /
144 acceleration;
145
146 units::second_t decelTime =
147 (-decelVelocity +
148 units::math::sqrt(units::math::abs(decelVelocity * decelVelocity +
149 2 * deceleration * decelDist))) /
150 deceleration;
151
152 units::second_t fullSpeedTime = fullSpeedDist / m_constraints.maxVelocity;
153
154 return accelTime + fullSpeedTime + decelTime;
155}
