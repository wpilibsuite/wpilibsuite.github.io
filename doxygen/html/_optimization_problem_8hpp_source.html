30/**
31 * This class allows the user to pose a constrained nonlinear optimization
32 * problem in natural mathematical notation and solve it.
33 *
34 * This class supports problems of the form:
35@verbatim
36 minₓ f(x)
37subject to cₑ(x) = 0
38 cᵢ(x) ≥ 0
39@endverbatim
40 *
41 * where f(x) is the scalar cost function, x is the vector of decision variables
42 * (variables the solver can tweak to minimize the cost function), cᵢ(x) are the
43 * inequality constraints, and cₑ(x) are the equality constraints. Constraints
44 * are equations or inequalities of the decision variables that constrain what
45 * values the solver is allowed to use when searching for an optimal solution.
46 *
47 * The nice thing about this class is users don't have to put their system in
48 * the form shown above manually; they can write it in natural mathematical form
49 * and it'll be converted for them.
50 */
51class SLEIPNIR_DLLEXPORT OptimizationProblem {
52 public:
53 /**
54 * Construct the optimization problem.
55 */
56 OptimizationProblem() noexcept = default;
57
58 /**
59 * Create a decision variable in the optimization problem.
60 */
61 [[nodiscard]]
62 Variable DecisionVariable() {
63 m_decisionVariables.emplace_back();
64 return m_decisionVariables.back();
65 }
66
67 /**
68 * Create a matrix of decision variables in the optimization problem.
69 *
70 * @param rows Number of matrix rows.
71 * @param cols Number of matrix columns.
72 */
73 [[nodiscard]]
74 VariableMatrix DecisionVariable(int rows, int cols = 1) {
75 m_decisionVariables.reserve(m_decisionVariables.size() + rows * cols);
76
77 VariableMatrix vars{rows, cols};
78
79 for (int row = 0; row < rows; ++row) {
80 for (int col = 0; col < cols; ++col) {
81 m_decisionVariables.emplace_back();
82 vars(row, col) = m_decisionVariables.back();
83 }
84 }
85
86 return vars;
87 }
88
89 /**
90 * Create a symmetric matrix of decision variables in the optimization
91 * problem.
92 *
93 * Variable instances are reused across the diagonal, which helps reduce
94 * problem dimensionality.
95 *
96 * @param rows Number of matrix rows.
97 */
98 [[nodiscard]]
99 VariableMatrix SymmetricDecisionVariable(int rows) {
100 // We only need to store the lower triangle of an n x n symmetric matrix;
101 // the other elements are duplicates. The lower triangle has (n² + n)/2
102 // elements.
103 //
104 // n
105 // Σ k = (n² + n)/2
106 // k=1
107 m_decisionVariables.reserve(m_decisionVariables.size() +
108 (rows * rows + rows) / 2);
109
110 VariableMatrix vars{rows, rows};
111
112 for (int row = 0; row < rows; ++row) {
113 for (int col = 0; col <= row; ++col) {
114 m_decisionVariables.emplace_back();
115 vars(row, col) = m_decisionVariables.back();
116 vars(col, row) = m_decisionVariables.back();
117 }
118 }
119
120 return vars;
121 }
122
123 /**
124 * Tells the solver to minimize the output of the given cost function.
125 *
126 * Note that this is optional. If only constraints are specified, the solver
127 * will find the closest solution to the initial conditions that's in the
128 * feasible set.
129 *
130 * @param cost The cost function to minimize.
131 */
132 void Minimize(const Variable& cost) {
133 m_f = cost;
134 status.costFunctionType = m_f.value().Type();
135 }
136
137 /**
138 * Tells the solver to minimize the output of the given cost function.
139 *
140 * Note that this is optional. If only constraints are specified, the solver
141 * will find the closest solution to the initial conditions that's in the
142 * feasible set.
143 *
144 * @param cost The cost function to minimize.
145 */
146 void Minimize(Variable&& cost) {
147 m_f = std::move(cost);
148 status.costFunctionType = m_f.value().Type();
149 }
150
151 /**
152 * Tells the solver to maximize the output of the given objective function.
153 *
154 * Note that this is optional. If only constraints are specified, the solver
155 * will find the closest solution to the initial conditions that's in the
156 * feasible set.
157 *
158 * @param objective The objective function to maximize.
159 */
160 void Maximize(const Variable& objective) {
161 // Maximizing a cost function is the same as minimizing its negative
162 m_f = -objective;
163 status.costFunctionType = m_f.value().Type();
164 }
165
166 /**
167 * Tells the solver to maximize the output of the given objective function.
168 *
169 * Note that this is optional. If only constraints are specified, the solver
170 * will find the closest solution to the initial conditions that's in the
171 * feasible set.
172 *
173 * @param objective The objective function to maximize.
174 */
175 void Maximize(Variable&& objective) {
176 // Maximizing a cost function is the same as minimizing its negative
177 m_f = -std::move(objective);
178 status.costFunctionType = m_f.value().Type();
179 }

181 /**
182 * Tells the solver to solve the problem while satisfying the given equality
183 * constraint.
184 *
185 * @param constraint The constraint to satisfy.
186 */
187 void SubjectTo(const EqualityConstraints& constraint) {
188 // Get the highest order equality constraint expression type
189 for (const auto& c : constraint.constraints) {
190 status.equalityConstraintType =
191 std::max(status.equalityConstraintType, c.Type());
192 }
193
194 m_equalityConstraints.reserve(m_equalityConstraints.size() +
195 constraint.constraints.size());
196 std::copy(constraint.constraints.begin(), constraint.constraints.end(),
197 std::back_inserter(m_equalityConstraints));
198 }
199
200 /**
201 * Tells the solver to solve the problem while satisfying the given equality
202 * constraint.
203 *
204 * @param constraint The constraint to satisfy.
205 */
206 void SubjectTo(EqualityConstraints&& constraint) {
207 // Get the highest order equality constraint expression type
208 for (const auto& c : constraint.constraints) {
209 status.equalityConstraintType =
210 std::max(status.equalityConstraintType, c.Type());
211 }
212
213 m_equalityConstraints.reserve(m_equalityConstraints.size() +
214 constraint.constraints.size());
215 std::copy(constraint.constraints.begin(), constraint.constraints.end(),
216 std::back_inserter(m_equalityConstraints));
217 }

244 void SubjectTo(InequalityConstraints&& constraint) {
245 // Get the highest order inequality constraint expression type

248 std::max(status.inequalityConstraintType, c.Type());

257 /**
258 * Solve the optimization problem. The solution will be stored in the original
259 * variables used to construct the problem.
260 *
261 * @param config Configuration options for the solver.
262 */
263 SolverStatus Solve(const SolverConfig& config = SolverConfig{}) {
264 // Create the initial value column vector
265 Eigen::VectorXd x{m_decisionVariables.size()};
266 for (size_t i = 0; i < m_decisionVariables.size(); ++i) {
267 x(i) = m_decisionVariables[i].Value();
268 }
269
270 status.exitCondition = SolverExitCondition::kSuccess;
271
272 // If there's no cost function, make it zero and continue
273 if (!m_f.has_value()) {
274 m_f = Variable();
275 }
276
277 if (config.diagnostics) {
278 constexpr std::array kExprTypeToName{"empty", "constant", "linear",
279 "quadratic", "nonlinear"};
280
281 // Print cost function and constraint expression types
282 sleipnir::println(
283 "The cost function is {}.",
284 kExprTypeToName[static_cast<int>(status.costFunctionType)]);
285 sleipnir::println(
286 "The equality constraints are {}.",
287 kExprTypeToName[static_cast<int>(status.equalityConstraintType)]);
288 sleipnir::println(
289 "The inequality constraints are {}.",
290 kExprTypeToName[static_cast<int>(status.inequalityConstraintType)]);
291 sleipnir::println("");
292
293 // Print problem dimensionality
294 sleipnir::println("Number of decision variables: {}",
295 m_decisionVariables.size());
296 sleipnir::println("Number of equality constraints: {}",
297 m_equalityConstraints.size());
298 sleipnir::println("Number of inequality constraints: {}\n",
299 m_inequalityConstraints.size());
300 }
301
302 // If the problem is empty or constant, there's nothing to do
303 if (status.costFunctionType <= ExpressionType::kConstant &&
304 status.equalityConstraintType <= ExpressionType::kConstant &&
305 status.inequalityConstraintType <= ExpressionType::kConstant) {
306 return status;
307 }
308
309 // Solve the optimization problem
310 Eigen::VectorXd s = Eigen::VectorXd::Ones(m_inequalityConstraints.size());
311 InteriorPoint(m_decisionVariables, m_equalityConstraints,
312 m_inequalityConstraints, m_f.value(), m_callback, config,
313 false, x, s, &status);
314
315 if (config.diagnostics) {
316 sleipnir::println("Exit condition: {}", ToMessage(status.exitCondition));
317 }
318
319 // Assign the solution to the original Variable instances
320 VariableMatrix{m_decisionVariables}.SetValue(x);
321
322 return status;
323 }

325 /**
326 * Sets a callback to be called at each solver iteration.
327 *
328 * The callback for this overload should return void.
329 *
330 * @param callback The callback.
331 */
332 template <typename F>
333 requires std::invocable<F, const SolverIterationInfo&> &&
334 std::same_as<std::invoke_result_t<F, const SolverIterationInfo&>,
335 void>
336 void Callback(F&& callback) {
337 m_callback = [=, callback = std::forward<F>(callback)](
338 const SolverIterationInfo& info) {
339 callback(info);
340 return false;
341 };
342 }
343
344 /**
345 * Sets a callback to be called at each solver iteration.
346 *
347 * The callback for this overload should return bool.
348 *
349 * @param callback The callback. Returning true from the callback causes the
350 * solver to exit early with the solution it has so far.
351 */
352 template <typename F>
353 requires std::invocable<F, const SolverIterationInfo&> &&
354 std::same_as<std::invoke_result_t<F, const SolverIterationInfo&>,
355 bool>
356 void Callback(F&& callback) {
357 m_callback = std::forward<F>(callback);
358 }
359
360 private:
361 // The list of decision variables, which are the root of the problem's
362 // expression tree
363 wpi::SmallVector<Variable> m_decisionVariables;
364
365 // The cost function: f(x)
366 std::optional<Variable> m_f;
367
368 // The list of equality constraints: cₑ(x) = 0
369 wpi::SmallVector<Variable> m_equalityConstraints;
370
371 // The list of inequality constraints: cᵢ(x) ≥ 0
372 wpi::SmallVector<Variable> m_inequalityConstraints;
373
374 // The user callback
375 std::function<bool(const SolverIterationInfo&)> m_callback =
376 [](const SolverIterationInfo&) { return false; };
377
378 // The solver status
379 SolverStatus status;
380};
