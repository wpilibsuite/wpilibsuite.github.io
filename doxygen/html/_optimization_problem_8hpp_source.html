28/**
29 * This class allows the user to pose a constrained nonlinear optimization
30 * problem in natural mathematical notation and solve it.
31 *
32 * This class supports problems of the form:
33@verbatim
34 minₓ f(x)
35subject to cₑ(x) = 0
36 cᵢ(x) ≥ 0
37@endverbatim
38 *
39 * where f(x) is the scalar cost function, x is the vector of decision variables
40 * (variables the solver can tweak to minimize the cost function), cᵢ(x) are the
41 * inequality constraints, and cₑ(x) are the equality constraints. Constraints
42 * are equations or inequalities of the decision variables that constrain what
43 * values the solver is allowed to use when searching for an optimal solution.
44 *
45 * The nice thing about this class is users don't have to put their system in
46 * the form shown above manually; they can write it in natural mathematical form
47 * and it'll be converted for them.
48 */
49class SLEIPNIR_DLLEXPORT OptimizationProblem {
50 public:
51 /**
52 * Construct the optimization problem.
53 */
54 OptimizationProblem() noexcept = default;
55
56 /**
57 * Create a decision variable in the optimization problem.
58 */
59 [[nodiscard]]
60 Variable DecisionVariable() {
61 m_decisionVariables.emplace_back();
62 return m_decisionVariables.back();
63 }
64
65 /**
66 * Create a matrix of decision variables in the optimization problem.
67 *
68 * @param rows Number of matrix rows.
69 * @param cols Number of matrix columns.
70 */
71 [[nodiscard]]
72 VariableMatrix DecisionVariable(int rows, int cols = 1) {
73 m_decisionVariables.reserve(m_decisionVariables.size() + rows * cols);
74
75 VariableMatrix vars{rows, cols};
76
77 for (int row = 0; row < rows; ++row) {
78 for (int col = 0; col < cols; ++col) {
79 m_decisionVariables.emplace_back();
80 vars(row, col) = m_decisionVariables.back();
81 }
82 }
83
84 return vars;
85 }
86
87 /**
88 * Create a symmetric matrix of decision variables in the optimization
89 * problem.
90 *
91 * Variable instances are reused across the diagonal, which helps reduce
92 * problem dimensionality.
93 *
94 * @param rows Number of matrix rows.
95 */
96 [[nodiscard]]
97 VariableMatrix SymmetricDecisionVariable(int rows) {
98 // We only need to store the lower triangle of an n x n symmetric matrix;
99 // the other elements are duplicates. The lower triangle has (n² + n)/2
100 // elements.
101 //
102 // n
103 // Σ k = (n² + n)/2
104 // k=1
105 m_decisionVariables.reserve(m_decisionVariables.size() +
106 (rows * rows + rows) / 2);
107
108 VariableMatrix vars{rows, rows};
109
110 for (int row = 0; row < rows; ++row) {
111 for (int col = 0; col <= row; ++col) {
112 m_decisionVariables.emplace_back();
113 vars(row, col) = m_decisionVariables.back();
114 vars(col, row) = m_decisionVariables.back();
115 }
116 }
117
118 return vars;
119 }
120
121 /**
122 * Tells the solver to minimize the output of the given cost function.
123 *
124 * Note that this is optional. If only constraints are specified, the solver
125 * will find the closest solution to the initial conditions that's in the
126 * feasible set.
127 *
128 * @param cost The cost function to minimize.
129 */
130 void Minimize(const Variable& cost) {
131 m_f = cost;
132 status.costFunctionType = m_f.value().Type();
133 }
134
135 /**
136 * Tells the solver to minimize the output of the given cost function.
137 *
138 * Note that this is optional. If only constraints are specified, the solver
139 * will find the closest solution to the initial conditions that's in the
140 * feasible set.
141 *
142 * @param cost The cost function to minimize.
143 */
144 void Minimize(Variable&& cost) {
145 m_f = std::move(cost);
146 status.costFunctionType = m_f.value().Type();
147 }
148
149 /**
150 * Tells the solver to maximize the output of the given objective function.
151 *
152 * Note that this is optional. If only constraints are specified, the solver
153 * will find the closest solution to the initial conditions that's in the
154 * feasible set.
155 *
156 * @param objective The objective function to maximize.
157 */
158 void Maximize(const Variable& objective) {
159 // Maximizing a cost function is the same as minimizing its negative
160 m_f = -objective;
161 status.costFunctionType = m_f.value().Type();
162 }
163
164 /**
165 * Tells the solver to maximize the output of the given objective function.
166 *
167 * Note that this is optional. If only constraints are specified, the solver
168 * will find the closest solution to the initial conditions that's in the
169 * feasible set.
170 *
171 * @param objective The objective function to maximize.
172 */
173 void Maximize(Variable&& objective) {
174 // Maximizing a cost function is the same as minimizing its negative
175 m_f = -std::move(objective);
176 status.costFunctionType = m_f.value().Type();
177 }

179 /**
180 * Tells the solver to solve the problem while satisfying the given equality
181 * constraint.
182 *
183 * @param constraint The constraint to satisfy.
184 */
185 void SubjectTo(const EqualityConstraints& constraint) {
186 // Get the highest order equality constraint expression type
187 for (const auto& c : constraint.constraints) {
188 status.equalityConstraintType =
189 std::max(status.equalityConstraintType, c.Type());
190 }
191
192 m_equalityConstraints.reserve(m_equalityConstraints.size() +
193 constraint.constraints.size());
194 std::copy(constraint.constraints.begin(), constraint.constraints.end(),
195 std::back_inserter(m_equalityConstraints));
196 }
197
198 /**
199 * Tells the solver to solve the problem while satisfying the given equality
200 * constraint.
201 *
202 * @param constraint The constraint to satisfy.
203 */
204 void SubjectTo(EqualityConstraints&& constraint) {
205 // Get the highest order equality constraint expression type
206 for (const auto& c : constraint.constraints) {
207 status.equalityConstraintType =
208 std::max(status.equalityConstraintType, c.Type());
209 }
210
211 m_equalityConstraints.reserve(m_equalityConstraints.size() +

242 void SubjectTo(InequalityConstraints&& constraint) {
243 // Get the highest order inequality constraint expression type
244 for (const auto& c : constraint.constraints) {
245 status.inequalityConstraintType =

255 /**
256 * Solve the optimization problem. The solution will be stored in the original
257 * variables used to construct the problem.
258 *
259 * @param config Configuration options for the solver.
260 */
261 SolverStatus Solve(const SolverConfig& config = SolverConfig{}) {
262 // Create the initial value column vector
263 Eigen::VectorXd x{m_decisionVariables.size()};
264 for (size_t i = 0; i < m_decisionVariables.size(); ++i) {
265 x(i) = m_decisionVariables[i].Value();
266 }
267
268 status.exitCondition = SolverExitCondition::kSuccess;
269
270 // If there's no cost function, make it zero and continue
271 if (!m_f.has_value()) {
272 m_f = Variable();
273 }
274
275 if (config.diagnostics) {
276 constexpr std::array kExprTypeToName{"empty", "constant", "linear",
277 "quadratic", "nonlinear"};
278
279 // Print cost function and constraint expression types
280 sleipnir::println(
281 "The cost function is {}.",
282 kExprTypeToName[static_cast<int>(status.costFunctionType)]);
283 sleipnir::println(
284 "The equality constraints are {}.",
285 kExprTypeToName[static_cast<int>(status.equalityConstraintType)]);
286 sleipnir::println(
287 "The inequality constraints are {}.",
288 kExprTypeToName[static_cast<int>(status.inequalityConstraintType)]);
289 sleipnir::println("");
290
291 // Print problem dimensionality
292 sleipnir::println("Number of decision variables: {}",
293 m_decisionVariables.size());
294 sleipnir::println("Number of equality constraints: {}",
295 m_equalityConstraints.size());
296 sleipnir::println("Number of inequality constraints: {}\n",
297 m_inequalityConstraints.size());
298 }
299
300 // If the problem is empty or constant, there's nothing to do
301 if (status.costFunctionType <= ExpressionType::kConstant &&
302 status.equalityConstraintType <= ExpressionType::kConstant &&
303 status.inequalityConstraintType <= ExpressionType::kConstant) {
304 return status;
305 }
306
307 // Solve the optimization problem
308 Eigen::VectorXd s = Eigen::VectorXd::Ones(m_inequalityConstraints.size());
309 InteriorPoint(m_decisionVariables, m_equalityConstraints,
310 m_inequalityConstraints, m_f.value(), m_callback, config,
311 false, x, s, &status);
312
313 if (config.diagnostics) {
314 sleipnir::println("Exit condition: {}", ToMessage(status.exitCondition));
315 }
316
317 // Assign the solution to the original Variable instances
318 VariableMatrix{m_decisionVariables}.SetValue(x);
319
320 return status;
321 }

323 /**
324 * Sets a callback to be called at each solver iteration.
325 *
326 * The callback for this overload should return void.
327 *
328 * @param callback The callback.
329 */
330 template <typename F>
331 requires requires(F callback, const SolverIterationInfo& info) {
332 { callback(info) } -> std::same_as<void>;
333 }
334 void Callback(F&& callback) {
335 m_callback = [=, callback = std::forward<F>(callback)](
336 const SolverIterationInfo& info) {
337 callback(info);
338 return false;
339 };
340 }
341
342 /**
343 * Sets a callback to be called at each solver iteration.
344 *
345 * The callback for this overload should return bool.
346 *
347 * @param callback The callback. Returning true from the callback causes the
348 * solver to exit early with the solution it has so far.
349 */
350 template <typename F>
351 requires requires(F callback, const SolverIterationInfo& info) {
352 { callback(info) } -> std::same_as<bool>;
353 }
354 void Callback(F&& callback) {
355 m_callback = std::forward<F>(callback);
356 }
357
358 private:
359 // The list of decision variables, which are the root of the problem's
360 // expression tree
361 wpi::SmallVector<Variable> m_decisionVariables;
362
363 // The cost function: f(x)
364 std::optional<Variable> m_f;
365
366 // The list of equality constraints: cₑ(x) = 0
367 wpi::SmallVector<Variable> m_equalityConstraints;
368
369 // The list of inequality constraints: cᵢ(x) ≥ 0
370 wpi::SmallVector<Variable> m_inequalityConstraints;
371
372 // The user callback
373 std::function<bool(const SolverIterationInfo& info)> m_callback =
374 [](const SolverIterationInfo&) { return false; };
375
376 // The solver status
377 SolverStatus status;
378};
