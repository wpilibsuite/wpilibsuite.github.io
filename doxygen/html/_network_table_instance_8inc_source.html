268template <wpi::ProtobufSerializable T>
269void NetworkTableInstance::AddProtobufSchema(wpi::ProtobufMessage<T>& msg) {
270 msg.ForEachProtobufDescriptor(
271 [this](auto typeString) { return HasSchema(typeString); },
272 [this](auto typeString, auto schema) {
273 AddSchema(typeString, "proto:FileDescriptorProto", schema);
274 });
275}
