19/**
20 * Persistent alert to be sent via NetworkTables. Alerts are tagged with a type
21 * of kError, kWarning, or kInfo to denote urgency. See Alert::AlertType for
22 * suggested usage of each type. Alerts can be displayed on supported
23 * dashboards, and are shown in a priority order based on type and recency of
24 * activation.
25 *
26 * Alerts should be created once and stored persistently, then updated to
27 * "active" or "inactive" as necessary. Set(bool) can be safely called
28 * periodically.
29 *
30 * This API is new for 2025, but is likely to change in future seasons to
31 * facilitate deeper integration with the robot control system.
32 *
33 * <pre>
34 * class Robot {
35 * frc::Alert alert{"Something went wrong", frc::Alert::AlertType::kWarning};
36 * }
37 *
38 * Robot::periodic() {
39 * alert.Set(...);
40 * }
41 * </pre>
42 */
43class Alert {
44 public:
45 /**
46 * Represents an alert's level of urgency.
47 */
48 enum class AlertType {
49 /**
50 * High priority alert - displayed first on the dashboard with a red "X"
51 * symbol. Use this type for problems which will seriously affect the
52 * robot's functionality and thus require immediate attention.
53 */
54 kError,
55
56 /**
57 * Medium priority alert - displayed second on the dashboard with a yellow
58 * "!" symbol. Use this type for problems which could affect the robot's
59 * functionality but do not necessarily require immediate attention.
60 */
61 kWarning,
62
63 /**
64 * Low priority alert - displayed last on the dashboard with a green "i"
65 * symbol. Use this type for problems which are unlikely to affect the
66 * robot's functionality, or any other alerts which do not fall under the
67 * other categories.
68 */
69 kInfo
70 };
71
72 /**
73 * Creates a new alert in the default group - "Alerts". If this is the first
74 * to be instantiated, the appropriate entries will be added to NetworkTables.
75 *
76 * @param text Text to be displayed when the alert is active.
77 * @param type Alert urgency level.
78 */
79 Alert(std::string_view text, AlertType type);
80
81 /**
82 * Creates a new alert. If this is the first to be instantiated in its group,
83 * the appropriate entries will be added to NetworkTables.
84 *
85 * @param group Group identifier, used as the entry name in NetworkTables.
86 * @param text Text to be displayed when the alert is active.
87 * @param type Alert urgency level.
88 */
89 Alert(std::string_view group, std::string_view text, AlertType type);
90
91 /**
92 * Sets whether the alert should currently be displayed. This method can be
93 * safely called periodically.
94 *
95 * @param active Whether to display the alert.
96 */
97 void Set(bool active);
98
99 /**
100 * Updates current alert text. Use this method to dynamically change the
101 * displayed alert, such as including more details about the detected problem.
102 *
103 * @param text Text to be displayed when the alert is active.
104 */
105 void SetText(std::string_view text);
106
107 private:
108 AlertType m_type;
109 bool m_active = false;
110 units::second_t m_activeStartTime;
111 std::string_view m_text;
112

118
119 private:
120 std::vector<std::string> GetStrings(AlertType type) const;
121 };
122
123 static wpi::StringMap<SendableAlerts> groups;
124};
