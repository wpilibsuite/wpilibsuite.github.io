12/**
13 * A Exponential-shaped velocity profile.
14 *
15 * While this class can be used for a profiled movement from start to finish,
16 * the intended usage is to filter a reference's dynamics based on
17 * ExponentialProfile velocity constraints. To compute the reference obeying
18 * this constraint, do the following.
19 *
20 * Initialization:
21 * @code{.cpp}
22 * ExponentialProfile::Constraints constraints{kMaxV, kV, kA};
23 * State previousProfiledReference = {initialReference, 0_mps};
24 * @endcode
25 *
26 * Run on update:
27 * @code{.cpp}
28 * previousProfiledReference = profile.Calculate(timeSincePreviousUpdate,
29 * previousProfiledReference, unprofiledReference);
30 * @endcode
31 *
32 * where `unprofiledReference` is free to change between calls. Note that when
33 * the unprofiled reference is within the constraints, `Calculate()` returns the
34 * unprofiled reference unchanged.
35 *
36 * Otherwise, a timer can be started to provide monotonic values for
37 * `Calculate()` and to determine when the profile has completed via
38 * `IsFinished()`.
39 */
40template <class Distance, class Input>
41class ExponentialProfile {
42 public:
43 using Distance_t = units::unit_t<Distance>;
44 using Velocity =
45 units::compound_unit<Distance, units::inverse<units::seconds>>;
46 using Velocity_t = units::unit_t<Velocity>;
47 using Acceleration =
48 units::compound_unit<Velocity, units::inverse<units::seconds>>;
49 using Input_t = units::unit_t<Input>;
50 using A_t = units::unit_t<units::inverse<units::seconds>>;
51 using B_t =
52 units::unit_t<units::compound_unit<Acceleration, units::inverse<Input>>>;
53 using KV = units::compound_unit<Input, units::inverse<Velocity>>;
54 using kV_t = units::unit_t<KV>;
55 using KA = units::compound_unit<Input, units::inverse<Acceleration>>;
56 using kA_t = units::unit_t<KA>;

79 * Profile constraints.

82 public:
83 /**
84 * Constructs constraints for an ExponentialProfile.
85 *
86 * @param maxInput maximum unsigned input voltage
87 * @param A The State-Space 1x1 system matrix.
88 * @param B The State-Space 1x1 input matrix.

93 /**
94 * Constructs constraints for an ExponentialProfile from characteristics.
95 *
96 * @param maxInput maximum unsigned input voltage
97 * @param kV The velocity gain.
98 * @param kA The acceleration gain.

100 Constraints(Input_t maxInput, kV_t kV, kA_t kA)
101 : maxInput{maxInput}, A{-kV / kA}, B{1 / kA} {}

104 * Computes the max achievable velocity for an Exponential Profile.
105 *
106 * @return The steady-state velocity achieved by this profile.

111 Input_t maxInput{0};

113 /// The State-Space 1x1 system matrix.
114 A_t A{0};
115
116 /// The State-Space 1x1 input matrix.
117 B_t B{0};
118 };

145 /**
146 * Calculates the position and velocity for the profile at a time t where the
147 * current state is at time t = 0.
148 *
149 * @param t How long to advance from the current state toward the desired
150 * state.
151 * @param current The current state.
152 * @param goal The desired state when the profile is complete.
153 * @return The position and velocity of the profile at time t.
154 */
155 State Calculate(const units::second_t& t, const State& current,
156 const State& goal) const {
157 auto direction = ShouldFlipInput(current, goal) ? -1 : 1;
158 auto u = direction * m_constraints.maxInput;
159
160 auto inflectionPoint = CalculateInflectionPoint(current, goal, u);
161 auto timing = CalculateProfileTiming(current, inflectionPoint, goal, u);
162
163 if (t < 0_s) {
164 return current;
165 } else if (t < timing.inflectionTime) {
166 return {ComputeDistanceFromTime(t, u, current),
167 ComputeVelocityFromTime(t, u, current)};
168 } else if (t < timing.totalTime) {
169 return {ComputeDistanceFromTime(t - timing.totalTime, -u, goal),
170 ComputeVelocityFromTime(t - timing.totalTime, -u, goal)};
171 } else {
172 return goal;
173 }
174 }
175
176 /**
177 * Calculates the point after which the fastest way to reach the goal state is
178 * to apply input in the opposite direction.
179 *
180 * @param current The current state.
181 * @param goal The desired state when the profile is complete.
182 * @return The position and velocity of the profile at the inflection point.
183 */
184 State CalculateInflectionPoint(const State& current,
185 const State& goal) const {
186 auto direction = ShouldFlipInput(current, goal) ? -1 : 1;
187 auto u = direction * m_constraints.maxInput;
188
189 return CalculateInflectionPoint(current, goal, u);
190 }

192 /**
193 * Calculates the time it will take for this profile to reach the goal state.
194 *
195 * @param current The current state.
196 * @param goal The desired state when the profile is complete.
197 * @return The total duration of this profile.
198 */
199 units::second_t TimeLeftUntil(const State& current, const State& goal) const {

201

205 /**
206 * Calculates the time it will take for this profile to reach the inflection
207 * point, and the time it will take for this profile to reach the goal state.
208 *
209 * @param current The current state.
210 * @param goal The desired state when the profile is complete.
211 * @return The timing information for this profile.
212 */
213 ProfileTiming CalculateProfileTiming(const State& current,
214 const State& goal) const {
215 auto direction = ShouldFlipInput(current, goal) ? -1 : 1;
216 auto u = direction * m_constraints.maxInput;
217
218 auto inflectionPoint = CalculateInflectionPoint(current, goal, u);
219 return CalculateProfileTiming(current, inflectionPoint, goal, u);
220 }
221
222 private:
223 /**
224 * Calculates the point after which the fastest way to reach the goal state is
225 * to apply input in the opposite direction.
226 *
227 * @param current The current state.
228 * @param goal The desired state when the profile is complete.
229 * @param input The signed input applied to this profile from the current
230 * state.
231 * @return The position and velocity of the profile at the inflection point.
232 */
233 State CalculateInflectionPoint(const State& current, const State& goal,
234 const Input_t& input) const {
235 auto u = input;
236
237 if (current == goal) {
238 return current;
239 }
240
241 auto inflectionVelocity = SolveForInflectionVelocity(u, current, goal);
242 auto inflectionPosition =
243 ComputeDistanceFromVelocity(inflectionVelocity, -u, goal);
244
245 return {inflectionPosition, inflectionVelocity};
246 }
247
248 /**
249 * Calculates the time it will take for this profile to reach the inflection
250 * point, and the time it will take for this profile to reach the goal state.
251 *
252 * @param current The current state.
253 * @param inflectionPoint The inflection point of this profile.
254 * @param goal The desired state when the profile is complete.
255 * @param input The signed input applied to this profile from the current
256 * state.
257 * @return The timing information for this profile.
258 */
259 ProfileTiming CalculateProfileTiming(const State& current,
260 const State& inflectionPoint,
261 const State& goal,
262 const Input_t& input) const {
263 auto u = input;
264 auto u_dir = units::math::abs(u) / u;
265
266 units::second_t inflectionT_forward;
267
268 // We need to handle 5 cases here:
269 //
270 // - Approaching -maxVelocity from below
271 // - Approaching -maxVelocity from above
272 // - Approaching maxVelocity from below
273 // - Approaching maxVelocity from above
274 // - At +-maxVelocity
275 //
276 // For cases 1 and 3, we want to subtract epsilon from the inflection point
277 // velocity For cases 2 and 4, we want to add epsilon to the inflection
278 // point velocity. For case 5, we have reached inflection point velocity.
279 auto epsilon = Velocity_t(1e-9);
280 if (units::math::abs(u_dir * m_constraints.MaxVelocity() -
281 inflectionPoint.velocity) < epsilon) {
282 auto solvableV = inflectionPoint.velocity;
283 units::second_t t_to_solvable_v;
284 Distance_t x_at_solvable_v;
285 if (units::math::abs(current.velocity - inflectionPoint.velocity) <
286 epsilon) {
287 t_to_solvable_v = 0_s;
288 x_at_solvable_v = current.position;
289 } else {
290 if (units::math::abs(current.velocity) > m_constraints.MaxVelocity()) {
291 solvableV += u_dir * epsilon;
292 } else {
293 solvableV -= u_dir * epsilon;
294 }
295
296 t_to_solvable_v =
297 ComputeTimeFromVelocity(solvableV, u, current.velocity);
298 x_at_solvable_v = ComputeDistanceFromVelocity(solvableV, u, current);
299 }
300
301 inflectionT_forward =
302 t_to_solvable_v + u_dir *
303 (inflectionPoint.position - x_at_solvable_v) /
304 m_constraints.MaxVelocity();
305 } else {
306 inflectionT_forward = ComputeTimeFromVelocity(inflectionPoint.velocity, u,
307 current.velocity);
308 }
309
310 auto inflectionT_backward =
311 ComputeTimeFromVelocity(inflectionPoint.velocity, -u, goal.velocity);
312
313 return {inflectionT_forward, inflectionT_forward - inflectionT_backward};
314 }
315
316 /**
317 * Calculates the position reached after t seconds when applying an input from
318 * the initial state.
319 *
320 * @param t The time since the initial state.
321 * @param input The signed input applied to this profile from the initial
322 * state.
323 * @param initial The initial state.
324 * @return The distance travelled by this profile.
325 */
326 Distance_t ComputeDistanceFromTime(const units::second_t& time,
327 const Input_t& input,
328 const State& initial) const {
329 auto A = m_constraints.A;
330 auto B = m_constraints.B;
331 auto u = input;
332
333 return initial.position +
334 (-B * u * time +
335 (initial.velocity + B * u / A) * (units::math::exp(A * time) - 1)) /
336 A;
337 }
338
339 /**
340 * Calculates the velocity reached after t seconds when applying an input from
341 * the initial state.
342 *
343 * @param t The time since the initial state.
344 * @param input The signed input applied to this profile from the initial
345 * state.
346 * @param initial The initial state.
347 * @return The distance travelled by this profile.
348 */
349 Velocity_t ComputeVelocityFromTime(const units::second_t& time,
350 const Input_t& input,
351 const State& initial) const {
352 auto A = m_constraints.A;
353 auto B = m_constraints.B;
354 auto u = input;
355
356 return (initial.velocity + B * u / A) * units::math::exp(A * time) -
357 B * u / A;
358 }
359
360 /**
361 * Calculates the time required to reach a specified velocity given the
362 * initial velocity.
363 *
364 * @param velocity The goal velocity.
365 * @param input The signed input applied to this profile from the initial
366 * state.
367 * @param initial The initial velocity.
368 * @return The time required to reach the goal velocity.
369 */
370 units::second_t ComputeTimeFromVelocity(const Velocity_t& velocity,
371 const Input_t& input,
372 const Velocity_t& initial) const {
373 auto A = m_constraints.A;
374 auto B = m_constraints.B;
375 auto u = input;
376
377 return units::math::log((A * velocity + B * u) / (A * initial + B * u)) / A;
378 }
379
380 /**
381 * Calculates the distance reached at the same time as the given velocity when
382 * applying the given input from the initial state.
383 *
384 * @param velocity The velocity reached by this profile
385 * @param input The signed input applied to this profile from the initial
386 * state.
387 * @param initial The initial state.
388 * @return The distance reached when the given velocity is reached.
389 */
390 Distance_t ComputeDistanceFromVelocity(const Velocity_t& velocity,
391 const Input_t& input,
392 const State& initial) const {
393 auto A = m_constraints.A;
394 auto B = m_constraints.B;
395 auto u = input;
396
397 return initial.position + (velocity - initial.velocity) / A -
398 B * u / (A * A) *
399 units::math::log((A * velocity + B * u) /
400 (A * initial.velocity + B * u));
401 }
402
403 /**
404 * Calculates the velocity at which input should be reversed in order to reach
405 * the goal state from the current state.
406 *
407 * @param input The signed input applied to this profile from the current
408 * state.
409 * @param current The current state.
410 * @param goal The goal state.
411 * @return The inflection velocity.
412 */
413 Velocity_t SolveForInflectionVelocity(const Input_t& input,
414 const State& current,
415 const State& goal) const {
416 auto A = m_constraints.A;
417 auto B = m_constraints.B;
418 auto u = input;
419
420 auto u_dir = u / units::math::abs(u);
421
422 auto position_delta = goal.position - current.position;
423 auto velocity_delta = goal.velocity - current.velocity;
424
425 auto scalar = (A * current.velocity + B * u) * (A * goal.velocity - B * u);
426 auto power = -A / B / u * (A * position_delta - velocity_delta);
427
428 auto a = -A * A;
429 auto c = B * B * u * u + scalar * units::math::exp(power);
430
431 if (-1e-9 < c.value() && c.value() < 0) {
432 // numeric instability - the heuristic gets it right but c is around
433 // -1e-13
434 return Velocity_t(0);
435 }
436
437 return u_dir * units::math::sqrt(-c / a);
438 }
439
440 /**
441 * Returns true if the profile should be inverted.
442 *
443 * The profile is inverted if we should first apply negative input in order to
444 * reach the goal state.
445 *
446 * @param current The initial state (usually the current state).
447 * @param goal The desired state when the profile is complete.
448 */
449 bool ShouldFlipInput(const State& current, const State& goal) const {
450 auto u = m_constraints.maxInput;
451
452 auto v0 = current.velocity;
453 auto xf = goal.position;
454 auto vf = goal.velocity;
455
456 auto x_forward = ComputeDistanceFromVelocity(vf, u, current);
457 auto x_reverse = ComputeDistanceFromVelocity(vf, -u, current);
458
459 if (v0 >= m_constraints.MaxVelocity()) {
460 return xf < x_reverse;
461 }
462
463 if (v0 <= -m_constraints.MaxVelocity()) {
464 return xf < x_forward;
465 }
466
467 auto a = v0 >= Velocity_t(0);
468 auto b = vf >= Velocity_t(0);
469 auto c = xf >= x_forward;
470 auto d = xf >= x_reverse;
471
472 return (a && !d) || (b && !c) || (!c && !d);
473 }
