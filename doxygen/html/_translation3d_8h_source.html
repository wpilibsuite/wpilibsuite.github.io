18/**
19 * Represents a translation in 3D space.
20 * This object can be used to represent a point or a vector.
21 *
22 * This assumes that you are using conventional mathematical axes. When the
23 * robot is at the origin facing in the positive X direction, forward is
24 * positive X, left is positive Y, and up is positive Z.
25 */
26class WPILIB_DLLEXPORT Translation3d {
27 public:
28 /**
29 * Constructs a Translation3d with X, Y, and Z components equal to zero.
30 */
31 constexpr Translation3d() = default;
32
33 /**
34 * Constructs a Translation3d with the X, Y, and Z components equal to the
35 * provided values.
36 *
37 * @param x The x component of the translation.
38 * @param y The y component of the translation.
39 * @param z The z component of the translation.
40 */
41 constexpr Translation3d(units::meter_t x, units::meter_t y, units::meter_t z)
42 : m_x{x}, m_y{y}, m_z{z} {}
43
44 /**
45 * Constructs a Translation3d with the provided distance and angle. This is
46 * essentially converting from polar coordinates to Cartesian coordinates.
47 *
48 * @param distance The distance from the origin to the end of the translation.
49 * @param angle The angle between the x-axis and the translation vector.
50 */
51 constexpr Translation3d(units::meter_t distance, const Rotation3d& angle) {
52 auto rectangular = Translation3d{distance, 0_m, 0_m}.RotateBy(angle);
53 m_x = rectangular.X();
54 m_y = rectangular.Y();
55 m_z = rectangular.Z();
56 }
57
58 /**
59 * Constructs a Translation3d from the provided translation vector's X, Y, and
60 * Z components. The values are assumed to be in meters.
61 *
62 * @param vector The translation vector to represent.
63 */
64 constexpr explicit Translation3d(const Eigen::Vector3d& vector)
65 : m_x{units::meter_t{vector.x()}},
66 m_y{units::meter_t{vector.y()}},
67 m_z{units::meter_t{vector.z()}} {}
68
69 /**
70 * Calculates the distance between two translations in 3D space.
71 *
72 * The distance between translations is defined as
73 * √((x₂−x₁)²+(y₂−y₁)²+(z₂−z₁)²).
74 *
75 * @param other The translation to compute the distance to.
76 *
77 * @return The distance between the two translations.
78 */
79 constexpr units::meter_t Distance(const Translation3d& other) const {
80 return units::math::sqrt(units::math::pow<2>(other.m_x - m_x) +
81 units::math::pow<2>(other.m_y - m_y) +
82 units::math::pow<2>(other.m_z - m_z));
83 }
84
85 /**
86 * Returns the X component of the translation.
87 *
88 * @return The Z component of the translation.
89 */
90 constexpr units::meter_t X() const { return m_x; }
91
92 /**
93 * Returns the Y component of the translation.
94 *
95 * @return The Y component of the translation.
96 */
97 constexpr units::meter_t Y() const { return m_y; }
98
99 /**
100 * Returns the Z component of the translation.
101 *
102 * @return The Z component of the translation.
103 */
104 constexpr units::meter_t Z() const { return m_z; }
105
106 /**
107 * Returns a vector representation of this translation.
108 *
109 * @return A Vector representation of this translation.
110 */
111 constexpr Eigen::Vector3d ToVector() const {
112 return Eigen::Vector3d{{m_x.value(), m_y.value(), m_z.value()}};
113 }
114
115 /**
116 * Returns the norm, or distance from the origin to the translation.
117 *
118 * @return The norm of the translation.
119 */
120 constexpr units::meter_t Norm() const {
121 return units::math::sqrt(m_x * m_x + m_y * m_y + m_z * m_z);
122 }
123
124 /**
125 * Applies a rotation to the translation in 3D space.
126 *
127 * For example, rotating a Translation3d of &lt;2, 0, 0&gt; by 90 degrees
128 * around the Z axis will return a Translation3d of &lt;0, 2, 0&gt;.
129 *
130 * @param other The rotation to rotate the translation by.
131 *
132 * @return The new rotated translation.
133 */
134 constexpr Translation3d RotateBy(const Rotation3d& other) const {
135 Quaternion p{0.0, m_x.value(), m_y.value(), m_z.value()};
136 auto qprime = other.GetQuaternion() * p * other.GetQuaternion().Inverse();
137 return Translation3d{units::meter_t{qprime.X()}, units::meter_t{qprime.Y()},
138 units::meter_t{qprime.Z()}};
139 }
140
141 /**
142 * Returns a Translation2d representing this Translation3d projected into the
143 * X-Y plane.
144 */
145 constexpr Translation2d ToTranslation2d() const {
146 return Translation2d{m_x, m_y};
147 }
148
149 /**
150 * Returns the sum of two translations in 3D space.
151 *
152 * For example, Translation3d{1.0, 2.5, 3.5} + Translation3d{2.0, 5.5, 7.5} =
153 * Translation3d{3.0, 8.0, 11.0}.
154 *
155 * @param other The translation to add.
156 *
157 * @return The sum of the translations.
158 */
159 constexpr Translation3d operator+(const Translation3d& other) const {
160 return {X() + other.X(), Y() + other.Y(), Z() + other.Z()};
161 }
162
163 /**
164 * Returns the difference between two translations.
165 *
166 * For example, Translation3d{5.0, 4.0, 3.0} - Translation3d{1.0, 2.0, 3.0} =
167 * Translation3d{4.0, 2.0, 0.0}.
168 *
169 * @param other The translation to subtract.
170 *
171 * @return The difference between the two translations.
172 */
173 constexpr Translation3d operator-(const Translation3d& other) const {
174 return operator+(-other);
175 }
176
177 /**
178 * Returns the inverse of the current translation. This is equivalent to
179 * negating all components of the translation.
180 *
181 * @return The inverse of the current translation.
182 */
183 constexpr Translation3d operator-() const { return {-m_x, -m_y, -m_z}; }
184
185 /**
186 * Returns the translation multiplied by a scalar.
187 *
188 * For example, Translation3d{2.0, 2.5, 4.5} * 2 = Translation3d{4.0, 5.0,
189 * 9.0}.
190 *
191 * @param scalar The scalar to multiply by.
192 *
193 * @return The scaled translation.
194 */
195 constexpr Translation3d operator*(double scalar) const {
196 return {scalar * m_x, scalar * m_y, scalar * m_z};
197 }
198
199 /**
200 * Returns the translation divided by a scalar.
201 *
202 * For example, Translation3d{2.0, 2.5, 4.5} / 2 = Translation3d{1.0, 1.25,
203 * 2.25}.
204 *
205 * @param scalar The scalar to divide by.
206 *
207 * @return The scaled translation.
208 */
209 constexpr Translation3d operator/(double scalar) const {
210 return operator*(1.0 / scalar);
211 }
212
213 /**
214 * Checks equality between this Translation3d and another object.
215 *
216 * @param other The other object.
217 * @return Whether the two objects are equal.
218 */
219 constexpr bool operator==(const Translation3d& other) const {
220 return units::math::abs(m_x - other.m_x) < 1E-9_m &&
221 units::math::abs(m_y - other.m_y) < 1E-9_m &&
222 units::math::abs(m_z - other.m_z) < 1E-9_m;
223 }
224
225 private:
226 units::meter_t m_x = 0_m;
227 units::meter_t m_y = 0_m;
228 units::meter_t m_z = 0_m;
229};
