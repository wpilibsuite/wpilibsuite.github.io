14template <typename Derived>
15HttpWebSocketServerConnection<Derived>::HttpWebSocketServerConnection(
16 std::shared_ptr<uv::Stream> stream,
17 std::span<const std::string_view> protocols)
18 : HttpServerConnection{stream},
19 m_helper{m_request},
20 m_protocols{protocols.begin(), protocols.end()} {
21 // Handle upgrade event
22 m_helper.upgrade.connect([this] {
23 // Negotiate sub-protocol
24 SmallVector<std::string_view, 2> protocols{m_protocols.begin(),
25 m_protocols.end()};
26 std::string_view protocol = m_helper.MatchProtocol(protocols).second;
27
28 // Check that the upgrade is valid
29 if (!IsValidWsUpgrade(protocol)) {
30 return;
31 }
32
33 // Disconnect HttpServerConnection header reader
34 m_dataConn.disconnect();
35 m_messageCompleteConn.disconnect();
36
37 // Accepting the stream may destroy this (as it replaces the stream user
38 // data), so grab a shared pointer first.
39 auto self = this->shared_from_this();
40
41 // Accept the upgrade
42 auto ws = m_helper.Accept(m_stream, protocol);
43
44 // Set this as the websocket user data to keep it around
45 ws->SetData(self);
46
47 // Store in member
48 m_websocket = ws.get();
49
50 // Call derived class function
51 ProcessWsUpgrade();
52 });
53}
54
55} // namespace wpi
