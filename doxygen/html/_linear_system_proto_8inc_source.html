23template <int States, int Inputs, int Outputs>
24frc::LinearSystem<States, Inputs, Outputs>
25wpi::Protobuf<frc::LinearSystem<States, Inputs, Outputs>>::Unpack(
26 const google::protobuf::Message& msg) {
27 auto m = static_cast<const wpi::proto::ProtobufLinearSystem*>(&msg);
28 if (m->num_states() != States || m->num_inputs() != Inputs ||
29 m->num_outputs() != Outputs) {
30 throw std::invalid_argument(fmt::format(
31 "Tried to unpack message with {} states and {} inputs and {} outputs "
32 "into LinearSystem with {} states and {} inputs and {} outputs",
33 m->num_states(), m->num_inputs(), m->num_outputs(), States, Inputs,
34 Outputs));
35 }
36 return frc::LinearSystem<States, Inputs, Outputs>{
37 wpi::UnpackProtobuf<frc::Matrixd<States, States>>(m->wpi_a()),
38 wpi::UnpackProtobuf<frc::Matrixd<States, Inputs>>(m->wpi_b()),
39 wpi::UnpackProtobuf<frc::Matrixd<Outputs, States>>(m->wpi_c()),
40 wpi::UnpackProtobuf<frc::Matrixd<Outputs, Inputs>>(m->wpi_d())};
41}

43template <int States, int Inputs, int Outputs>
44void wpi::Protobuf<frc::LinearSystem<States, Inputs, Outputs>>::Pack(
45 google::protobuf::Message* msg,
46 const frc::LinearSystem<States, Inputs, Outputs>& value) {

48 m->set_num_states(States);

50 m->set_num_outputs(Outputs);

52 wpi::PackProtobuf(m->mutable_b(), value.B());
53 wpi::PackProtobuf(m->mutable_c(), value.C());
54 wpi::PackProtobuf(m->mutable_d(), value.D());
55}
