16/**
17 * A helper class that converts Pose3d objects between different standard
18 * coordinate frames.
19 */
20class WPILIB_DLLEXPORT CoordinateSystem {
21 public:
22 /**
23 * Constructs a coordinate system with the given cardinal directions for each
24 * axis.
25 *
26 * @param positiveX The cardinal direction of the positive x-axis.
27 * @param positiveY The cardinal direction of the positive y-axis.
28 * @param positiveZ The cardinal direction of the positive z-axis.
29 * @throws std::domain_error if the coordinate system isn't special orthogonal
30 */
31 constexpr CoordinateSystem(const CoordinateAxis& positiveX,
32 const CoordinateAxis& positiveY,
33 const CoordinateAxis& positiveZ) {
34 // Construct a change of basis matrix from the source coordinate system to
35 // the NWU coordinate system. Each column vector in the change of basis
36 // matrix is one of the old basis vectors mapped to its representation in
37 // the new basis.
38 Eigen::Matrix3d R{{positiveX.m_axis.coeff(0), positiveY.m_axis.coeff(0),
39 positiveZ.m_axis.coeff(0)},
40 {positiveX.m_axis.coeff(1), positiveY.m_axis.coeff(1),
41 positiveZ.m_axis.coeff(1)},
42 {positiveX.m_axis.coeff(2), positiveY.m_axis.coeff(2),
43 positiveZ.m_axis.coeff(2)}};
44
45 // The change of basis matrix should be a pure rotation. The Rotation3d
46 // constructor will verify this by checking for special orthogonality.
47 m_rotation = Rotation3d{R};
48 }
49
50 /**
51 * Returns an instance of the North-West-Up (NWU) coordinate system.
52 *
53 * The +X axis is north, the +Y axis is west, and the +Z axis is up.
54 */
55 constexpr static CoordinateSystem NWU() {
56 return CoordinateSystem{CoordinateAxis::N(), CoordinateAxis::W(),
57 CoordinateAxis::U()};
58 }
59
60 /**
61 * Returns an instance of the East-Down-North (EDN) coordinate system.
62 *
63 * The +X axis is east, the +Y axis is down, and the +Z axis is north.
64 */
65 constexpr static CoordinateSystem EDN() {
66 return CoordinateSystem{CoordinateAxis::E(), CoordinateAxis::D(),
67 CoordinateAxis::N()};
68 }
69
70 /**
71 * Returns an instance of the NED coordinate system.
72 *
73 * The +X axis is north, the +Y axis is east, and the +Z axis is down.
74 */
75 constexpr static CoordinateSystem NED() {
76 return CoordinateSystem{CoordinateAxis::N(), CoordinateAxis::E(),
77 CoordinateAxis::D()};
78 }
79
80 /**
81 * Converts the given translation from one coordinate system to another.
82 *
83 * @param translation The translation to convert.
84 * @param from The coordinate system the translation starts in.
85 * @param to The coordinate system to which to convert.
86 * @return The given translation in the desired coordinate system.
87 */
88 constexpr static Translation3d Convert(const Translation3d& translation,
89 const CoordinateSystem& from,
90 const CoordinateSystem& to) {
91 return translation.RotateBy(from.m_rotation - to.m_rotation);
92 }
93
94 /**
95 * Converts the given rotation from one coordinate system to another.
96 *
97 * @param rotation The rotation to convert.
98 * @param from The coordinate system the rotation starts in.
99 * @param to The coordinate system to which to convert.
100 * @return The given rotation in the desired coordinate system.
101 */
102 constexpr static Rotation3d Convert(const Rotation3d& rotation,
103 const CoordinateSystem& from,
104 const CoordinateSystem& to) {
105 return rotation.RotateBy(from.m_rotation - to.m_rotation);
106 }
107
108 /**
109 * Converts the given pose from one coordinate system to another.
110 *
111 * @param pose The pose to convert.
112 * @param from The coordinate system the pose starts in.
113 * @param to The coordinate system to which to convert.
114 * @return The given pose in the desired coordinate system.
115 */
116 constexpr static Pose3d Convert(const Pose3d& pose,
117 const CoordinateSystem& from,
118 const CoordinateSystem& to) {
119 return Pose3d{Convert(pose.Translation(), from, to),
120 Convert(pose.Rotation(), from, to)};
121 }
122
123 /**
124 * Converts the given transform from one coordinate system to another.
125 *
126 * @param transform The transform to convert.
127 * @param from The coordinate system the transform starts in.
128 * @param to The coordinate system to which to convert.
129 * @return The given transform in the desired coordinate system.
130 */
131 constexpr static Transform3d Convert(const Transform3d& transform,
132 const CoordinateSystem& from,
133 const CoordinateSystem& to) {
134 const auto coordRot = from.m_rotation - to.m_rotation;
135 return Transform3d{
136 Convert(transform.Translation(), from, to),
137 (-coordRot).RotateBy(transform.Rotation().RotateBy(coordRot))};
138 }
139
140 private:
141 // Rotation from this coordinate system to NWU coordinate system
142 Rotation3d m_rotation;
143};
144
145} // namespace frc
