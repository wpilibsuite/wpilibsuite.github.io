23/**
24 * The SendableChooser class is a useful tool for presenting a selection of
25 * options to the SmartDashboard.
26 *
27 * For instance, you may wish to be able to select between multiple autonomous
28 * modes. You can do this by putting every possible Command you want to run as
29 * an autonomous into a SendableChooser and then put it into the SmartDashboard
30 * to have a list of options appear on the laptop. Once autonomous starts,
31 * simply ask the SendableChooser what the selected value is.
32 *
33 * @tparam T The type of values to be stored
34 * @see SmartDashboard
35 */
36template <class T>
37 requires std::copy_constructible<T> && std::default_initializable<T>
38class SendableChooser : public SendableChooserBase {
39 wpi::StringMap<T> m_choices;
40 std::function<void(T)> m_listener;
41 template <class U>
42 static U _unwrap_smart_ptr(const U& value) {
43 return value;
44 }
45
46 template <class U>
47 static std::weak_ptr<U> _unwrap_smart_ptr(const std::shared_ptr<U>& value) {
48 return value;
49 }
50
51 public:
52 using CopyType = decltype(_unwrap_smart_ptr(m_choices.lookup("")));
53
54 SendableChooser() = default;
55 ~SendableChooser() override = default;
56 SendableChooser(SendableChooser&& rhs) = default;
57 SendableChooser& operator=(SendableChooser&& rhs) = default;
58
59 /**
60 * Adds the given object to the list of options.
61 *
62 * On the SmartDashboard on the desktop, the object will appear as the given
63 * name.
64 *
65 * @param name the name of the option
66 * @param object the option
67 */
68 void AddOption(std::string_view name, T object) {
69 m_choices[name] = std::move(object);
70 }
71
72 /**
73 * Add the given object to the list of options and marks it as the default.
74 *
75 * Functionally, this is very close to AddOption() except that it will use
76 * this as the default option if none other is explicitly selected.
77 *
78 * @param name the name of the option
79 * @param object the option
80 */
81 void SetDefaultOption(std::string_view name, T object) {
82 m_defaultChoice = name;
83 AddOption(name, std::move(object));
84 }
85
86 /**
87 * Returns a copy of the selected option (a std::weak_ptr<U> if T =
88 * std::shared_ptr<U>).
89 *
90 * If there is none selected, it will return the default. If there is none
91 * selected and no default, then it will return a value-initialized instance.
92 * For integer types, this is 0. For container types like std::string, this is
93 * an empty string.
94 *
95 * @return The option selected
96 */
97 CopyType GetSelected() const {
98 std::string selected = m_defaultChoice;
99 {
100 std::scoped_lock lock(m_mutex);
101 if (m_haveSelected) {
102 selected = m_selected;
103 }
104 }
105 if (selected.empty()) {
106 return CopyType{};
107 } else {
108 return _unwrap_smart_ptr(m_choices.lookup(selected));
109 }
110 }
111
112 /**
113 * Bind a listener that's called when the selected value changes.
114 * Only one listener can be bound. Calling this function will replace the
115 * previous listener.
116 * @param listener The function to call that accepts the new value
117 */
118 void OnChange(std::function<void(T)> listener) {
119 std::scoped_lock lock(m_mutex);
120 m_listener = listener;
121 }
122
123 void InitSendable(wpi::SendableBuilder& builder) override {
124 builder.SetSmartDashboardType("String Chooser");
125 builder.PublishConstInteger(kInstance, m_instance);
126 builder.AddStringArrayProperty(
127 kOptions,
128 [=, this] {
129 std::vector<std::string> keys;
130 for (const auto& choice : m_choices) {
131 keys.emplace_back(choice.first());
132 }
133
134 // Unlike std::map, wpi::StringMap elements
135 // are not sorted
136 std::sort(keys.begin(), keys.end());
137
138 return keys;
139 },
140 nullptr);
141 builder.AddSmallStringProperty(
142 kDefault,
143 [=, this](wpi::SmallVectorImpl<char>&) -> std::string_view {
144 return m_defaultChoice;
145 },
146 nullptr);
147 builder.AddSmallStringProperty(
148 kActive,
149 [=, this](wpi::SmallVectorImpl<char>& buf) -> std::string_view {
150 std::scoped_lock lock(m_mutex);
151 if (m_haveSelected) {
152 buf.assign(m_selected.begin(), m_selected.end());
153 return {buf.data(), buf.size()};
154 } else {
155 return m_defaultChoice;
156 }
157 },
158 nullptr);
159 builder.AddStringProperty(kSelected, nullptr,
160 [=, this](std::string_view val) {
161 T choice{};
162 std::function<void(T)> listener;
163 {
164 std::scoped_lock lock(m_mutex);
165 m_haveSelected = true;
166 m_selected = val;
167 if (m_previousVal != val && m_listener) {
168 choice = m_choices[val];
169 listener = m_listener;
170 }
171 m_previousVal = val;
172 }
173 if (listener) {
174 listener(choice);
175 }
176 });
177 }
178};
