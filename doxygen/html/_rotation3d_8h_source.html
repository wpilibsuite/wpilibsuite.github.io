27/**
28 * A rotation in a 3D coordinate frame represented by a quaternion.
29 */
30class WPILIB_DLLEXPORT Rotation3d {
31 public:
32 /**
33 * Constructs a Rotation3d representing no rotation.
34 */
35 constexpr Rotation3d() = default;
36
37 /**
38 * Constructs a Rotation3d from a quaternion.
39 *
40 * @param q The quaternion.
41 */
42 constexpr explicit Rotation3d(const Quaternion& q) { m_q = q.Normalize(); }
43
44 /**
45 * Constructs a Rotation3d from extrinsic roll, pitch, and yaw.
46 *
47 * Extrinsic rotations occur in that order around the axes in the fixed global
48 * frame rather than the body frame.
49 *
50 * Angles are measured counterclockwise with the rotation axis pointing "out
51 * of the page". If you point your right thumb along the positive axis
52 * direction, your fingers curl in the direction of positive rotation.
53 *
54 * @param roll The counterclockwise rotation angle around the X axis (roll).
55 * @param pitch The counterclockwise rotation angle around the Y axis (pitch).
56 * @param yaw The counterclockwise rotation angle around the Z axis (yaw).
57 */
58 constexpr Rotation3d(units::radian_t roll, units::radian_t pitch,
59 units::radian_t yaw) {
60 // https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles#Euler_angles_to_quaternion_conversion
61 double cr = units::math::cos(roll * 0.5);
62 double sr = units::math::sin(roll * 0.5);
63
64 double cp = units::math::cos(pitch * 0.5);
65 double sp = units::math::sin(pitch * 0.5);
66
67 double cy = units::math::cos(yaw * 0.5);
68 double sy = units::math::sin(yaw * 0.5);
69
70 m_q = Quaternion{cr * cp * cy + sr * sp * sy, sr * cp * cy - cr * sp * sy,
71 cr * sp * cy + sr * cp * sy, cr * cp * sy - sr * sp * cy};
72 }
73
74 /**
75 * Constructs a Rotation3d with the given axis-angle representation. The axis
76 * doesn't have to be normalized.
77 *
78 * @param axis The rotation axis.
79 * @param angle The rotation around the axis.
80 */
81 constexpr Rotation3d(const Eigen::Vector3d& axis, units::radian_t angle) {
82 double norm = ct_matrix{axis}.norm();
83 if (norm == 0.0) {
84 return;
85 }
86
87 // https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles#Definition
88 Eigen::Vector3d v{{axis.coeff(0) / norm * units::math::sin(angle / 2.0),
89 axis.coeff(1) / norm * units::math::sin(angle / 2.0),
90 axis.coeff(2) / norm * units::math::sin(angle / 2.0)}};
91 m_q = Quaternion{units::math::cos(angle / 2.0), v.coeff(0), v.coeff(1),
92 v.coeff(2)};
93 }
94
95 /**
96 * Constructs a Rotation3d with the given rotation vector representation. This
97 * representation is equivalent to axis-angle, where the normalized axis is
98 * multiplied by the rotation around the axis in radians.
99 *
100 * @param rvec The rotation vector.
101 */
102 constexpr explicit Rotation3d(const Eigen::Vector3d& rvec)
103 : Rotation3d{rvec, units::radian_t{ct_matrix{rvec}.norm()}} {}
104
105 /**
106 * Constructs a Rotation3d from a rotation matrix.
107 *
108 * @param rotationMatrix The rotation matrix.
109 * @throws std::domain_error if the rotation matrix isn't special orthogonal.
110 */
111 constexpr explicit Rotation3d(const Eigen::Matrix3d& rotationMatrix) {
112 auto impl = []<typename Matrix3d>(const Matrix3d& R) -> Quaternion {
113 // Require that the rotation matrix is special orthogonal. This is true if
114 // the matrix is orthogonal (RRᵀ = I) and normalized (determinant is 1).
115 if ((R * R.transpose() - Matrix3d::Identity()).norm() > 1e-9) {
116 throw std::domain_error("Rotation matrix isn't orthogonal");
117 }
118 if (gcem::abs(R.determinant() - 1.0) > 1e-9) {
119 throw std::domain_error(
120 "Rotation matrix is orthogonal but not special orthogonal");
121 }
122
123 // Turn rotation matrix into a quaternion
124 // https://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/
125 double trace = R(0, 0) + R(1, 1) + R(2, 2);
126 double w;
127 double x;
128 double y;
129 double z;
130
131 if (trace > 0.0) {
132 double s = 0.5 / gcem::sqrt(trace + 1.0);
133 w = 0.25 / s;
134 x = (R(2, 1) - R(1, 2)) * s;
135 y = (R(0, 2) - R(2, 0)) * s;
136 z = (R(1, 0) - R(0, 1)) * s;
137 } else {
138 if (R(0, 0) > R(1, 1) && R(0, 0) > R(2, 2)) {
139 double s = 2.0 * gcem::sqrt(1.0 + R(0, 0) - R(1, 1) - R(2, 2));
140 w = (R(2, 1) - R(1, 2)) / s;
141 x = 0.25 * s;
142 y = (R(0, 1) + R(1, 0)) / s;
143 z = (R(0, 2) + R(2, 0)) / s;
144 } else if (R(1, 1) > R(2, 2)) {
145 double s = 2.0 * gcem::sqrt(1.0 + R(1, 1) - R(0, 0) - R(2, 2));
146 w = (R(0, 2) - R(2, 0)) / s;
147 x = (R(0, 1) + R(1, 0)) / s;
148 y = 0.25 * s;
149 z = (R(1, 2) + R(2, 1)) / s;
150 } else {
151 double s = 2.0 * gcem::sqrt(1.0 + R(2, 2) - R(0, 0) - R(1, 1));
152 w = (R(1, 0) - R(0, 1)) / s;
153 x = (R(0, 2) + R(2, 0)) / s;
154 y = (R(1, 2) + R(2, 1)) / s;
155 z = 0.25 * s;
156 }
157 }
158
159 return Quaternion{w, x, y, z};
160 };
161
162 if (std::is_constant_evaluated()) {
163 m_q = impl(ct_matrix3d{rotationMatrix});
164 } else {
165 m_q = impl(rotationMatrix);
166 }
167 }
168
169 /**
170 * Constructs a Rotation3d that rotates the initial vector onto the final
171 * vector.
172 *
173 * This is useful for turning a 3D vector (final) into an orientation relative
174 * to a coordinate system vector (initial).
175 *
176 * @param initial The initial vector.
177 * @param final The final vector.
178 */
179 constexpr Rotation3d(const Eigen::Vector3d& initial,
180 const Eigen::Vector3d& final) {
181 double dot = ct_matrix{initial}.dot(ct_matrix{final});
182 double normProduct = ct_matrix{initial}.norm() * ct_matrix{final}.norm();
183 double dotNorm = dot / normProduct;
184
185 if (dotNorm > 1.0 - 1E-9) {
186 // If the dot product is 1, the two vectors point in the same direction so
187 // there's no rotation. The default initialization of m_q will work.
188 return;
189 } else if (dotNorm < -1.0 + 1E-9) {
190 // If the dot product is -1, the two vectors are antiparallel, so a 180°
191 // rotation is required. Any other vector can be used to generate an
192 // orthogonal one.
193
194 double x = gcem::abs(initial.coeff(0));
195 double y = gcem::abs(initial.coeff(1));
196 double z = gcem::abs(initial.coeff(2));
197
198 // Find vector that is most orthogonal to initial vector
199 Eigen::Vector3d other;
200 if (x < y) {
201 if (x < z) {
202 // Use x-axis
203 other = Eigen::Vector3d{{1, 0, 0}};
204 } else {
205 // Use z-axis
206 other = Eigen::Vector3d{{0, 0, 1}};
207 }
208 } else {
209 if (y < z) {
210 // Use y-axis
211 other = Eigen::Vector3d{{0, 1, 0}};
212 } else {
213 // Use z-axis
214 other = Eigen::Vector3d{{0, 0, 1}};
215 }
216 }
217
218 auto axis = ct_matrix{initial}.cross(ct_matrix{other});
219
220 double axisNorm = axis.norm();
221 m_q = Quaternion{0.0, axis(0) / axisNorm, axis(1) / axisNorm,
222 axis(2) / axisNorm};
223 } else {
224 auto axis = ct_matrix{initial}.cross(final);
225
226 // https://stackoverflow.com/a/11741520
227 m_q =
228 Quaternion{normProduct + dot, axis(0), axis(1), axis(2)}.Normalize();
229 }
230 }

232 /**
233 * Adds two rotations together.
234 *
235 * @param other The rotation to add.
236 *
237 * @return The sum of the two rotations.
238 */
239 constexpr Rotation3d operator+(const Rotation3d& other) const {
240 return RotateBy(other);
241 }
242
243 /**
244 * Subtracts the new rotation from the current rotation and returns the new
245 * rotation.
246 *
247 * @param other The rotation to subtract.
248 *
249 * @return The difference between the two rotations.
250 */
251 constexpr Rotation3d operator-(const Rotation3d& other) const {
252 return *this + -other;
253 }
254
255 /**
256 * Takes the inverse of the current rotation.
257 *
258 * @return The inverse of the current rotation.
259 */
260 constexpr Rotation3d operator-() const { return Rotation3d{m_q.Inverse()}; }
261
262 /**
263 * Multiplies the current rotation by a scalar.
264 *
265 * @param scalar The scalar.
266 *
267 * @return The new scaled Rotation3d.
268 */
269 constexpr Rotation3d operator*(double scalar) const {
270 // https://en.wikipedia.org/wiki/Slerp#Quaternion_Slerp
271 if (m_q.W() >= 0.0) {
272 return Rotation3d{Eigen::Vector3d{{m_q.X(), m_q.Y(), m_q.Z()}},
273 2.0 * units::radian_t{scalar * gcem::acos(m_q.W())}};
274 } else {
275 return Rotation3d{Eigen::Vector3d{{-m_q.X(), -m_q.Y(), -m_q.Z()}},
276 2.0 * units::radian_t{scalar * gcem::acos(-m_q.W())}};
277 }
278 }
279
280 /**
281 * Divides the current rotation by a scalar.
282 *
283 * @param scalar The scalar.
284 *
285 * @return The new scaled Rotation3d.
286 */
287 constexpr Rotation3d operator/(double scalar) const {
288 return *this * (1.0 / scalar);
289 }
290
291 /**
292 * Checks equality between this Rotation3d and another object.
293 */
294 constexpr bool operator==(const Rotation3d& other) const {
295 return gcem::abs(gcem::abs(m_q.Dot(other.m_q)) -
296 m_q.Norm() * other.m_q.Norm()) < 1e-9;
297 }
298
299 /**
300 * Adds the new rotation to the current rotation. The other rotation is
301 * applied extrinsically, which means that it rotates around the global axes.
302 * For example, Rotation3d{90_deg, 0, 0}.RotateBy(Rotation3d{0, 45_deg, 0})
303 * rotates by 90 degrees around the +X axis and then by 45 degrees around the
304 * global +Y axis. (This is equivalent to Rotation3d{90_deg, 45_deg, 0})
305 *
306 * @param other The extrinsic rotation to rotate by.
307 *
308 * @return The new rotated Rotation3d.
309 */
310 constexpr Rotation3d RotateBy(const Rotation3d& other) const {
311 return Rotation3d{other.m_q * m_q};
312 }

314 /**
315 * Returns the quaternion representation of the Rotation3d.

317 constexpr const Quaternion& GetQuaternion() const { return m_q; }

319 /**
320 * Returns the counterclockwise rotation angle around the X axis (roll).
321 */
322 constexpr units::radian_t X() const {
323 double w = m_q.W();
324 double x = m_q.X();
325 double y = m_q.Y();
326 double z = m_q.Z();
327
328 // wpimath/algorithms.md
329 double cxcy = 1.0 - 2.0 * (x * x + y * y);
330 double sxcy = 2.0 * (w * x + y * z);
331 double cy_sq = cxcy * cxcy + sxcy * sxcy;
332 if (cy_sq > 1e-20) {
333 return units::radian_t{gcem::atan2(sxcy, cxcy)};
334 } else {
335 return 0_rad;
336 }
337 }
338
339 /**
340 * Returns the counterclockwise rotation angle around the Y axis (pitch).
341 */
342 constexpr units::radian_t Y() const {
343 double w = m_q.W();
344 double x = m_q.X();
345 double y = m_q.Y();
346 double z = m_q.Z();
347
348 // https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles#Quaternion_to_Euler_angles_(in_3-2-1_sequence)_conversion
349 double ratio = 2.0 * (w * y - z * x);
350 if (gcem::abs(ratio) >= 1.0) {
351 return units::radian_t{gcem::copysign(std::numbers::pi / 2.0, ratio)};
352 } else {
353 return units::radian_t{gcem::asin(ratio)};
354 }
355 }
356
357 /**
358 * Returns the counterclockwise rotation angle around the Z axis (yaw).
359 */
360 constexpr units::radian_t Z() const {
361 double w = m_q.W();
362 double x = m_q.X();
363 double y = m_q.Y();
364 double z = m_q.Z();
365
366 // wpimath/algorithms.md
367 double cycz = 1.0 - 2.0 * (y * y + z * z);
368 double cysz = 2.0 * (w * z + x * y);
369 double cy_sq = cycz * cycz + cysz * cysz;
370 if (cy_sq > 1e-20) {
371 return units::radian_t{gcem::atan2(cysz, cycz)};
372 } else {
373 return units::radian_t{gcem::atan2(2.0 * w * z, w * w - z * z)};
374 }
375 }
376
377 /**
378 * Returns the axis in the axis-angle representation of this rotation.
379 */
380 constexpr Eigen::Vector3d Axis() const {
381 double norm = gcem::hypot(m_q.X(), m_q.Y(), m_q.Z());
382 if (norm == 0.0) {
383 return Eigen::Vector3d{{0.0, 0.0, 0.0}};
384 } else {
385 return Eigen::Vector3d{{m_q.X() / norm, m_q.Y() / norm, m_q.Z() / norm}};
386 }
387 }
388
389 /**
390 * Returns the angle in the axis-angle representation of this rotation.
391 */
392 constexpr units::radian_t Angle() const {
393 double norm = gcem::hypot(m_q.X(), m_q.Y(), m_q.Z());
394 return units::radian_t{2.0 * gcem::atan2(norm, m_q.W())};
395 }
396
397 /**
398 * Returns a Rotation2d representing this Rotation3d projected into the X-Y
399 * plane.
400 */
401 constexpr Rotation2d ToRotation2d() const { return Rotation2d{Z()}; }
402
403 private:
404 Quaternion m_q;
405};
