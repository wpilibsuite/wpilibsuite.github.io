21/**
22 * Represents a translation in 2D space.
23 * This object can be used to represent a point or a vector.
24 *
25 * This assumes that you are using conventional mathematical axes.
26 * When the robot is at the origin facing in the positive X direction, forward
27 * is positive X and left is positive Y.
28 */
29class WPILIB_DLLEXPORT Translation2d {
30 public:
31 /**
32 * Constructs a Translation2d with X and Y components equal to zero.
33 */
34 constexpr Translation2d() = default;
35
36 /**
37 * Constructs a Translation2d with the X and Y components equal to the
38 * provided values.
39 *
40 * @param x The x component of the translation.
41 * @param y The y component of the translation.
42 */
43 constexpr Translation2d(units::meter_t x, units::meter_t y)
44 : m_x{x}, m_y{y} {}
45
46 /**
47 * Constructs a Translation2d with the provided distance and angle. This is
48 * essentially converting from polar coordinates to Cartesian coordinates.
49 *
50 * @param distance The distance from the origin to the end of the translation.
51 * @param angle The angle between the x-axis and the translation vector.
52 */
53 constexpr Translation2d(units::meter_t distance, const Rotation2d& angle)
54 : m_x{distance * angle.Cos()}, m_y{distance * angle.Sin()} {}
55
56 /**
57 * Constructs a Translation2d from the provided translation vector's X and Y
58 * components. The values are assumed to be in meters.
59 *
60 * @param vector The translation vector to represent.
61 */
62 constexpr explicit Translation2d(const Eigen::Vector2d& vector)
63 : m_x{units::meter_t{vector.coeff(0)}},
64 m_y{units::meter_t{vector.coeff(1)}} {}
65
66 /**
67 * Calculates the distance between two translations in 2D space.
68 *
69 * The distance between translations is defined as √((x₂−x₁)²+(y₂−y₁)²).
70 *
71 * @param other The translation to compute the distance to.
72 *
73 * @return The distance between the two translations.
74 */
75 constexpr units::meter_t Distance(const Translation2d& other) const {
76 return units::math::hypot(other.m_x - m_x, other.m_y - m_y);
77 }
78
79 /**
80 * Returns the X component of the translation.
81 *
82 * @return The X component of the translation.
83 */
84 constexpr units::meter_t X() const { return m_x; }
85
86 /**
87 * Returns the Y component of the translation.
88 *
89 * @return The Y component of the translation.
90 */
91 constexpr units::meter_t Y() const { return m_y; }
92
93 /**
94 * Returns a vector representation of this translation.
95 *
96 * @return A Vector representation of this translation.
97 */
98 constexpr Eigen::Vector2d ToVector() const {
99 return Eigen::Vector2d{{m_x.value(), m_y.value()}};
100 }
101
102 /**
103 * Returns the norm, or distance from the origin to the translation.
104 *
105 * @return The norm of the translation.
106 */
107 constexpr units::meter_t Norm() const { return units::math::hypot(m_x, m_y); }
108
109 /**
110 * Returns the angle this translation forms with the positive X axis.
111 *
112 * @return The angle of the translation
113 */
114 constexpr Rotation2d Angle() const {
115 return Rotation2d{m_x.value(), m_y.value()};
116 }
117
118 /**
119 * Applies a rotation to the translation in 2D space.
120 *
121 * This multiplies the translation vector by a counterclockwise rotation
122 * matrix of the given angle.
123 *
124 * <pre>
125 * [x_new] [other.cos, -other.sin][x]
126 * [y_new] = [other.sin, other.cos][y]
127 * </pre>
128 *
129 * For example, rotating a Translation2d of &lt;2, 0&gt; by 90 degrees will
130 * return a Translation2d of &lt;0, 2&gt;.
131 *
132 * @param other The rotation to rotate the translation by.
133 *
134 * @return The new rotated translation.
135 */
136 constexpr Translation2d RotateBy(const Rotation2d& other) const {
137 return {m_x * other.Cos() - m_y * other.Sin(),
138 m_x * other.Sin() + m_y * other.Cos()};
139 }
140
141 /**
142 * Rotates this translation around another translation in 2D space.
143 *
144 * <pre>
145 * [x_new] [rot.cos, -rot.sin][x - other.x] [other.x]
146 * [y_new] = [rot.sin, rot.cos][y - other.y] + [other.y]
147 * </pre>
148 *
149 * @param other The other translation to rotate around.
150 * @param rot The rotation to rotate the translation by.
151 * @return The new rotated translation.
152 */
153 constexpr Translation2d RotateAround(const Translation2d& other,
154 const Rotation2d& rot) const {
155 return {(m_x - other.X()) * rot.Cos() - (m_y - other.Y()) * rot.Sin() +
156 other.X(),
157 (m_x - other.X()) * rot.Sin() + (m_y - other.Y()) * rot.Cos() +
158 other.Y()};
159 }
160
161 /**
162 * Returns the sum of two translations in 2D space.
163 *
164 * For example, Translation3d{1.0, 2.5} + Translation3d{2.0, 5.5} =
165 * Translation3d{3.0, 8.0}.
166 *
167 * @param other The translation to add.
168 *
169 * @return The sum of the translations.
170 */
171 constexpr Translation2d operator+(const Translation2d& other) const {
172 return {X() + other.X(), Y() + other.Y()};
173 }
174
175 /**
176 * Returns the difference between two translations.
177 *
178 * For example, Translation2d{5.0, 4.0} - Translation2d{1.0, 2.0} =
179 * Translation2d{4.0, 2.0}.
180 *
181 * @param other The translation to subtract.
182 *
183 * @return The difference between the two translations.
184 */
185 constexpr Translation2d operator-(const Translation2d& other) const {
186 return *this + -other;
187 }
188
189 /**
190 * Returns the inverse of the current translation. This is equivalent to
191 * rotating by 180 degrees, flipping the point over both axes, or negating all
192 * components of the translation.
193 *
194 * @return The inverse of the current translation.
195 */
196 constexpr Translation2d operator-() const { return {-m_x, -m_y}; }
197
198 /**
199 * Returns the translation multiplied by a scalar.
200 *
201 * For example, Translation2d{2.0, 2.5} * 2 = Translation2d{4.0, 5.0}.
202 *
203 * @param scalar The scalar to multiply by.
204 *
205 * @return The scaled translation.
206 */
207 constexpr Translation2d operator*(double scalar) const {
208 return {scalar * m_x, scalar * m_y};
209 }
210
211 /**
212 * Returns the translation divided by a scalar.
213 *
214 * For example, Translation2d{2.0, 2.5} / 2 = Translation2d{1.0, 1.25}.
215 *
216 * @param scalar The scalar to divide by.
217 *
218 * @return The scaled translation.
219 */
220 constexpr Translation2d operator/(double scalar) const {
221 return operator*(1.0 / scalar);
222 }
223
224 /**
225 * Checks equality between this Translation2d and another object.
226 *
227 * @param other The other object.
228 * @return Whether the two objects are equal.
229 */
230 constexpr bool operator==(const Translation2d& other) const {
231 return units::math::abs(m_x - other.m_x) < 1E-9_m &&
232 units::math::abs(m_y - other.m_y) < 1E-9_m;
233 }
234
235 /**
236 * Returns the nearest Translation2d from a collection of translations
237 * @param translations The collection of translations.
238 * @return The nearest Translation2d from the collection.
239 */
240 constexpr Translation2d Nearest(
241 std::span<const Translation2d> translations) const {
242 return *std::min_element(translations.begin(), translations.end(),
243 [this](Translation2d a, Translation2d b) {
244 return this->Distance(a) < this->Distance(b);
245 });
246 }
247
248 /**
249 * Returns the nearest Translation2d from a collection of translations
250 * @param translations The collection of translations.
251 * @return The nearest Translation2d from the collection.
252 */
253 constexpr Translation2d Nearest(
254 std::initializer_list<Translation2d> translations) const {
255 return *std::min_element(translations.begin(), translations.end(),
256 [this](Translation2d a, Translation2d b) {
257 return this->Distance(a) < this->Distance(b);
258 });
259 }
260
261 private:
262 units::meter_t m_x = 0_m;
263 units::meter_t m_y = 0_m;
264};
