24constexpr Rotation2d::Rotation2d(double x, double y) {
25 double magnitude = gcem::hypot(x, y);
26 if (magnitude > 1e-6) {
27 m_sin = y / magnitude;
28 m_cos = x / magnitude;
29 } else {
30 m_sin = 0.0;
31 m_cos = 1.0;
32 if (!std::is_constant_evaluated()) {
33 wpi::math::MathSharedStore::ReportError(
34 "x and y components of Rotation2d are zero\n{}",
35 wpi::GetStackTrace(1));
36 }
37 }
38 m_value = units::radian_t{gcem::atan2(m_sin, m_cos)};
39}
