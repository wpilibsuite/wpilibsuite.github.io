23/**
24 * A Kalman filter combines predictions from a model and measurements to give an
25 * estimate of the true system state. This is useful because many states cannot
26 * be measured directly as a result of sensor noise, or because the state is
27 * "hidden".
28 *
29 * Kalman filters use a K gain matrix to determine whether to trust the model or
30 * measurements more. Kalman filter theory uses statistics to compute an optimal
31 * K gain which minimizes the sum of squares error in the state estimate. This K
32 * gain is used to correct the state estimate by some amount of the difference
33 * between the actual measurements and the measurements predicted by the model.
34 *
35 * An extended Kalman filter supports nonlinear state and measurement models. It
36 * propagates the error covariance by linearizing the models around the state
37 * estimate, then applying the linear Kalman filter equations.
38 *
39 * For more on the underlying math, read
40 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf chapter 9
41 * "Stochastic control theory".
42 *
43 * @tparam States Number of states.
44 * @tparam Inputs Number of inputs.
45 * @tparam Outputs Number of outputs.
46 */
47template <int States, int Inputs, int Outputs>
48class ExtendedKalmanFilter {
49 public:
50 using StateVector = Vectord<States>;
51 using InputVector = Vectord<Inputs>;
52 using OutputVector = Vectord<Outputs>;
53
54 using StateArray = wpi::array<double, States>;
55 using OutputArray = wpi::array<double, Outputs>;
56
57 using StateMatrix = Matrixd<States, States>;
58
59 /**
60 * Constructs an extended Kalman filter.
61 *
62 * See
63 * https://docs.wpilib.org/en/stable/docs/software/advanced-controls/state-space/state-space-observers.html#process-and-measurement-noise-covariance-matrices
64 * for how to select the standard deviations.
65 *
66 * @param f A vector-valued function of x and u that returns
67 * the derivative of the state vector.
68 * @param h A vector-valued function of x and u that returns
69 * the measurement vector.
70 * @param stateStdDevs Standard deviations of model states.
71 * @param measurementStdDevs Standard deviations of measurements.
72 * @param dt Nominal discretization timestep.
73 */
74 ExtendedKalmanFilter(
75 std::function<StateVector(const StateVector&, const InputVector&)> f,
76 std::function<OutputVector(const StateVector&, const InputVector&)> h,
77 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
78 units::second_t dt)
79 : m_f(std::move(f)), m_h(std::move(h)) {
80 m_contQ = MakeCovMatrix(stateStdDevs);
81 m_contR = MakeCovMatrix(measurementStdDevs);
82 m_residualFuncY = [](const OutputVector& a,
83 const OutputVector& b) -> OutputVector {
84 return a - b;
85 };
86 m_addFuncX = [](const StateVector& a, const StateVector& b) -> StateVector {
87 return a + b;
88 };
89 m_dt = dt;
90
91 StateMatrix contA = NumericalJacobianX<States, States, Inputs>(
92 m_f, m_xHat, InputVector::Zero());
93 Matrixd<Outputs, States> C = NumericalJacobianX<Outputs, States, Inputs>(
94 m_h, m_xHat, InputVector::Zero());
95
96 StateMatrix discA;
97 StateMatrix discQ;
98 DiscretizeAQ<States>(contA, m_contQ, dt, &discA, &discQ);
99
100 Matrixd<Outputs, Outputs> discR = DiscretizeR<Outputs>(m_contR, dt);
101
102 if (IsDetectable<States, Outputs>(discA, C) && Outputs <= States) {
103 m_initP =
104 DARE<States, Outputs>(discA.transpose(), C.transpose(), discQ, discR);
105 } else {
106 m_initP = StateMatrix::Zero();
107 }
108 m_P = m_initP;
109 }

111 /**
112 * Constructs an extended Kalman filter.
113 *
114 * See
115 * https://docs.wpilib.org/en/stable/docs/software/advanced-controls/state-space/state-space-observers.html#process-and-measurement-noise-covariance-matrices
116 * for how to select the standard deviations.
117 *
118 * @param f A vector-valued function of x and u that returns
119 * the derivative of the state vector.
120 * @param h A vector-valued function of x and u that returns
121 * the measurement vector.
122 * @param stateStdDevs Standard deviations of model states.
123 * @param measurementStdDevs Standard deviations of measurements.
124 * @param residualFuncY A function that computes the residual of two
125 * measurement vectors (i.e. it subtracts them.)
126 * @param addFuncX A function that adds two state vectors.
127 * @param dt Nominal discretization timestep.
128 */
129 ExtendedKalmanFilter(
130 std::function<StateVector(const StateVector&, const InputVector&)> f,
131 std::function<OutputVector(const StateVector&, const InputVector&)> h,
132 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
133 std::function<OutputVector(const OutputVector&, const OutputVector&)>
134 residualFuncY,
135 std::function<StateVector(const StateVector&, const StateVector&)>
136 addFuncX,
137 units::second_t dt)
138 : m_f(std::move(f)),
139 m_h(std::move(h)),
140 m_residualFuncY(std::move(residualFuncY)),
141 m_addFuncX(std::move(addFuncX)) {
142 m_contQ = MakeCovMatrix(stateStdDevs);
143 m_contR = MakeCovMatrix(measurementStdDevs);
144 m_dt = dt;
145
146 StateMatrix contA = NumericalJacobianX<States, States, Inputs>(
147 m_f, m_xHat, InputVector::Zero());
148 Matrixd<Outputs, States> C = NumericalJacobianX<Outputs, States, Inputs>(
149 m_h, m_xHat, InputVector::Zero());
150
151 StateMatrix discA;
152 StateMatrix discQ;
153 DiscretizeAQ<States>(contA, m_contQ, dt, &discA, &discQ);
154
155 Matrixd<Outputs, Outputs> discR = DiscretizeR<Outputs>(m_contR, dt);
156
157 if (IsDetectable<States, Outputs>(discA, C) && Outputs <= States) {
158 m_initP =
159 DARE<States, Outputs>(discA.transpose(), C.transpose(), discQ, discR);
160 } else {
161 m_initP = StateMatrix::Zero();
162 }
163 m_P = m_initP;
164 }

221 /**
222 * Project the model into the future with a new control input u.
223 *
224 * @param u New control input from controller.
225 * @param dt Timestep for prediction.
226 */
227 void Predict(const InputVector& u, units::second_t dt) {
228 // Find continuous A
229 StateMatrix contA =
230 NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
231
232 // Find discrete A and Q
233 StateMatrix discA;
234 StateMatrix discQ;
235 DiscretizeAQ<States>(contA, m_contQ, dt, &discA, &discQ);
236
237 m_xHat = RK4(m_f, m_xHat, u, dt);
238
239 // P = APA + Q
240 m_P = discA * m_P * discA.transpose() + discQ;
241
242 m_dt = dt;
243 }

269 /**
270 * Correct the state estimate x-hat using the measurements in y.
271 *
272 * This is useful for when the measurements available during a timestep's
273 * Correct() call vary. The h(x, u) passed to the constructor is used if one
274 * is not provided (the two-argument version of this function).
275 *
276 * @param u Same control input used in the predict step.
277 * @param y Measurement vector.
278 * @param h A vector-valued function of x and u that returns the measurement
279 * vector.
280 * @param R Continuous measurement noise covariance matrix.
281 */
282 template <int Rows>
283 void Correct(
284 const InputVector& u, const Vectord<Rows>& y,
285 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
286 const Matrixd<Rows, Rows>& R) {
287 auto residualFuncY = [](const Vectord<Rows>& a,
288 const Vectord<Rows>& b) -> Vectord<Rows> {
289 return a - b;
290 };
291 auto addFuncX = [](const StateVector& a,
292 const StateVector& b) -> StateVector { return a + b; };
293 Correct<Rows>(u, y, std::move(h), R, std::move(residualFuncY),
294 std::move(addFuncX));
295 }
296
297 /**
298 * Correct the state estimate x-hat using the measurements in y.
299 *
300 * This is useful for when the measurements available during a timestep's
301 * Correct() call vary. The h(x, u) passed to the constructor is used if one
302 * is not provided (the two-argument version of this function).
303 *
304 * @param u Same control input used in the predict step.
305 * @param y Measurement vector.
306 * @param h A vector-valued function of x and u that returns
307 * the measurement vector.
308 * @param R Continuous measurement noise covariance matrix.
309 * @param residualFuncY A function that computes the residual of two
310 * measurement vectors (i.e. it subtracts them.)
311 * @param addFuncX A function that adds two state vectors.
312 */
313 template <int Rows>
314 void Correct(
315 const InputVector& u, const Vectord<Rows>& y,
316 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
317 const Matrixd<Rows, Rows>& R,
318 std::function<Vectord<Rows>(const Vectord<Rows>&, const Vectord<Rows>&)>
319 residualFuncY,
320 std::function<StateVector(const StateVector&, const StateVector&)>
321 addFuncX) {
322 const Matrixd<Rows, States> C =
323 NumericalJacobianX<Rows, States, Inputs>(h, m_xHat, u);
324 const Matrixd<Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
325
326 Matrixd<Rows, Rows> S = C * m_P * C.transpose() + discR;
327
328 // We want to put K = PCS into Ax = b form so we can solve it more
329 // efficiently.
330 //
331 // K = PCS
332 // KS = PC
333 // (KS) = (PC)
334 // SK = CP
335 //
336 // The solution of Ax = b can be found via x = A.solve(b).
337 //
338 // K = S.solve(CP)
339 // K = (S.solve(CP))
340 Matrixd<States, Rows> K =
341 S.transpose().ldlt().solve(C * m_P.transpose()).transpose();
342
343 // x = x + K(y h(x, u))
344 m_xHat = addFuncX(m_xHat, K * residualFuncY(y, h(m_xHat, u)));
345
346 // P = (IKC)P(IKC) + KRK
347 // Use Joseph form for numerical stability
348 m_P = (StateMatrix::Identity() - K * C) * m_P *
349 (StateMatrix::Identity() - K * C).transpose() +
350 K * discR * K.transpose();
351 }
352
353 private:
354 std::function<StateVector(const StateVector&, const InputVector&)> m_f;
355 std::function<OutputVector(const StateVector&, const InputVector&)> m_h;
356 std::function<OutputVector(const OutputVector&, const OutputVector&)>
357 m_residualFuncY;
358 std::function<StateVector(const StateVector&, const StateVector&)> m_addFuncX;
359 StateVector m_xHat = StateVector::Zero();
360 StateMatrix m_P;
361 StateMatrix m_contQ;
362 Matrixd<Outputs, Outputs> m_contR;
363 units::second_t m_dt;
364
365 StateMatrix m_initP;
366};
