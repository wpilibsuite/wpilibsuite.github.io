25/**
26 * Represents a 2D pose containing translational and rotational elements.
27 */
28class WPILIB_DLLEXPORT Pose2d {
29 public:
30 /**
31 * Constructs a pose at the origin facing toward the positive X axis.
32 */
33 constexpr Pose2d() = default;
34
35 /**
36 * Constructs a pose with the specified translation and rotation.
37 *
38 * @param translation The translational component of the pose.
39 * @param rotation The rotational component of the pose.
40 */
41 constexpr Pose2d(Translation2d translation, Rotation2d rotation)
42 : m_translation{std::move(translation)},
43 m_rotation{std::move(rotation)} {}
44
45 /**
46 * Constructs a pose with x and y translations instead of a separate
47 * Translation2d.
48 *
49 * @param x The x component of the translational component of the pose.
50 * @param y The y component of the translational component of the pose.
51 * @param rotation The rotational component of the pose.
52 */
53 constexpr Pose2d(units::meter_t x, units::meter_t y, Rotation2d rotation)
54 : m_translation{x, y}, m_rotation{std::move(rotation)} {}
55
56 /**
57 * Transforms the pose by the given transformation and returns the new
58 * transformed pose.
59 *
60 * <pre>
61 * [x_new] [cos, -sin, 0][transform.x]
62 * [y_new] += [sin, cos, 0][transform.y]
63 * [t_new] [ 0, 0, 1][transform.t]
64 * </pre>
65 *
66 * @param other The transform to transform the pose by.
67 *
68 * @return The transformed pose.
69 */
70 constexpr Pose2d operator+(const Transform2d& other) const {
71 return TransformBy(other);
72 }
73
74 /**
75 * Returns the Transform2d that maps the one pose to another.
76 *
77 * @param other The initial pose of the transformation.
78 * @return The transform that maps the other pose to the current pose.
79 */
80 constexpr Transform2d operator-(const Pose2d& other) const;
81
82 /**
83 * Checks equality between this Pose2d and another object.
84 */
85 constexpr bool operator==(const Pose2d&) const = default;
86
87 /**
88 * Returns the underlying translation.
89 *
90 * @return Reference to the translational component of the pose.
91 */
92 constexpr const Translation2d& Translation() const { return m_translation; }
93
94 /**
95 * Returns the X component of the pose's translation.
96 *
97 * @return The x component of the pose's translation.
98 */
99 constexpr units::meter_t X() const { return m_translation.X(); }
100
101 /**
102 * Returns the Y component of the pose's translation.
103 *
104 * @return The y component of the pose's translation.
105 */
106 constexpr units::meter_t Y() const { return m_translation.Y(); }
107
108 /**
109 * Returns the underlying rotation.
110 *
111 * @return Reference to the rotational component of the pose.
112 */
113 constexpr const Rotation2d& Rotation() const { return m_rotation; }
114
115 /**
116 * Multiplies the current pose by a scalar.
117 *
118 * @param scalar The scalar.
119 *
120 * @return The new scaled Pose2d.
121 */
122 constexpr Pose2d operator*(double scalar) const {
123 return Pose2d{m_translation * scalar, m_rotation * scalar};
124 }
125
126 /**
127 * Divides the current pose by a scalar.
128 *
129 * @param scalar The scalar.
130 *
131 * @return The new scaled Pose2d.
132 */
133 constexpr Pose2d operator/(double scalar) const {
134 return *this * (1.0 / scalar);
135 }
136
137 /**
138 * Rotates the pose around the origin and returns the new pose.
139 *
140 * @param other The rotation to transform the pose by.
141 *
142 * @return The rotated pose.
143 */
144 constexpr Pose2d RotateBy(const Rotation2d& other) const {
145 return {m_translation.RotateBy(other), m_rotation.RotateBy(other)};
146 }
147
148 /**
149 * Transforms the pose by the given transformation and returns the new pose.
150 * See + operator for the matrix multiplication performed.
151 *
152 * @param other The transform to transform the pose by.
153 *
154 * @return The transformed pose.
155 */
156 constexpr Pose2d TransformBy(const Transform2d& other) const;
157
158 /**
159 * Returns the current pose relative to the given pose.
160 *
161 * This function can often be used for trajectory tracking or pose
162 * stabilization algorithms to get the error between the reference and the
163 * current pose.
164 *
165 * @param other The pose that is the origin of the new coordinate frame that
166 * the current pose will be converted into.
167 *
168 * @return The current pose relative to the new origin pose.
169 */
170 constexpr Pose2d RelativeTo(const Pose2d& other) const;
171
172 /**
173 * Obtain a new Pose2d from a (constant curvature) velocity.
174 *
175 * See https://file.tavsys.net/control/controls-engineering-in-frc.pdf section
176 * 10.2 "Pose exponential" for a derivation.
177 *
178 * The twist is a change in pose in the robot's coordinate frame since the
179 * previous pose update. When the user runs exp() on the previous known
180 * field-relative pose with the argument being the twist, the user will
181 * receive the new field-relative pose.
182 *
183 * "Exp" represents the pose exponential, which is solving a differential
184 * equation moving the pose forward in time.
185 *
186 * @param twist The change in pose in the robot's coordinate frame since the
187 * previous pose update. For example, if a non-holonomic robot moves forward
188 * 0.01 meters and changes angle by 0.5 degrees since the previous pose
189 * update, the twist would be Twist2d{0.01_m, 0_m, 0.5_deg}.
190 *
191 * @return The new pose of the robot.
192 */
193 constexpr Pose2d Exp(const Twist2d& twist) const;
194
195 /**
196 * Returns a Twist2d that maps this pose to the end pose. If c is the output
197 * of a.Log(b), then a.Exp(c) would yield b.
198 *
199 * @param end The end pose for the transformation.
200 *
201 * @return The twist that maps this to end.
202 */
203 constexpr Twist2d Log(const Pose2d& end) const;
204
205 /**
206 * Returns the nearest Pose2d from a collection of poses
207 * @param poses The collection of poses.
208 * @return The nearest Pose2d from the collection.
209 */
210 constexpr Pose2d Nearest(std::span<const Pose2d> poses) const {
211 return *std::min_element(
212 poses.begin(), poses.end(), [this](const Pose2d& a, const Pose2d& b) {
213 auto aDistance = this->Translation().Distance(a.Translation());
214 auto bDistance = this->Translation().Distance(b.Translation());
215
216 // If the distances are equal sort by difference in rotation
217 if (aDistance == bDistance) {
218 return gcem::abs(
219 (this->Rotation() - a.Rotation()).Radians().value()) <
220 gcem::abs(
221 (this->Rotation() - b.Rotation()).Radians().value());
222 }
223 return aDistance < bDistance;
224 });
225 }
226
227 /**
228 * Returns the nearest Pose2d from a collection of poses
229 * @param poses The collection of poses.
230 * @return The nearest Pose2d from the collection.
231 */
232 constexpr Pose2d Nearest(std::initializer_list<Pose2d> poses) const {
233 return *std::min_element(
234 poses.begin(), poses.end(), [this](const Pose2d& a, const Pose2d& b) {
235 auto aDistance = this->Translation().Distance(a.Translation());
236 auto bDistance = this->Translation().Distance(b.Translation());
237
238 // If the distances are equal sort by difference in rotation
239 if (aDistance == bDistance) {
240 return gcem::abs(
241 (this->Rotation() - a.Rotation()).Radians().value()) <
242 gcem::abs(
243 (this->Rotation() - b.Rotation()).Radians().value());
244 }
245 return aDistance < bDistance;
246 });
247 }
248
249 private:
250 Translation2d m_translation;
251 Rotation2d m_rotation;
252};

286constexpr Pose2d Pose2d::Exp(const Twist2d& twist) const {
287 const auto dx = twist.dx;
288 const auto dy = twist.dy;
289 const auto dtheta = twist.dtheta.value();
290
291 const auto sinTheta = gcem::sin(dtheta);
292 const auto cosTheta = gcem::cos(dtheta);
293
294 double s, c;
295 if (gcem::abs(dtheta) < 1E-9) {
296 s = 1.0 - 1.0 / 6.0 * dtheta * dtheta;
297 c = 0.5 * dtheta;
298 } else {
299 s = sinTheta / dtheta;
300 c = (1 - cosTheta) / dtheta;
301 }
302
303 const Transform2d transform{Translation2d{dx * s - dy * c, dx * c + dy * s},
304 Rotation2d{cosTheta, sinTheta}};
305
306 return *this + transform;
307}

309constexpr Twist2d Pose2d::Log(const Pose2d& end) const {
310 const auto transform = end.RelativeTo(*this);
311 const auto dtheta = transform.Rotation().Radians().value();
312 const auto halfDtheta = dtheta / 2.0;
313
314 const auto cosMinusOne = transform.Rotation().Cos() - 1;
315
316 double halfThetaByTanOfHalfDtheta;
317
318 if (gcem::abs(cosMinusOne) < 1E-9) {
319 halfThetaByTanOfHalfDtheta = 1.0 - 1.0 / 12.0 * dtheta * dtheta;
320 } else {
321 halfThetaByTanOfHalfDtheta =
322 -(halfDtheta * transform.Rotation().Sin()) / cosMinusOne;
323 }
324
325 const Translation2d translationPart =
326 transform.Translation().RotateBy(
327 {halfThetaByTanOfHalfDtheta, -halfDtheta}) *
328 gcem::hypot(halfThetaByTanOfHalfDtheta, halfDtheta);
329
330 return {translationPart.X(), translationPart.Y(), units::radian_t{dtheta}};
331}
