32template <typename T>
33void HttpRequest::SetPath(std::string_view path_, const T& params) {
34 // Build location including query string
35 raw_svector_ostream pathOs{path};
36 pathOs << path_;
37 bool first = true;
38 for (const auto& param : params) {
39 if (first) {
40 pathOs << '?';
41 first = false;
42 } else {
43 pathOs << '&';
44 }
45 SmallString<64> escapeBuf;
46 pathOs << EscapeURI(GetFirst(param), escapeBuf, false);
47 if (!GetSecond(param).empty()) {
48 pathOs << '=' << EscapeURI(GetSecond(param), escapeBuf, false);
49 }
50 }
51}
