16/**
17 * Class for getting voltage, current, temperature, power and energy from the
18 * CTRE Power Distribution Panel (PDP) or REV Power Distribution Hub (PDH).
19 */
20class PowerDistribution : public wpi::Sendable,
21 public wpi::SendableHelper<PowerDistribution> {
22 public:
23 /// Default module number.
24 static constexpr int kDefaultModule = -1;
25
26 /**
27 * Power distribution module type.
28 */
29 enum class ModuleType {
30 /// CTRE (Cross The Road Electronics) CTRE Power Distribution Panel (PDP).
31 kCTRE = 1,
32 /// REV Power Distribution Hub (PDH).
33 kRev = 2
34 };
35
36 /**
37 * Constructs a PowerDistribution object.
38 *
39 * Detects the connected PDP/PDH using the default CAN ID (0 for CTRE and 1
40 * for REV).
41 */
42 PowerDistribution();
43
44 /**
45 * Constructs a PowerDistribution object.
46 *
47 * @param module The CAN ID of the PDP/PDH
48 * @param moduleType The type of module
49 */
50 PowerDistribution(int module, ModuleType moduleType);
51
52 PowerDistribution(PowerDistribution&&) = default;
53 PowerDistribution& operator=(PowerDistribution&&) = default;
54
55 ~PowerDistribution() override = default;
56
57 /**
58 * Gets the number of channels for this power distribution object.
59 *
60 * @return Number of output channels (16 for PDP, 24 for PDH).
61 */
62 int GetNumChannels() const;
63
64 /**
65 * Query the input voltage of the PDP/PDH.
66 *
67 * @return The input voltage in volts
68 */
69 double GetVoltage() const;
70
71 /**
72 * Query the temperature of the PDP.
73 *
74 * Not supported on the Rev PDH and returns 0.
75 *
76 *
77 * @return The temperature in degrees Celsius
78 */
79 double GetTemperature() const;
80
81 /**
82 * Query the current of a single channel of the PDP/PDH.
83 *
84 * @param channel the channel to query (0-15 for PDP, 0-23 for PDH)
85 * @return The current of the channel in Amperes
86 */
87 double GetCurrent(int channel) const;
88
89 /**
90 * Query all currents of the PDP.
91 *
92 * @return The current of each channel in Amperes
93 */
94 std::vector<double> GetAllCurrents() const;
95
96 /**
97 * Query the total current of all monitored PDP/PDH channels.
98 *
99 * @return The total current drawn from all channels in Amperes
100 */
101 double GetTotalCurrent() const;
102
103 /**
104 * Query the total power drawn from all monitored PDP channels.
105 *
106 * Not supported on the Rev PDH and returns 0.
107 *
108 * @return The total power drawn in Watts
109 */
110 double GetTotalPower() const;
111
112 /**
113 * Query the total energy drawn from the monitored PDP channels.
114 *
115 * Not supported on the Rev PDH and returns 0.
116 *
117 * @return The total energy drawn in Joules
118 */
119 double GetTotalEnergy() const;
120
121 /**
122 * Reset the total energy drawn from the PDP.
123 *
124 * Not supported on the Rev PDH and does nothing.
125 *
126 * @see PowerDistribution#GetTotalEnergy
127 */
128 void ResetTotalEnergy();
129
130 /**
131 * Remove all of the fault flags on the PDP/PDH.
132 */
133 void ClearStickyFaults();
134
135 /**
136 * Gets module number (CAN ID).
137 */
138 int GetModule() const;
139
140 /**
141 * Gets module type.
142 */
143 ModuleType GetType() const;
144
145 /**
146 * Gets whether the PDH switchable channel is turned on or off. Returns false
147 * with the CTRE PDP.
148 *
149 * @return The output state of the PDH switchable channel
150 */
151 bool GetSwitchableChannel() const;
152
153 /**
154 * Sets the PDH switchable channel on or off. Does nothing with the CTRE PDP.
155 *
156 * @param enabled Whether to turn the PDH switchable channel on or off
157 */
158 void SetSwitchableChannel(bool enabled);
