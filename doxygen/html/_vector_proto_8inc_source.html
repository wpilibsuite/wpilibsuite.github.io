22template <int Size, int Options, int MaxRows, int MaxCols>
23frc::Matrixd<Size, 1, Options, MaxRows, MaxCols>
24wpi::Protobuf<frc::Matrixd<Size, 1, Options, MaxRows, MaxCols>>::Unpack(
25 const google::protobuf::Message& msg) {
26 auto m = static_cast<const wpi::proto::ProtobufVector*>(&msg);
27 if (m->rows_size() != Size) {
28 throw std::invalid_argument(
29 fmt::format("Tried to unpack message with {} elements in rows into "
30 "Vector with {} rows",
31 m->rows_size(), Size));
32 }
33 frc::Matrixd<Size, 1, Options, MaxRows, MaxCols> vec;
34 for (int i = 0; i < Size; i++) {
35 vec(i) = m->rows(i);
36 }
37 return vec;
38}
