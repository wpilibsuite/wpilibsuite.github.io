33constexpr Translation2d Translation2d::RotateAround(
34 const Translation2d& other, const Rotation2d& rot) const {
35 return {
36 (m_x - other.X()) * rot.Cos() - (m_y - other.Y()) * rot.Sin() + other.X(),
37 (m_x - other.X()) * rot.Sin() + (m_y - other.Y()) * rot.Cos() +
38 other.Y()};
39}
