42// This implementation of bit_cast is different from the C++20 one in two ways:
43// - It isn't constexpr because that requires compiler support.
44// - It requires trivially-constructible To, to avoid UB in the implementation.
45template <
46 typename To, typename From,
47 typename = std::enable_if_t<sizeof(To) == sizeof(From)>,
48 typename = std::enable_if_t<std::is_trivially_constructible<To>::value>,
49 typename = std::enable_if_t<std::is_trivially_copyable<To>::value>,
50 typename = std::enable_if_t<std::is_trivially_copyable<From>::value>>
51[[nodiscard]] inline To bit_cast(const From &from) noexcept {
52#if __has_builtin(__builtin_bit_cast)
53 return __builtin_bit_cast(To, from);
54#else
55 To to;
56 std::memcpy(&to, &from, sizeof(To));
57 return to;
58#endif
59}

61/// Reverses the bytes in the given integer value V.
62template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
63[[nodiscard]] constexpr T byteswap(T V) noexcept {
64 if constexpr (sizeof(T) == 1) {
65 return V;
66 } else if constexpr (sizeof(T) == 2) {
67 uint16_t UV = V;
68#if defined(_MSC_VER) && !defined(_DEBUG)
69 // The DLL version of the runtime lacks these functions (bug!?), but in a
70 // release build they're replaced with BSWAP instructions anyway.
71 return _byteswap_ushort(UV);
72#else
73 uint16_t Hi = UV << 8;
74 uint16_t Lo = UV >> 8;
75 return Hi | Lo;
76#endif
77 } else if constexpr (sizeof(T) == 4) {
78 uint32_t UV = V;
79#if __has_builtin(__builtin_bswap32)
80 return __builtin_bswap32(UV);
81#elif defined(_MSC_VER) && !defined(_DEBUG)
82 return _byteswap_ulong(UV);
83#else
84 uint32_t Byte0 = UV & 0x000000FF;
85 uint32_t Byte1 = UV & 0x0000FF00;
86 uint32_t Byte2 = UV & 0x00FF0000;
87 uint32_t Byte3 = UV & 0xFF000000;
88 return (Byte0 << 24) | (Byte1 << 8) | (Byte2 >> 8) | (Byte3 >> 24);
89#endif
90 } else if constexpr (sizeof(T) == 8) {
91 uint64_t UV = V;
92#if __has_builtin(__builtin_bswap64)
93 return __builtin_bswap64(UV);
94#elif defined(_MSC_VER) && !defined(_DEBUG)
95 return _byteswap_uint64(UV);
96#else
97 uint64_t Hi = wpi::byteswap<uint32_t>(UV);
98 uint32_t Lo = wpi::byteswap<uint32_t>(UV >> 32);
99 return (Hi << 32) | Lo;
100#endif
101 } else {
102 static_assert(!sizeof(T *), "Don't know how to handle the given type.");
103 return 0;
104 }
105}
