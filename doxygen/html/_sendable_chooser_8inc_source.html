58template <class T>
59 requires std::copy_constructible<T> && std::default_initializable<T>
60void SendableChooser<T>::InitSendable(wpi::SendableBuilder& builder) {
61 builder.SetSmartDashboardType("String Chooser");
62 builder.PublishConstInteger(kInstance, m_instance);
63 builder.AddStringArrayProperty(
64 kOptions,
65 [=, this] {
66 std::vector<std::string> keys;
67 for (const auto& choice : m_choices) {
68 keys.emplace_back(choice.first());
69 }
70
71 // Unlike std::map, wpi::StringMap elements
72 // are not sorted
73 std::sort(keys.begin(), keys.end());
74
75 return keys;
76 },
77 nullptr);
78 builder.AddSmallStringProperty(
79 kDefault,
80 [=, this](wpi::SmallVectorImpl<char>&) -> std::string_view {
81 return m_defaultChoice;
82 },
83 nullptr);
84 builder.AddSmallStringProperty(
85 kActive,
86 [=, this](wpi::SmallVectorImpl<char>& buf) -> std::string_view {
87 std::scoped_lock lock(m_mutex);
88 if (m_haveSelected) {
89 buf.assign(m_selected.begin(), m_selected.end());
90 return {buf.data(), buf.size()};
91 } else {
92 return m_defaultChoice;
93 }
94 },
95 nullptr);
96 builder.AddStringProperty(kSelected, nullptr,
97 [=, this](std::string_view val) {
98 T choice{};
99 std::function<void(T)> listener;
100 {
101 std::scoped_lock lock(m_mutex);
102 m_haveSelected = true;
103 m_selected = val;
104 if (m_previousVal != val && m_listener) {
105 choice = m_choices[val];
106 listener = m_listener;
107 }
108 m_previousVal = val;
109 }
110 if (listener) {
111 listener(choice);
112 }
113 });
114}
