13template <typename WheelSpeeds, typename WheelPositions>
14PoseEstimator<WheelSpeeds, WheelPositions>::PoseEstimator(
15 Kinematics<WheelSpeeds, WheelPositions>& kinematics,
16 Odometry<WheelSpeeds, WheelPositions>& odometry,
17 const wpi::array<double, 3>& stateStdDevs,
18 const wpi::array<double, 3>& visionMeasurementStdDevs)
19 : m_odometry(odometry), m_poseEstimate(m_odometry.GetPose()) {
20 for (size_t i = 0; i < 3; ++i) {
21 m_q[i] = stateStdDevs[i] * stateStdDevs[i];
22 }
23
24 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
25}

27template <typename WheelSpeeds, typename WheelPositions>
28void PoseEstimator<WheelSpeeds, WheelPositions>::SetVisionMeasurementStdDevs(
29 const wpi::array<double, 3>& visionMeasurementStdDevs) {
30 wpi::array<double, 3> r{wpi::empty_array};
31 for (size_t i = 0; i < 3; ++i) {
32 r[i] = visionMeasurementStdDevs[i] * visionMeasurementStdDevs[i];
33 }
34
35 // Solve for closed form Kalman gain for continuous Kalman filter with A = 0
36 // and C = I. See wpimath/algorithms.md.
37 for (size_t row = 0; row < 3; ++row) {
38 if (m_q[row] == 0.0) {
39 m_visionK(row, row) = 0.0;
40 } else {
41 m_visionK(row, row) =
42 m_q[row] / (m_q[row] + std::sqrt(m_q[row] * r[row]));
43 }
44 }
45}

89template <typename WheelSpeeds, typename WheelPositions>
90std::optional<Pose2d> PoseEstimator<WheelSpeeds, WheelPositions>::SampleAt(
91 units::second_t timestamp) const {
92 // Step 0: If there are no odometry updates to sample, skip.
93 if (m_odometryPoseBuffer.GetInternalBuffer().empty()) {
94 return std::nullopt;
95 }
96
97 // Step 1: Make sure timestamp matches the sample from the odometry pose
98 // buffer. (When sampling, the buffer will always use a timestamp
99 // between the first and last timestamps)
100 units::second_t oldestOdometryTimestamp =
101 m_odometryPoseBuffer.GetInternalBuffer().front().first;
102 units::second_t newestOdometryTimestamp =
103 m_odometryPoseBuffer.GetInternalBuffer().back().first;
104 timestamp =
105 std::clamp(timestamp, oldestOdometryTimestamp, newestOdometryTimestamp);
106
107 // Step 2: If there are no applicable vision updates, use the odometry-only
108 // information.
109 if (m_visionUpdates.empty() || timestamp < m_visionUpdates.begin()->first) {
110 return m_odometryPoseBuffer.Sample(timestamp);
111 }
112
113 // Step 3: Get the latest vision update from before or at the timestamp to
114 // sample at.
115 // First, find the iterator past the sample timestamp, then go back one. Note
116 // that upper_bound() won't return begin() because we check begin() earlier.
117 auto floorIter = m_visionUpdates.upper_bound(timestamp);
118 --floorIter;
119 auto visionUpdate = floorIter->second;
120
121 // Step 4: Get the pose measured by odometry at the time of the sample.
122 auto odometryEstimate = m_odometryPoseBuffer.Sample(timestamp);
123
124 // Step 5: Apply the vision compensation to the odometry pose.
125 // TODO Replace with std::optional::transform() in C++23
126 if (odometryEstimate) {
127 return visionUpdate.Compensate(*odometryEstimate);
128 }
129 return std::nullopt;
130}

132template <typename WheelSpeeds, typename WheelPositions>
133void PoseEstimator<WheelSpeeds, WheelPositions>::CleanUpVisionUpdates() {
134 // Step 0: If there are no odometry samples, skip.
135 if (m_odometryPoseBuffer.GetInternalBuffer().empty()) {
136 return;
137 }
138
139 // Step 1: Find the oldest timestamp that needs a vision update.
140 units::second_t oldestOdometryTimestamp =
141 m_odometryPoseBuffer.GetInternalBuffer().front().first;
142
143 // Step 2: If there are no vision updates before that timestamp, skip.
144 if (m_visionUpdates.empty() ||
145 oldestOdometryTimestamp < m_visionUpdates.begin()->first) {
146 return;
147 }
148
149 // Step 3: Find the newest vision update timestamp before or at the oldest
150 // timestamp.
151 // First, find the iterator past the oldest odometry timestamp, then go
152 // back one. Note that upper_bound() won't return begin() because we check
153 // begin() earlier.
154 auto newestNeededVisionUpdate =
155 m_visionUpdates.upper_bound(oldestOdometryTimestamp);
156 --newestNeededVisionUpdate;
157
158 // Step 4: Remove all entries strictly before the newest timestamp we need.
159 m_visionUpdates.erase(m_visionUpdates.begin(), newestNeededVisionUpdate);
160}

162template <typename WheelSpeeds, typename WheelPositions>
163void PoseEstimator<WheelSpeeds, WheelPositions>::AddVisionMeasurement(
164 const Pose2d& visionRobotPose, units::second_t timestamp) {
165 // Step 0: If this measurement is old enough to be outside the pose buffer's
166 // timespan, skip.
167 if (m_odometryPoseBuffer.GetInternalBuffer().empty() ||
168 m_odometryPoseBuffer.GetInternalBuffer().front().first - kBufferDuration >
169 timestamp) {
170 return;
171 }
172
173 // Step 1: Clean up any old entries
174 CleanUpVisionUpdates();
175
176 // Step 2: Get the pose measured by odometry at the moment the vision
177 // measurement was made.
178 auto odometrySample = m_odometryPoseBuffer.Sample(timestamp);
179
180 if (!odometrySample) {
181 return;
182 }
183
184 // Step 3: Get the vision-compensated pose estimate at the moment the vision
185 // measurement was made.
186 auto visionSample = SampleAt(timestamp);
187
188 if (!visionSample) {
189 return;
190 }
191
192 // Step 4: Measure the twist between the old pose estimate and the vision
193 // pose.
194 auto twist = visionSample.value().Log(visionRobotPose);
195
196 // Step 5: We should not trust the twist entirely, so instead we scale this
197 // twist by a Kalman gain matrix representing how much we trust vision
198 // measurements compared to our current pose.
199 Eigen::Vector3d k_times_twist =
200 m_visionK *
201 Eigen::Vector3d{twist.dx.value(), twist.dy.value(), twist.dtheta.value()};
202
203 // Step 6: Convert back to Twist2d.
204 Twist2d scaledTwist{units::meter_t{k_times_twist(0)},
205 units::meter_t{k_times_twist(1)},
206 units::radian_t{k_times_twist(2)}};
207
208 // Step 7: Calculate and record the vision update.
209 VisionUpdate visionUpdate{visionSample->Exp(scaledTwist), *odometrySample};
210 m_visionUpdates[timestamp] = visionUpdate;
211
212 // Step 8: Remove later vision measurements. (Matches previous behavior)
213 auto firstAfter = m_visionUpdates.upper_bound(timestamp);
214 m_visionUpdates.erase(firstAfter, m_visionUpdates.end());
215
216 // Step 9: Update latest pose estimate. Since we cleared all updates after
217 // this vision update, it's guaranteed to be the latest vision update.
218 m_poseEstimate = visionUpdate.Compensate(m_odometry.GetPose());
219}

228template <typename WheelSpeeds, typename WheelPositions>
229Pose2d PoseEstimator<WheelSpeeds, WheelPositions>::UpdateWithTime(
230 units::second_t currentTime, const Rotation2d& gyroAngle,
231 const WheelPositions& wheelPositions) {
232 auto odometryEstimate = m_odometry.Update(gyroAngle, wheelPositions);
233
234 m_odometryPoseBuffer.AddSample(currentTime, odometryEstimate);
235
236 if (m_visionUpdates.empty()) {
237 m_poseEstimate = odometryEstimate;
238 } else {
239 auto visionUpdate = m_visionUpdates.rbegin()->second;
240 m_poseEstimate = visionUpdate.Compensate(odometryEstimate);
241 }
242
243 return GetEstimatedPosition();
244}
