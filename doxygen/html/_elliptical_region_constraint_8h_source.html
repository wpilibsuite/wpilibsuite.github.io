18/**
19 * Enforces a particular constraint only within an elliptical region.
20 */
21template <std::derived_from<TrajectoryConstraint> Constraint>
22class EllipticalRegionConstraint : public TrajectoryConstraint {
23 public:
24 /**
25 * Constructs a new EllipticalRegionConstraint.
26 *
27 * @param center The center of the ellipse in which to enforce the constraint.
28 * @param xWidth The width of the ellipse in which to enforce the constraint.
29 * @param yWidth The height of the ellipse in which to enforce the constraint.
30 * @param rotation The rotation to apply to all radii around the origin.
31 * @param constraint The constraint to enforce when the robot is within the
32 * region.
33 * @deprecated Use constructor taking Ellipse2d instead.
34 */
35 [[deprecated("Use constructor taking Ellipse2d instead.")]]
36 EllipticalRegionConstraint(const Translation2d& center, units::meter_t xWidth,
37 units::meter_t yWidth, const Rotation2d& rotation,
38 const Constraint& constraint)
39 : m_ellipse{Pose2d{center, rotation}, xWidth / 2.0, yWidth / 2.0},
40 m_constraint(constraint) {}
41
42 /**
43 * Constructs a new EllipticalRegionConstraint.
44 *
45 * @param ellipse The ellipse in which to enforce the constraint.
46 * @param constraint The constraint to enforce when the robot is within the
47 * region.
48 */
49 EllipticalRegionConstraint(const Ellipse2d& ellipse,
50 const Constraint& constraint)
51 : m_ellipse{ellipse}, m_constraint{constraint} {}
52
53 units::meters_per_second_t MaxVelocity(
54 const Pose2d& pose, units::curvature_t curvature,
55 units::meters_per_second_t velocity) const override {
56 if (m_ellipse.Contains(pose.Translation())) {
57 return m_constraint.MaxVelocity(pose, curvature, velocity);
58 } else {
59 return units::meters_per_second_t{
60 std::numeric_limits<double>::infinity()};
61 }
62 }
63
64 MinMax MinMaxAcceleration(const Pose2d& pose, units::curvature_t curvature,
65 units::meters_per_second_t speed) const override {
66 if (m_ellipse.Contains(pose.Translation())) {
67 return m_constraint.MinMaxAcceleration(pose, curvature, speed);
68 } else {
69 return {};
70 }
71 }
72
73 private:
74 Ellipse2d m_ellipse;
75 Constraint m_constraint;
76};
77
78} // namespace frc
