17namespace frc::sim {
18/**
19 * This class helps simulate linear systems. To use this class, do the following
20 * in the simulationPeriodic() method.
21 *
22 * Call the SetInput() method with the inputs to your system (generally
23 * voltage). Call the Update() method to update the simulation. Set simulated
24 * sensor readings with the simulated positions in the GetOutput() method.
25 *
26 * @tparam States Number of states of the system.
27 * @tparam Inputs Number of inputs to the system.
28 * @tparam Outputs Number of outputs of the system.
29 */
30template <int States, int Inputs, int Outputs>
31class LinearSystemSim {
32 public:
33 /**
34 * Creates a simulated generic linear system.
35 *
36 * @param system The system to simulate.
37 * @param measurementStdDevs The standard deviations of the measurements.
38 */
39 explicit LinearSystemSim(
40 const LinearSystem<States, Inputs, Outputs>& system,
41 const std::array<double, Outputs>& measurementStdDevs = {})
42 : m_plant(system), m_measurementStdDevs(measurementStdDevs) {
43 m_x = Vectord<States>::Zero();
44 m_y = Vectord<Outputs>::Zero();
45 m_u = Vectord<Inputs>::Zero();
46 }
47
48 virtual ~LinearSystemSim() = default;
49
50 /**
51 * Updates the simulation.
52 *
53 * @param dt The time between updates.
54 */
55 void Update(units::second_t dt) {
56 // Update x. By default, this is the linear system dynamics x_k+1 = Ax_k +
57 // Bu_k
58 m_x = UpdateX(m_x, m_u, dt);
59
60 // y = Cx + Du
61 m_y = m_plant.CalculateY(m_x, m_u);
62
63 // Add noise. If the user did not pass a noise vector to the
64 // constructor, then this method will not do anything because
65 // the standard deviations default to zero.
66 m_y += frc::MakeWhiteNoiseVector<Outputs>(m_measurementStdDevs);
67 }
68
69 /**
70 * Returns the current output of the plant.
71 *
72 * @return The current output of the plant.
73 */
74 const Vectord<Outputs>& GetOutput() const { return m_y; }
75
76 /**
77 * Returns an element of the current output of the plant.
78 *
79 * @param row The row to return.
80 * @return An element of the current output of the plant.
81 */
82 double GetOutput(int row) const { return m_y(row); }
83
84 /**
85 * Sets the system inputs (usually voltages).
86 *
87 * @param u The system inputs.
88 */
89 void SetInput(const Vectord<Inputs>& u) { m_u = u; }
90
91 /**
92 * Sets the system inputs.
93 *
94 * @param row The row in the input matrix to set.
95 * @param value The value to set the row to.
96 */
97 void SetInput(int row, double value) { m_u(row, 0) = value; }
98
99 /**
100 * Returns the current input of the plant.
101 *
102 * @return The current input of the plant.
103 */
104 const Vectord<Inputs>& GetInput() const { return m_u; }
105
106 /**
107 * Returns an element of the current input of the plant.
108 *
109 * @param row The row to return.
110 * @return An element of the current input of the plant.
111 */
112 double GetInput(int row) const { return m_u(row); }
113
114 /**
115 * Sets the system state.
116 *
117 * @param state The new state.
118 */
119 void SetState(const Vectord<States>& state) { m_x = state; }
120
121 protected:
122 /**
123 * Updates the state estimate of the system.
124 *
125 * @param currentXhat The current state estimate.
126 * @param u The system inputs (usually voltage).
127 * @param dt The time difference between controller updates.
128 */
129 virtual Vectord<States> UpdateX(const Vectord<States>& currentXhat,
130 const Vectord<Inputs>& u,
131 units::second_t dt) {
132 return m_plant.CalculateX(currentXhat, u, dt);
133 }
134
135 /**
136 * Clamp the input vector such that no element exceeds the given voltage. If
137 * any does, the relative magnitudes of the input will be maintained.
138 *
139 * @param maxInput The maximum magnitude of the input vector after clamping.
140 */
141 void ClampInput(double maxInput) {
142 m_u = frc::DesaturateInputVector<Inputs>(m_u, maxInput);
143 }
144
145 /// The plant that represents the linear system.
146 LinearSystem<States, Inputs, Outputs> m_plant;
147
148 /// State vector.
149 Vectord<States> m_x;
150
151 /// Input vector.
152 Vectord<Inputs> m_u;
153
154 /// Output vector.
155 Vectord<Outputs> m_y;
156
157 /// The standard deviations of measurements, used for adding noise to the
158 /// measurements.
159 std::array<double, Outputs> m_measurementStdDevs;
160};
161} // namespace frc::sim
