58/// Read a value of a particular endianness from memory.
59template <typename value_type, std::size_t alignment = unaligned>
60[[nodiscard]] inline value_type read(const void *memory, endianness endian) {
61 value_type ret;
62
63 memcpy(&ret,
64 LLVM_ASSUME_ALIGNED(
65 memory, (detail::PickAlignment<value_type, alignment>::value)),
66 sizeof(value_type));
67 return byte_swap<value_type>(ret, endian);
68}

110/// Read a value of a particular endianness from memory, for a location
111/// that starts at the given bit offset within the first byte.
112template <typename value_type, endianness endian, std::size_t alignment>
113[[nodiscard]] inline value_type readAtBitAlignment(const void *memory,
114 uint64_t startBit) {
115 assert(startBit < 8);
116 if (startBit == 0)
117 return read<value_type, endian, alignment>(memory);
118 else {
119 // Read two values and compose the result from them.
120 value_type val[2];
121 memcpy(&val[0],
122 LLVM_ASSUME_ALIGNED(
123 memory, (detail::PickAlignment<value_type, alignment>::value)),
124 sizeof(value_type) * 2);
125 val[0] = byte_swap<value_type, endian>(val[0]);
126 val[1] = byte_swap<value_type, endian>(val[1]);
127
128 // Shift bits from the lower value into place.
129 make_unsigned_t<value_type> lowerVal = val[0] >> startBit;
130 // Mask off upper bits after right shift in case of signed type.
131 make_unsigned_t<value_type> numBitsFirstVal =
132 (sizeof(value_type) * 8) - startBit;
133 lowerVal &= ((make_unsigned_t<value_type>)1 << numBitsFirstVal) - 1;
134
135 // Get the bits from the upper value.
136 make_unsigned_t<value_type> upperVal =
137 val[1] & (((make_unsigned_t<value_type>)1 << startBit) - 1);
138 // Shift them in to place.
139 upperVal <<= numBitsFirstVal;
140
141 return lowerVal | upperVal;
142 }
143}

145/// Write a value to memory with a particular endianness, for a location
146/// that starts at the given bit offset within the first byte.
147template <typename value_type, endianness endian, std::size_t alignment>
148inline void writeAtBitAlignment(void *memory, value_type value,
149 uint64_t startBit) {
150 assert(startBit < 8);
151 if (startBit == 0)
152 write<value_type, endian, alignment>(memory, value);
153 else {
154 // Read two values and shift the result into them.
155 value_type val[2];
156 memcpy(&val[0],
157 LLVM_ASSUME_ALIGNED(
158 memory, (detail::PickAlignment<value_type, alignment>::value)),
159 sizeof(value_type) * 2);
160 val[0] = byte_swap<value_type, endian>(val[0]);
161 val[1] = byte_swap<value_type, endian>(val[1]);
162
163 // Mask off any existing bits in the upper part of the lower value that
164 // we want to replace.
165 val[0] &= ((make_unsigned_t<value_type>)1 << startBit) - 1;
166 make_unsigned_t<value_type> numBitsFirstVal =
167 (sizeof(value_type) * 8) - startBit;
168 make_unsigned_t<value_type> lowerVal = value;
169 if (startBit > 0) {
170 // Mask off the upper bits in the new value that are not going to go into
171 // the lower value. This avoids a left shift of a negative value, which
172 // is undefined behavior.
173 lowerVal &= (((make_unsigned_t<value_type>)1 << numBitsFirstVal) - 1);
174 // Now shift the new bits into place
175 lowerVal <<= startBit;
176 }
177 val[0] |= lowerVal;
178
179 // Mask off any existing bits in the lower part of the upper value that
180 // we want to replace.
181 val[1] &= ~(((make_unsigned_t<value_type>)1 << startBit) - 1);
182 // Next shift the bits that go into the upper value into position.
183 make_unsigned_t<value_type> upperVal = value >> numBitsFirstVal;
184 // Mask off upper bits after right shift in case of signed type.
185 upperVal &= ((make_unsigned_t<value_type>)1 << startBit) - 1;
186 val[1] |= upperVal;
187
188 // Finally, rewrite values.
189 val[0] = byte_swap<value_type, endian>(val[0]);
190 val[1] = byte_swap<value_type, endian>(val[1]);
191 memcpy(LLVM_ASSUME_ALIGNED(
192 memory, (detail::PickAlignment<value_type, alignment>::value)),
193 &val[0], sizeof(value_type) * 2);
194 }
195}

201template <typename ValueType, endianness Endian, std::size_t Alignment,
202 std::size_t ALIGN = PickAlignment<ValueType, Alignment>::value>
203struct packed_endian_specific_integral {
204 using value_type = ValueType;
205 static constexpr endianness endian = Endian;
206 static constexpr std::size_t alignment = Alignment;
207
208 packed_endian_specific_integral() = default;
209
210 explicit packed_endian_specific_integral(value_type val) { *this = val; }
211
212 operator value_type() const {
213 return endian::read<value_type, endian, alignment>(
214 (const void*)Value.buffer);
215 }
216
217 void operator=(value_type newValue) {
218 endian::write<value_type, endian, alignment>(
219 (void*)Value.buffer, newValue);
220 }
221
222 packed_endian_specific_integral &operator+=(value_type newValue) {
223 *this = *this + newValue;
224 return *this;
225 }
226
227 packed_endian_specific_integral &operator-=(value_type newValue) {
228 *this = *this - newValue;
229 return *this;
230 }
231
232 packed_endian_specific_integral &operator|=(value_type newValue) {
233 *this = *this | newValue;
234 return *this;
235 }
236
237 packed_endian_specific_integral &operator&=(value_type newValue) {
238 *this = *this & newValue;
239 return *this;
240 }
241
242private:
243 struct {
244 alignas(ALIGN) char buffer[sizeof(value_type)];
245 } Value;
246
247public:
248 struct ref {
249 explicit ref(void *Ptr) : Ptr(Ptr) {}
250
251 operator value_type() const {
252 return endian::read<value_type, endian, alignment>(Ptr);
253 }
254
255 void operator=(value_type NewValue) {
256 endian::write<value_type, endian, alignment>(Ptr, NewValue);
257 }
258
259 private:
260 void *Ptr;
261 };
262};
