12namespace frc {
13/**
14 * Helper class that converts a chassis velocity (dx, dy, and dtheta components)
15 * into individual wheel speeds. Robot code should not use this directly-
16 * Instead, use the particular type for your drivetrain (e.g.,
17 * DifferentialDriveKinematics).
18 *
19 * Inverse kinematics converts a desired chassis speed into wheel speeds whereas
20 * forward kinematics converts wheel speeds into chassis speed.
21 */
22template <typename WheelSpeeds, typename WheelPositions>
23 requires std::copy_constructible<WheelPositions> &&
24 std::assignable_from<WheelPositions&, const WheelPositions&>
25class WPILIB_DLLEXPORT Kinematics {
26 public:
27 /**
28 * Performs forward kinematics to return the resulting chassis speed from the
29 * wheel speeds. This method is often used for odometry -- determining the
30 * robot's position on the field using data from the real-world speed of each
31 * wheel on the robot.
32 *
33 * @param wheelSpeeds The speeds of the wheels.
34 * @return The chassis speed.
35 */
36 virtual ChassisSpeeds ToChassisSpeeds(
37 const WheelSpeeds& wheelSpeeds) const = 0;
38
39 /**
40 * Performs inverse kinematics to return the wheel speeds from a desired
41 * chassis velocity. This method is often used to convert joystick values into
42 * wheel speeds.
43 *
44 * @param chassisSpeeds The desired chassis speed.
45 * @return The wheel speeds.
46 */
47 virtual WheelSpeeds ToWheelSpeeds(
48 const ChassisSpeeds& chassisSpeeds) const = 0;
49
50 /**
51 * Performs forward kinematics to return the resulting Twist2d from the given
52 * change in wheel positions. This method is often used for odometry --
53 * determining the robot's position on the field using changes in the distance
54 * driven by each wheel on the robot.
55 *
56 * @param start The starting distances driven by the wheels.
57 * @param end The ending distances driven by the wheels.
58 *
59 * @return The resulting Twist2d in the robot's movement.
60 */
61 virtual Twist2d ToTwist2d(const WheelPositions& start,
62 const WheelPositions& end) const = 0;
63
64 /**
65 * Performs interpolation between two values.
66 *
67 * @param start The value to start at.
68 * @param end The value to end at.
69 * @param t How far between the two values to interpolate. This should be
70 * bounded to [0, 1].
71 * @return The interpolated value.
72 */
73 virtual WheelPositions Interpolate(const WheelPositions& start,
74 const WheelPositions& end,
75 double t) const = 0;
76};
77} // namespace frc
