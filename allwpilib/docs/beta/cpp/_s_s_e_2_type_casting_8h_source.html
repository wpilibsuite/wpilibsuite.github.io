104 float f1 = static_cast<float>(h);
105 h = raw_uint16_to_half(static_cast<unsigned short>(a64 >> 16));
106 float f2 = static_cast<float>(h);

108 float f3 = static_cast<float>(h);

110 float f4 = static_cast<float>(h);
