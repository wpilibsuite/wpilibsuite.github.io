21 /// A \concept{concept_segregatable,Segregatable} that allocates until a maximum size.
22 /// \ingroup adapter
23 template <class RawAllocator>
24 class threshold_segregatable : WPI_EBO(allocator_traits<RawAllocator>::allocator_type)
25 {
26 public:
27 using allocator_type = typename allocator_traits<RawAllocator>::allocator_type;
28
29 /// \effects Creates it by passing the maximum size it will allocate
30 /// and the allocator it uses.
31 explicit threshold_segregatable(std::size_t max_size,
32 allocator_type alloc = allocator_type())
33 : allocator_type(detail::move(alloc)), max_size_(max_size)
34 {
35 }
36
37 /// \returns `true` if `size` is less then or equal to the maximum size,
38 /// `false` otherwise.
39 /// \note A return value of `true` means that the allocator will be used for the allocation.
40 bool use_allocate_node(std::size_t size, std::size_t) noexcept
41 {
42 return size <= max_size_;
43 }
44
45 /// \returns `true` if `count * size` is less then or equal to the maximum size,
46 /// `false` otherwise.
47 /// \note A return value of `true` means that the allocator will be used for the allocation.
48 bool use_allocate_array(std::size_t count, std::size_t size, std::size_t) noexcept
49 {
50 return count * size <= max_size_;
51 }
52
53 /// @{
54 /// \returns A reference to the allocator it owns.
55 allocator_type& get_allocator() noexcept
56 {
57 return *this;
58 }
59
60 const allocator_type& get_allocator() const noexcept
61 {
62 return *this;
63 }
64 /// @}
65
66 private:
67 std::size_t max_size_;
68 };

120 /// A \concept{concept_rawallocator,RawAllocator} that either uses the \concept{concept_segregatable,Segregatable} or the other `RawAllocator`.
121 /// It is a faster alternative to \ref fallback_allocator that doesn't require a composable allocator
122 /// and decides about the allocator to use purely with the `Segregatable` based on size and alignment.
123 /// \ingroup adapter
124 template <class Segregatable, class RawAllocator>
125 class binary_segregator
126 : WPI_EBO(
127 detail::ebo_storage<1, typename allocator_traits<RawAllocator>::allocator_type>)
128 {
129 using segregatable_traits = allocator_traits<typename Segregatable::allocator_type>;
130 using fallback_traits = allocator_traits<RawAllocator>;
131
132 public:
133 using segregatable = Segregatable;
134 using segregatable_allocator_type = typename segregatable::allocator_type;
135 using fallback_allocator_type = typename allocator_traits<RawAllocator>::allocator_type;
136
137 /// \effects Creates it by giving the \concept{concept_segregatable,Segregatable}
138 /// and the \concept{concept_rawallocator,RawAllocator}.
139 explicit binary_segregator(segregatable s,
140 fallback_allocator_type fallback = fallback_allocator_type())
141 : detail::ebo_storage<1, fallback_allocator_type>(detail::move(fallback)),
142 s_(detail::move(s))
143 {
144 }
145
146 /// @{
147 /// \effects Uses the \concept{concept_segregatable,Segregatable} to decide which allocator to use.
148 /// Then forwards to the chosen allocator.
149 void* allocate_node(std::size_t size, std::size_t alignment)
150 {
151 if (get_segregatable().use_allocate_node(size, alignment))
152 return segregatable_traits::allocate_node(get_segregatable_allocator(), size,
153 alignment);
154 else
155 return fallback_traits::allocate_node(get_fallback_allocator(), size,
156 alignment);
157 }
158
159 void deallocate_node(void* ptr, std::size_t size, std::size_t alignment) noexcept
160 {
161 if (get_segregatable().use_allocate_node(size, alignment))
162 segregatable_traits::deallocate_node(get_segregatable_allocator(), ptr, size,
163 alignment);
164 else
165 fallback_traits::deallocate_node(get_fallback_allocator(), ptr, size,
166 alignment);
167 }
168
169 void* allocate_array(std::size_t count, std::size_t size, std::size_t alignment)
170 {
171 if (get_segregatable().use_allocate_array(count, size, alignment))
172 return segregatable_traits::allocate_array(get_segregatable_allocator(), count,
173 size, alignment);
174 else
175 return fallback_traits::allocate_array(get_fallback_allocator(), count, size,
176 alignment);
177 }
178
179 void deallocate_array(void* array, std::size_t count, std::size_t size,
180 std::size_t alignment) noexcept
181 {
182 if (get_segregatable().use_allocate_array(count, size, alignment))
183 segregatable_traits::deallocate_array(get_segregatable_allocator(), array,
184 count, size, alignment);
185 else
186 fallback_traits::deallocate_array(get_fallback_allocator(), array, count, size,
187 alignment);
188 }
189 /// @}
190
191 /// @{
192 /// \returns The maximum value of the fallback.
193 /// \note It assumes that the fallback will be used for larger allocations,
194 /// and the `Segregatable` for smaller ones.
195 std::size_t max_node_size() const
196 {
197 return fallback_traits::max_node_size(get_fallback_allocator());
198 }
199
200 std::size_t max_array_size() const
201 {
202 return fallback_traits::max_array_size(get_fallback_allocator());
203 }
204
205 std::size_t max_alignemnt() const
206 {
207 return fallback_traits::max_alignment(get_fallback_allocator());
208 }
209 /// @}
210
211 /// @{
212 /// \returns A reference to the segregatable allocator.
213 /// This is the one primarily used.
214 segregatable_allocator_type& get_segregatable_allocator() noexcept
215 {
216 return get_segregatable().get_allocator();
217 }
218
219 const segregatable_allocator_type& get_segregatable_allocator() const noexcept
220 {
221 return get_segregatable().get_allocator();
222 }
223 /// @}
224
225 /// @{
226 /// \returns A reference to the fallback allocator.
227 /// It will be used if the \concept{concept_segregator,Segregator} doesn't want the alloction.
228 fallback_allocator_type& get_fallback_allocator() noexcept
229 {
230 return detail::ebo_storage<1, fallback_allocator_type>::get();
231 }
232
233 const fallback_allocator_type& get_fallback_allocator() const noexcept
234 {
235 return detail::ebo_storage<1, fallback_allocator_type>::get();
236 }
237 /// @}
238
239 private:
240 segregatable& get_segregatable() noexcept
241 {
242 return s_;
243 }
244
245 segregatable s_;
246 };

369 /// Creates multiple nested \ref binary_segregator.
370 /// If you pass one type, it must be a \concept{concept_segregatable,Segregatable}.
371 /// Then the result is a \ref binary_segregator with that `Segregatable` and \ref null_allocator as fallback.
372 /// If you pass two types, the first one must be a `Segregatable`,
373 /// the second one a \concept{concept_rawallocator,RawAllocator}.
374 /// Then the result is a simple \ref binary_segregator with those arguments.
375 /// If you pass more than one, the last one must be a `RawAllocator` all others `Segregatable`,
376 /// the result is `binary_segregator<Head, segregator<Tail...>>`.
377 /// \note It will result in an allocator that tries each `Segregatable` in the order specified
378 /// using the last parameter as final fallback.
379 /// \ingroup adapter
380 template <class... Allocators>
381 WPI_ALIAS_TEMPLATE(segregator,
382 typename detail::make_segregator_t<Allocators...>::type);
383
384 /// \returns A \ref segregator created from the allocators `args`.
385 /// \relates segregator
386 template <typename... Args>
387 auto make_segregator(Args&&... args) -> segregator<typename std::decay<Args>::type...>
388 {
389 return detail::make_segregator(std::forward<Args>(args)...);
390 }
