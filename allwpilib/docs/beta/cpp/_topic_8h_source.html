193 /**
194 * Create a new publisher to the topic.
195 *
196 * The publisher is only active as long as the returned object
197 * is not destroyed.
198 *
199 * @note It is not possible to publish two different data types to the same
200 * topic. Conflicts between publishers are typically resolved by the
201 * server on a first-come, first-served basis. Any published values that
202 * do not match the topic's data type are dropped (ignored). To determine
203 * if the data type matches, use the appropriate Topic functions.
204 *
205 * @param typeString type string
206 * @param options publish options
207 * @return publisher
208 */
209 [[nodiscard]] GenericPublisher GenericPublish(

212 /**
213 * Create a new publisher to the topic, with type string and initial
214 * properties.
215 *
216 * The publisher is only active as long as the returned object
217 * is not destroyed.
218 *
219 * @note It is not possible to publish two different data types to the same
220 * topic. Conflicts between publishers are typically resolved by the
221 * server on a first-come, first-served basis. Any published values that
222 * do not match the topic's data type are dropped (ignored). To determine
223 * if the data type matches, use the appropriate Topic functions.
224 *
225 * @param typeString type string
226 * @param properties JSON properties
227 * @param options publish options
228 * @return publisher
229 */

231 std::string_view typeString, const wpi::json& properties,
232 std::span<const PubSubOption> options = {});
233
234 /**
235 * Create a new generic entry for the topic.
236 *
237 * Entries act as a combination of a subscriber and a weak publisher. The
238 * subscriber is active as long as the entry is not destroyed. The publisher
239 * is created when the entry is first written to, and remains active until
240 * either Unpublish() is called or the entry is destroyed.
241 *
242 * @note It is not possible to use two different data types with the same
243 * topic. Conflicts between publishers are typically resolved by the
244 * server on a first-come, first-served basis. Any published values that
245 * do not match the topic's data type are dropped (ignored), and the entry
246 * will show no new values if the data type does not match. To determine
247 * if the data type matches, use the appropriate Topic functions.
248 *
249 * @param options publish and/or subscribe options
250 * @return entry
251 */
252 [[nodiscard]] GenericEntry GetGenericEntry(
253 std::span<const PubSubOption> options = {});
254
255 /**
256 * Create a new generic entry for the topic.
257 *
258 * Entries act as a combination of a subscriber and a weak publisher. The
259 * subscriber is active as long as the entry is not destroyed. The publisher
260 * is created when the entry is first written to, and remains active until
261 * either Unpublish() is called or the entry is destroyed.
262 *
263 * @note It is not possible to use two different data types with the same
264 * topic. Conflicts between publishers are typically resolved by the
265 * server on a first-come, first-served basis. Any published values that
266 * do not match the topic's data type are dropped (ignored), and the entry
267 * will show no new values if the data type does not match. To determine
268 * if the data type matches, use the appropriate Topic functions.
269 *
270 * @param typeString type string
271 * @param options publish and/or subscribe options
272 * @return entry
273 */
274 [[nodiscard]] GenericEntry GetGenericEntry(
275 std::string_view typeString, std::span<const PubSubOption> options = {});
276
277 /**
