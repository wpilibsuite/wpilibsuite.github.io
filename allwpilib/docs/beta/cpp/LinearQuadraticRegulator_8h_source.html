59 LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
60 const Eigen::Matrix<double, States, Inputs>& B,
61 const wpi::array<double, States>& Qelems,
62 const wpi::array<double, Inputs>& Relems,

76 LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
77 const Eigen::Matrix<double, States, Inputs>& B,
78 const Eigen::Matrix<double, States, States>& Q,
79 const Eigen::Matrix<double, Inputs, Inputs>& R,
80 units::second_t dt) {
81 Eigen::Matrix<double, States, States> discA;
82 Eigen::Matrix<double, States, Inputs> discB;
83 DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
84
85 Eigen::Matrix<double, States, States> S =
86 drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R);
87
88 // K = (BᵀSB + R)⁻¹BᵀSA
89 m_K = (discB.transpose() * S * discB + R)
90 .llt()
91 .solve(discB.transpose() * S * discA);
92
93 Reset();
94 }
95
106 LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
107 const Eigen::Matrix<double, States, Inputs>& B,
108 const Eigen::Matrix<double, States, States>& Q,
109 const Eigen::Matrix<double, Inputs, Inputs>& R,
110 const Eigen::Matrix<double, States, Inputs>& N,
111 units::second_t dt) {
112 Eigen::Matrix<double, States, States> discA;
113 Eigen::Matrix<double, States, Inputs> discB;
114 DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
115
116 Eigen::Matrix<double, States, States> S =
117 drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R, N);
118
119 // K = (BᵀSB + R)⁻¹(BᵀSA + Nᵀ)
120 m_K = (B.transpose() * S * B + R)
121 .llt()
122 .solve(discB.transpose() * S * discA + N.transpose());
123
124 Reset();
125 }

129 default;

189 Eigen::Vector<double, Inputs> Calculate(
190 const Eigen::Vector<double, States>& x) {
191 m_u = m_K * (m_r - x);
192 return m_u;
193 }
194
201 Eigen::Vector<double, Inputs> Calculate(
202 const Eigen::Vector<double, States>& x,
203 const Eigen::Vector<double, States>& nextR) {
204 m_r = nextR;
205 return Calculate(x);
206 }

224 template <int Outputs>
225 void LatencyCompensate(const LinearSystem<States, Inputs, Outputs>& plant,
226 units::second_t dt, units::second_t inputDelay) {
227 Eigen::Matrix<double, States, States> discA;
228 Eigen::Matrix<double, States, Inputs> discB;
229 DiscretizeAB<States, Inputs>(plant.A(), plant.B(), dt, &discA, &discB);
230
231 m_K = m_K * (discA - discB * m_K).pow(inputDelay / dt);
232 }
233
234 private:
235 // Current reference
236 Eigen::Vector<double, States> m_r;
237
238 // Computed controller output
239 Eigen::Vector<double, Inputs> m_u;
240
241 // Controller gain
242 Eigen::Matrix<double, Inputs, States> m_K;
