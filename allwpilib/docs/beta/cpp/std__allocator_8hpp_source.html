187 //=== allocation/deallocation ===//
188 /// \effects Allocates memory using the underlying \concept{concept_rawallocator,RawAllocator}.
189 /// If \c n is \c 1, it will call <tt>allocate_node(sizeof(T), alignof(T))</tt>,
190 /// otherwise <tt>allocate_array(n, sizeof(T), alignof(T))</tt>.
191 /// \returns A pointer to a memory block suitable for \c n objects of type \c T.
192 /// \throws Anything thrown by the \c RawAllocator.
193 pointer allocate(size_type n, void* = nullptr)
194 {
195 return static_cast<pointer>(allocate_impl(is_any{}, n));
196 }
197
198 /// \effects Deallcoates memory using the underlying \concept{concept_rawallocator,RawAllocator}.
199 /// It will forward to the deallocation function in the same way as in \ref allocate().
200 /// \requires The pointer must come from a previous call to \ref allocate() with the same \c n on this object or any copy of it.
201 void deallocate(pointer p, size_type n) noexcept
202 {
203 deallocate_impl(is_any{}, p, n);
204 }
