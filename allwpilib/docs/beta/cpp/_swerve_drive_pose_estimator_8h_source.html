22namespace frc {
23/**
24 * This class wraps an Unscented Kalman Filter to fuse latency-compensated
25 * vision measurements with swerve drive encoder velocity measurements. It will
26 * correct for noisy measurements and encoder drift. It is intended to be an
27 * easy but more accurate drop-in for SwerveDriveOdometry.
28 *
29 * Update() should be called every robot loop. If your loops are faster or
30 * slower than the default of 0.02s, then you should change the nominal delta
31 * time by specifying it in the constructor.
32 *
33 * AddVisionMeasurement() can be called as infrequently as you want; if you
34 * never call it, then this class will behave mostly like regular encoder
35 * odometry.
36 *
37 * The state-space system used internally has the following states (x), inputs
38 * (u), and outputs (y):
39 *
40 * <strong> x = [x, y, theta]ᵀ </strong> in the field coordinate system
41 * containing x position, y position, and heading.
42 *
43 * <strong> u = [v_l, v_r, dtheta]ᵀ </strong> containing left wheel velocity,
44 * right wheel velocity, and change in gyro heading.
45 *
46 * <strong> y = [x, y, theta]ᵀ </strong> from vision containing x position, y
47 * position, and heading; or <strong> y = [theta]ᵀ </strong> containing gyro
48 * heading.
49 */
50template <size_t NumModules>
51class SwerveDrivePoseEstimator {
52 public:
53 /**
54 * Constructs a SwerveDrivePoseEstimator.
55 *
56 * @param gyroAngle The current gyro angle.
57 * @param initialPose The starting pose estimate.
58 * @param kinematics A correctly-configured kinematics object
59 * for your drivetrain.
60 * @param stateStdDevs Standard deviations of model states.
61 * Increase these numbers to trust your
62 * model's state estimates less. This matrix
63 * is in the form [x, y, theta]ᵀ, with units
64 * in meters and radians.
65 * @param localMeasurementStdDevs Standard deviations of the encoder and gyro
66 * measurements. Increase these numbers to
67 * trust sensor readings from encoders
68 * and gyros less. This matrix is in the form
69 * [theta], with units in radians.
70 * @param visionMeasurementStdDevs Standard deviations of the vision
71 * measurements. Increase these numbers to
72 * trust global measurements from vision
73 * less. This matrix is in the form
74 * [x, y, theta]ᵀ, with units in meters and
75 * radians.
76 * @param nominalDt The time in seconds between each robot
77 * loop.
78 */
79 SwerveDrivePoseEstimator(
80 const Rotation2d& gyroAngle, const Pose2d& initialPose,
81 SwerveDriveKinematics<NumModules>& kinematics,
82 const wpi::array<double, 3>& stateStdDevs,
83 const wpi::array<double, 1>& localMeasurementStdDevs,
84 const wpi::array<double, 3>& visionMeasurementStdDevs,
85 units::second_t nominalDt = 0.02_s)
86 : m_observer([](const Eigen::Vector<double, 3>& x,
87 const Eigen::Vector<double, 3>& u) { return u; },
88 [](const Eigen::Vector<double, 3>& x,
89 const Eigen::Vector<double, 3>& u) {
90 return x.block<1, 1>(2, 0);
91 },
92 stateStdDevs, localMeasurementStdDevs,
93 frc::AngleMean<3, 3>(2), frc::AngleMean<1, 3>(0),
94 frc::AngleResidual<3>(2), frc::AngleResidual<1>(0),
95 frc::AngleAdd<3>(2), nominalDt),
96 m_kinematics(kinematics),
97 m_nominalDt(nominalDt) {
98 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
99
100 // Create correction mechanism for vision measurements.
101 m_visionCorrect = [&](const Eigen::Vector<double, 3>& u,
102 const Eigen::Vector<double, 3>& y) {
103 m_observer.Correct<3>(
104 u, y,
105 [](const Eigen::Vector<double, 3>& x,
106 const Eigen::Vector<double, 3>& u) { return x; },
107 m_visionContR, frc::AngleMean<3, 3>(2), frc::AngleResidual<3>(2),
108 frc::AngleResidual<3>(2), frc::AngleAdd<3>(2));
109 };
110
111 // Set initial state.
112 m_observer.SetXhat(PoseTo3dVector(initialPose));
113
114 // Calculate offsets.
115 m_gyroOffset = initialPose.Rotation() - gyroAngle;
116 m_previousAngle = initialPose.Rotation();
117 }
118
119 /**
120 * Resets the robot's position on the field.
121 *
122 * You NEED to reset your encoders (to zero) when calling this method.
123 *
124 * The gyroscope angle does not need to be reset in the user's robot code.
125 * The library automatically takes care of offsetting the gyro angle.
126 *
127 * @param pose The position on the field that your robot is at.
128 * @param gyroAngle The angle reported by the gyroscope.
129 */
130 void ResetPosition(const Pose2d& pose, const Rotation2d& gyroAngle) {
131 // Reset state estimate and error covariance
132 m_observer.Reset();
133 m_latencyCompensator.Reset();
134
135 m_observer.SetXhat(PoseTo3dVector(pose));
136
137 m_gyroOffset = pose.Rotation() - gyroAngle;
138 m_previousAngle = pose.Rotation();
139 }
140
141 /**
142 * Gets the pose of the robot at the current time as estimated by the Extended
143 * Kalman Filter.
144 *
145 * @return The estimated robot pose in meters.
146 */
147 Pose2d GetEstimatedPosition() const {
148 return Pose2d(m_observer.Xhat(0) * 1_m, m_observer.Xhat(1) * 1_m,
149 Rotation2d(units::radian_t{m_observer.Xhat(2)}));
150 }
151
152 /**
153 * Sets the pose estimator's trust of global measurements. This might be used
154 * to change trust in vision measurements after the autonomous period, or to
155 * change trust as distance to a vision target increases.
156 *
157 * @param visionMeasurementStdDevs Standard deviations of the vision
158 * measurements. Increase these numbers to
159 * trust global measurements from vision
160 * less. This matrix is in the form
161 * [x, y, theta]ᵀ, with units in meters and
162 * radians.
163 */
164 void SetVisionMeasurementStdDevs(
165 const wpi::array<double, 3>& visionMeasurementStdDevs) {
166 // Create R (covariances) for vision measurements.
167 m_visionContR = frc::MakeCovMatrix(visionMeasurementStdDevs);
168 }
169
170 /**
171 * Add a vision measurement to the Unscented Kalman Filter. This will correct
172 * the odometry pose estimate while still accounting for measurement noise.
173 *
174 * This method can be called as infrequently as you want, as long as you are
175 * calling Update() every loop.
176 *
177 * @param visionRobotPose The pose of the robot as measured by the vision
178 * camera.
179 * @param timestamp The timestamp of the vision measurement in seconds.
180 * Note that if you don't use your own time source by
181 * calling UpdateWithTime() then you must use a
182 * timestamp with an epoch since FPGA startup
183 * (i.e. the epoch of this timestamp is the same
184 * epoch as Timer#GetFPGATimestamp.) This means
185 * that you should use Timer#GetFPGATimestamp as your
186 * time source or sync the epochs.
187 */
188 void AddVisionMeasurement(const Pose2d& visionRobotPose,
189 units::second_t timestamp) {
190 m_latencyCompensator.ApplyPastGlobalMeasurement<3>(
191 &m_observer, m_nominalDt, PoseTo3dVector(visionRobotPose),
192 m_visionCorrect, timestamp);
193 }
194
195 /**
196 * Adds a vision measurement to the Unscented Kalman Filter. This will correct
197 * the odometry pose estimate while still accounting for measurement noise.
198 *
199 * This method can be called as infrequently as you want, as long as you are
200 * calling Update() every loop.
201 *
202 * Note that the vision measurement standard deviations passed into this
203 * method will continue to apply to future measurements until a subsequent
204 * call to SetVisionMeasurementStdDevs() or this method.
205 *
206 * @param visionRobotPose The pose of the robot as measured by the
207 * vision camera.
208 * @param timestamp The timestamp of the vision measurement in
209 * seconds. Note that if you don't use your
210 * own time source by calling
211 * UpdateWithTime(), then you must use a
212 * timestamp with an epoch since FPGA startup
213 * (i.e. the epoch of this timestamp is the
214 * same epoch as
215 * frc::Timer::GetFPGATimestamp(). This means
216 * that you should use
217 * frc::Timer::GetFPGATimestamp() as your
218 * time source in this case.
219 * @param visionMeasurementStdDevs Standard deviations of the vision
220 * measurements. Increase these numbers to
221 * trust global measurements from vision
222 * less. This matrix is in the form
223 * [x, y, theta]ᵀ, with units in meters and
224 * radians.
225 */
226 void AddVisionMeasurement(
227 const Pose2d& visionRobotPose, units::second_t timestamp,
228 const wpi::array<double, 3>& visionMeasurementStdDevs) {
229 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
230 AddVisionMeasurement(visionRobotPose, timestamp);
231 }
232
233 /**
234 * Updates the the Unscented Kalman Filter using only wheel encoder
235 * information. This should be called every loop, and the correct loop period
236 * must be passed into the constructor of this class.
237 *
238 * @param gyroAngle The current gyro angle.
239 * @param moduleStates The current velocities and rotations of the swerve
240 * modules.
241 * @return The estimated pose of the robot in meters.
242 */
243 template <typename... ModuleState>
244 Pose2d Update(const Rotation2d& gyroAngle, ModuleState&&... moduleStates) {
245 return UpdateWithTime(units::microsecond_t(wpi::Now()), gyroAngle,
246 moduleStates...);
247 }
248
249 /**
250 * Updates the the Unscented Kalman Filter using only wheel encoder
251 * information. This should be called every loop, and the correct loop period
252 * must be passed into the constructor of this class.
253 *
254 * @param currentTime Time at which this method was called, in seconds.
255 * @param gyroAngle The current gyroscope angle.
256 * @param moduleStates The current velocities and rotations of the swerve
257 * modules.
258 * @return The estimated pose of the robot in meters.
259 */
260 template <typename... ModuleState>
261 Pose2d UpdateWithTime(units::second_t currentTime,
262 const Rotation2d& gyroAngle,
263 ModuleState&&... moduleStates) {
264 auto dt = m_prevTime >= 0_s ? currentTime - m_prevTime : m_nominalDt;
265 m_prevTime = currentTime;
266
267 auto angle = gyroAngle + m_gyroOffset;
268 auto omega = (angle - m_previousAngle).Radians() / dt;
269
270 auto chassisSpeeds = m_kinematics.ToChassisSpeeds(moduleStates...);
271 auto fieldRelativeSpeeds =
272 Translation2d(chassisSpeeds.vx * 1_s, chassisSpeeds.vy * 1_s)
273 .RotateBy(angle);
274
275 Eigen::Vector<double, 3> u{fieldRelativeSpeeds.X().value(),
276 fieldRelativeSpeeds.Y().value(), omega.value()};
277
278 Eigen::Vector<double, 1> localY{angle.Radians().value()};
279 m_previousAngle = angle;
280
281 m_latencyCompensator.AddObserverState(m_observer, u, localY, currentTime);
282
283 m_observer.Predict(u, dt);
284 m_observer.Correct(u, localY);
285
286 return GetEstimatedPosition();
287 }
288
289 private:
290 UnscentedKalmanFilter<3, 3, 1> m_observer;
291 SwerveDriveKinematics<NumModules>& m_kinematics;
292 KalmanFilterLatencyCompensator<3, 3, 1, UnscentedKalmanFilter<3, 3, 1>>
293 m_latencyCompensator;
294 std::function<void(const Eigen::Vector<double, 3>& u,
295 const Eigen::Vector<double, 3>& y)>
296 m_visionCorrect;
297
298 Eigen::Matrix3d m_visionContR;
299
300 units::second_t m_nominalDt;
301 units::second_t m_prevTime = -1_s;
302
303 Rotation2d m_gyroOffset;
304 Rotation2d m_previousAngle;
305
306 template <int Dim>
307 static wpi::array<double, Dim> StdDevMatrixToArray(
308 const Eigen::Vector<double, Dim>& vector) {
309 wpi::array<double, Dim> array;
310 for (size_t i = 0; i < Dim; ++i) {
311 array[i] = vector(i);
312 }
313 return array;
314 }
315};
