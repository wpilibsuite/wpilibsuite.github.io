20template <int States>
21void DiscretizeA(const Eigen::Matrix<double, States, States>& contA,
22 units::second_t dt,
23 Eigen::Matrix<double, States, States>* discA) {
24 *discA = (contA * dt.to<double>()).exp();
25}

36template <int States, int Inputs>
37void DiscretizeAB(const Eigen::Matrix<double, States, States>& contA,
38 const Eigen::Matrix<double, States, Inputs>& contB,
39 units::second_t dt,
40 Eigen::Matrix<double, States, States>* discA,
41 Eigen::Matrix<double, States, Inputs>* discB) {
42 // Matrices are blocked here to minimize matrix exponentiation calculations
43 Eigen::Matrix<double, States + Inputs, States + Inputs> Mcont;
44 Mcont.setZero();
45 Mcont.template block<States, States>(0, 0) = contA * dt.to<double>();
46 Mcont.template block<States, Inputs>(0, States) = contB * dt.to<double>();
47
48 // Discretize A and B with the given timestep
49 Eigen::Matrix<double, States + Inputs, States + Inputs> Mdisc = Mcont.exp();
50 *discA = Mdisc.template block<States, States>(0, 0);
51 *discB = Mdisc.template block<States, Inputs>(0, States);
52}

63template <int States>
64void DiscretizeAQ(const Eigen::Matrix<double, States, States>& contA,
65 const Eigen::Matrix<double, States, States>& contQ,
66 units::second_t dt,
67 Eigen::Matrix<double, States, States>* discA,
68 Eigen::Matrix<double, States, States>* discQ) {
69 // Make continuous Q symmetric if it isn't already
70 Eigen::Matrix<double, States, States> Q = (contQ + contQ.transpose()) / 2.0;
71
72 // Set up the matrix M = [[-A, Q], [0, A.T]]
73 Eigen::Matrix<double, 2 * States, 2 * States> M;
74 M.template block<States, States>(0, 0) = -contA;
75 M.template block<States, States>(0, States) = Q;
76 M.template block<States, States>(States, 0).setZero();
77 M.template block<States, States>(States, States) = contA.transpose();
78
79 Eigen::Matrix<double, 2 * States, 2 * States> phi =
80 (M * dt.to<double>()).exp();
81
82 // Phi12 = phi[0:States, States:2*States]
83 // Phi22 = phi[States:2*States, States:2*States]
84 Eigen::Matrix<double, States, States> phi12 =
85 phi.block(0, States, States, States);
86 Eigen::Matrix<double, States, States> phi22 =
87 phi.block(States, States, States, States);
88
89 *discA = phi22.transpose();
90
91 Q = *discA * phi12;
92
93 // Make discrete Q symmetric if it isn't already
94 *discQ = (Q + Q.transpose()) / 2.0;
95}

115template <int States>
116void DiscretizeAQTaylor(const Eigen::Matrix<double, States, States>& contA,
117 const Eigen::Matrix<double, States, States>& contQ,
118 units::second_t dt,
119 Eigen::Matrix<double, States, States>* discA,
120 Eigen::Matrix<double, States, States>* discQ) {
121 // Make continuous Q symmetric if it isn't already
122 Eigen::Matrix<double, States, States> Q = (contQ + contQ.transpose()) / 2.0;
123
124 Eigen::Matrix<double, States, States> lastTerm = Q;
125 double lastCoeff = dt.to<double>();
126
127 // Aᵀⁿ
128 Eigen::Matrix<double, States, States> Atn = contA.transpose();
129
130 Eigen::Matrix<double, States, States> phi12 = lastTerm * lastCoeff;
131
132 // i = 6 i.e. 5th order should be enough precision
133 for (int i = 2; i < 6; ++i) {
134 lastTerm = -contA * lastTerm + Q * Atn;
135 lastCoeff *= dt.to<double>() / static_cast<double>(i);
136
137 phi12 += lastTerm * lastCoeff;
138
139 Atn *= contA.transpose();
140 }
141
142 DiscretizeA<States>(contA, dt, discA);
143 Q = *discA * phi12;
144
145 // Make discrete Q symmetric if it isn't already
146 *discQ = (Q + Q.transpose()) / 2.0;
147}

156template <int Outputs>
157Eigen::Matrix<double, Outputs, Outputs> DiscretizeR(
158 const Eigen::Matrix<double, Outputs, Outputs>& R, units::second_t dt) {
159 return R / dt.to<double>();
