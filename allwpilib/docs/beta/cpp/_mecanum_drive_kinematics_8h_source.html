18/**
19 * Helper class that converts a chassis velocity (dx, dy, and dtheta components)
20 * into individual wheel speeds.
21 *
22 * The inverse kinematics (converting from a desired chassis velocity to
23 * individual wheel speeds) uses the relative locations of the wheels with
24 * respect to the center of rotation. The center of rotation for inverse
25 * kinematics is also variable. This means that you can set your set your center
26 * of rotation in a corner of the robot to perform special evasion maneuvers.
27 *
28 * Forward kinematics (converting an array of wheel speeds into the overall
29 * chassis motion) is performs the exact opposite of what inverse kinematics
30 * does. Since this is an overdetermined system (more equations than variables),
31 * we use a least-squares approximation.
32 *
33 * The inverse kinematics: [wheelSpeeds] = [wheelLocations] * [chassisSpeeds]
34 * We take the Moore-Penrose pseudoinverse of [wheelLocations] and then
35 * multiply by [wheelSpeeds] to get our chassis speeds.
36 *
37 * Forward kinematics is also used for odometry -- determining the position of
38 * the robot on the field using encoders and a gyro.
39 */
40class WPILIB_DLLEXPORT MecanumDriveKinematics {
41 public:
42 /**
43 * Constructs a mecanum drive kinematics object.
44 *
45 * @param frontLeftWheel The location of the front-left wheel relative to the
46 * physical center of the robot.
47 * @param frontRightWheel The location of the front-right wheel relative to
48 * the physical center of the robot.
49 * @param rearLeftWheel The location of the rear-left wheel relative to the
50 * physical center of the robot.
51 * @param rearRightWheel The location of the rear-right wheel relative to the
52 * physical center of the robot.
53 */
54 explicit MecanumDriveKinematics(Translation2d frontLeftWheel,
55 Translation2d frontRightWheel,
56 Translation2d rearLeftWheel,
57 Translation2d rearRightWheel)
58 : m_frontLeftWheel{frontLeftWheel},
59 m_frontRightWheel{frontRightWheel},
60 m_rearLeftWheel{rearLeftWheel},
61 m_rearRightWheel{rearRightWheel} {
62 SetInverseKinematics(frontLeftWheel, frontRightWheel, rearLeftWheel,
63 rearRightWheel);
64 m_forwardKinematics = m_inverseKinematics.householderQr();
65 wpi::math::MathSharedStore::ReportUsage(
66 wpi::math::MathUsageId::kKinematics_MecanumDrive, 1);
67 }
68
69 MecanumDriveKinematics(const MecanumDriveKinematics&) = default;
70
71 /**
72 * Performs inverse kinematics to return the wheel speeds from a desired
73 * chassis velocity. This method is often used to convert joystick values into
74 * wheel speeds.
75 *
76 * This function also supports variable centers of rotation. During normal
77 * operations, the center of rotation is usually the same as the physical
78 * center of the robot; therefore, the argument is defaulted to that use case.
79 * However, if you wish to change the center of rotation for evasive
80 * maneuvers, vision alignment, or for any other use case, you can do so.
81 *
82 * @param chassisSpeeds The desired chassis speed.
83 * @param centerOfRotation The center of rotation. For example, if you set the
84 * center of rotation at one corner of the robot and
85 * provide a chassis speed that only has a dtheta
86 * component, the robot will rotate around that
87 * corner.
88 *
89 * @return The wheel speeds. Use caution because they are not normalized.
90 * Sometimes, a user input may cause one of the wheel speeds to go
91 * above the attainable max velocity. Use the
92 * MecanumDriveWheelSpeeds::Normalize() function to rectify this
93 * issue. In addition, you can leverage the power of C++17 to directly
94 * assign the wheel speeds to variables:
95 *
96 * @code{.cpp}
97 * auto [fl, fr, bl, br] = kinematics.ToWheelSpeeds(chassisSpeeds);
98 * @endcode
99 */
100 MecanumDriveWheelSpeeds ToWheelSpeeds(
101 const ChassisSpeeds& chassisSpeeds,
102 const Translation2d& centerOfRotation = Translation2d()) const;
103
104 /**
105 * Performs forward kinematics to return the resulting chassis state from the
106 * given wheel speeds. This method is often used for odometry -- determining
107 * the robot's position on the field using data from the real-world speed of
108 * each wheel on the robot.
109 *
110 * @param wheelSpeeds The current mecanum drive wheel speeds.
111 *
112 * @return The resulting chassis speed.
113 */
114 ChassisSpeeds ToChassisSpeeds(
115 const MecanumDriveWheelSpeeds& wheelSpeeds) const;
116
117 private:
118 mutable Eigen::Matrix<double, 4, 3> m_inverseKinematics;
119 Eigen::HouseholderQR<Eigen::Matrix<double, 4, 3>> m_forwardKinematics;
120 Translation2d m_frontLeftWheel;
121 Translation2d m_frontRightWheel;
122 Translation2d m_rearLeftWheel;
123 Translation2d m_rearRightWheel;
124
125 mutable Translation2d m_previousCoR;
126
127 /**
128 * Construct inverse kinematics matrix from wheel locations.
129 *
130 * @param fl The location of the front-left wheel relative to the physical
131 * center of the robot.
132 * @param fr The location of the front-right wheel relative to the physical
133 * center of the robot.
134 * @param rl The location of the rear-left wheel relative to the physical
135 * center of the robot.
136 * @param rr The location of the rear-right wheel relative to the physical
137 * center of the robot.
138 */
139 void SetInverseKinematics(Translation2d fl, Translation2d fr,
140 Translation2d rl, Translation2d rr) const;
141};
