18 template <typename Constraint, typename = std::enable_if_t<std::is_base_of_v<
19  TrajectoryConstraint, Constraint>>>
20 class EllipticalRegionConstraint : public TrajectoryConstraint {
21  public:
32  EllipticalRegionConstraint(const Translation2d& center, units::meter_t xWidth,
33  units::meter_t yWidth, const Rotation2d& rotation,
34  const Constraint& constraint)
35  : m_center(center),
36  m_radii(xWidth / 2.0, yWidth / 2.0),
37  m_constraint(constraint) {
38  m_radii = m_radii.RotateBy(rotation);
39  }
40 
41  units::meters_per_second_t MaxVelocity(
42  const Pose2d& pose, units::curvature_t curvature,
43  units::meters_per_second_t velocity) const override {
44  if (IsPoseInRegion(pose)) {
45  return m_constraint.MaxVelocity(pose, curvature, velocity);
46  } else {
47  return units::meters_per_second_t(
48  std::numeric_limits<double>::infinity());
49  }
50  }
51 
52  MinMax MinMaxAcceleration(const Pose2d& pose, units::curvature_t curvature,
53  units::meters_per_second_t speed) const override {
54  if (IsPoseInRegion(pose)) {
55  return m_constraint.MinMaxAcceleration(pose, curvature, speed);
56  } else {
57  return {};
58  }
59  }
60 
68  bool IsPoseInRegion(const Pose2d& pose) const {
69  // The region (disk) bounded by the ellipse is given by the equation:
70  // ((x-h)^2)/Rx^2) + ((y-k)^2)/Ry^2) <= 1
71  // If the inequality is satisfied, then it is inside the ellipse; otherwise
72  // it is outside the ellipse.
73  // Both sides have been multiplied by Rx^2 * Ry^2 for efficiency reasons.
74  return units::math::pow<2>(pose.X() - m_center.X()) *
75  units::math::pow<2>(m_radii.Y()) +
76  units::math::pow<2>(pose.Y() - m_center.Y()) *
77  units::math::pow<2>(m_radii.X()) <=
78  units::math::pow<2>(m_radii.X()) * units::math::pow<2>(m_radii.Y());
79  }
80 
81  private:
82  Translation2d m_center;
83  Translation2d m_radii;
84  Constraint m_constraint;
85 };
