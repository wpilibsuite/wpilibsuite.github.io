56  PoseWithCurvature GetPoint(double t) const {
57  Eigen::Matrix<double, Degree + 1, 1> polynomialBases;
58 
59  // Populate the polynomial bases
60  for (int i = 0; i <= Degree; i++) {
61  polynomialBases(i) = std::pow(t, Degree - i);
62  }
63 
64  // This simply multiplies by the coefficients. We need to divide out t some
65  // n number of times where n is the derivative we want to take.
66  Eigen::Matrix<double, 6, 1> combined = Coefficients() * polynomialBases;
67 
68  double dx, dy, ddx, ddy;
69 
70  // If t = 0, all other terms in the equation cancel out to zero. We can use
71  // the last x^0 term in the equation.
72  if (t == 0.0) {
73  dx = Coefficients()(2, Degree - 1);
74  dy = Coefficients()(3, Degree - 1);
75  ddx = Coefficients()(4, Degree - 2);
76  ddy = Coefficients()(5, Degree - 2);
77  } else {
78  // Divide out t for first derivative.
79  dx = combined(2) / t;
80  dy = combined(3) / t;
81 
82  // Divide out t for second derivative.
83  ddx = combined(4) / t / t;
84  ddy = combined(5) / t / t;
85  }
86 
87  // Find the curvature.
88  const auto curvature =
89  (dx * ddy - ddx * dy) / ((dx * dx + dy * dy) * std::hypot(dx, dy));
90 
91  return {
92  {FromVector(combined.template block<2, 1>(0, 0)), Rotation2d(dx, dy)},
93  units::curvature_t(curvature)};
94  }
95

102  virtual Eigen::Matrix<double, 6, Degree + 1> Coefficients() const = 0;
