14/**
15 * Subtracts a and b while normalizing the resulting value in the selected row
16 * as if it were an angle.
17 *
18 * @tparam States The number of states.
19 * @param a A vector to subtract from.
20 * @param b A vector to subtract with.
21 * @param angleStateIdx The row containing angles to be normalized.
22 */
23template <int States>
24Eigen::Vector<double, States> AngleResidual(
25 const Eigen::Vector<double, States>& a,
26 const Eigen::Vector<double, States>& b, int angleStateIdx) {
27 Eigen::Vector<double, States> ret = a - b;
28 ret[angleStateIdx] =
29 AngleModulus(units::radian_t{ret[angleStateIdx]}).value();
30 return ret;
31}

49/**
50 * Adds a and b while normalizing the resulting value in the selected row as an
51 * angle.
52 *
53 * @tparam States The number of states.
54 * @param a A vector to add with.
55 * @param b A vector to add with.
56 * @param angleStateIdx The row containing angles to be normalized.
57 */
58template <int States>
59Eigen::Vector<double, States> AngleAdd(const Eigen::Vector<double, States>& a,
60 const Eigen::Vector<double, States>& b,
61 int angleStateIdx) {
62 Eigen::Vector<double, States> ret = a + b;
63 ret[angleStateIdx] =
64 InputModulus(ret[angleStateIdx], -wpi::numbers::pi, wpi::numbers::pi);
65 return ret;

82/**
83 * Computes the mean of sigmas with the weights Wm while computing a special
84 * angle mean for a select row.
85 *
86 * @tparam CovDim Dimension of covariance of sigma points after passing through
87 * the transform.
88 * @tparam States The number of states.
89 * @param sigmas Sigma points.
90 * @param Wm Weights for the mean.
91 * @param angleStatesIdx The row containing the angles.
92 */
93template <int CovDim, int States>
94Eigen::Vector<double, CovDim> AngleMean(
95 const Eigen::Matrix<double, CovDim, 2 * States + 1>& sigmas,
96 const Eigen::Vector<double, 2 * States + 1>& Wm, int angleStatesIdx) {
97 double sumSin = sigmas.row(angleStatesIdx)
98 .unaryExpr([](auto it) { return std::sin(it); })
99 .sum();
100 double sumCos = sigmas.row(angleStatesIdx)
101 .unaryExpr([](auto it) { return std::cos(it); })
102 .sum();
103
104 Eigen::Vector<double, CovDim> ret = sigmas * Wm;
105 ret[angleStatesIdx] = std::atan2(sumSin, sumCos);
106 return ret;
107}

112 *
113 * @tparam CovDim Dimension of covariance of sigma points after passing through
114 * the transform.
115 * @tparam States The number of states.
116 * @param angleStateIdx The row containing the angles.
117 */
118template <int CovDim, int States>
119std::function<Eigen::Vector<double, CovDim>(

121 const Eigen::Vector<double, 2 * States + 1>&)>
122AngleMean(int angleStateIdx) {
123 return [=](auto sigmas, auto Wm) {
124 return AngleMean<CovDim, States>(sigmas, Wm, angleStateIdx);
