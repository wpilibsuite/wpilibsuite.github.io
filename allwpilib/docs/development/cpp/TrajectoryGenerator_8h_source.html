93 template <typename Spline>
94 static std::vector<PoseWithCurvature> SplinePointsFromSplines(
95 const std::vector<Spline>& splines) {
96 // Create the vector of spline points.
97 std::vector<PoseWithCurvature> splinePoints;
98
99 // Add the first point to the vector.
100 splinePoints.push_back(splines.front().GetPoint(0.0));
101
102 // Iterate through the vector and parameterize each spline, adding the
103 // parameterized points to the final vector.
104 for (auto&& spline : splines) {
105 auto points = SplineParameterizer::Parameterize(spline);
106 // Append the array of poses to the vector. We are removing the first
107 // point because it's a duplicate of the last point from the previous
108 // spline.
109 splinePoints.insert(std::end(splinePoints), std::begin(points) + 1,
110 std::end(points));
111 }
112 return splinePoints;
113 }
