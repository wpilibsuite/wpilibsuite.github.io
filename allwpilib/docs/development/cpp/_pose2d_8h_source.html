19/**
20 * Represents a 2d pose containing translational and rotational elements.
21 */
22class WPILIB_DLLEXPORT Pose2d {
23 public:
24 /**
25 * Constructs a pose at the origin facing toward the positive X axis.
26 * (Translation2d{0, 0} and Rotation{0})
27 */
28 constexpr Pose2d() = default;
29
30 /**
31 * Constructs a pose with the specified translation and rotation.
32 *
33 * @param translation The translational component of the pose.
34 * @param rotation The rotational component of the pose.
35 */
36 Pose2d(Translation2d translation, Rotation2d rotation);
37
38 /**
39 * Convenience constructors that takes in x and y values directly instead of
40 * having to construct a Translation2d.
41 *
42 * @param x The x component of the translational component of the pose.
43 * @param y The y component of the translational component of the pose.
44 * @param rotation The rotational component of the pose.
45 */
46 Pose2d(units::meter_t x, units::meter_t y, Rotation2d rotation);
47
48 /**
49 * Transforms the pose by the given transformation and returns the new
50 * transformed pose.
51 *
52 * [x_new] [cos, -sin, 0][transform.x]
53 * [y_new] += [sin, cos, 0][transform.y]
54 * [t_new] [0, 0, 1][transform.t]
55 *
56 * @param other The transform to transform the pose by.
57 *
58 * @return The transformed pose.
59 */
60 Pose2d operator+(const Transform2d& other) const;
61
62 /**
63 * Returns the Transform2d that maps the one pose to another.
64 *
65 * @param other The initial pose of the transformation.
66 * @return The transform that maps the other pose to the current pose.
67 */
68 Transform2d operator-(const Pose2d& other) const;
69
70 /**
71 * Checks equality between this Pose2d and another object.
72 *
73 * @param other The other object.
74 * @return Whether the two objects are equal.
75 */
76 bool operator==(const Pose2d& other) const;
77
78 /**
79 * Checks inequality between this Pose2d and another object.
80 *
81 * @param other The other object.
82 * @return Whether the two objects are not equal.
83 */
84 bool operator!=(const Pose2d& other) const;
85
86 /**
87 * Returns the underlying translation.
88 *
89 * @return Reference to the translational component of the pose.
90 */
91 const Translation2d& Translation() const { return m_translation; }
92
93 /**
94 * Returns the X component of the pose's translation.
95 *
96 * @return The x component of the pose's translation.
97 */
98 units::meter_t X() const { return m_translation.X(); }
99
100 /**
101 * Returns the Y component of the pose's translation.
102 *
103 * @return The y component of the pose's translation.
104 */
105 units::meter_t Y() const { return m_translation.Y(); }
106
107 /**
108 * Returns the underlying rotation.
109 *
110 * @return Reference to the rotational component of the pose.
111 */
112 const Rotation2d& Rotation() const { return m_rotation; }
113
114 /**
115 * Transforms the pose by the given transformation and returns the new pose.
116 * See + operator for the matrix multiplication performed.
117 *
118 * @param other The transform to transform the pose by.
119 *
120 * @return The transformed pose.
121 */
122 Pose2d TransformBy(const Transform2d& other) const;
123
124 /**
125 * Returns the other pose relative to the current pose.
126 *
127 * This function can often be used for trajectory tracking or pose
128 * stabilization algorithms to get the error between the reference and the
129 * current pose.
130 *
131 * @param other The pose that is the origin of the new coordinate frame that
132 * the current pose will be converted into.
133 *
134 * @return The current pose relative to the new origin pose.
135 */
136 Pose2d RelativeTo(const Pose2d& other) const;
137
138 /**
139 * Obtain a new Pose2d from a (constant curvature) velocity.
140 *
141 * See https://file.tavsys.net/control/controls-engineering-in-frc.pdf section
142 * 10.2 "Pose exponential" for a derivation.
143 *
144 * The twist is a change in pose in the robot's coordinate frame since the
145 * previous pose update. When the user runs exp() on the previous known
146 * field-relative pose with the argument being the twist, the user will
147 * receive the new field-relative pose.
148 *
149 * "Exp" represents the pose exponential, which is solving a differential
150 * equation moving the pose forward in time.
151 *
152 * @param twist The change in pose in the robot's coordinate frame since the
153 * previous pose update. For example, if a non-holonomic robot moves forward
154 * 0.01 meters and changes angle by 0.5 degrees since the previous pose
155 * update, the twist would be Twist2d{0.01, 0.0, toRadians(0.5)}
156 *
157 * @return The new pose of the robot.
158 */
159 Pose2d Exp(const Twist2d& twist) const;
160
161 /**
162 * Returns a Twist2d that maps this pose to the end pose. If c is the output
163 * of a.Log(b), then a.Exp(c) would yield b.
164 *
165 * @param end The end pose for the transformation.
166 *
167 * @return The twist that maps this to end.
168 */
169 Twist2d Log(const Pose2d& end) const;
170
171 private:
172 Translation2d m_translation;
173 Rotation2d m_rotation;
174};
