28/**
29 * Contains the controller coefficients and logic for a linear-quadratic
30 * regulator (LQR).
31 * LQRs use the control law u = K(r - x).
32 *
33 * For more on the underlying math, read
34 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf.
35 *
36 * @tparam States Number of states.
37 * @tparam Inputs Number of inputs.
38 */
39template <int States, int Inputs>
40class LinearQuadraticRegulatorImpl {
41 public:
42 /**
43 * Constructs a controller with the given coefficients and plant.
44 *
45 * @param plant The plant being controlled.
46 * @param Qelems The maximum desired error tolerance for each state.
47 * @param Relems The maximum desired control effort for each input.
48 * @param dt Discretization timestep.
49 */
50 template <int Outputs>
51 LinearQuadraticRegulatorImpl(
52 const LinearSystem<States, Inputs, Outputs>& plant,
53 const wpi::array<double, States>& Qelems,
54 const wpi::array<double, Inputs>& Relems, units::second_t dt)
55 : LinearQuadraticRegulatorImpl(plant.A(), plant.B(), Qelems, Relems, dt) {
56 }
57
58 /**
59 * Constructs a controller with the given coefficients and plant.
60 *
61 * @param A Continuous system matrix of the plant being controlled.
62 * @param B Continuous input matrix of the plant being controlled.
63 * @param Qelems The maximum desired error tolerance for each state.
64 * @param Relems The maximum desired control effort for each input.
65 * @param dt Discretization timestep.
66 */
67 LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
68 const Eigen::Matrix<double, States, Inputs>& B,
69 const wpi::array<double, States>& Qelems,
70 const wpi::array<double, Inputs>& Relems,
71 units::second_t dt)
72 : LinearQuadraticRegulatorImpl(A, B, MakeCostMatrix(Qelems),
73 MakeCostMatrix(Relems), dt) {}
74
75 /**
76 * Constructs a controller with the given coefficients and plant.
77 *
78 * @param A Continuous system matrix of the plant being controlled.
79 * @param B Continuous input matrix of the plant being controlled.
80 * @param Q The state cost matrix.
81 * @param R The input cost matrix.
82 * @param dt Discretization timestep.
83 */
84 LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
85 const Eigen::Matrix<double, States, Inputs>& B,
86 const Eigen::Matrix<double, States, States>& Q,
87 const Eigen::Matrix<double, Inputs, Inputs>& R,
88 units::second_t dt) {
89 Eigen::Matrix<double, States, States> discA;
90 Eigen::Matrix<double, States, Inputs> discB;
91 DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
92
93 if (!IsStabilizable<States, Inputs>(discA, discB)) {
94 std::string msg = fmt::format(
95 "The system passed to the LQR is uncontrollable!\n\nA =\n{}\nB "
96 "=\n{}\n",
97 discA, discB);
98
99 wpi::math::MathSharedStore::ReportError(msg);
100 throw std::invalid_argument(msg);
101 }
102
103 Eigen::Matrix<double, States, States> S =
104 drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R);
105
106 // K = (BᵀSB + R)⁻¹BᵀSA
107 m_K = (discB.transpose() * S * discB + R)
108 .llt()
109 .solve(discB.transpose() * S * discA);
110
111 Reset();
112 }
113
114 /**
115 * Constructs a controller with the given coefficients and plant.
116 *
117 * @param A Continuous system matrix of the plant being controlled.
118 * @param B Continuous input matrix of the plant being controlled.
119 * @param Q The state cost matrix.
120 * @param R The input cost matrix.
121 * @param N The state-input cross-term cost matrix.
122 * @param dt Discretization timestep.
123 */
124 LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
125 const Eigen::Matrix<double, States, Inputs>& B,
126 const Eigen::Matrix<double, States, States>& Q,
127 const Eigen::Matrix<double, Inputs, Inputs>& R,
128 const Eigen::Matrix<double, States, Inputs>& N,
129 units::second_t dt) {
130 Eigen::Matrix<double, States, States> discA;
131 Eigen::Matrix<double, States, Inputs> discB;
132 DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
133
134 Eigen::Matrix<double, States, States> S =
135 drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R, N);
136
137 // K = (BᵀSB + R)⁻¹(BᵀSA + Nᵀ)
138 m_K = (B.transpose() * S * B + R)
139 .llt()
140 .solve(discB.transpose() * S * discA + N.transpose());
141
142 Reset();
143 }
144
145 LinearQuadraticRegulatorImpl(LinearQuadraticRegulatorImpl&&) = default;
146 LinearQuadraticRegulatorImpl& operator=(LinearQuadraticRegulatorImpl&&) =
147 default;
148
149 /**
150 * Returns the controller matrix K.
151 */
152 const Eigen::Matrix<double, Inputs, States>& K() const { return m_K; }
153
154 /**
155 * Returns an element of the controller matrix K.
156 *
157 * @param i Row of K.
158 * @param j Column of K.
159 */
160 double K(int i, int j) const { return m_K(i, j); }
161
162 /**
163 * Returns the reference vector r.
164 *
165 * @return The reference vector.
166 */
167 const Eigen::Vector<double, States>& R() const { return m_r; }
168
169 /**
170 * Returns an element of the reference vector r.
171 *
172 * @param i Row of r.
173 *
174 * @return The row of the reference vector.
175 */
176 double R(int i) const { return m_r(i, 0); }
177
178 /**
179 * Returns the control input vector u.
180 *
181 * @return The control input.
182 */
183 const Eigen::Vector<double, Inputs>& U() const { return m_u; }
184
185 /**
186 * Returns an element of the control input vector u.
187 *
188 * @param i Row of u.
189 *
190 * @return The row of the control input vector.
191 */
192 double U(int i) const { return m_u(i, 0); }
193
194 /**
195 * Resets the controller.
196 */
197 void Reset() {
198 m_r.setZero();
199 m_u.setZero();
200 }
201
202 /**
203 * Returns the next output of the controller.
204 *
205 * @param x The current state x.
206 */
207 Eigen::Vector<double, Inputs> Calculate(
208 const Eigen::Vector<double, States>& x) {
209 m_u = m_K * (m_r - x);
210 return m_u;
211 }
212
213 /**
214 * Returns the next output of the controller.
215 *
216 * @param x The current state x.
217 * @param nextR The next reference vector r.
218 */
219 Eigen::Vector<double, Inputs> Calculate(
220 const Eigen::Vector<double, States>& x,
221 const Eigen::Vector<double, States>& nextR) {
222 m_r = nextR;
223 return Calculate(x);
224 }
225
226 /**
227 * Adjusts LQR controller gain to compensate for a pure time delay in the
228 * input.
229 *
230 * Linear-Quadratic regulator controller gains tend to be aggressive. If
231 * sensor measurements are time-delayed too long, the LQR may be unstable.
232 * However, if we know the amount of delay, we can compute the control based
233 * on where the system will be after the time delay.
234 *
235 * See https://file.tavsys.net/control/controls-engineering-in-frc.pdf
236 * appendix C.4 for a derivation.
237 *
238 * @param plant The plant being controlled.
239 * @param dt Discretization timestep.
240 * @param inputDelay Input time delay.
241 */
242 template <int Outputs>
243 void LatencyCompensate(const LinearSystem<States, Inputs, Outputs>& plant,
244 units::second_t dt, units::second_t inputDelay) {
245 Eigen::Matrix<double, States, States> discA;
246 Eigen::Matrix<double, States, Inputs> discB;
247 DiscretizeAB<States, Inputs>(plant.A(), plant.B(), dt, &discA, &discB);
248
249 m_K = m_K * (discA - discB * m_K).pow(inputDelay / dt);
250 }
251
252 private:
253 // Current reference
254 Eigen::Vector<double, States> m_r;
255
256 // Computed controller output
257 Eigen::Vector<double, Inputs> m_u;
258
259 // Controller gain
260 Eigen::Matrix<double, Inputs, States> m_K;
261};
