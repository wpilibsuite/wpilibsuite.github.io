571enum uv_udp_flags {
572 /* Disables dual stack mode. */
573 UV_UDP_IPV6ONLY = 1,
574 /*
575 * Indicates message was truncated because read buffer was too small. The
576 * remainder was discarded by the OS. Used in uv_udp_recv_cb.
577 */
578 UV_UDP_PARTIAL = 2,
579 /*
580 * Indicates if SO_REUSEADDR will be set when binding the handle.
581 * This sets the SO_REUSEPORT socket flag on the BSDs and OS X. On other
582 * Unix platforms, it sets the SO_REUSEADDR flag. What that means is that
583 * multiple threads or processes can bind to the same address without error
584 * (provided they all set the flag) but only the last one to bind will receive
585 * any traffic, in effect "stealing" the port from the previous listener.
586 */
587 UV_UDP_REUSEADDR = 4
588};
