30 template <typename T, class RawAllocator, typename... Args>
31 auto allocate_unique(allocator_reference<RawAllocator> alloc, Args&&... args)
32 -> std::unique_ptr<T, allocator_deleter<T, RawAllocator>>
33 {
34 using raw_ptr = std::unique_ptr<T, allocator_deallocator<T, RawAllocator>>;
35
36 auto memory = alloc.allocate_node(sizeof(T), alignof(T));
37 // raw_ptr deallocates memory in case of constructor exception
38 raw_ptr result(static_cast<T*>(memory), {alloc});
39 // call constructor
40 ::new (memory) T(detail::forward<Args>(args)...);
41 // pass ownership to return value using a deleter that calls destructor
42 return {result.release(), {alloc}};
43 }

45 template <typename T, typename... Args>
46 void construct(std::true_type, T* cur, T* end, Args&&... args)
47 {
48 for (; cur != end; ++cur)
49 ::new (static_cast<void*>(cur)) T(detail::forward<Args>(args)...);
50 }

52 template <typename T, typename... Args>
53 void construct(std::false_type, T* begin, T* end, Args&&... args)
54 {
55#if WPI_HAS_EXCEPTION_SUPPORT
56 auto cur = begin;
57 try
58 {
59 for (; cur != end; ++cur)
60 ::new (static_cast<void*>(cur)) T(detail::forward<Args>(args)...);
61 }
62 catch (...)
63 {
64 for (auto el = begin; el != cur; ++el)
65 el->~T();
66 throw;
67 }
68#else
69 construct(std::true_type{}, begin, end, detail::forward<Args>(args)...);
70#endif
71 }

73 template <typename T, class RawAllocator>
74 auto allocate_array_unique(std::size_t size, allocator_reference<RawAllocator> alloc)
75 -> std::unique_ptr<T[], allocator_deleter<T[], RawAllocator>>
76 {
77 using raw_ptr = std::unique_ptr<T[], allocator_deallocator<T[], RawAllocator>>;
78
79 auto memory = alloc.allocate_array(size, sizeof(T), alignof(T));
80 // raw_ptr deallocates memory in case of constructor exception
81 raw_ptr result(static_cast<T*>(memory), {alloc, size});
82 construct(std::integral_constant<bool, noexcept(T())>{}, result.get(),
83 result.get() + size);
84 // pass ownership to return value using a deleter that calls destructor
85 return {result.release(), {alloc, size}};
86 }

109 /// Creates a \c std::unique_ptr using a \concept{concept_rawallocator,RawAllocator} for the allocation.
110 /// \effects Allocates memory for the given type using the allocator
111 /// and creates a new object inside it passing the given arguments to its constructor.
112 /// \returns A \c std::unique_ptr owning that memory.
113 /// \note If the allocator is stateful a reference to the \c RawAllocator will be stored inside the deleter,
114 /// the caller has to ensure that the object lives as long as the smart pointer.
115 /// \ingroup adapter
116 template <typename T, class RawAllocator, typename... Args>
117 auto allocate_unique(RawAllocator&& alloc, Args&&... args) -> WPI_REQUIRES_RET(
118 !std::is_array<T>::value,
119 std::unique_ptr<T, allocator_deleter<T, typename std::decay<RawAllocator>::type>>)
120 {
121 return detail::allocate_unique<T>(make_allocator_reference(
122 detail::forward<RawAllocator>(alloc)),
123 detail::forward<Args>(args)...);
124 }

181 /// Creates a \c std::shared_ptr using a \concept{concept_rawallocator,RawAllocator} for the allocation.
182 /// It is similar to \c std::allocate_shared but uses a \c RawAllocator (and thus also supports any \c Allocator).
183 /// \effects Calls \ref std_allocator::make_std_allocator to wrap the allocator and forwards to \c std::allocate_shared.
184 /// \returns A \c std::shared_ptr created using \c std::allocate_shared.
185 /// \note If the allocator is stateful a reference to the \c RawAllocator will be stored inside the shared pointer,
186 /// the caller has to ensure that the object lives as long as the smart pointer.
187 /// \ingroup adapter
188 template <typename T, class RawAllocator, typename... Args>
189 std::shared_ptr<T> allocate_shared(RawAllocator&& alloc, Args&&... args)
190 {
191 return std::allocate_shared<T>(make_std_allocator<T>(
192 detail::forward<RawAllocator>(alloc)),
193 detail::forward<Args>(args)...);
194 }
