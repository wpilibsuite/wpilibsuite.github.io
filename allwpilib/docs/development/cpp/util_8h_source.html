22template <typename T>
23int AddrToName(const sockaddr_in& addr, T* ip, unsigned int* port) {
24 char name[128];
25 int err = uv_ip4_name(&addr, name, 128);
26 if (err == 0) {
27 ip->assign(name, name + std::strlen(name));
28 *port = ntohs(addr.sin_port);
29 } else {
30 ip->assign(name, name);
31 }
32 return err;
33}

62template <typename T>
63int AddrToName(const sockaddr_storage& addr, T* ip, unsigned int* port) {
64 if (addr.ss_family == AF_INET) {
65 return AddrToName(reinterpret_cast<const sockaddr_in&>(addr), ip, port);
66 }
67 if (addr.ss_family == AF_INET6) {
68 return AddrToName(reinterpret_cast<const sockaddr_in6&>(addr), ip, port);
69 }
70 char name[1];
71 ip->assign(name, name);
72 return -1;
73}
