21/**
22 * The linear time-varying unicycle controller has a similar form to the LQR,
23 * but the model used to compute the controller gain is the nonlinear model
24 * linearized around the drivetrain's current state.
25 *
26 * See section 8.9 in Controls Engineering in FRC for a derivation of the
27 * control law we used shown in theorem 8.9.1.
28 */
29class WPILIB_DLLEXPORT LTVUnicycleController {
30 public:
31 /**
32 * Constructs a linear time-varying unicycle controller with default maximum
33 * desired error tolerances of (0.0625 m, 0.125 m, 2 rad) and default maximum
34 * desired control effort of (1 m/s, 2 rad/s).
35 *
36 * @param dt Discretization timestep.
37 * @param maxVelocity The maximum velocity for the controller gain lookup
38 * table.
39 */
40 explicit LTVUnicycleController(
41 units::second_t dt, units::meters_per_second_t maxVelocity = 9_mps);
42
43 /**
44 * Constructs a linear time-varying unicycle controller.
45 *
46 * @param Qelems The maximum desired error tolerance for each state.
47 * @param Relems The maximum desired control effort for each input.
48 * @param dt Discretization timestep.
49 * @param maxVelocity The maximum velocity for the controller gain lookup
50 * table.
51 */
52 LTVUnicycleController(const wpi::array<double, 3>& Qelems,
53 const wpi::array<double, 2>& Relems, units::second_t dt,
54 units::meters_per_second_t maxVelocity = 9_mps);
55
56 /**
57 * Move constructor.
58 */
59 LTVUnicycleController(LTVUnicycleController&&) = default;
60
61 /**
62 * Move assignment operator.
63 */
64 LTVUnicycleController& operator=(LTVUnicycleController&&) = default;
65
66 /**
67 * Returns true if the pose error is within tolerance of the reference.
68 */
69 bool AtReference() const;
70
71 /**
72 * Sets the pose error which is considered tolerable for use with
73 * AtReference().
74 *
75 * @param poseTolerance Pose error which is tolerable.
76 */
77 void SetTolerance(const Pose2d& poseTolerance);
78
79 /**
80 * Returns the linear and angular velocity outputs of the LTV controller.
81 *
82 * The reference pose, linear velocity, and angular velocity should come from
83 * a drivetrain trajectory.
84 *
85 * @param currentPose The current pose.
86 * @param poseRef The desired pose.
87 * @param linearVelocityRef The desired linear velocity.
88 * @param angularVelocityRef The desired angular velocity.
89 */
90 ChassisSpeeds Calculate(const Pose2d& currentPose, const Pose2d& poseRef,
91 units::meters_per_second_t linearVelocityRef,
92 units::radians_per_second_t angularVelocityRef);
93
94 /**
95 * Returns the linear and angular velocity outputs of the LTV controller.
96 *
97 * The reference pose, linear velocity, and angular velocity should come from
98 * a drivetrain trajectory.
99 *
100 * @param currentPose The current pose.
101 * @param desiredState The desired pose, linear velocity, and angular velocity
102 * from a trajectory.
103 */
104 ChassisSpeeds Calculate(const Pose2d& currentPose,
105 const Trajectory::State& desiredState);
106
107 /**
108 * Enables and disables the controller for troubleshooting purposes.
109 *
110 * @param enabled If the controller is enabled or not.
111 */
112 void SetEnabled(bool enabled);
113
114 private:
115 // LUT from drivetrain linear velocity to LQR gain
116 wpi::interpolating_map<units::meters_per_second_t, Matrixd<2, 3>> m_table;
117
118 Pose2d m_poseError;
119 Pose2d m_poseTolerance;
120 bool m_enabled = true;
121};
122
123} // namespace frc
