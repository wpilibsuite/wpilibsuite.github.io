36 /// That means that for each possible size up to an upper bound there will be a seperate free list.
37 /// Allocating a node will not waste any memory.
38 /// \ingroup allocator

41 using type = detail::identity_access_policy;
42 };

44 /// A \c BucketDistribution for \ref memory_pool_collection defining that there is a bucket, i.e. pool, for each power of two.
45 /// That means for each power of two up to an upper bound there will be a separate free list.
46 /// Allocating a node will only waste half of the memory.

53 /// A stateful \concept{concept_rawallocator,RawAllocator} that behaves as a collection of multiple \ref memory_pool objects.
54 /// It maintains a list of multiple free lists, whose types are controlled via the \c PoolType tags defined in \ref memory_pool_type.hpp,
55 /// each of a different size as defined in the \c BucketDistribution (\ref identity_buckets or \ref log2_buckets).
56 /// Allocating a node of given size will use the appropriate free list.<br>
57 /// This allocator is ideal for \concept{concept_node,node} allocations in any order but with a predefined set of sizes,
58 /// not only one size like \ref memory_pool.
59 /// \ingroup allocator
60 template <class PoolType, class BucketDistribution,
61 class BlockOrRawAllocator = default_allocator>
62 class memory_pool_collection
63 : WPI_EBO(detail::default_leak_checker<detail::memory_pool_collection_leak_handler>)
64 {
65 using free_list_array =
66 detail::free_list_array<typename PoolType::type, typename BucketDistribution::type>;
67 using leak_checker =
68 detail::default_leak_checker<detail::memory_pool_collection_leak_handler>;
69
70 public:
71 using allocator_type = make_block_allocator_t<BlockOrRawAllocator>;
72 using pool_type = PoolType;
73 using bucket_distribution = BucketDistribution;

75 /// \effects Creates it by giving it the maximum node size it should be able to allocate,
76 /// the size of the initial memory block and other constructor arguments for the \concept{concept_blockallocator,BlockAllocator}.
77 /// The \c BucketDistribution controls how many free lists are created,
78 /// but unlike in \ref memory_pool all free lists are initially empty and the first memory block queued.
79 /// \requires \c max_node_size must be a valid \concept{concept_node,node} size
80 /// and \c block_size must be non-zero.
81 template <typename... Args>
82 memory_pool_collection(std::size_t max_node_size, std::size_t block_size,
83 Args&&... args)
84 : arena_(block_size, detail::forward<Args>(args)...),
85 stack_(allocate_block()),
86 pools_(stack_, block_end(), max_node_size)
87 {
88 }

90 /// \effects Destroys the \ref memory_pool_collection by returning all memory blocks,
91 /// regardless of properly deallocated back to the \concept{concept_blockallocator,BlockAllocator}.
92 ~memory_pool_collection() noexcept = default;

94 /// @{
95 /// \effects Moving a \ref memory_pool_collection object transfers ownership over the free lists,
96 /// i.e. the moved from pool is completely empty and the new one has all its memory.
97 /// That means that it is not allowed to call \ref deallocate_node() on a moved-from allocator
98 /// even when passing it memory that was previously allocated by this object.
99 memory_pool_collection(memory_pool_collection&& other) noexcept
100 : leak_checker(detail::move(other)),
101 arena_(detail::move(other.arena_)),
102 stack_(detail::move(other.stack_)),
103 pools_(detail::move(other.pools_))
104 {
105 }
106
107 memory_pool_collection& operator=(memory_pool_collection&& other) noexcept
108 {
109 leak_checker::operator=(detail::move(other));
110 arena_ = detail::move(other.arena_);
111 stack_ = detail::move(other.stack_);
112 pools_ = detail::move(other.pools_);
113 return *this;
114 }

117 /// \effects Allocates a \concept{concept_node,node} of given size.
118 /// It first finds the appropriate free list as defined in the \c BucketDistribution.
119 /// If it is empty, it will use an implementation defined amount of memory from the arena
120 /// and inserts it in it.
121 /// If the arena is empty too, it will request a new memory block from the \concept{concept_blockallocator,BlockAllocator}
122 /// of size \ref next_capacity() and puts part of it onto this free list.
123 /// Then it removes a node from it.
124 /// \returns A \concept{concept_node,node} of given size suitable aligned,
125 /// i.e. suitable for any type where <tt>sizeof(T) < node_size</tt>.
126 /// \throws Anything thrown by the \concept{concept_blockallocator,BlockAllocator} if a growth is needed or a \ref bad_node_size exception if the node size is too big.
127 void* allocate_node(std::size_t node_size)
128 {
129 detail::check_allocation_size<bad_node_size>(
130 node_size, [&] { return max_node_size(); }, info());
131 auto& pool = pools_.get(node_size);
132 if (pool.empty())
133 {
134 auto block = reserve_memory(pool, def_capacity());
135 pool.insert(block.memory, block.size);
136 }
137
138 auto mem = pool.allocate();
139 WPI_MEMORY_ASSERT(mem);
140 return mem;
141 }
142
143 /// \effects Allocates a \concept{concept_node,node} of given size.
144 /// It is similar to \ref allocate_node() but will return `nullptr` on any failure,
145 /// instead of growing the arnea and possibly throwing.
146 /// \returns A \concept{concept_node,node} of given size suitable aligned
147 /// or `nullptr` in case of failure.
148 void* try_allocate_node(std::size_t node_size) noexcept
149 {
150 if (node_size > max_node_size())
151 return nullptr;
152 auto& pool = pools_.get(node_size);
153 if (pool.empty())
154 {
155 try_reserve_memory(pool, def_capacity());
156 return pool.empty() ? nullptr : pool.allocate();
157 }
158 else
159 return pool.allocate();
160 }
161
162 /// \effects Allocates an \concept{concept_array,array} of nodes by searching for \c n continuous nodes on the appropriate free list and removing them.
163 /// Depending on the \c PoolType this can be a slow operation or not allowed at all.
164 /// This can sometimes lead to a growth on the free list, even if technically there is enough continuous memory on the free list.
165 /// Otherwise has the same behavior as \ref allocate_node().
166 /// \returns An array of \c n nodes of size \c node_size suitable aligned.
167 /// \throws Anything thrown by the used \concept{concept_blockallocator,BlockAllocator}'s allocation function if a growth is needed,
168 /// or a \ref bad_allocation_size exception.
169 /// \requires \c count must be valid \concept{concept_array,array count} and
170 /// \c node_size must be valid \concept{concept_node,node size}.
171 void* allocate_array(std::size_t count, std::size_t node_size)
172 {
173 detail::check_allocation_size<bad_node_size>(
174 node_size, [&] { return max_node_size(); }, info());
175
176 auto& pool = pools_.get(node_size);
177
178 // try allocating if not empty
179 // for pools without array allocation support, allocate() will always return nullptr
180 auto mem = pool.empty() ? nullptr : pool.allocate(count * node_size);
181 if (!mem)
182 {
183 // reserve more memory
184 auto block = reserve_memory(pool, def_capacity());
185 pool.insert(block.memory, block.size);
186
187 mem = pool.allocate(count * node_size);
188 if (!mem)
189 {
190 // reserve more then the default capacity if that didn't work either
191 detail::check_allocation_size<bad_array_size>(
192 count * node_size,
193 [&] { return next_capacity() - pool.alignment() + 1; }, info());
194
195 block = reserve_memory(pool, count * node_size);
196 pool.insert(block.memory, block.size);
197
198 mem = pool.allocate(count * node_size);
199 WPI_MEMORY_ASSERT(mem);
200 }
201 }
202
203 return mem;
204 }

225 /// \effects Deallocates a \concept{concept_node,node} by putting it back onto the appropriate free list.
226 /// \requires \c ptr must be a result from a previous call to \ref allocate_node() with the same size on the same free list,
227 /// i.e. either this allocator object or a new object created by moving this to it.
228 void deallocate_node(void* ptr, std::size_t node_size) noexcept

233 /// \effects Deallocates a \concept{concept_node,node} similar to \ref deallocate_node().

265 /// \effects Inserts more memory on the free list for nodes of given size.
266 /// It will try to put \c capacity_left bytes from the arena onto the free list defined over the \c BucketDistribution,
267 /// if the arena is empty, a new memory block is requested from the \concept{concept_blockallocator,BlockAllocator}
268 /// and it will be used.
269 /// \throws Anything thrown by the \concept{concept_blockallocator,BlockAllocator} if a growth is needed.
270 /// \requires \c node_size must be valid \concept{concept_node,node size} less than or equal to \ref max_node_size(),
271 /// \c capacity_left must be less than \ref next_capacity().
272 void reserve(std::size_t node_size, std::size_t capacity)
273 {
274 WPI_MEMORY_ASSERT_MSG(node_size <= max_node_size(), "node_size too big");
275 auto& pool = pools_.get(node_size);
276 reserve_memory(pool, capacity);
277 }

286 /// \returns The amount of nodes available in the free list for nodes of given size
287 /// as defined over the \c BucketDistribution.
288 /// This is the number of nodes that can be allocated without the free list requesting more memory from the arena.
289 /// \note Array allocations may lead to a growth even if the capacity_left is big enough.

291 {
292 WPI_MEMORY_ASSERT_MSG(node_size <= max_node_size(), "node_size too big");
293 return pools_.get(node_size).capacity();
294 }

296 /// \returns The amount of memory available in the arena not inside the free lists.
297 /// This is the number of bytes that can be inserted into the free lists
298 /// without requesting more memory from the \concept{concept_blockallocator,BlockAllocator}.
299 /// \note Array allocations may lead to a growth even if the capacity is big enough.
300 std::size_t capacity_left() const noexcept
301 {
302 return std::size_t(block_end() - stack_.top());
303 }

305 /// \returns The size of the next memory block after \ref capacity_left() arena grows.
306 /// This is the amount of memory that can be distributed in the pools.
307 /// \note If the `PoolType` is \ref small_node_pool, the exact usable memory is lower than that.

342 bool insert_rest(typename pool_type::type& pool) noexcept
343 {
344 if (auto remaining = std::size_t(block_end() - stack_.top()))
345 {
346 auto offset = detail::align_offset(stack_.top(), detail::max_alignment);
347 if (offset < remaining)
348 {
349 detail::debug_fill(stack_.top(), offset, debug_magic::alignment_memory);
350 pool.insert(stack_.top() + offset, remaining - offset);
351 return true;
352 }
353 }
354
355 return false;
356 }
357
358 void try_reserve_memory(typename pool_type::type& pool, std::size_t capacity) noexcept
359 {
360 auto mem = stack_.allocate(block_end(), capacity, detail::max_alignment);
361 if (!mem)
362 insert_rest(pool);
363 else
364 pool.insert(mem, capacity);
365 }
366
367 memory_block reserve_memory(typename pool_type::type& pool, std::size_t capacity)
368 {
369 auto mem = stack_.allocate(block_end(), capacity, detail::max_alignment);
370 if (!mem)
371 {
372 insert_rest(pool);
373 // get new block
374 stack_ = allocate_block();
375
376 // allocate ensuring alignment
377 mem = stack_.allocate(block_end(), capacity, detail::max_alignment);
378 WPI_MEMORY_ASSERT(mem);
379 }
380 return {mem, capacity};
381 }

410 /// Specialization of the \ref allocator_traits for \ref memory_pool_collection classes.
411 /// \note It is not allowed to mix calls through the specialization and through the member functions,
412 /// i.e. \ref memory_pool_collection::allocate_node() and this \c allocate_node().
413 /// \ingroup allocator
414 template <class Pool, class BucketDist, class RawAllocator>
415 class allocator_traits<memory_pool_collection<Pool, BucketDist, RawAllocator>>
416 {
417 public:
418 using allocator_type = memory_pool_collection<Pool, BucketDist, RawAllocator>;
419 using is_stateful = std::true_type;
420
421 /// \returns The result of \ref memory_pool_collection::allocate_node().
422 /// \throws Anything thrown by the pool allocation function
423 /// or a \ref bad_allocation_size exception if \c size / \c alignment exceeds \ref max_node_size() / the suitable alignment value,
424 /// i.e. the node is over-aligned.
425 static void* allocate_node(allocator_type& state, std::size_t size,
426 std::size_t alignment)
427 {
428 // node already checked
429 detail::check_allocation_size<bad_alignment>(
430 alignment, [&] { return detail::alignment_for(size); }, state.info());
431 auto mem = state.allocate_node(size);
432 state.on_allocate(size);
433 return mem;
434 }
435
436 /// \returns The result of \ref memory_pool_collection::allocate_array().
437 /// \throws Anything thrown by the pool allocation function or a \ref bad_allocation_size exception.
438 /// \requires The \ref memory_pool_collection has to support array allocations.
439 static void* allocate_array(allocator_type& state, std::size_t count, std::size_t size,
440 std::size_t alignment)
441 {
442 // node and array already checked
443 detail::check_allocation_size<bad_alignment>(
444 alignment, [&] { return detail::alignment_for(size); }, state.info());
445 auto mem = state.allocate_array(count, size);
446 state.on_allocate(count * size);
447 return mem;
448 }
449
450 /// \effects Calls \ref memory_pool_collection::deallocate_node().
451 static void deallocate_node(allocator_type& state, void* node, std::size_t size,
452 std::size_t) noexcept
453 {
454 state.deallocate_node(node, size);
455 state.on_deallocate(size);
456 }
457
458 /// \effects Calls \ref memory_pool_collection::deallocate_array().
459 /// \requires The \ref memory_pool_collection has to support array allocations.
460 static void deallocate_array(allocator_type& state, void* array, std::size_t count,
461 std::size_t size, std::size_t) noexcept
462 {
463 state.deallocate_array(array, count, size);
464 state.on_deallocate(count * size);
465 }
466
467 /// \returns The maximum size of each node which is \ref memory_pool_collection::max_node_size().

471 }
472

479 /// \returns Just \c alignof(std::max_align_t) since the actual maximum alignment depends on the node size,

482 {

484 }
