253  double Calculate(Distance_t measurement) {
254  if (m_controller.IsContinuousInputEnabled()) {
255  // Get error which is smallest distance between goal and measurement
256  auto goalMinDistance = frc::InputModulus<Distance_t>(
257  m_goal.position - measurement, m_minimumInput, m_maximumInput);
258  auto setpointMinDistance = frc::InputModulus<Distance_t>(
259  m_setpoint.position - measurement, m_minimumInput, m_maximumInput);
260 
261  // Recompute the profile goal with the smallest error, thus giving the
262  // shortest path. The goal may be outside the input range after this
263  // operation, but that's OK because the controller will still go there and
264  // report an error of zero. In other words, the setpoint only needs to be
265  // offset from the measurement by the input range modulus; they don't need
266  // to be equal.
267  m_goal.position = goalMinDistance + measurement;
268  m_setpoint.position = setpointMinDistance + measurement;
269  }
270 
271  frc::TrapezoidProfile<Distance> profile{m_constraints, m_goal, m_setpoint};
272  m_setpoint = profile.Calculate(GetPeriod());
273  return m_controller.Calculate(measurement.template to<double>(),
274  m_setpoint.position.template to<double>());
275  }
