104/**
105 * Configures the analog trigger to use the averaged vs. raw values.
106 *
107 * If the value is true, then the averaged value is selected for the analog
108 * trigger, otherwise the immediate value is used.
109 *
110 * This is not allowed to be used if filtered mode is set.
111 * This is not allowed to be used with Duty Cycle based inputs.
112 *
113 * @param[in] analogTriggerHandle the trigger handle
114 * @param[in] useAveragedValue true to use averaged values, false for raw
115 * @param[out] status Error status variable. 0 on success.
116 */
117void HAL_SetAnalogTriggerAveraged(HAL_AnalogTriggerHandle analogTriggerHandle,
118 HAL_Bool useAveragedValue, int32_t* status);
119
120/**
121 * Configures the analog trigger to use a filtered value.
122 *
123 * The analog trigger will operate with a 3 point average rejection filter. This
124 * is designed to help with 360 degree pot applications for the period where the
125 * pot crosses through zero.
126 *
127 * This is not allowed to be used if averaged mode is set.
128 *
129 * @param[in] analogTriggerHandle the trigger handle
130 * @param[in] useFilteredValue true to use filtered values, false for average
131 * or raw
132 * @param[out] status Error status variable. 0 on success.
133 */
134void HAL_SetAnalogTriggerFiltered(HAL_AnalogTriggerHandle analogTriggerHandle,
135 HAL_Bool useFilteredValue, int32_t* status);
136
