18/**
19 * Implements a PID control loop.
20 */
21class WPILIB_DLLEXPORT PIDController
22 : public wpi::Sendable,
23 public wpi::SendableHelper<PIDController> {
24 public:
25 /**
26 * Allocates a PIDController with the given constants for Kp, Ki, and Kd.
27 *
28 * @param Kp The proportional coefficient.
29 * @param Ki The integral coefficient.
30 * @param Kd The derivative coefficient.
31 * @param period The period between controller updates in seconds. The
32 * default is 20 milliseconds. Must be non-zero and positive.
33 */
34 PIDController(double Kp, double Ki, double Kd,
35 units::second_t period = 20_ms);
36
37 ~PIDController() override = default;
38
39 PIDController(const PIDController&) = default;
40 PIDController& operator=(const PIDController&) = default;
41 PIDController(PIDController&&) = default;
42 PIDController& operator=(PIDController&&) = default;
43
44 /**
45 * Sets the PID Controller gain parameters.
46 *
47 * Sets the proportional, integral, and differential coefficients.
48 *
49 * @param Kp Proportional coefficient
50 * @param Ki Integral coefficient
51 * @param Kd Differential coefficient
52 */
53 void SetPID(double Kp, double Ki, double Kd);
54
55 /**
56 * Sets the proportional coefficient of the PID controller gain.
57 *
58 * @param Kp proportional coefficient
59 */
60 void SetP(double Kp);
61
62 /**
63 * Sets the integral coefficient of the PID controller gain.
64 *
65 * @param Ki integral coefficient
66 */
67 void SetI(double Ki);
68
69 /**
70 * Sets the differential coefficient of the PID controller gain.
71 *
72 * @param Kd differential coefficient
73 */
74 void SetD(double Kd);
75
76 /**
77 * Gets the proportional coefficient.
78 *
79 * @return proportional coefficient
80 */
81 double GetP() const;
82
83 /**
84 * Gets the integral coefficient.
85 *
86 * @return integral coefficient
87 */
88 double GetI() const;
89
90 /**
91 * Gets the differential coefficient.
92 *
93 * @return differential coefficient
94 */
95 double GetD() const;
96
97 /**
98 * Gets the period of this controller.
99 *
100 * @return The period of the controller.
101 */
102 units::second_t GetPeriod() const;
103
104 /**
105 * Sets the setpoint for the PIDController.
106 *
107 * @param setpoint The desired setpoint.
108 */
109 void SetSetpoint(double setpoint);
110
111 /**
112 * Returns the current setpoint of the PIDController.
113 *
114 * @return The current setpoint.
115 */
116 double GetSetpoint() const;
117
118 /**
119 * Returns true if the error is within the tolerance of the setpoint.
120 *
121 * This will return false until at least one input value has been computed.
122 */
123 bool AtSetpoint() const;
124
125 /**
126 * Enables continuous input.
127 *
128 * Rather then using the max and min input range as constraints, it considers
129 * them to be the same point and automatically calculates the shortest route
130 * to the setpoint.
131 *
132 * @param minimumInput The minimum value expected from the input.
133 * @param maximumInput The maximum value expected from the input.
134 */
135 void EnableContinuousInput(double minimumInput, double maximumInput);
136
137 /**
138 * Disables continuous input.
139 */
140 void DisableContinuousInput();
141
142 /**
143 * Returns true if continuous input is enabled.
144 */
145 bool IsContinuousInputEnabled() const;
146
147 /**
148 * Sets the minimum and maximum values for the integrator.
149 *
150 * When the cap is reached, the integrator value is added to the controller
151 * output rather than the integrator value times the integral gain.
152 *
153 * @param minimumIntegral The minimum value of the integrator.
154 * @param maximumIntegral The maximum value of the integrator.
155 */
156 void SetIntegratorRange(double minimumIntegral, double maximumIntegral);
157
158 /**
159 * Sets the error which is considered tolerable for use with AtSetpoint().
160 *
161 * @param positionTolerance Position error which is tolerable.
162 * @param velocityTolerance Velocity error which is tolerable.
163 */
164 void SetTolerance(
165 double positionTolerance,
166 double velocityTolerance = std::numeric_limits<double>::infinity());
167
168 /**
169 * Returns the difference between the setpoint and the measurement.
170 */
171 double GetPositionError() const;
172
173 /**
174 * Returns the velocity error.
175 */
176 double GetVelocityError() const;
177
178 /**
179 * Returns the next output of the PID controller.
180 *
181 * @param measurement The current measurement of the process variable.
182 */
183 double Calculate(double measurement);
184
185 /**
186 * Returns the next output of the PID controller.
187 *
188 * @param measurement The current measurement of the process variable.
189 * @param setpoint The new setpoint of the controller.
190 */
191 double Calculate(double measurement, double setpoint);
192
193 /**
194 * Reset the previous error, the integral term, and disable the controller.
195 */
196 void Reset();
197
198 void InitSendable(wpi::SendableBuilder& builder) override;
199
200 private:
201 // Factor for "proportional" control
202 double m_Kp;
203
204 // Factor for "integral" control
205 double m_Ki;
206
207 // Factor for "derivative" control
208 double m_Kd;
209
210 // The period (in seconds) of the control loop running this controller
211 units::second_t m_period;
212
213 double m_maximumIntegral = 1.0;
214
215 double m_minimumIntegral = -1.0;
216
217 double m_maximumInput = 0;
218
219 double m_minimumInput = 0;
220
221 // Do the endpoints wrap around? eg. Absolute encoder
222 bool m_continuous = false;
223
224 // The error at the time of the most recent call to Calculate()
225 double m_positionError = 0;
226 double m_velocityError = 0;
227
228 // The error at the time of the second-most-recent call to Calculate() (used
229 // to compute velocity)
230 double m_prevError = 0;
231
232 // The sum of the errors for use in the integral calc
233 double m_totalError = 0;
234
235 // The error that is considered at setpoint.
236 double m_positionTolerance = 0.05;
237 double m_velocityTolerance = std::numeric_limits<double>::infinity();
238
239 double m_setpoint = 0;
240 double m_measurement = 0;
241};
