24 template <class Robot>
25 void RunRobot(wpi::mutex& m, Robot** robot) {
26  try {
27  static Robot theRobot;
28  {
29  std::scoped_lock lock{m};
30  *robot = &theRobot;
31  }
32  theRobot.StartCompetition();
33  } catch (const frc::RuntimeError& e) {
34  e.Report();
35  FRC_ReportError(
36  err::Error, "{}",
37  "The robot program quit unexpectedly."
38  " This is usually due to a code error.\n"
39  " The above stacktrace can help determine where the error occurred.\n"
40  " See https://wpilib.org/stacktrace for more information.\n");
41  throw;
42  } catch (const std::exception& e) {
43  HAL_SendError(1, err::Error, 0, e.what(), "", "", 1);
44  throw;
45  }
46 }

50 template <class Robot>
51 int StartRobot() {
52  int halInit = RunHALInitialization();
53  if (halInit != 0) {
54  return halInit;
55  }
56 
57  static wpi::mutex m;
58  static wpi::condition_variable cv;
59  static Robot* robot = nullptr;
60  static bool exited = false;
61 
62  if (HAL_HasMain()) {
63  std::thread thr([] {
64  try {
65  impl::RunRobot<Robot>(m, &robot);
66  } catch (...) {
67  HAL_ExitMain();
68  {
69  std::scoped_lock lock{m};
70  robot = nullptr;
71  exited = true;
72  }
73  cv.notify_all();
74  throw;
75  }
76 
77  HAL_ExitMain();
78  {
79  std::scoped_lock lock{m};
80  robot = nullptr;
81  exited = true;
82  }
83  cv.notify_all();
84  });
85 
86  HAL_RunMain();
87 
88  // signal loop to exit
89  if (robot) {
90  robot->EndCompetition();
91  }
92 
93  // prefer to join, but detach to exit if it doesn't exit in a timely manner
94  using namespace std::chrono_literals;
95  std::unique_lock lock{m};
96  if (cv.wait_for(lock, 1s, [] { return exited; })) {
97  thr.join();
98  } else {
99  thr.detach();
100  }
101  } else {
102  impl::RunRobot<Robot>(m, &robot);
103  }
104 
105  HAL_Shutdown();
106 
107  return 0;
108 }
