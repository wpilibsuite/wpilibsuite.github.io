26template <class Robot>
27void RunRobot(wpi::mutex& m, Robot** robot) {
28 try {
29 static Robot theRobot;
30 {
31 std::scoped_lock lock{m};
32 *robot = &theRobot;
33 }
34 theRobot.StartCompetition();
35 } catch (const frc::RuntimeError& e) {
36 e.Report();
37 FRC_ReportError(
38 err::Error, "{}",
39 "The robot program quit unexpectedly."
40 " This is usually due to a code error.\n"
41 " The above stacktrace can help determine where the error occurred.\n"
42 " See https://wpilib.org/stacktrace for more information.\n");
43 throw;
44 } catch (const std::exception& e) {
45 HAL_SendError(1, err::Error, 0, e.what(), "", "", 1);
46 throw;
47 }
48}

52template <class Robot>
53int StartRobot() {
54 int halInit = RunHALInitialization();
55 if (halInit != 0) {
56 return halInit;
57 }
58
59 static wpi::mutex m;
60 static wpi::condition_variable cv;
61 static Robot* robot = nullptr;
62 static bool exited = false;
63
64 if (HAL_HasMain()) {
65 std::thread thr([] {
66 try {
67 impl::RunRobot<Robot>(m, &robot);
68 } catch (...) {
69 HAL_ExitMain();
70 {
71 std::scoped_lock lock{m};
72 robot = nullptr;
73 exited = true;
74 }
75 cv.notify_all();
76 throw;
77 }
78
79 HAL_ExitMain();
80 {
81 std::scoped_lock lock{m};
82 robot = nullptr;
83 exited = true;
84 }
85 cv.notify_all();
86 });
87
88 HAL_RunMain();
89
90 // signal loop to exit
91 if (robot) {
92 robot->EndCompetition();
93 }
94
95 // prefer to join, but detach to exit if it doesn't exit in a timely manner
96 using namespace std::chrono_literals;
97 std::unique_lock lock{m};
98 if (cv.wait_for(lock, 1s, [] { return exited; })) {
99 thr.join();
100 } else {
101 thr.detach();
102 }
103 } else {
104 impl::RunRobot<Robot>(m, &robot);
105 }
106
107 HAL_Shutdown();
108
109 return 0;
110}
