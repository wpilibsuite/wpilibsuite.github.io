19template <int Rows, int Cols, typename F>
20auto NumericalJacobian(F&& f, const Eigen::Vector<double, Cols>& x) {
21 constexpr double kEpsilon = 1e-5;
22 Eigen::Matrix<double, Rows, Cols> result;
23 result.setZero();
24
25 // It's more expensive, but +- epsilon will be more accurate
26 for (int i = 0; i < Cols; ++i) {
27 Eigen::Vector<double, Cols> dX_plus = x;
28 dX_plus(i) += kEpsilon;
29 Eigen::Vector<double, Cols> dX_minus = x;
30 dX_minus(i) -= kEpsilon;
31 result.col(i) = (f(dX_plus) - f(dX_minus)) / (kEpsilon * 2.0);
32 }
33
34 return result;
35}

49template <int Rows, int States, int Inputs, typename F, typename... Args>
50auto NumericalJacobianX(F&& f, const Eigen::Vector<double, States>& x,
51 const Eigen::Vector<double, Inputs>& u,
52 Args&&... args) {
53 return NumericalJacobian<Rows, States>(
54 [&](const Eigen::Vector<double, States>& x) { return f(x, u, args...); },
55 x);
56}
