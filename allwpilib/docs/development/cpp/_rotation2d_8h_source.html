17/**
18 * A rotation in a 2D coordinate frame represented by a point on the unit circle
19 * (cosine and sine).
20 */
21class WPILIB_DLLEXPORT Rotation2d {
22 public:
23 /**
24 * Constructs a Rotation2d with a default angle of 0 degrees.
25 */
26 constexpr Rotation2d() = default;
27
28 /**
29 * Constructs a Rotation2d with the given radian value.
30 *
31 * @param value The value of the angle in radians.
32 */
33 constexpr Rotation2d(units::radian_t value); // NOLINT
34
35 /**
36 * Constructs a Rotation2d with the given degree value.
37 *
38 * @param value The value of the angle in degrees.
39 */
40 constexpr Rotation2d(units::degree_t value); // NOLINT
41
42 /**
43 * Constructs a Rotation2d with the given x and y (cosine and sine)
44 * components. The x and y don't have to be normalized.
45 *
46 * @param x The x component or cosine of the rotation.
47 * @param y The y component or sine of the rotation.
48 */
49 constexpr Rotation2d(double x, double y);
50
51 /**
52 * Adds two rotations together, with the result being bounded between -pi and
53 * pi.
54 *
55 * For example, <code>Rotation2d{30_deg} + Rotation2d{60_deg}</code> equals
56 * <code>Rotation2d{units::radian_t{std::numbers::pi/2.0}}</code>
57 *
58 * @param other The rotation to add.
59 *
60 * @return The sum of the two rotations.
61 */
62 constexpr Rotation2d operator+(const Rotation2d& other) const;
63
64 /**
65 * Subtracts the new rotation from the current rotation and returns the new
66 * rotation.
67 *
68 * For example, <code>Rotation2d{10_deg} - Rotation2d{100_deg}</code> equals
69 * <code>Rotation2d{units::radian_t{-std::numbers::pi/2.0}}</code>
70 *
71 * @param other The rotation to subtract.
72 *
73 * @return The difference between the two rotations.
74 */
75 constexpr Rotation2d operator-(const Rotation2d& other) const;
76
77 /**
78 * Takes the inverse of the current rotation. This is simply the negative of
79 * the current angular value.
80 *
81 * @return The inverse of the current rotation.
82 */
83 constexpr Rotation2d operator-() const;
84
85 /**
86 * Multiplies the current rotation by a scalar.
87 *
88 * @param scalar The scalar.
89 *
90 * @return The new scaled Rotation2d.
91 */
92 constexpr Rotation2d operator*(double scalar) const;
93
94 /**
95 * Divides the current rotation by a scalar.
96 *
97 * @param scalar The scalar.
98 *
99 * @return The new scaled Rotation2d.
100 */
101 constexpr Rotation2d operator/(double scalar) const;
102
103 /**
104 * Checks equality between this Rotation2d and another object.
105 *
106 * @param other The other object.
107 * @return Whether the two objects are equal.
108 */
109 constexpr bool operator==(const Rotation2d& other) const;
110
111 /**
112 * Checks inequality between this Rotation2d and another object.
113 *
114 * @param other The other object.
115 * @return Whether the two objects are not equal.
116 */
117 constexpr bool operator!=(const Rotation2d& other) const;
118
119 /**
120 * Adds the new rotation to the current rotation using a rotation matrix.
121 *
122 * <pre>
123 * [cos_new] [other.cos, -other.sin][cos]
124 * [sin_new] = [other.sin, other.cos][sin]
125 * value_new = std::atan2(sin_new, cos_new)
126 * </pre>
127 *
128 * @param other The rotation to rotate by.
129 *
130 * @return The new rotated Rotation2d.
131 */
132 constexpr Rotation2d RotateBy(const Rotation2d& other) const;
133
134 /**
135 * Returns the radian value of the rotation within (-pi, pi].
136 *
137 * @return The radian value of the rotation.
138 */
139 constexpr units::radian_t Radians() const;
140
141 /**
142 * Returns the degree value of the rotation within (-180, 180].
143 *
144 * @return The degree value of the rotation.
145 */
146 constexpr units::degree_t Degrees() const;
147
148 /**
149 * Returns the cosine of the rotation.
150 *
151 * @return The cosine of the rotation.
152 */
153 constexpr double Cos() const { return m_cos; }
154
155 /**
156 * Returns the sine of the rotation.
157 *
158 * @return The sine of the rotation.
159 */
160 constexpr double Sin() const { return m_sin; }
161
162 /**
163 * Returns the tangent of the rotation.
164 *
165 * @return The tangent of the rotation.
166 */
167 constexpr double Tan() const { return Sin() / Cos(); }
168
169 private:
170 double m_cos = 1;
171 double m_sin = 0;
172};
