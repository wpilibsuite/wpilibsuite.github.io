33/**
34 * A state machine representing a complete action to be performed by the robot.
35 * Commands are run by the CommandScheduler, and can be composed into
36 * CommandGroups to allow users to build complicated multi-step actions without
37 * the need to roll the state machine logic themselves.
38 *
39 * <p>Commands are run synchronously from the main robot loop; no multithreading
40 * is used, unless specified explicitly from the command implementation.
41 *
42 * <p>Note: ALWAYS create a subclass by extending CommandHelper<Base, Subclass>,
43 * or decorators will not function!
44 *
45 * @see CommandScheduler
46 * @see CommandHelper
47 */
48class Command {
49 public:
50 Command() = default;
51 virtual ~Command();
52
53 Command(const Command&) = default;
54 Command& operator=(const Command& rhs);
55 Command(Command&&) = default;
56 Command& operator=(Command&&) = default;
57
58 /**
59 * The initial subroutine of a command. Called once when the command is
60 * initially scheduled.
61 */
62 virtual void Initialize();
63
64 /**
65 * The main body of a command. Called repeatedly while the command is
66 * scheduled.
67 */
68 virtual void Execute();
69
70 /**
71 * The action to take when the command ends. Called when either the command
72 * finishes normally, or when it interrupted/canceled.
73 *
74 * @param interrupted whether the command was interrupted/canceled
75 */
76 virtual void End(bool interrupted);
77
78 /**
79 * Whether the command has finished. Once a command finishes, the scheduler
80 * will call its end() method and un-schedule it.
81 *
82 * @return whether the command has finished.
83 */
84 virtual bool IsFinished() { return false; }
85
86 /**
87 * Specifies the set of subsystems used by this command. Two commands cannot
88 * use the same subsystem at the same time. If the command is scheduled as
89 * interruptible and another command is scheduled that shares a requirement,
90 * the command will be interrupted. Else, the command will not be scheduled.
91 * If no subsystems are required, return an empty set.
92 *
93 * <p>Note: it is recommended that user implementations contain the
94 * requirements as a field, and return that field here, rather than allocating
95 * a new set every time this is called.
96 *
97 * @return the set of subsystems that are required
98 */
99 virtual wpi::SmallSet<Subsystem*, 4> GetRequirements() const = 0;
100
101 /**
102 * Decorates this command with a timeout. If the specified timeout is
103 * exceeded before the command finishes normally, the command will be
104 * interrupted and un-scheduled. Note that the timeout only applies to the
105 * command returned by this method; the calling command is not itself changed.
106 *
107 * @param duration the timeout duration
108 * @return the command with the timeout added
109 */
110 virtual ParallelRaceGroup WithTimeout(units::second_t duration) &&;
111
112 /**
113 * Decorates this command with an interrupt condition. If the specified
114 * condition becomes true before the command finishes normally, the command
115 * will be interrupted and un-scheduled. Note that this only applies to the
116 * command returned by this method; the calling command is not itself changed.
117 *
118 * @param condition the interrupt condition
119 * @return the command with the interrupt condition added
120 */
121 virtual ParallelRaceGroup WithInterrupt(std::function<bool()> condition) &&;
122
123 /**
124 * Decorates this command with a runnable to run before this command starts.
125 *
126 * @param toRun the Runnable to run
127 * @param requirements the required subsystems
128 * @return the decorated command
129 */
130 virtual SequentialCommandGroup BeforeStarting(
131 std::function<void()> toRun,
132 std::initializer_list<Subsystem*> requirements) &&;
133
134 /**
135 * Decorates this command with a runnable to run before this command starts.
136 *
137 * @param toRun the Runnable to run
138 * @param requirements the required subsystems
139 * @return the decorated command
140 */
141 virtual SequentialCommandGroup BeforeStarting(
142 std::function<void()> toRun,
143 wpi::span<Subsystem* const> requirements = {}) &&;
144
145 /**
146 * Decorates this command with a runnable to run after the command finishes.
147 *
148 * @param toRun the Runnable to run
149 * @param requirements the required subsystems
150 * @return the decorated command
151 */
152 virtual SequentialCommandGroup AndThen(
153 std::function<void()> toRun,
154 std::initializer_list<Subsystem*> requirements) &&;
155
156 /**
157 * Decorates this command with a runnable to run after the command finishes.
158 *
159 * @param toRun the Runnable to run
160 * @param requirements the required subsystems
161 * @return the decorated command
162 */
163 virtual SequentialCommandGroup AndThen(
164 std::function<void()> toRun,
165 wpi::span<Subsystem* const> requirements = {}) &&;
166
167 /**
168 * Decorates this command to run perpetually, ignoring its ordinary end
169 * conditions. The decorated command can still be interrupted or canceled.
170 *
171 * @return the decorated command
172 */
173 virtual PerpetualCommand Perpetually() &&;
174
175 /**
176 * Decorates this command to run "by proxy" by wrapping it in a
177 * ProxyScheduleCommand. This is useful for "forking off" from command groups
178 * when the user does not wish to extend the command's requirements to the
179 * entire command group.
180 *
181 * @return the decorated command
182 */
183 virtual ProxyScheduleCommand AsProxy();
184
185 /**
186 * Schedules this command.
187 *
188 * @param interruptible whether this command can be interrupted by another
189 * command that shares one of its requirements
190 */
191 void Schedule(bool interruptible);
192
193 /**
194 * Schedules this command, defaulting to interruptible.
195 */
196 void Schedule() { Schedule(true); }
197
198 /**
199 * Cancels this command. Will call the command's interrupted() method.
200 * Commands will be canceled even if they are not marked as interruptible.
201 */
202 void Cancel();
203
204 /**
205 * Whether or not the command is currently scheduled. Note that this does not
206 * detect whether the command is being run by a CommandGroup, only whether it
207 * is directly being run by the scheduler.
208 *
209 * @return Whether the command is scheduled.
210 */
211 bool IsScheduled() const;
212
213 /**
214 * Whether the command requires a given subsystem. Named "HasRequirement"
215 * rather than "requires" to avoid confusion with Command::Requires(Subsystem)
216 * -- this may be able to be changed in a few years.
217 *
218 * @param requirement the subsystem to inquire about
219 * @return whether the subsystem is required
220 */
221 bool HasRequirement(Subsystem* requirement) const;
222
223 /**
224 * Whether the command is currently grouped in a command group. Used as extra
225 * insurance to prevent accidental independent use of grouped commands.
226 */
227 bool IsGrouped() const;
228
229 /**
230 * Sets whether the command is currently grouped in a command group. Can be
231 * used to "reclaim" a command if a group is no longer going to use it. NOT
232 * ADVISED!
233 */
234 void SetGrouped(bool grouped);
235
236 /**
237 * Whether the given command should run when the robot is disabled. Override
238 * to return true if the command should run when disabled.
239 *
240 * @return whether the command should run when the robot is disabled
241 */
242 virtual bool RunsWhenDisabled() const { return false; }
243
244 virtual std::string GetName() const;
245
246 protected:
247 /**
248 * Transfers ownership of this command to a unique pointer. Used for
249 * decorator methods.
250 */
251 virtual std::unique_ptr<Command> TransferOwnership() && = 0;
252
253 bool m_isGrouped = false;
254};
255
256/**
257 * Checks if two commands have disjoint requirement sets.
258 *
259 * @param first The first command to check.
260 * @param second The second command to check.
261 * @return False if first and second share a requirement.
262 */
263bool RequirementsDisjoint(Command* first, Command* second);
264} // namespace frc2
