29/**
30 * A class for driving differential drive/skid-steer drive platforms such as
31 * the Kit of Parts drive base, "tank drive", or West Coast Drive.
32 *
33 * These drive bases typically have drop-center / skid-steer with two or more
34 * wheels per side (e.g., 6WD or 8WD). This class takes a MotorController per
35 * side. For four and six motor drivetrains, construct and pass in
36 * MotorControllerGroup instances as follows.
37 *
38 * Four motor drivetrain:
39 * @code{.cpp}
40 * class Robot {
41 * public:
42 * frc::PWMVictorSPX m_frontLeft{1};
43 * frc::PWMVictorSPX m_rearLeft{2};
44 * frc::MotorControllerGroup m_left{m_frontLeft, m_rearLeft};
45 *
46 * frc::PWMVictorSPX m_frontRight{3};
47 * frc::PWMVictorSPX m_rearRight{4};
48 * frc::MotorControllerGroup m_right{m_frontRight, m_rearRight};
49 *
50 * frc::DifferentialDrive m_drive{m_left, m_right};
51 * };
52 * @endcode
53 *
54 * Six motor drivetrain:
55 * @code{.cpp}
56 * class Robot {
57 * public:
58 * frc::PWMVictorSPX m_frontLeft{1};
59 * frc::PWMVictorSPX m_midLeft{2};
60 * frc::PWMVictorSPX m_rearLeft{3};
61 * frc::MotorControllerGroup m_left{m_frontLeft, m_midLeft, m_rearLeft};
62 *
63 * frc::PWMVictorSPX m_frontRight{4};
64 * frc::PWMVictorSPX m_midRight{5};
65 * frc::PWMVictorSPX m_rearRight{6};
66 * frc::MotorControllerGroup m_right{m_frontRight, m_midRight, m_rearRight};
67 *
68 * frc::DifferentialDrive m_drive{m_left, m_right};
69 * };
70 * @endcode
71 *
72 * A differential drive robot has left and right wheels separated by an
73 * arbitrary width.
74 *
75 * Drive base diagram:
76 * <pre>
77 * |_______|
78 * | | | |
79 * | |
80 * |_|___|_|
81 * | |
82 * </pre>
83 *
84 * Each Drive() function provides different inverse kinematic relations for a
85 * differential drive robot. Motor outputs for the right side are negated, so
86 * motor direction inversion by the user is usually unnecessary.
87 *
88 * This library uses the NED axes convention (North-East-Down as external
89 * reference in the world frame):
90 * http://www.nuclearprojects.com/ins/images/axis_big.png.
91 *
92 * The positive X axis points ahead, the positive Y axis points to the right,
93 * and the positive Z axis points down. Rotations follow the right-hand rule, so
94 * clockwise rotation around the Z axis is positive.
95 *
96 * Inputs smaller then 0.02 will be set to 0, and larger values will be scaled
97 * so that the full range is still used. This deadband value can be changed
98 * with SetDeadband().
99 */

109 /**
110 * Construct a DifferentialDrive.

112 * To pass multiple motors per side, use a MotorControllerGroup. If a motor

115 DifferentialDrive(SpeedController& leftMotor, SpeedController& rightMotor);

119 DifferentialDrive(DifferentialDrive&&) = default;

122 /**
123 * Arcade drive method for differential drive platform.
124 *
125 * Note: Some drivers may prefer inverted rotation controls. This can be done
126 * by negating the value passed for rotation.
127 *
128 * @param xSpeed The speed at which the robot should drive along the X
129 * axis [-1.0..1.0]. Forward is positive.
130 * @param zRotation The rotation rate of the robot around the Z axis
131 * [-1.0..1.0]. Clockwise is positive.
132 * @param squareInputs If set, decreases the input sensitivity at low speeds.
133 */
134 void ArcadeDrive(double xSpeed, double zRotation, bool squareInputs = true);
135
136 /**
137 * Curvature drive method for differential drive platform.
138 *
139 * The rotation argument controls the curvature of the robot's path rather
140 * than its rate of heading change. This makes the robot more controllable at
141 * high speeds.
142 *
143 * @param xSpeed The robot's speed along the X axis [-1.0..1.0].
144 * Forward is positive.
145 * @param zRotation The robot's rotation rate around the Z axis
146 * [-1.0..1.0]. Clockwise is positive.
147 * @param allowTurnInPlace If set, overrides constant-curvature turning for
148 * turn-in-place maneuvers.
149 */
150 void CurvatureDrive(double xSpeed, double zRotation, bool allowTurnInPlace);
151
152 /**
153 * Tank drive method for differential drive platform.
154 *
155 * @param leftSpeed The robot left side's speed along the X axis
156 * [-1.0..1.0]. Forward is positive.
157 * @param rightSpeed The robot right side's speed along the X axis
158 * [-1.0..1.0]. Forward is positive.
159 * @param squareInputs If set, decreases the input sensitivity at low speeds.
160 */
161 void TankDrive(double leftSpeed, double rightSpeed, bool squareInputs = true);
162
163 /**
164 * Arcade drive inverse kinematics for differential drive platform.
165 *
166 * Note: Some drivers may prefer inverted rotation controls. This can be done
167 * by negating the value passed for rotation.
168 *
169 * @param xSpeed The speed at which the robot should drive along the X
170 * axis [-1.0..1.0]. Forward is positive.
171 * @param zRotation The rotation rate of the robot around the Z axis
172 * [-1.0..1.0]. Clockwise is positive.
173 * @param squareInputs If set, decreases the input sensitivity at low speeds.
174 */
175 static WheelSpeeds ArcadeDriveIK(double xSpeed, double zRotation,
176 bool squareInputs = true);
177
178 /**
179 * Curvature drive inverse kinematics for differential drive platform.
180 *
181 * The rotation argument controls the curvature of the robot's path rather
182 * than its rate of heading change. This makes the robot more controllable at
183 * high speeds.
184 *
185 * @param xSpeed The robot's speed along the X axis [-1.0..1.0].
186 * Forward is positive.
187 * @param zRotation The robot's rotation rate around the Z axis
188 * [-1.0..1.0]. Clockwise is positive.
189 * @param allowTurnInPlace If set, overrides constant-curvature turning for
190 * turn-in-place maneuvers.
191 */
192 static WheelSpeeds CurvatureDriveIK(double xSpeed, double zRotation,
193 bool allowTurnInPlace);
194
195 /**
196 * Tank drive inverse kinematics for differential drive platform.
197 *
198 * @param leftSpeed The robot left side's speed along the X axis
199 * [-1.0..1.0]. Forward is positive.
200 * @param rightSpeed The robot right side's speed along the X axis
201 * [-1.0..1.0]. Forward is positive.
202 * @param squareInputs If set, decreases the input sensitivity at low speeds.
203 */
204 static WheelSpeeds TankDriveIK(double leftSpeed, double rightSpeed,
205 bool squareInputs = true);
206
207 void StopMotor() override;
208 std::string GetDescription() const override;
209
210 void InitSendable(wpi::SendableBuilder& builder) override;
211
212 private:
213 SpeedController* m_leftMotor;
214 SpeedController* m_rightMotor;
215};
