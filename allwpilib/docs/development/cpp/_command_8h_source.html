37/**
38 * A state machine representing a complete action to be performed by the robot.
39 * Commands are run by the CommandScheduler, and can be composed into
40 * CommandGroups to allow users to build complicated multi-step actions without
41 * the need to roll the state machine logic themselves.
42 *
43 * <p>Commands are run synchronously from the main robot loop; no
44 * multithreading is used, unless specified explicitly from the command
45 * implementation.
46 *
47 * <p>Note: ALWAYS create a subclass by extending CommandHelper<Base, Subclass>,
48 * or decorators will not function!
49 *
50 * This class is provided by the NewCommands VendorDep
51 *
52 * @see CommandScheduler
53 * @see CommandHelper
54 */
55class Command {
56 public:
57 Command() = default;
58 virtual ~Command();
59
60 Command(const Command&) = default;
61 Command& operator=(const Command& rhs);
62 Command(Command&&) = default;
63 Command& operator=(Command&&) = default;
64
65 /**
66 * The initial subroutine of a command. Called once when the command is
67 * initially scheduled.
68 */
69 virtual void Initialize();
70
71 /**
72 * The main body of a command. Called repeatedly while the command is
73 * scheduled.
74 */
75 virtual void Execute();
76
77 /**
78 * The action to take when the command ends. Called when either the command
79 * finishes normally, or when it interrupted/canceled.
80 *
81 * @param interrupted whether the command was interrupted/canceled
82 */
83 virtual void End(bool interrupted);
84
85 /**
86 * Whether the command has finished. Once a command finishes, the scheduler
87 * will call its end() method and un-schedule it.
88 *
89 * @return whether the command has finished.
90 */
91 virtual bool IsFinished() { return false; }
92
93 /**
94 * Specifies the set of subsystems used by this command. Two commands cannot
95 * use the same subsystem at the same time. If the command is scheduled as
96 * interruptible and another command is scheduled that shares a requirement,
97 * the command will be interrupted. Else, the command will not be scheduled.
98 * If no subsystems are required, return an empty set.
99 *
100 * <p>Note: it is recommended that user implementations contain the
101 * requirements as a field, and return that field here, rather than allocating
102 * a new set every time this is called.
103 *
104 * @return the set of subsystems that are required
105 */
106 virtual wpi::SmallSet<Subsystem*, 4> GetRequirements() const = 0;
107
108 /**
109 * Decorates this command with a timeout. If the specified timeout is
110 * exceeded before the command finishes normally, the command will be
111 * interrupted and un-scheduled. Note that the timeout only applies to the
112 * command returned by this method; the calling command is not itself changed.
113 *
114 * @param duration the timeout duration
115 * @return the command with the timeout added
116 */
117 virtual ParallelRaceGroup WithTimeout(units::second_t duration) &&;
118
119 /**
120 * Decorates this command with an interrupt condition. If the specified
121 * condition becomes true before the command finishes normally, the command
122 * will be interrupted and un-scheduled. Note that this only applies to the
123 * command returned by this method; the calling command is not itself changed.
124 *
125 * @param condition the interrupt condition
126 * @return the command with the interrupt condition added
127 */
128 virtual ParallelRaceGroup Until(std::function<bool()> condition) &&;
129
130 /**
131 * Decorates this command with an interrupt condition. If the specified
132 * condition becomes true before the command finishes normally, the command
133 * will be interrupted and un-scheduled. Note that this only applies to the
134 * command returned by this method; the calling command is not itself changed.
135 *
136 * @param condition the interrupt condition
137 * @return the command with the interrupt condition added
138 */
139 virtual ParallelRaceGroup WithInterrupt(std::function<bool()> condition) &&;
140
141 /**
142 * Decorates this command with a runnable to run before this command starts.
143 *
144 * @param toRun the Runnable to run
145 * @param requirements the required subsystems
146 * @return the decorated command
147 */
148 virtual SequentialCommandGroup BeforeStarting(
149 std::function<void()> toRun,
150 std::initializer_list<Subsystem*> requirements) &&;
151
152 /**
153 * Decorates this command with a runnable to run before this command starts.
154 *
155 * @param toRun the Runnable to run
156 * @param requirements the required subsystems
157 * @return the decorated command
158 */
159 virtual SequentialCommandGroup BeforeStarting(
160 std::function<void()> toRun,
161 wpi::span<Subsystem* const> requirements = {}) &&;
162
163 /**
164 * Decorates this command with a runnable to run after the command finishes.
165 *
166 * @param toRun the Runnable to run
167 * @param requirements the required subsystems
168 * @return the decorated command
169 */
170 virtual SequentialCommandGroup AndThen(
171 std::function<void()> toRun,
172 std::initializer_list<Subsystem*> requirements) &&;
173
174 /**
175 * Decorates this command with a runnable to run after the command finishes.
176 *
177 * @param toRun the Runnable to run
178 * @param requirements the required subsystems
179 * @return the decorated command
180 */
181 virtual SequentialCommandGroup AndThen(
182 std::function<void()> toRun,
183 wpi::span<Subsystem* const> requirements = {}) &&;
184
185 /**
186 * Decorates this command to run perpetually, ignoring its ordinary end
187 * conditions. The decorated command can still be interrupted or canceled.
188 *
189 * @return the decorated command
190 * @deprecated replace with EndlessCommand
191 */
192 WPI_DEPRECATED("Replace with Endlessly()")
193 virtual PerpetualCommand Perpetually() &&;
194
195 /**
196 * Decorates this command to run endlessly, ignoring its ordinary end
197 * conditions. The decorated command can still be interrupted or canceled.
198 *
199 * @return the decorated command
200 */
201 virtual EndlessCommand Endlessly() &&;
202
203 /**
204 * Decorates this command to run repeatedly, restarting it when it ends, until
205 * this command is interrupted. The decorated command can still be canceled.
206 *
207 * @return the decorated command
208 */
209 virtual RepeatCommand Repeat() &&;
210
211 /**
212 * Decorates this command to run "by proxy" by wrapping it in a
213 * ProxyScheduleCommand. This is useful for "forking off" from command groups
214 * when the user does not wish to extend the command's requirements to the
215 * entire command group.
216 *
217 * @return the decorated command
218 */
219 virtual ProxyScheduleCommand AsProxy();
220
221 /**
222 * Decorates this command to only run if this condition is not met. If the
223 * command is already running and the condition changes to true, the command
224 * will not stop running. The requirements of this command will be kept for
225 * the new conditonal command.
226 *
227 * @param condition the condition that will prevent the command from running
228 * @return the decorated command
229 */
230 virtual ConditionalCommand Unless(std::function<bool()> condition) &&;
231
232 /**
233 * Decorates this command to run or stop when disabled.
234 *
235 * @param doesRunWhenDisabled true to run when disabled.
236 * @return the decorated command
237 */
238 virtual std::unique_ptr<Command> IgnoringDisable(bool doesRunWhenDisabled) &&;
239
240 /**
241 * Schedules this command.
242 *
243 * @param interruptible whether this command can be interrupted by another
244 * command that shares one of its requirements
245 */
246 void Schedule(bool interruptible);
247
248 /**
249 * Schedules this command, defaulting to interruptible.
250 */
251 void Schedule() { Schedule(true); }
252
253 /**
254 * Cancels this command. Will call the command's interrupted() method.
255 * Commands will be canceled even if they are not marked as interruptible.
256 */
257 void Cancel();
258
259 /**
260 * Whether or not the command is currently scheduled. Note that this does not
261 * detect whether the command is being run by a CommandGroup, only whether it
262 * is directly being run by the scheduler.
263 *
264 * @return Whether the command is scheduled.
265 */
266 bool IsScheduled() const;
267
268 /**
269 * Whether the command requires a given subsystem. Named "HasRequirement"
270 * rather than "requires" to avoid confusion with Command::Requires(Subsystem)
271 * -- this may be able to be changed in a few years.
272 *
273 * @param requirement the subsystem to inquire about
274 * @return whether the subsystem is required
275 */
276 bool HasRequirement(Subsystem* requirement) const;
277
278 /**
279 * Whether the command is currently grouped in a command group. Used as extra
280 * insurance to prevent accidental independent use of grouped commands.
281 */
282 bool IsGrouped() const;
283
284 /**
285 * Sets whether the command is currently grouped in a command group. Can be
286 * used to "reclaim" a command if a group is no longer going to use it. NOT
287 * ADVISED!
288 */
289 void SetGrouped(bool grouped);
290
291 /**
292 * Whether the given command should run when the robot is disabled. Override
293 * to return true if the command should run when disabled.
294 *
295 * @return whether the command should run when the robot is disabled
296 */
297 virtual bool RunsWhenDisabled() const { return false; }
298
299 virtual std::string GetName() const;
300
301 protected:
302 /**
303 * Transfers ownership of this command to a unique pointer. Used for
304 * decorator methods.
305 */
306 virtual std::unique_ptr<Command> TransferOwnership() && = 0;
307
308 bool m_isGrouped = false;
309};

311/**
312 * Checks if two commands have disjoint requirement sets.

314 * @param first The first command to check.
315 * @param second The second command to check.
316 * @return False if first and second share a requirement.
317 */
318bool RequirementsDisjoint(Command* first, Command* second);
319} // namespace frc2
