19/**
20 * A class for driving addressable LEDs, such as WS2812s and NeoPixels.
21 *
22 * <p>Only 1 LED driver is currently supported by the roboRIO.
23 */
24class AddressableLED {
25 public:
26 class LEDData : public HAL_AddressableLEDData {
27 public:
28 LEDData() : LEDData(0, 0, 0) {}
29 LEDData(int _r, int _g, int _b) {
30 r = _r;
31 g = _g;
32 b = _b;
33 padding = 0;
34 }
35
36 /**
37 * A helper method to set all values of the LED.
38 *
39 * @param r the r value [0-255]
40 * @param g the g value [0-255]
41 * @param b the b value [0-255]
42 */
43 void SetRGB(int r, int g, int b) {
44 this->r = r;
45 this->g = g;
46 this->b = b;
47 }
48
49 /**
50 * A helper method to set all values of the LED.
51 *
52 * @param h the h value [0-180]
53 * @param s the s value [0-255]
54 * @param v the v value [0-255]
55 */
56 void SetHSV(int h, int s, int v);
57
58 /*
59 * Sets a specific LED in the buffer.
60 *
61 * @param color The color of the LED
62 */
63 void SetLED(const Color& color) {
64 this->r = color.red * 255;
65 this->g = color.green * 255;
66 this->b = color.blue * 255;
67 }
68
69 /*
70 * Sets a specific LED in the buffer.
71 *
72 * @param color The color of the LED
73 */
74 void SetLED(const Color8Bit& color) {
75 this->r = color.red;
76 this->g = color.green;
77 this->b = color.blue;
78 }
79 };
80
81 /**
82 * Constructs a new driver for a specific port.
83 *
84 * @param port the output port to use (Must be a PWM header)
85 */
86 explicit AddressableLED(int port);
87
88 ~AddressableLED();
89
90 /**
91 * Sets the length of the LED strip.
92 *
93 * <p>Calling this is an expensive call, so its best to call it once, then
94 * just update data.
95 *
96 * <p>The max length is 5460 LEDs.
97 *
98 * @param length the strip length
99 */
100 void SetLength(int length);
101
102 /**
103 * Sets the led output data.
104 *
105 * <p>If the output is enabled, this will start writing the next data cycle.
106 * It is safe to call, even while output is enabled.
107 *
108 * @param ledData the buffer to write
109 */
110 void SetData(wpi::span<const LEDData> ledData);
111
112 /**
113 * Sets the led output data.
114 *
115 * <p>If the output is enabled, this will start writing the next data cycle.
116 * It is safe to call, even while output is enabled.
117 *
118 * @param ledData the buffer to write
119 */
120 void SetData(std::initializer_list<LEDData> ledData);
121
122 /**
123 * Sets the bit timing.
124 *
125 * <p>By default, the driver is set up to drive WS2812s, so nothing needs to
126 * be set for those.
127 *
128 * @param lowTime0 low time for 0 bit
129 * @param highTime0 high time for 0 bit
130 * @param lowTime1 low time for 1 bit
131 * @param highTime1 high time for 1 bit
132 */
133 void SetBitTiming(units::nanosecond_t lowTime0, units::nanosecond_t highTime0,
134 units::nanosecond_t lowTime1,
135 units::nanosecond_t highTime1);
136
137 /**
138 * Sets the sync time.
139 *
140 * <p>The sync time is the time to hold output so LEDs enable. Default set for
141 * WS2812.
142 *
143 * @param syncTime the sync time
144 */
145 void SetSyncTime(units::microsecond_t syncTime);
146
147 /**
148 * Starts the output.
149 *
150 * <p>The output writes continuously.
151 */
152 void Start();
153
154 /**
155 * Stops the output.
156 */
157 void Stop();
158
159 private:
160 hal::Handle<HAL_DigitalHandle> m_pwmHandle;
161 hal::Handle<HAL_AddressableLEDHandle> m_handle;
162 int m_port;
163};
164} // namespace frc
