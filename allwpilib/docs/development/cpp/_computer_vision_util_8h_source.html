18/**
19 * Algorithm from
20 * https://docs.limelightvision.io/en/latest/cs_estimating_distance.html
21 * Estimates range to a target using the target's elevation. This method can
22 * produce more stable results than SolvePNP when well tuned, if the full 6d
23 * robot pose is not required.
24 *
25 * @param cameraHeight The height of the camera off the floor.
26 * @param targetHeight The height of the target off the floor.
27 * @param cameraPitch The pitch of the camera from the horizontal plane.
28 * Positive values up.
29 * @param targetPitch The pitch of the target in the camera's lens. Positive
30 * values up.
31 * @param targetYaw The yaw of the target in the camera's lens.
32 * @return The estimated distance to the target.
33 */
34WPILIB_DLLEXPORT
35units::meter_t CalculateDistanceToTarget(units::meter_t cameraHeight,
36 units::meter_t targetHeight,
37 units::radian_t cameraPitch,
38 units::radian_t targetPitch,
39 units::radian_t targetYaw);

41/**
42 * Estimate the position of the robot in the field.
43 *
44 * @param cameraHeight The physical height of the camera off the floor.
45 * @param targetHeight The physical height of the target off the floor. This
46 * should be the height of whatever is being targeted (i.e.
47 * if the targeting region is set to top, this should be the
48 * height of the top of the target).
49 * @param cameraPitch The pitch of the camera from the horizontal plane.
50 * Positive values up.
51 * @param targetPitch The pitch of the target in the camera's lens. Positive
52 * values up.
53 * @param targetYaw The observed yaw of the target. Note that this *must* be
54 * CCW-positive, and Photon returns CW-positive.
55 * @param gyroAngle The current robot gyro angle, likely from odometry.
56 * @param fieldToTarget A Pose3d representing the target position in the field
57 * coordinate system.
58 * @param cameraToRobot The position of the robot relative to the camera. If the
59 * camera was mounted 3 inches behind the "origin" (usually
60 * physical center) of the robot, this would be
61 * frc::Transform3d{3_in, 0_in, 0_in, frc::Rotation3d{}}.
62 * @return The position of the robot in the field.
63 */
64WPILIB_DLLEXPORT
65frc::Pose3d EstimateFieldToRobot(
66 units::meter_t cameraHeight, units::meter_t targetHeight,
67 units::radian_t cameraPitch, units::radian_t targetPitch,
68 const frc::Rotation2d& targetYaw, const frc::Rotation2d& gyroAngle,
69 const frc::Pose3d& fieldToTarget, const frc::Transform3d& cameraToRobot);

71/**
72 * Estimates the pose of the robot in the field coordinate system, given the
73 * position of the target relative to the camera, the target relative to the
74 * field, and the robot relative to the camera.
75 *
76 * @param cameraToTarget The position of the target relative to the camera.
77 * @param fieldToTarget The position of the target in the field.
78 * @param cameraToRobot The position of the robot relative to the camera. If
79 * the camera was mounted 3 inches behind the "origin"
80 * (usually physical center) of the robot, this would be
81 * frc::Transform3d{3_in, 0_in, 0_in, frc::Rotation3d{}}.
82 * @return The position of the robot in the field.
83 */
84WPILIB_DLLEXPORT
85frc::Pose3d EstimateFieldToRobot(const frc::Transform3d& cameraToTarget,
86 const frc::Pose3d& fieldToTarget,
87 const frc::Transform3d& cameraToRobot);

89/**
90 * Estimates a Transform3d that maps the camera position to the target position,
91 * using the robot's gyro. Note that the gyro angle provided *must* line up with
92 * the field coordinate system -- that is, it should read zero degrees when
93 * pointed towards the opposing alliance station, and increase as the robot
94 * rotates CCW.
95 *
96 * @param cameraToTargetTranslation A Translation3d that encodes the x/y
97 * position of the target relative to the
98 * camera.
99 * @param fieldToTarget A Pose3d representing the target position in the field
100 * coordinate system.
101 * @param gyroAngle The current robot gyro angle, likely from odometry.
102 * @return A Transform3d that takes us from the camera to the target.
103 */
104WPILIB_DLLEXPORT
105frc::Transform3d EstimateCameraToTarget(
106 const frc::Translation3d& cameraToTargetTranslation,
107 const frc::Pose3d& fieldToTarget, const frc::Rotation2d& gyroAngle);
108
109/**
110 * Estimates the pose of the camera in the field coordinate system, given the
111 * position of the target relative to the camera, and the target relative to
112 * the field. This *only* tracks the position of the camera, not the position
113 * of the robot itself.
114 *
115 * @param cameraToTarget The position of the target relative to the camera.
116 * @param fieldToTarget The position of the target in the field.
117 * @return The position of the camera in the field.
118 */
119WPILIB_DLLEXPORT
120frc::Pose3d EstimateFieldToCamera(const frc::Transform3d& cameraToTarget,
121 const frc::Pose3d& fieldToTarget);
