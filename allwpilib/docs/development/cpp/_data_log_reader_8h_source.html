19/**
20 * Data contained in a start control record as created by DataLog::Start() when
21 * writing the log. This can be read by calling DataLogRecord::GetStartData().
22 */
23struct StartRecordData {
24 /** Entry ID; this will be used for this entry in future records. */
25 int entry;
26
27 /** Entry name. */
28 std::string_view name;
29
30 /** Type of the stored data for this entry, as a string, e.g. "double". */
31 std::string_view type;
32
33 /** Initial metadata. */
34 std::string_view metadata;
35};

350
351inline DataLogIterator& DataLogIterator::operator++() {
352 if (!m_reader->GetNextRecord(&m_pos)) {
353 m_pos = SIZE_MAX;
354 }
355 m_valid = false;
356 return *this;
357}
358
359inline DataLogIterator::reference DataLogIterator::operator*() const {
360 if (!m_valid) {
361 size_t pos = m_pos;
362 if (m_reader->GetRecord(&pos, &m_value)) {
363 m_valid = true;
364 }
365 }
366 return m_value;
367}
