43 static LinearSystem<2, 1, 1> ElevatorSystem(DCMotor motor,
44 units::kilogram_t m,
45 units::meter_t r, double G) {
46 Eigen::Matrix<double, 2, 2> A{
47 {0.0, 1.0},
48 {0.0, (-std::pow(G, 2) * motor.Kt /
49 (motor.R * units::math::pow<2>(r) * m * motor.Kv))
50 .to<double>()}};
51 Eigen::Matrix<double, 2, 1> B{
52 0.0, (G * motor.Kt / (motor.R * r * m)).to<double>()};
53 Eigen::Matrix<double, 1, 2> C{1.0, 0.0};
54 Eigen::Matrix<double, 1, 1> D{0.0};
55
56 return LinearSystem<2, 1, 1>(A, B, C, D);
57 }

170 static LinearSystem<2, 2, 2> IdentifyDrivetrainSystem(
171 decltype(1_V / 1_mps) kVlinear, decltype(1_V / 1_mps_sq) kAlinear,
172 decltype(1_V / 1_mps) kVangular, decltype(1_V / 1_mps_sq) kAangular) {
173 double A1 = -(kVlinear.to<double>() / kAlinear.to<double>() +
174 kVangular.to<double>() / kAangular.to<double>());
175 double A2 = -(kVlinear.to<double>() / kAlinear.to<double>() -
176 kVangular.to<double>() / kAangular.to<double>());
177 double B1 = 1.0 / kAlinear.to<double>() + 1.0 / kAangular.to<double>();
178 double B2 = 1.0 / kAlinear.to<double>() - 1.0 / kAangular.to<double>();
179
180 Eigen::Matrix<double, 2, 2> A =
181 0.5 * Eigen::Matrix<double, 2, 2>{{A1, A2}, {A2, A1}};
182 Eigen::Matrix<double, 2, 2> B =
183 0.5 * Eigen::Matrix<double, 2, 2>{{B1, B2}, {B2, B1}};
184 Eigen::Matrix<double, 2, 2> C{{1.0, 0.0}, {0.0, 1.0}};
185 Eigen::Matrix<double, 2, 2> D{{0.0, 0.0}, {0.0, 0.0}};
186
187 return LinearSystem<2, 2, 2>(A, B, C, D);
188 }

207 static LinearSystem<2, 2, 2> IdentifyDrivetrainSystem(
208 decltype(1_V / 1_mps) kVlinear, decltype(1_V / 1_mps_sq) kAlinear,
209 decltype(1_V / 1_rad_per_s) kVangular,
210 decltype(1_V / 1_rad_per_s_sq) kAangular, units::meter_t trackwidth) {
211 // We want to find a factor to include in Kv,angular that will convert
212 // `u = Kv,angular omega` to `u = Kv,angular v`.
213 //
214 // v = omega r
215 // omega = v/r
216 // omega = 1/r v
217 // omega = 1/(trackwidth/2) v
218 // omega = 2/trackwidth v
219 //
220 // So multiplying by 2/trackwidth converts the angular gains from V/(rad/s)
221 // to V/m/s).
222 return IdentifyDrivetrainSystem(kVlinear, kAlinear,
223 kVangular * 2.0 / trackwidth * 1_rad,
224 kAangular * 2.0 / trackwidth * 1_rad);
225 }

264 static LinearSystem<2, 2, 2> DrivetrainVelocitySystem(
265 const DCMotor& motor, units::kilogram_t m, units::meter_t r,
266 units::meter_t rb, units::kilogram_square_meter_t J, double G) {
267 auto C1 = -std::pow(G, 2) * motor.Kt /
268 (motor.Kv * motor.R * units::math::pow<2>(r));
269 auto C2 = G * motor.Kt / (motor.R * r);
270
271 Eigen::Matrix<double, 2, 2> A{
272 {((1 / m + units::math::pow<2>(rb) / J) * C1).to<double>(),
273 ((1 / m - units::math::pow<2>(rb) / J) * C1).to<double>()},
274 {((1 / m - units::math::pow<2>(rb) / J) * C1).to<double>(),
275 ((1 / m + units::math::pow<2>(rb) / J) * C1).to<double>()}};
276 Eigen::Matrix<double, 2, 2> B{
277 {((1 / m + units::math::pow<2>(rb) / J) * C2).to<double>(),
278 ((1 / m - units::math::pow<2>(rb) / J) * C2).to<double>()},
279 {((1 / m - units::math::pow<2>(rb) / J) * C2).to<double>(),
280 ((1 / m + units::math::pow<2>(rb) / J) * C2).to<double>()}};
281 Eigen::Matrix<double, 2, 2> C{{1.0, 0.0}, {0.0, 1.0}};
282 Eigen::Matrix<double, 2, 2> D{{0.0, 0.0}, {0.0, 0.0}};
283
284 return LinearSystem<2, 2, 2>(A, B, C, D);
285 }
