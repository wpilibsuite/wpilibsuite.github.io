16/**
17 * A helper class for use with objects that add themselves to SendableRegistry.
18 * It takes care of properly calling Move() and Remove() on move and
19 * destruction. No action is taken if the object is copied.
20 * Use public inheritance with CRTP when using this class.
21 * @tparam CRTP derived class
22 */
23template <typename Derived>
24class SendableHelper {
25 public:
26 SendableHelper(const SendableHelper& rhs) = default;
27 SendableHelper& operator=(const SendableHelper& rhs) = default;
28
29#if !defined(_MSC_VER) && (defined(__llvm__) || __GNUC__ > 7)
30 // See https://bugzilla.mozilla.org/show_bug.cgi?id=1442819
31 __attribute__((no_sanitize("vptr")))
32#endif
33 SendableHelper(SendableHelper&& rhs) {
34 // it is safe to call Move() multiple times with the same rhs
35 SendableRegistry::Move(static_cast<Derived*>(this),
36 static_cast<Derived*>(&rhs));
37 }
38
39#if !defined(_MSC_VER) && (defined(__llvm__) || __GNUC__ > 7)
40 // See https://bugzilla.mozilla.org/show_bug.cgi?id=1442819
41 __attribute__((no_sanitize("vptr")))
42#endif
43 SendableHelper&
44 operator=(SendableHelper&& rhs) {
45 // it is safe to call Move() multiple times with the same rhs
46 SendableRegistry::Move(static_cast<Derived*>(this),
47 static_cast<Derived*>(&rhs));
48 return *this;
49 }
