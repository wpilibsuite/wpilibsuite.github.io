47  : m_f(f), m_h(h) {

49  m_contR = MakeCovMatrix(measurementStdDevs);
50  m_dt = dt;
51 
52  Reset();
53 
54  Eigen::Matrix<double, States, States> contA =
55  NumericalJacobianX<States, States, Inputs>(
56  m_f, m_xHat, Eigen::Matrix<double, Inputs, 1>::Zero());
57  Eigen::Matrix<double, Outputs, States> C =
58  NumericalJacobianX<Outputs, States, Inputs>(
59  m_h, m_xHat, Eigen::Matrix<double, Inputs, 1>::Zero());
60 
61  Eigen::Matrix<double, States, States> discA;
62  Eigen::Matrix<double, States, States> discQ;
63  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
64 
65  Eigen::Matrix<double, Outputs, Outputs> discR =
66  DiscretizeR<Outputs>(m_contR, dt);
67 
68  // IsStabilizable(A^T, C^T) will tell us if the system is observable.
69  bool isObservable =
70  IsStabilizable<States, Outputs>(discA.transpose(), C.transpose());
71  if (isObservable && Outputs <= States) {
72  m_initP = drake::math::DiscreteAlgebraicRiccatiEquation(
73  discA.transpose(), C.transpose(), discQ, discR);
74  } else {
75  m_initP = Eigen::Matrix<double, States, States>::Zero();
76  }
77  m_P = m_initP;
78  }

141  void Predict(const Eigen::Matrix<double, Inputs, 1>& u, units::second_t dt) {
142  m_dt = dt;
143 
144  // Find continuous A
145  Eigen::Matrix<double, States, States> contA =
146  NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
147 
148  // Find discrete A and Q
149  Eigen::Matrix<double, States, States> discA;
150  Eigen::Matrix<double, States, States> discQ;
151  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
152 
153  m_xHat = RK4(m_f, m_xHat, u, dt);
154  m_P = discA * m_P * discA.transpose() + discQ;
155  }

181  template <int Rows>
182  void Correct(const Eigen::Matrix<double, Inputs, 1>& u,
183  const Eigen::Matrix<double, Rows, 1>& y,
184  std::function<Eigen::Matrix<double, Rows, 1>(
185  const Eigen::Matrix<double, States, 1>&,
186  const Eigen::Matrix<double, Inputs, 1>&)>
187  h,
188  const Eigen::Matrix<double, Rows, Rows>& R) {
189  const Eigen::Matrix<double, Rows, States> C =
190  NumericalJacobianX<Rows, States, Inputs>(h, m_xHat, u);
191  const Eigen::Matrix<double, Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
192 
193  Eigen::Matrix<double, Rows, Rows> S = C * m_P * C.transpose() + discR;
194 
195  // We want to put K = PC^T S^-1 into Ax = b form so we can solve it more
196  // efficiently.
197  //
198  // K = PC^T S^-1
199  // KS = PC^T
200  // (KS)^T = (PC^T)^T
201  // S^T K^T = CP^T
202  //
203  // The solution of Ax = b can be found via x = A.solve(b).
204  //
205  // K^T = S^T.solve(CP^T)
206  // K = (S^T.solve(CP^T))^T
207  Eigen::Matrix<double, States, Rows> K =
208  S.transpose().ldlt().solve(C * m_P.transpose()).transpose();
209 
210  m_xHat += K * (y - h(m_xHat, u));
211  m_P = (Eigen::Matrix<double, States, States>::Identity() - K * C) * m_P;
212  }
