39  f,

43  h,
44  const wpi::array<double, States>& stateStdDevs,
45  const wpi::array<double, Outputs>& measurementStdDevs,
46  units::second_t dt)
47  : m_f(f), m_h(h) {
48  m_contQ = MakeCovMatrix(stateStdDevs);
49  m_contR = MakeCovMatrix(measurementStdDevs);
50  m_residualFuncY = [](auto a, auto b) -> Eigen::Matrix<double, Outputs, 1> {
51  return a - b;
52  };
53  m_addFuncX = [](auto a, auto b) -> Eigen::Matrix<double, States, 1> {
54  return a + b;
55  };
56  m_dt = dt;
57 
58  Reset();
59 
60  Eigen::Matrix<double, States, States> contA =
61  NumericalJacobianX<States, States, Inputs>(
62  m_f, m_xHat, Eigen::Matrix<double, Inputs, 1>::Zero());
63  Eigen::Matrix<double, Outputs, States> C =
64  NumericalJacobianX<Outputs, States, Inputs>(
65  m_h, m_xHat, Eigen::Matrix<double, Inputs, 1>::Zero());
66 
67  Eigen::Matrix<double, States, States> discA;
68  Eigen::Matrix<double, States, States> discQ;
69  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
70 
71  Eigen::Matrix<double, Outputs, Outputs> discR =
72  DiscretizeR<Outputs>(m_contR, dt);
73 
74  // IsStabilizable(A^T, C^T) will tell us if the system is observable.
75  bool isObservable =
76  IsStabilizable<States, Outputs>(discA.transpose(), C.transpose());
77  if (isObservable && Outputs <= States) {
78  m_initP = drake::math::DiscreteAlgebraicRiccatiEquation(
79  discA.transpose(), C.transpose(), discQ, discR);
80  } else {
81  m_initP = Eigen::Matrix<double, States, States>::Zero();
82  }
83  m_P = m_initP;
84  }

213  void Predict(const Eigen::Matrix<double, Inputs, 1>& u, units::second_t dt) {
214  m_dt = dt;
215 
216  // Find continuous A
217  Eigen::Matrix<double, States, States> contA =
218  NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
219 
220  // Find discrete A and Q
221  Eigen::Matrix<double, States, States> discA;
222  Eigen::Matrix<double, States, States> discQ;
223  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
224 
225  m_xHat = RK4(m_f, m_xHat, u, dt);
226  m_P = discA * m_P * discA.transpose() + discQ;
227  }

240  template <int Rows>
241  void Correct(const Eigen::Matrix<double, Inputs, 1>& u,
242  const Eigen::Matrix<double, Rows, 1>& y,
243  std::function<Eigen::Matrix<double, Rows, 1>(
244  const Eigen::Matrix<double, States, 1>&,
245  const Eigen::Matrix<double, Inputs, 1>&)>
246  h,
247  const Eigen::Matrix<double, Rows, Rows>& R) {
248  auto residualFuncY = [](auto a, auto b) -> Eigen::Matrix<double, Rows, 1> {
249  return a - b;
250  };
251  auto addFuncX = [](auto a, auto b) -> Eigen::Matrix<double, States, 1> {
252  return a + b;
253  };
254  Correct<Rows>(u, y, h, R, residualFuncY, addFuncX);

273  template <int Rows>
274  void Correct(const Eigen::Matrix<double, Inputs, 1>& u,
275  const Eigen::Matrix<double, Rows, 1>& y,
276  std::function<Eigen::Matrix<double, Rows, 1>(
277  const Eigen::Matrix<double, States, 1>&,
278  const Eigen::Matrix<double, Inputs, 1>&)>
279  h,
280  const Eigen::Matrix<double, Rows, Rows>& R,
281  std::function<Eigen::Matrix<double, Rows, 1>(
282  const Eigen::Matrix<double, Rows, 1>&,
283  const Eigen::Matrix<double, Rows, 1>&)>
284  residualFuncY,
285  std::function<Eigen::Matrix<double, States, 1>(
286  const Eigen::Matrix<double, States, 1>&,
287  const Eigen::Matrix<double, States, 1>)>
288  addFuncX) {
289  const Eigen::Matrix<double, Rows, States> C =
290  NumericalJacobianX<Rows, States, Inputs>(h, m_xHat, u);
291  const Eigen::Matrix<double, Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
292 
293  Eigen::Matrix<double, Rows, Rows> S = C * m_P * C.transpose() + discR;
294 
295  // We want to put K = PC^T S^-1 into Ax = b form so we can solve it more
296  // efficiently.
297  //
298  // K = PC^T S^-1
299  // KS = PC^T
300  // (KS)^T = (PC^T)^T
301  // S^T K^T = CP^T
302  //
303  // The solution of Ax = b can be found via x = A.solve(b).
304  //
305  // K^T = S^T.solve(CP^T)
306  // K = (S^T.solve(CP^T))^T
307  Eigen::Matrix<double, States, Rows> K =
308  S.transpose().ldlt().solve(C * m_P.transpose()).transpose();
309 
310  m_xHat = addFuncX(m_xHat, K * residualFuncY(y, h(m_xHat, u)));
311  m_P = (Eigen::Matrix<double, States, States>::Identity() - K * C) * m_P;
312  }
