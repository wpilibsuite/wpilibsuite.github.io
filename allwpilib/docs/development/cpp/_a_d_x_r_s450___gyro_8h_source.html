18/**
19 * Use a rate gyro to return the robots heading relative to a starting position.
20 *
21 * The %Gyro class tracks the robots heading based on the starting position. As
22 * the robot rotates the new heading is computed by integrating the rate of
23 * rotation returned by the sensor. When the class is instantiated, it does a
24 * short calibration routine where it samples the gyro while at rest to
25 * determine the default offset. This is subtracted from each sample to
26 * determine the heading.
27 *
28 * This class is for the digital ADXRS450 gyro sensor that connects via SPI.
29 * Only one instance of an ADXRS %Gyro is supported.
30 */
31class ADXRS450_Gyro : public Gyro,
32 public wpi::Sendable,
33 public wpi::SendableHelper<ADXRS450_Gyro> {
34 public:
35 /**
36 * %Gyro constructor on onboard CS0.
37 */
38 ADXRS450_Gyro();
39
40 /**
41 * %Gyro constructor on the specified SPI port.
42 *
43 * @param port The SPI port the gyro is attached to.
44 */
45 explicit ADXRS450_Gyro(SPI::Port port);
46
47 ~ADXRS450_Gyro() override = default;
48
49 ADXRS450_Gyro(ADXRS450_Gyro&&) = default;
50 ADXRS450_Gyro& operator=(ADXRS450_Gyro&&) = default;
51
52 /**
53 * Return the actual angle in degrees that the robot is currently facing.
54 *
55 * The angle is based on integration of the returned rate from the gyro.
56 * The angle is continuous, that is it will continue from 360->361 degrees.
57 * This allows algorithms that wouldn't want to see a discontinuity in the
58 * gyro output as it sweeps from 360 to 0 on the second time around.
59 *
60 * @return the current heading of the robot in degrees.
61 */
62 double GetAngle() const override;
63
64 /**
65 * Return the rate of rotation of the gyro
66 *
67 * The rate is based on the most recent reading of the gyro.
68 *
69 * @return the current rate in degrees per second
70 */
71 double GetRate() const override;
72
73 /**
74 * Reset the gyro.
75 *
76 * Resets the gyro to a heading of zero. This can be used if there is
77 * significant drift in the gyro and it needs to be recalibrated after it has
78 * been running.
79 */
80 void Reset() override;
81
82 /**
83 * Initialize the gyro.
84 *
85 * Calibrate the gyro by running for a number of samples and computing the
86 * center value. Then use the center value as the Accumulator center value for
87 * subsequent measurements.
88 *
89 * It's important to make sure that the robot is not moving while the
90 * centering calculations are in progress, this is typically done when the
91 * robot is first turned on while it's sitting at rest before the competition
92 * starts.
93 */
94 void Calibrate() final;
95
96 /**
97 * Get the SPI port number.
98 *
99 * @return The SPI port number.
100 */
101 int GetPort() const;
102
103 void InitSendable(wpi::SendableBuilder& builder) override;
104
105 private:
106 SPI m_spi;
107 SPI::Port m_port;
108
109 hal::SimDevice m_simDevice;
110 hal::SimDouble m_simAngle;
111 hal::SimDouble m_simRate;
112
113 uint16_t ReadRegister(int reg);
114};
