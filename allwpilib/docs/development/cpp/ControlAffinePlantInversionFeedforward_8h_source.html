166 Eigen::Matrix<double, Inputs, 1> Calculate(
167 const Eigen::Matrix<double, States, 1>& r,
168 const Eigen::Matrix<double, States, 1>& nextR) {
169 Eigen::Matrix<double, States, 1> rDot = (nextR - r) / m_dt.to<double>();
170
171 m_uff = m_B.householderQr().solve(
