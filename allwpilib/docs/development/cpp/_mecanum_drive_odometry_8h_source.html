17/**
18 * Class for mecanum drive odometry. Odometry allows you to track the robot's
19 * position on the field over a course of a match using readings from your
20 * mecanum wheel encoders.
21 *
22 * Teams can use odometry during the autonomous period for complex tasks like
23 * path following. Furthermore, odometry can be used for latency compensation
24 * when using computer-vision systems.
25 */
26class WPILIB_DLLEXPORT MecanumDriveOdometry {
27 public:
28 /**
29 * Constructs a MecanumDriveOdometry object.
30 *
31 * @param kinematics The mecanum drive kinematics for your drivetrain.
32 * @param gyroAngle The angle reported by the gyroscope.
33 * @param wheelPositions The current distances measured by each wheel.
34 * @param initialPose The starting position of the robot on the field.
35 */
36 explicit MecanumDriveOdometry(MecanumDriveKinematics kinematics,
37 const Rotation2d& gyroAngle,
38 const MecanumDriveWheelPositions wheelPositions,
39 const Pose2d& initialPose = Pose2d{});
40
41 /**
42 * Resets the robot's position on the field.
43 *
44 * The gyroscope angle does not need to be reset here on the user's robot
45 * code. The library automatically takes care of offsetting the gyro angle.
46 *
47 * @param pose The position on the field that your robot is at.
48 * @param gyroAngle The angle reported by the gyroscope.
49 * @param wheelPositions The current distances measured by each wheel.
50 */
51 void ResetPosition(const Pose2d& pose, const Rotation2d& gyroAngle,
52 const MecanumDriveWheelPositions wheelPositions) {
53 m_pose = pose;
54 m_previousAngle = pose.Rotation();
55 m_gyroOffset = m_pose.Rotation() - gyroAngle;
56 m_previousWheelPositions = wheelPositions;
57 }
58
59 /**
60 * Returns the position of the robot on the field.
61 * @return The pose of the robot.
62 */
63 const Pose2d& GetPose() const { return m_pose; }
64
65 /**
66 * Updates the robot's position on the field using forward kinematics and
67 * integration of the pose over time. This method takes in an angle parameter
68 * which is used instead of the angular rate that is calculated from forward
69 * kinematics, in addition to the current distance measurement at each wheel.
70 *
71 * @param gyroAngle The angle reported by the gyroscope.
72 * @param wheelPositions The current distances measured by each wheel.
73 *
74 * @return The new pose of the robot.
75 */
76 const Pose2d& Update(const Rotation2d& gyroAngle,
77 const MecanumDriveWheelPositions wheelPositions);
78
79 private:
80 MecanumDriveKinematics m_kinematics;
81 Pose2d m_pose;
82
83 MecanumDriveWheelPositions m_previousWheelPositions;
84 Rotation2d m_previousAngle;
85 Rotation2d m_gyroOffset;
86};
