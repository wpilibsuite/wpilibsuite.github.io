17/**
18 * Class for mecanum drive odometry. Odometry allows you to track the robot's
19 * position on the field over a course of a match using readings from your
20 * mecanum wheel encoders.
21 *
22 * Teams can use odometry during the autonomous period for complex tasks like
23 * path following. Furthermore, odometry can be used for latency compensation
24 * when using computer-vision systems.
25 */
26class WPILIB_DLLEXPORT MecanumDriveOdometry {
27 public:
28 /**
29 * Constructs a MecanumDriveOdometry object.
30 *
31 * @param kinematics The mecanum drive kinematics for your drivetrain.
32 * @param gyroAngle The angle reported by the gyroscope.
33 * @param initialPose The starting position of the robot on the field.
34 */
35 explicit MecanumDriveOdometry(MecanumDriveKinematics kinematics,
36 const Rotation2d& gyroAngle,
37 const Pose2d& initialPose = Pose2d());
38
39 /**
40 * Resets the robot's position on the field.
41 *
42 * The gyroscope angle does not need to be reset here on the user's robot
43 * code. The library automatically takes care of offsetting the gyro angle.
44 *
45 * @param pose The position on the field that your robot is at.
46 * @param gyroAngle The angle reported by the gyroscope.
47 */
48 void ResetPosition(const Pose2d& pose, const Rotation2d& gyroAngle) {
49 m_pose = pose;
50 m_previousAngle = pose.Rotation();
51 m_gyroOffset = m_pose.Rotation() - gyroAngle;
52 }
53
54 /**
55 * Returns the position of the robot on the field.
56 * @return The pose of the robot.
57 */
58 const Pose2d& GetPose() const { return m_pose; }
59
60 /**
61 * Updates the robot's position on the field using forward kinematics and
62 * integration of the pose over time. This method takes in the current time as
63 * a parameter to calculate period (difference between two timestamps). The
64 * period is used to calculate the change in distance from a velocity. This
65 * also takes in an angle parameter which is used instead of the
66 * angular rate that is calculated from forward kinematics.
67 *
68 * @param currentTime The current time.
69 * @param gyroAngle The angle reported by the gyroscope.
70 * @param wheelSpeeds The current wheel speeds.
71 *
72 * @return The new pose of the robot.
73 */
74 const Pose2d& UpdateWithTime(units::second_t currentTime,
75 const Rotation2d& gyroAngle,
76 MecanumDriveWheelSpeeds wheelSpeeds);
77
78 /**
79 * Updates the robot's position on the field using forward kinematics and
80 * integration of the pose over time. This method automatically calculates
81 * the current time to calculate period (difference between two timestamps).
82 * The period is used to calculate the change in distance from a velocity.
83 * This also takes in an angle parameter which is used instead of the
84 * angular rate that is calculated from forward kinematics.
85 *
86 * @param gyroAngle The angle reported by the gyroscope.
87 * @param wheelSpeeds The current wheel speeds.
88 *
89 * @return The new pose of the robot.
90 */
91 const Pose2d& Update(const Rotation2d& gyroAngle,
92 MecanumDriveWheelSpeeds wheelSpeeds) {
93 return UpdateWithTime(wpi::Now() * 1.0e-6_s, gyroAngle, wheelSpeeds);
94 }
95
96 private:
97 MecanumDriveKinematics m_kinematics;
98 Pose2d m_pose;
99
100 units::second_t m_previousTime = -1_s;
101 Rotation2d m_previousAngle;
102 Rotation2d m_gyroOffset;
103};
