17/**
18 * A base for CommandGroups. Statically tracks commands that have been
19 * allocated to groups to ensure those commands are not also used independently,
20 * which can result in inconsistent command state and unpredictable execution.
21 *
22 * This class is provided by the NewCommands VendorDep
23 */
24class CommandGroupBase : public CommandBase {
25 public:
26 /**
27 * Requires that the specified command not have been already allocated to a
28 * CommandGroup. Reports an error if the command is already grouped.
29 *
30 * @param command The command to check
31 * @return True if all the command is ungrouped.
32 */
33 static bool RequireUngrouped(const Command& command);
34
35 /**
36 * Requires that the specified command not have been already allocated to a
37 * CommandGroup. Reports an error if the command is already grouped.
38 *
39 * @param command The command to check
40 * @return True if all the command is ungrouped.
41 */
42 static bool RequireUngrouped(const Command* command);
43
44 /**
45 * Requires that the specified commands not have been already allocated to a
46 * CommandGroup. Reports an error if any of the commands are already grouped.
47 *
48 * @param commands The commands to check
49 * @return True if all the commands are ungrouped.
50 */
51 static bool RequireUngrouped(
52 wpi::span<const std::unique_ptr<Command>> commands);
53
54 /**
55 * Requires that the specified commands not have been already allocated to a
56 * CommandGroup. Reports an error if any of the commands are already grouped.
57 *
58 * @param commands The commands to check
59 * @return True if all the commands are ungrouped.
60 */
61 static bool RequireUngrouped(std::initializer_list<const Command*> commands);
62
63 /**
64 * Adds the given commands to the command group.
65 *
66 * @param commands The commands to add.
67 */
68 virtual void AddCommands(
69 std::vector<std::unique_ptr<Command>>&& commands) = 0;
70};
71} // namespace frc2
