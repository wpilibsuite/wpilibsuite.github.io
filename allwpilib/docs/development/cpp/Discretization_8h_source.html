37 template <int States, int Inputs>
38 void DiscretizeAB(const Eigen::Matrix<double, States, States>& contA,
39  const Eigen::Matrix<double, States, Inputs>& contB,
40  units::second_t dt,
41  Eigen::Matrix<double, States, States>* discA,
42  Eigen::Matrix<double, States, Inputs>* discB) {
43  // Matrices are blocked here to minimize matrix exponentiation calculations
44  Eigen::Matrix<double, States + Inputs, States + Inputs> Mcont;
45  Mcont.setZero();
46  Mcont.template block<States, States>(0, 0) = contA * dt.to<double>();
47  Mcont.template block<States, Inputs>(0, States) = contB * dt.to<double>();
48 
49  // Discretize A and B with the given timestep
50  Eigen::Matrix<double, States + Inputs, States + Inputs> Mdisc = Mcont.exp();
51  *discA = Mdisc.template block<States, States>(0, 0);
52  *discB = Mdisc.template block<States, Inputs>(0, States);
53 }

64 template <int States>
65 void DiscretizeAQ(const Eigen::Matrix<double, States, States>& contA,
66  const Eigen::Matrix<double, States, States>& contQ,
67  units::second_t dt,
68  Eigen::Matrix<double, States, States>* discA,
69  Eigen::Matrix<double, States, States>* discQ) {
70  // Make continuous Q symmetric if it isn't already
71  Eigen::Matrix<double, States, States> Q = (contQ + contQ.transpose()) / 2.0;
72 
73  // Set up the matrix M = [[-A, Q], [0, A.T]]
74  Eigen::Matrix<double, 2 * States, 2 * States> M;
75  M.template block<States, States>(0, 0) = -contA;
76  M.template block<States, States>(0, States) = Q;
77  M.template block<States, States>(States, 0).setZero();
78  M.template block<States, States>(States, States) = contA.transpose();
79 
80  Eigen::Matrix<double, 2 * States, 2 * States> phi =
81  (M * dt.to<double>()).exp();
82 
83  // Phi12 = phi[0:States, States:2*States]
84  // Phi22 = phi[States:2*States, States:2*States]
85  Eigen::Matrix<double, States, States> phi12 =
86  phi.block(0, States, States, States);
87  Eigen::Matrix<double, States, States> phi22 =
88  phi.block(States, States, States, States);
89 
90  *discA = phi22.transpose();
91 
92  Q = *discA * phi12;
93 
94  // Make discrete Q symmetric if it isn't already
95  *discQ = (Q + Q.transpose()) / 2.0;
96 }

116 template <int States>
117 void DiscretizeAQTaylor(const Eigen::Matrix<double, States, States>& contA,
118  const Eigen::Matrix<double, States, States>& contQ,
119  units::second_t dt,
120  Eigen::Matrix<double, States, States>* discA,
121  Eigen::Matrix<double, States, States>* discQ) {
122  // Make continuous Q symmetric if it isn't already
123  Eigen::Matrix<double, States, States> Q = (contQ + contQ.transpose()) / 2.0;
124 
125  Eigen::Matrix<double, States, States> lastTerm = Q;
126  double lastCoeff = dt.to<double>();
127 
128  // A^T^n
129  Eigen::Matrix<double, States, States> Atn = contA.transpose();
130 
131  Eigen::Matrix<double, States, States> phi12 = lastTerm * lastCoeff;
132 
133  // i = 6 i.e. 5th order should be enough precision
134  for (int i = 2; i < 6; ++i) {
135  lastTerm = -contA * lastTerm + Q * Atn;
136  lastCoeff *= dt.to<double>() / static_cast<double>(i);
137 
138  phi12 += lastTerm * lastCoeff;
139 
140  Atn *= contA.transpose();
141  }
142 
143  DiscretizeA<States>(contA, dt, discA);
144  Q = *discA * phi12;
145 
146  // Make discrete Q symmetric if it isn't already
147  *discQ = (Q + Q.transpose()) / 2.0;
148 }
