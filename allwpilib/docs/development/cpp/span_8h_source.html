113template <typename ElementType, std::size_t Extent>
114class span {
115 static_assert(std::is_object<ElementType>::value,
116 "A span's ElementType must be an object type (not a "
117 "reference type or void)");
118 static_assert(detail::is_complete<ElementType>::value,
119 "A span's ElementType must be a complete type (not a forward "
120 "declaration)");
121 static_assert(!std::is_abstract<ElementType>::value,
122 "A span's ElementType cannot be an abstract class type");
123
124 using storage_type = detail::span_storage<ElementType, Extent>;
125
126public:
127 // constants and types
128 using element_type = ElementType;
129 using value_type = typename std::remove_cv<ElementType>::type;
130 using size_type = std::size_t;
131 using difference_type = std::ptrdiff_t;
132 using pointer = element_type*;
133 using const_pointer = const element_type*;
134 using reference = element_type&;
135 using const_reference = const element_type&;
136 using iterator = pointer;
137 using reverse_iterator = std::reverse_iterator<iterator>;

139 static constexpr size_type extent = Extent;

141 // [span.cons], span constructors, copy, assignment, and destructor
142 template <

144 typename std::enable_if<(E == dynamic_extent || E <= 0), int>::type = 0>
145 constexpr span() noexcept
146 {}
147
148 constexpr span(pointer ptr, size_type count)
149 : storage_(ptr, count)
150 {
151 assert(extent == dynamic_extent || count == extent);
152 }
153
154 constexpr span(pointer first_elem, pointer last_elem)
155 : storage_(first_elem, last_elem - first_elem)
156 {
157 assert(extent == dynamic_extent ||
158 last_elem - first_elem ==

188 : storage_(arr.data(), N)

192 typename Container, std::size_t E = Extent,
