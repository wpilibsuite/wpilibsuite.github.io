63 static std::shared_ptr<Async> Create(const std::shared_ptr<Loop>& loop) {
64 auto h = std::make_shared<Async>(loop, private_init{});
65 int err =
66 uv_async_init(loop->GetRaw(), h->GetRaw(), [](uv_async_t* handle) {
67 auto& h = *static_cast<Async*>(handle->data);
68 std::scoped_lock lock(h.m_mutex);
69 for (auto&& v : h.m_data) {
70 std::apply(h.wakeup, v);
71 }
72 h.m_data.clear();
73 });
74 if (err < 0) {
75 loop->ReportError(err);
76 return nullptr;
77 }
78 h->Keep();
79 return h;
80 }

88 template <typename... U>
89 void Send(U&&... u) {
90 auto loop = m_loop.lock();
91 if (loop && loop->GetThreadId() == std::this_thread::get_id()) {
92 // called from within the loop, just call the function directly
93 wakeup(std::forward<U>(u)...);
94 return;
95 }
96
97 {
98 std::scoped_lock lock(m_mutex);
99 m_data.emplace_back(std::forward_as_tuple(std::forward<U>(u)...));
100 }
101 if (loop) {
102 this->Invoke(&uv_async_send, this->GetRaw());
103 }
104 }
105
109 sig::Signal<T...> wakeup;
110
111 private:
112 wpi::mutex m_mutex;
113 std::vector<std::tuple<T...>> m_data;
114 std::weak_ptr<Loop> m_loop;
115};
