12namespace frc {
13class DigitalSource;
14
15/**
16 * Class for handling synchronous (blocking) interrupts.
17 *
18 * <p> By default, interrupts will occur on rising edge.
19 *
20 * <p> Asynchronous interrupts are handled by the AsynchronousInterrupt class.
21 */
22class SynchronousInterrupt {
23 public:
24 enum WaitResult {
25 kTimeout = 0x0,
26 kRisingEdge = 0x1,
27 kFallingEdge = 0x100,
28 kBoth = 0x101,
29 };
30
31 /**
32 * Construct a Synchronous Interrupt from a Digital Source.
33 *
34 * @param source the DigitalSource the interrupts are triggered from
35 */
36 explicit SynchronousInterrupt(DigitalSource& source);
37
38 /**
39 * Construct a Synchronous Interrupt from a Digital Source.
40 *
41 * @param source the DigitalSource the interrupts are triggered from
42 */
43 explicit SynchronousInterrupt(DigitalSource* source);
44
45 /**
46 * Construct a Synchronous Interrupt from a Digital Source.
47 *
48 * @param source the DigitalSource the interrupts are triggered from
49 */
50 explicit SynchronousInterrupt(std::shared_ptr<DigitalSource> source);
51
52 ~SynchronousInterrupt();
53
54 SynchronousInterrupt(SynchronousInterrupt&&) = default;
55 SynchronousInterrupt& operator=(SynchronousInterrupt&&) = default;
56
57 /**
58 * Wait for an interrupt to occur.
59 *
60 * <p> Both rising and falling edge can be returned if both a rising and
61 * falling happened between calls, and ignorePrevious is false.
62 *
63 * @param timeout The timeout to wait for. 0s or less will return immediately.
64 * @param ignorePrevious True to ignore any previous interrupts, false to
65 * return interrupt value if an interrupt has occured since last call.
66 * @return The edge(s) that were triggered, or timeout.
67 */
68 WaitResult WaitForInterrupt(units::second_t timeout,
69 bool ignorePrevious = true);
70
71 /**
72 * Set which edges cause an interrupt to occur.
73 *
74 * @param risingEdge true to trigger on rising edge, false otherwise.
75 * @param fallingEdge true to trigger on falling edge, false otherwise
76 */
77 void SetInterruptEdges(bool risingEdge, bool fallingEdge);
78
79 /**
80 * Get the timestamp (relative to FPGA Time) of the last rising edge.
81 *
82 * @return the timestamp in seconds relative to getFPGATime
83 */
84 units::second_t GetRisingTimestamp();
85
86 /**
87 * Get the timestamp of the last falling edge.
88 *
89 * <p>This function does not require the interrupt to be enabled to work.
90 *
91 * <p>This only works if falling edge was configured using setInterruptEdges.
92 * @return the timestamp in seconds relative to getFPGATime
93 */
94 units::second_t GetFallingTimestamp();
95
96 /**
97 * Wake up an existing wait call. Can be called from any thread.
98 */
99 void WakeupWaitingInterrupt();
100
101 private:
102 void InitSynchronousInterrupt();
103 std::shared_ptr<DigitalSource> m_source;
104 hal::Handle<HAL_InterruptHandle> m_handle;
105};
106} // namespace frc
