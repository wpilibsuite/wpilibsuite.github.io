12namespace frc2 {
13class Command;
14/**
15 * A robot subsystem. Subsystems are the basic unit of robot organization in
16 * the Command-based framework; they encapsulate low-level hardware objects
17 * (motor controllers, sensors, etc) and provide methods through which they can
18 * be used by Commands. Subsystems are used by the CommandScheduler's resource
19 * management system to ensure multiple robot actions are not "fighting" over
20 * the same hardware; Commands that use a subsystem should include that
21 * subsystem in their GetRequirements() method, and resources used within a
22 * subsystem should generally remain encapsulated and not be shared by other
23 * parts of the robot.
24 *
25 * <p>Subsystems must be registered with the scheduler with the
26 * CommandScheduler.RegisterSubsystem() method in order for the
27 * Periodic() method to be called. It is recommended that this method be called
28 * from the constructor of users' Subsystem implementations. The
29 * SubsystemBase class offers a simple base for user implementations
30 * that handles this.
31 *
32 * @see Command
33 * @see CommandScheduler
34 * @see SubsystemBase
35 */
36class Subsystem {
37 public:
38 ~Subsystem();
39 /**
40 * This method is called periodically by the CommandScheduler. Useful for
41 * updating subsystem-specific state that you don't want to offload to a
42 * Command. Teams should try to be consistent within their own codebases
43 * about which responsibilities will be handled by Commands, and which will be
44 * handled here.
45 */
46 virtual void Periodic();
47
48 /**
49 * This method is called periodically by the CommandScheduler. Useful for
50 * updating subsystem-specific state that needs to be maintained for
51 * simulations, such as for updating simulation classes and setting simulated
52 * sensor readings.
53 */
54 virtual void SimulationPeriodic();
55
56 /**
57 * Sets the default Command of the subsystem. The default command will be
58 * automatically scheduled when no other commands are scheduled that require
59 * the subsystem. Default commands should generally not end on their own, i.e.
60 * their IsFinished() method should always return false. Will automatically
61 * register this subsystem with the CommandScheduler.
62 *
63 * @param defaultCommand the default command to associate with this subsystem
64 */
65 template <class T, typename = std::enable_if_t<std::is_base_of_v<
66 Command, std::remove_reference_t<T>>>>
67 void SetDefaultCommand(T&& defaultCommand) {
68 CommandScheduler::GetInstance().SetDefaultCommand(
69 this, std::forward<T>(defaultCommand));
70 }
71
72 /**
73 * Gets the default command for this subsystem. Returns null if no default
74 * command is currently associated with the subsystem.
75 *
76 * @return the default command associated with this subsystem
77 */
78 Command* GetDefaultCommand() const;
79
80 /**
81 * Returns the command currently running on this subsystem. Returns null if
82 * no command is currently scheduled that requires this subsystem.
83 *
84 * @return the scheduled command currently requiring this subsystem
85 */
86 Command* GetCurrentCommand() const;
87
88 /**
89 * Registers this subsystem with the CommandScheduler, allowing its
90 * Periodic() method to be called when the scheduler runs.
91 */
92 void Register();
93};
94} // namespace frc2
