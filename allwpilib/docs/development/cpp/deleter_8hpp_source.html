21 /// A deleter class that deallocates the memory through a specified \concept{concept_rawallocator,RawAllocator}.
22 ///
23 /// It deallocates memory for a specified type but does not call its destructors.
24 /// \ingroup adapter
25 template <typename Type, class RawAllocator>
26 class allocator_deallocator : WPI_EBO(allocator_reference<RawAllocator>)
27 {
28 static_assert(!std::is_abstract<Type>::value,
29 "use allocator_polymorphic_deallocator for storing base classes");
30
31 public:
32 using allocator_type = typename allocator_reference<RawAllocator>::allocator_type;
33 using value_type = Type;
34
35 /// \effects Creates it without any associated allocator.
36 /// The deallocator must not be used if that is the case.
37 /// \notes This functions is useful if you have want to create an empty smart pointer without giving it an allocator.
38 allocator_deallocator() noexcept = default;
39
40 /// \effects Creates it by passing it an \ref allocator_reference.
41 /// It will store the reference to it and uses the referenced allocator object for the deallocation.
42 allocator_deallocator(allocator_reference<RawAllocator> alloc) noexcept
43 : allocator_reference<RawAllocator>(alloc)
44 {
45 }
46
47 /// \effects Deallocates the memory given to it.
48 /// Calls \c deallocate_node(pointer, sizeof(value_type), alignof(value_type)) on the referenced allocator object.
49 /// \requires The deallocator must not have been created by the default constructor.
50 void operator()(value_type* pointer) noexcept
51 {
52 this->deallocate_node(pointer, sizeof(value_type), alignof(value_type));
53 }
54
55 /// \returns The reference to the allocator.
56 /// It has the same type as the call to \ref allocator_reference::get_allocator().
57 /// \requires The deallocator must not be created by the default constructor.
58 auto get_allocator() const noexcept
59 -> decltype(std::declval<allocator_reference<allocator_type>>().get_allocator())
60 {
61 return this->allocator_reference<allocator_type>::get_allocator();
62 }
63 };

121 ///
122 /// It can only be created from a \ref allocator_deallocator and thus must only be used for smart pointers initialized by derived-to-base conversion of the pointer.
123 /// \ingroup adapter
124 template <typename BaseType, class RawAllocator>

129 using value_type = BaseType;

132 /// It will deallocate the memory as if done by the derived type.

144 void operator()(value_type* pointer) noexcept

146 this->deallocate_node(pointer, derived_size_, derived_alignment_);
147 }
