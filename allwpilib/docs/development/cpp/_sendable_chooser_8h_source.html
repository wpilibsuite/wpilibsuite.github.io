17/**
18 * The SendableChooser class is a useful tool for presenting a selection of
19 * options to the SmartDashboard.
20 *
21 * For instance, you may wish to be able to select between multiple autonomous
22 * modes. You can do this by putting every possible Command you want to run as
23 * an autonomous into a SendableChooser and then put it into the SmartDashboard
24 * to have a list of options appear on the laptop. Once autonomous starts,
25 * simply ask the SendableChooser what the selected value is.
26 *
27 * @tparam T The type of values to be stored
28 * @see SmartDashboard
29 */
30template <class T>
31class SendableChooser : public SendableChooserBase {
32 wpi::StringMap<T> m_choices;
33
34 template <class U>
35 static U _unwrap_smart_ptr(const U& value);
36
37 template <class U>
38 static U* _unwrap_smart_ptr(const std::unique_ptr<U>& value);
39
40 template <class U>
41 static std::weak_ptr<U> _unwrap_smart_ptr(const std::shared_ptr<U>& value);
42
43 public:
44 SendableChooser() = default;
45 ~SendableChooser() override = default;
46 SendableChooser(SendableChooser&& rhs) = default;
47 SendableChooser& operator=(SendableChooser&& rhs) = default;
48
49 /**
50 * Adds the given object to the list of options.
51 *
52 * On the SmartDashboard on the desktop, the object will appear as the given
53 * name.
54 *
55 * @param name the name of the option
56 * @param object the option
57 */
58 void AddOption(std::string_view name, T object);
59
60 /**
61 * Add the given object to the list of options and marks it as the default.
62 *
63 * Functionally, this is very close to AddOption() except that it will use
64 * this as the default option if none other is explicitly selected.
65 *
66 * @param name the name of the option
67 * @param object the option
68 */
69 void SetDefaultOption(std::string_view name, T object);
70
71 /**
72 * Adds the given object to the list of options.
73 *
74 * On the SmartDashboard on the desktop, the object will appear as the given
75 * name.
76 *
77 * @deprecated use AddOption(std::string_view name, T object) instead.
78 *
79 * @param name the name of the option
80 * @param object the option
81 */
82 WPI_DEPRECATED("use AddOption() instead")
83 void AddObject(std::string_view name, T object) { AddOption(name, object); }
84
85 /**
86 * Add the given object to the list of options and marks it as the default.
87 *
88 * Functionally, this is very close to AddOption() except that it will use
89 * this as the default option if none other is explicitly selected.
90 *
91 * @deprecated use SetDefaultOption(std::string_view name, T object) instead.
92 *
93 * @param name the name of the option
94 * @param object the option
95 */
96 WPI_DEPRECATED("use SetDefaultOption() instead")
97 void AddDefault(std::string_view name, T object) {
98 SetDefaultOption(name, object);
99 }
100
101 /**
102 * Returns a copy of the selected option (a raw pointer U* if T =
103 * std::unique_ptr<U> or a std::weak_ptr<U> if T = std::shared_ptr<U>).
104 *
105 * If there is none selected, it will return the default. If there is none
106 * selected and no default, then it will return a value-initialized instance.
107 * For integer types, this is 0. For container types like std::string, this is
108 * an empty string.
109 *
110 * @return The option selected
111 */
112 auto GetSelected() -> decltype(_unwrap_smart_ptr(m_choices[""]));
113
114 void InitSendable(nt::NTSendableBuilder& builder) override;
115};
