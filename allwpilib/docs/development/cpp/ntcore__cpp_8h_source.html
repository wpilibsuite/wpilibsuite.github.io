73/** NetworkTables Connection Information */
74struct ConnectionInfo {
75 /**
76 * The remote identifier (as set on the remote node by
77 * NetworkTableInstance::SetNetworkIdentity() or nt::SetNetworkIdentity()).
78 */
79 std::string remote_id;
80
81 /** The IP address of the remote node. */
82 std::string remote_ip;
83
84 /** The port number of the remote node. */
85 unsigned int remote_port{0};
86
87 /**
88 * The last time any update was received from the remote node (same scale as
89 * returned by nt::Now()).
90 */
91 int64_t last_update{0};
92
93 /**
94 * The protocol version being used for this connection. This in protocol
95 * layer format, so 0x0200 = 2.0, 0x0300 = 3.0).
96 */
97 unsigned int protocol_version{0};
98
99 friend void swap(ConnectionInfo& first, ConnectionInfo& second) {
100 using std::swap;
101 swap(first.remote_id, second.remote_id);
102 swap(first.remote_ip, second.remote_ip);
103 swap(first.remote_port, second.remote_port);
104 swap(first.last_update, second.last_update);
105 swap(first.protocol_version, second.protocol_version);
106 }
107};

239/** NetworkTables publish/subscribe option. */
240class PubSubOption {
241 public:
242 constexpr PubSubOption(NT_PubSubOptionType type, double value)
243 : type{type}, value{value} {}
244
245 /**
246 * How frequently changes will be sent over the network. NetworkTables may
247 * send more frequently than this (e.g. use a combined minimum period for all
248 * values) or apply a restricted range to this value. The default if
249 * unspecified (and the immediate flag is false) is 100 ms. This option and
250 * the immediate option override each other.
251 *
252 * @param time time between updates, in seconds
253 * @return option
254 */
255 static constexpr PubSubOption Periodic(double time) {
256 return PubSubOption{NT_PUBSUB_PERIODIC, time};
257 }
258
259 /**
260 * If enabled, sends all value changes over the network even if only sent
261 * periodically. This option defaults to disabled.
262 *
263 * @param enabled True to enable, false to disable
264 * @return option
265 */
266 static constexpr PubSubOption SendAll(bool enabled) {
267 return PubSubOption{NT_PUBSUB_SENDALL, enabled ? 1.0 : 0.0};
268 }
269
270 /**
271 * If enabled, no value changes are sent over the network. This option
272 * defaults to disabled.
273 *
274 * @param enabled True to enable, false to disable
275 * @return option
276 */
277 static constexpr PubSubOption TopicsOnly(bool enabled) {
278 return PubSubOption{NT_PUBSUB_TOPICSONLY, enabled ? 1.0 : 0.0};
279 }
280
281 /**
282 * If enabled, preserves duplicate value changes (rather than ignoring them).
283 * This option defaults to disabled.
284 *
285 * @param enabled True to enable, false to disable
286 * @return option
287 */
288 static constexpr PubSubOption KeepDuplicates(bool enabled) {
289 return PubSubOption{NT_PUBSUB_KEEPDUPLICATES, enabled ? 1.0 : 0.0};
290 }
291
292 /**
293 * Polling storage for subscription. Specifies the maximum number of updates
294 * NetworkTables should store between calls to the subscriber's poll()
295 * function. Defaults to 1 if logging is false, 20 if logging is true.
296 *
297 * @param depth number of entries to save for polling.
298 * @return option
299 */
300 static constexpr PubSubOption PollStorage(int depth) {
301 return PubSubOption{NT_PUBSUB_POLLSTORAGE, static_cast<double>(depth)};
302 }
303
304 NT_PubSubOptionType type;
305 double value;
306};
