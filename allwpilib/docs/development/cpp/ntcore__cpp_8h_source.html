44/**
45 * Event notification flags.
46 *
47 * The flags are a bitmask and must be OR'ed together to indicate the
48 * combination of events desired to be received.
49 *
50 */
51struct EventFlags {
52 EventFlags() = delete;
53
54 static constexpr unsigned int kNone = NT_EVENT_NONE;
55 /**
56 * Initial listener addition.
57 * Set this flag to receive immediate notification of matches to the
58 * flag criteria.
59 */
60 static constexpr unsigned int kImmediate = NT_EVENT_IMMEDIATE;
61 /** Client connected (on server, any client connected). */
62 static constexpr unsigned int kConnected = NT_EVENT_CONNECTED;
63 /** Client disconnected (on server, any client disconnected). */
64 static constexpr unsigned int kDisconnected = NT_EVENT_DISCONNECTED;
65 /** Any connection event (connect or disconnect). */
66 static constexpr unsigned int kConnection = kConnected | kDisconnected;
67 /** New topic published. */
68 static constexpr unsigned int kPublish = NT_EVENT_PUBLISH;
69 /** Topic unpublished. */
70 static constexpr unsigned int kUnpublish = NT_EVENT_UNPUBLISH;
71 /** Topic properties changed. */
72 static constexpr unsigned int kProperties = NT_EVENT_PROPERTIES;
73 /** Any topic event (publish, unpublish, or properties changed). */
74 static constexpr unsigned int kTopic = kPublish | kUnpublish | kProperties;
75 /** Topic value updated (via network). */
76 static constexpr unsigned int kValueRemote = NT_EVENT_VALUE_REMOTE;
77 /** Topic value updated (local). */
78 static constexpr unsigned int kValueLocal = NT_EVENT_VALUE_LOCAL;
79 /** Topic value updated (network or local). */
80 static constexpr unsigned int kValueAll = kValueRemote | kValueLocal;
81 /** Log message. */
82 static constexpr unsigned int kLogMessage = NT_EVENT_LOGMESSAGE;
83};

190class Event {
191 public:
192 Event() = default;
193 Event(NT_Listener listener, unsigned int flags, ConnectionInfo info)
194 : listener{listener}, flags{flags}, data{std::move(info)} {}
195 Event(NT_Listener listener, unsigned int flags, TopicInfo info)
196 : listener{listener}, flags{flags}, data{std::move(info)} {}
197 Event(NT_Listener listener, unsigned int flags, ValueEventData data)
198 : listener{listener}, flags{flags}, data{std::move(data)} {}
199 Event(NT_Listener listener, unsigned int flags, LogMessage msg)
200 : listener{listener}, flags{flags}, data{std::move(msg)} {}
201 Event(NT_Listener listener, unsigned int flags, NT_Topic topic,
202 NT_Handle subentry, Value value)
203 : listener{listener},
204 flags{flags},
205 data{ValueEventData{topic, subentry, std::move(value)}} {}
206 Event(NT_Listener listener, unsigned int flags, unsigned int level,
207 std::string_view filename, unsigned int line, std::string_view message)
208 : listener{listener},
209 flags{flags},
210 data{LogMessage{level, filename, line, message}} {}
211
212 /** Listener that triggered this event. */
213 NT_Listener listener{0};
214
215 /**
216 * Event flags (NT_EventFlags). Also indicates the data included with the
217 * event:
218 * - NT_NOTIFY_CONNECTED or NT_NOTIFY_DISCONNECTED: GetConnectionInfo()

224 unsigned int flags{0};
225
226 /** Event data; content depends on flags. */
227 std::variant<ConnectionInfo, TopicInfo, ValueEventData, LogMessage> data;

229 const ConnectionInfo* GetConnectionInfo() const {
230 return std::get_if<nt::ConnectionInfo>(&data);
231 }

249 return std::get_if<nt::LogMessage>(&data);
250 }

254/** NetworkTables publish/subscribe option. */
255class PubSubOption {
256 public:
257 constexpr PubSubOption(NT_PubSubOptionType type, double value)
258 : type{type}, value{value} {}
259
260 /**
261 * How frequently changes will be sent over the network. NetworkTables may
262 * send more frequently than this (e.g. use a combined minimum period for all
263 * values) or apply a restricted range to this value. The default if
264 * unspecified (and the immediate flag is false) is 100 ms. This option and
265 * the immediate option override each other.
266 *
267 * @param time time between updates, in seconds
268 * @return option
269 */
270 static constexpr PubSubOption Periodic(double time) {
271 return PubSubOption{NT_PUBSUB_PERIODIC, time};
272 }
273
274 /**
275 * If enabled, sends all value changes over the network even if only sent
276 * periodically. This option defaults to disabled.
277 *
278 * @param enabled True to enable, false to disable
279 * @return option
280 */
281 static constexpr PubSubOption SendAll(bool enabled) {
282 return PubSubOption{NT_PUBSUB_SENDALL, enabled ? 1.0 : 0.0};
283 }
284
285 /**
286 * If enabled, no value changes are sent over the network. This option
287 * defaults to disabled.
288 *
289 * @param enabled True to enable, false to disable
290 * @return option
291 */
292 static constexpr PubSubOption TopicsOnly(bool enabled) {
293 return PubSubOption{NT_PUBSUB_TOPICSONLY, enabled ? 1.0 : 0.0};
294 }
295
296 /**
297 * If enabled, preserves duplicate value changes (rather than ignoring them).
298 * This option defaults to disabled.
299 *
300 * @param enabled True to enable, false to disable
301 * @return option
302 */
303 static constexpr PubSubOption KeepDuplicates(bool enabled) {
304 return PubSubOption{NT_PUBSUB_KEEPDUPLICATES, enabled ? 1.0 : 0.0};
305 }
306
307 /**
308 * Polling storage for subscription. Specifies the maximum number of updates
309 * NetworkTables should store between calls to the subscriber's poll()
310 * function. Defaults to 1 if SendAll is false, 20 if SendAll is true.
311 *
312 * @param depth number of entries to save for polling.
313 * @return option
314 */
315 static constexpr PubSubOption PollStorage(int depth) {
316 return PubSubOption{NT_PUBSUB_POLLSTORAGE, static_cast<double>(depth)};
317 }
318
319 NT_PubSubOptionType type;
320 double value;
321};
