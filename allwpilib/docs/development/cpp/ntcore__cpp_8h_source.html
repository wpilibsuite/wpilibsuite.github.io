34/** NetworkTables Entry Information */
35struct EntryInfo {
36 /** Entry handle */
37 NT_Entry entry;
38
39 /** Entry name */
40 std::string name;
41
42 /** Entry type */
43 NT_Type type;
44
45 /** Entry flags */
46 unsigned int flags;
47
48 /** Timestamp of last change to entry (type or value). */
49 uint64_t last_change;
50
51 friend void swap(EntryInfo& first, EntryInfo& second) {
52 using std::swap;
53 swap(first.entry, second.entry);
54 swap(first.name, second.name);
55 swap(first.type, second.type);
56 swap(first.flags, second.flags);
57 swap(first.last_change, second.last_change);
58 }
59};

61/** NetworkTables Connection Information */
62struct ConnectionInfo {
63 /**
64 * The remote identifier (as set on the remote node by
65 * NetworkTableInstance::SetNetworkIdentity() or nt::SetNetworkIdentity()).
66 */
67 std::string remote_id;
68
69 /** The IP address of the remote node. */
70 std::string remote_ip;
71
72 /** The port number of the remote node. */
73 unsigned int remote_port{0};
74
75 /**
76 * The last time any update was received from the remote node (same scale as
77 * returned by nt::Now()).
78 */
79 uint64_t last_update{0};
80
81 /**
82 * The protocol version being used for this connection. This in protocol
83 * layer format, so 0x0200 = 2.0, 0x0300 = 3.0).
84 */
85 unsigned int protocol_version{0};
86
87 friend void swap(ConnectionInfo& first, ConnectionInfo& second) {
88 using std::swap;
89 swap(first.remote_id, second.remote_id);
90 swap(first.remote_ip, second.remote_ip);
91 swap(first.remote_port, second.remote_port);
92 swap(first.last_update, second.last_update);
93 swap(first.protocol_version, second.protocol_version);
94 }
95};

125/** NetworkTables Remote Procedure Call (Server Side) */
126class RpcAnswer {
127 public:
128 RpcAnswer() = default;
129 RpcAnswer(NT_Entry entry_, NT_RpcCall call_, std::string_view name_,
130 std::string_view params_, ConnectionInfo conn_)
131 : entry(entry_),
132 call(call_),
133 name(name_),
134 params(params_),
135 conn(std::move(conn_)) {}
136
137 /** Entry handle. */
138 NT_Entry entry{0};
139
140 /** Call handle. */
141 mutable NT_RpcCall call{0};
142
143 /** Entry name. */
144 std::string name;
145
146 /** Call raw parameters. */
147 std::string params;
148
149 /** Connection that called the RPC. */
150 ConnectionInfo conn;
151
152 /**
153 * Determines if the native handle is valid.
154 * @return True if the native handle is valid, false otherwise.
155 */
156 explicit operator bool() const { return call != 0; }
157
158 /**
159 * Post RPC response (return value) for a polled RPC.
160 * @param result result raw data that will be provided to remote caller
161 * @return True if posting the response is valid, otherwise false
162 */
163 bool PostResponse(std::string_view result) const;
164
165 friend void swap(RpcAnswer& first, RpcAnswer& second) {
166 using std::swap;
167 swap(first.entry, second.entry);
168 swap(first.call, second.call);
169 swap(first.name, second.name);
170 swap(first.params, second.params);
171 swap(first.conn, second.conn);
172 }
173};
