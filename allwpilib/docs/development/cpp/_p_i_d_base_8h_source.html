23/**
24 * Class implements a PID Control Loop.
25 *
26 * Creates a separate thread which reads the given PIDSource and takes care of
27 * the integral calculations, as well as writing the given PIDOutput.
28 *
29 * This feedback controller runs in discrete time, so time deltas are not used
30 * in the integral and derivative calculations. Therefore, the sample rate
31 * affects the controller's behavior for a given set of PID constants.
32 *
33 * @deprecated All APIs which use this have been deprecated.
34 */
35class PIDBase : public PIDInterface,
36 public PIDOutput,
37 public wpi::Sendable,
38 public wpi::SendableHelper<PIDBase> {
39 public:
40 /**
41 * Allocate a PID object with the given constants for P, I, D.
42 *
43 * @param p the proportional coefficient
44 * @param i the integral coefficient
45 * @param d the derivative coefficient
46 * @param source The PIDSource object that is used to get values
47 * @param output The PIDOutput object that is set to the output value
48 */
49 WPI_DEPRECATED("All APIs which use this have been deprecated.")
50 PIDBase(double p, double i, double d, PIDSource& source, PIDOutput& output);
51
52 /**
53 * Allocate a PID object with the given constants for P, I, D.

59 * @param source The PIDSource object that is used to get values

64 PIDOutput& output);

125 virtual void SetPID(double p, double i, double d, double f);

260 /**
261 * Set the number of previous error samples to average for tolerancing. When
262 * determining whether a mechanism is on target, the user may want to use a
263 * rolling average of previous measurements instead of a precise position or
264 * velocity. This is useful for noisy sensors which return a few erroneous
265 * measurements when the mechanism is on target. However, the mechanism will
266 * not register as on target for at least the specified bufLength cycles.
267 *
268 * @param bufLength Number of previous cycles to average. Defaults to 1.
269 */
270 WPI_DEPRECATED("Use a LinearDigitalFilter as the input.")

273 /**
274 * Return true if the error is within the percentage of the total input range,
275 * determined by SetTolerance. This asssumes that the maximum and minimum
276 * input were set using SetInput.
277 *
278 * Currently this just reports on target as the actual value passes through
279 * the setpoint. Ideally it should be based on being within the tolerance for
280 * some period of time.
281 *
282 * This will return false until at least one input value has been computed.
283 */
284 virtual bool OnTarget() const;

294 * PIDControllers can be nested by passing a PIDController as another
295 * PIDController's output. In that case, the output of the parent controller

297 *
298 * It is the caller's responsibility to put the data into a valid form for
299 * SetSetpoint().
300 */
301 void PIDWrite(double output) override;

323 virtual void Calculate();

325 /**
326 * Calculate the feed forward term.
327 *
328 * Both of the provided feed forward calculations are velocity feed forwards.
329 * If a different feed forward calculation is desired, the user can override
330 * this function and provide his or her own. This function does no
331 * synchronization because the PIDBase class only calls it in synchronized
332 * code, so be careful if calling it oneself.
333 *
334 * If a velocity PID controller is being used, the F term should be set to 1
335 * over the maximum setpoint for the output. If a position PID controller is
336 * being used, the F term should be set to 1 over the maximum speed for the
337 * output measured in setpoint units per this controller's update period (see
338 * the default period in this class's constructor).
339 */
340 virtual double CalculateFeedForward();
341
342 /**
343 * Wraps error around for continuous inputs. The original error is returned if

345 *
346 * @param error The current error of the PID controller.
347 * @return Error for continuous inputs.
348 */
349 double GetContinuousError(double error) const;
