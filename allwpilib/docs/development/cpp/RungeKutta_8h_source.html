19 template <typename F, typename T>
20 T RungeKutta(F&& f, T x, units::second_t dt) {
21  const auto halfDt = 0.5 * dt;
22  T k1 = f(x);
23  T k2 = f(x + k1 * halfDt.to<double>());
24  T k3 = f(x + k2 * halfDt.to<double>());
25  T k4 = f(x + k3 * dt.to<double>());
26  return x + dt.to<double>() / 6.0 * (k1 + 2.0 * k2 + 2.0 * k3 + k4);
27 }

56 T RungeKuttaTimeVarying(F&& f, T x, units::second_t t, units::second_t dt) {
57  const auto halfDt = 0.5 * dt;
58  T k1 = f(t, x);
59  T k2 = f(t + halfDt, x + k1 / 2.0);
60  T k3 = f(t + halfDt, x + k2 / 2.0);
61  T k4 = f(t + dt, x + k3);
