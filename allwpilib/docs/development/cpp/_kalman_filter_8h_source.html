27/**
28 * A Kalman filter combines predictions from a model and measurements to give an
29 * estimate of the true system state. This is useful because many states cannot
30 * be measured directly as a result of sensor noise, or because the state is
31 * "hidden".
32 *
33 * Kalman filters use a K gain matrix to determine whether to trust the model or
34 * measurements more. Kalman filter theory uses statistics to compute an optimal
35 * K gain which minimizes the sum of squares error in the state estimate. This K
36 * gain is used to correct the state estimate by some amount of the difference
37 * between the actual measurements and the measurements predicted by the model.
38 *
39 * For more on the underlying math, read
40 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf chapter 9
41 * "Stochastic control theory".
42 *
43 * @tparam States The number of states.
44 * @tparam Inputs The number of inputs.
45 * @tparam Outputs The number of outputs.
46 */
47template <int States, int Inputs, int Outputs>
48class KalmanFilterImpl {
49 public:
50 /**
51 * Constructs a state-space observer with the given plant.
52 *
53 * @param plant The plant used for the prediction step.
54 * @param stateStdDevs Standard deviations of model states.
55 * @param measurementStdDevs Standard deviations of measurements.
56 * @param dt Nominal discretization timestep.
57 */
58 KalmanFilterImpl(LinearSystem<States, Inputs, Outputs>& plant,
59 const wpi::array<double, States>& stateStdDevs,
60 const wpi::array<double, Outputs>& measurementStdDevs,
61 units::second_t dt) {
62 m_plant = &plant;
63
64 auto contQ = MakeCovMatrix(stateStdDevs);
65 auto contR = MakeCovMatrix(measurementStdDevs);
66
67 Eigen::Matrix<double, States, States> discA;
68 Eigen::Matrix<double, States, States> discQ;
69 DiscretizeAQTaylor<States>(plant.A(), contQ, dt, &discA, &discQ);
70
71 auto discR = DiscretizeR<Outputs>(contR, dt);
72
73 const auto& C = plant.C();
74
75 if (!IsDetectable<States, Outputs>(discA, C)) {
76 std::string msg = fmt::format(
77 "The system passed to the Kalman filter is "
78 "unobservable!\n\nA =\n{}\nC =\n{}\n",
79 discA, C);
80
81 wpi::math::MathSharedStore::ReportError(msg);
82 throw std::invalid_argument(msg);
83 }
84
85 Eigen::Matrix<double, States, States> P =
86 drake::math::DiscreteAlgebraicRiccatiEquation(
87 discA.transpose(), C.transpose(), discQ, discR);
88
89 // S = CPCᵀ + R
90 Eigen::Matrix<double, Outputs, Outputs> S = C * P * C.transpose() + discR;
91
92 // We want to put K = PCᵀS⁻¹ into Ax = b form so we can solve it more
93 // efficiently.
94 //
95 // K = PCᵀS⁻¹
96 // KS = PCᵀ
97 // (KS)ᵀ = (PCᵀ)ᵀ
98 // SᵀKᵀ = CPᵀ
99 //
100 // The solution of Ax = b can be found via x = A.solve(b).
101 //
102 // Kᵀ = Sᵀ.solve(CPᵀ)
103 // K = (Sᵀ.solve(CPᵀ))ᵀ
104 m_K = S.transpose().ldlt().solve(C * P.transpose()).transpose();
105
106 Reset();
107 }
108
109 KalmanFilterImpl(KalmanFilterImpl&&) = default;
110 KalmanFilterImpl& operator=(KalmanFilterImpl&&) = default;
111
112 /**
113 * Returns the steady-state Kalman gain matrix K.
114 */
115 const Eigen::Matrix<double, States, Outputs>& K() const { return m_K; }
116
117 /**
118 * Returns an element of the steady-state Kalman gain matrix K.
119 *
120 * @param i Row of K.
121 * @param j Column of K.
122 */
123 double K(int i, int j) const { return m_K(i, j); }
124
125 /**
126 * Returns the state estimate x-hat.
127 */
128 const Eigen::Vector<double, States>& Xhat() const { return m_xHat; }
129
130 /**
131 * Returns an element of the state estimate x-hat.
132 *
133 * @param i Row of x-hat.
134 */
135 double Xhat(int i) const { return m_xHat(i); }
136
137 /**
138 * Set initial state estimate x-hat.
139 *
140 * @param xHat The state estimate x-hat.
141 */
142 void SetXhat(const Eigen::Vector<double, States>& xHat) { m_xHat = xHat; }
143
144 /**
145 * Set an element of the initial state estimate x-hat.
146 *
147 * @param i Row of x-hat.
148 * @param value Value for element of x-hat.
149 */
150 void SetXhat(int i, double value) { m_xHat(i) = value; }
151
152 /**
153 * Resets the observer.
154 */
155 void Reset() { m_xHat.setZero(); }
156
157 /**
158 * Project the model into the future with a new control input u.
159 *
160 * @param u New control input from controller.
161 * @param dt Timestep for prediction.
162 */
163 void Predict(const Eigen::Vector<double, Inputs>& u, units::second_t dt) {
164 m_xHat = m_plant->CalculateX(m_xHat, u, dt);
165 }
166
167 /**
168 * Correct the state estimate x-hat using the measurements in y.
169 *
170 * @param u Same control input used in the last predict step.
171 * @param y Measurement vector.
172 */
173 void Correct(const Eigen::Vector<double, Inputs>& u,
174 const Eigen::Vector<double, Outputs>& y) {
175 // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − (Cx̂ₖ₊₁⁻ + Duₖ₊₁))
176 m_xHat += m_K * (y - (m_plant->C() * m_xHat + m_plant->D() * u));
177 }
178
179 private:
180 LinearSystem<States, Inputs, Outputs>* m_plant;
181
182 /**
183 * The steady-state Kalman gain matrix.
184 */
185 Eigen::Matrix<double, States, Outputs> m_K;
186
187 /**
188 * The state estimate.
189 */
190 Eigen::Vector<double, States> m_xHat;
191};

195template <int States, int Inputs, int Outputs>
196class KalmanFilter : public detail::KalmanFilterImpl<States, Inputs, Outputs> {
197 public:
198 /**
199 * Constructs a state-space observer with the given plant.
200 *
201 * @param plant The plant used for the prediction step.
202 * @param stateStdDevs Standard deviations of model states.
203 * @param measurementStdDevs Standard deviations of measurements.
204 * @param dt Nominal discretization timestep.
205 */
206 KalmanFilter(LinearSystem<States, Inputs, Outputs>& plant,
207 const wpi::array<double, States>& stateStdDevs,
208 const wpi::array<double, Outputs>& measurementStdDevs,
209 units::second_t dt)
210 : detail::KalmanFilterImpl<States, Inputs, Outputs>{
211 plant, stateStdDevs, measurementStdDevs, dt} {}
212
213 KalmanFilter(KalmanFilter&&) = default;

215};
