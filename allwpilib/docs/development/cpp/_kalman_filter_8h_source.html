16/**
17 * A Kalman filter combines predictions from a model and measurements to give an
18 * estimate of the true system state. This is useful because many states cannot
19 * be measured directly as a result of sensor noise, or because the state is
20 * "hidden".
21 *
22 * Kalman filters use a K gain matrix to determine whether to trust the model or
23 * measurements more. Kalman filter theory uses statistics to compute an optimal
24 * K gain which minimizes the sum of squares error in the state estimate. This K
25 * gain is used to correct the state estimate by some amount of the difference
26 * between the actual measurements and the measurements predicted by the model.
27 *
28 * For more on the underlying math, read
29 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf chapter 9
30 * "Stochastic control theory".
31 *
32 * @tparam States The number of states.
33 * @tparam Inputs The number of inputs.
34 * @tparam Outputs The number of outputs.
35 */
36template <int States, int Inputs, int Outputs>
37class KalmanFilter {
38 public:
39 using StateVector = Vectord<States>;
40 using InputVector = Vectord<Inputs>;
41 using OutputVector = Vectord<Outputs>;
42
43 using StateArray = wpi::array<double, States>;
44 using OutputArray = wpi::array<double, Outputs>;
45
46 /**
47 * Constructs a state-space observer with the given plant.
48 *
49 * @param plant The plant used for the prediction step.
50 * @param stateStdDevs Standard deviations of model states.
51 * @param measurementStdDevs Standard deviations of measurements.
52 * @param dt Nominal discretization timestep.
53 */
54 KalmanFilter(LinearSystem<States, Inputs, Outputs>& plant,
55 const StateArray& stateStdDevs,
56 const OutputArray& measurementStdDevs, units::second_t dt);
57
58 KalmanFilter(KalmanFilter&&) = default;
59 KalmanFilter& operator=(KalmanFilter&&) = default;
60
61 /**
62 * Returns the steady-state Kalman gain matrix K.
63 */
64 const Matrixd<States, Outputs>& K() const { return m_K; }
65
66 /**
67 * Returns an element of the steady-state Kalman gain matrix K.
68 *
69 * @param i Row of K.
70 * @param j Column of K.
71 */
72 double K(int i, int j) const { return m_K(i, j); }
73
74 /**
75 * Returns the state estimate x-hat.
76 */
77 const StateVector& Xhat() const { return m_xHat; }
78
79 /**
80 * Returns an element of the state estimate x-hat.
81 *
82 * @param i Row of x-hat.
83 */
84 double Xhat(int i) const { return m_xHat(i); }
85
86 /**
87 * Set initial state estimate x-hat.
88 *
89 * @param xHat The state estimate x-hat.
90 */
91 void SetXhat(const StateVector& xHat) { m_xHat = xHat; }
92
93 /**
94 * Set an element of the initial state estimate x-hat.
95 *
96 * @param i Row of x-hat.
97 * @param value Value for element of x-hat.
98 */
99 void SetXhat(int i, double value) { m_xHat(i) = value; }
100
101 /**
102 * Resets the observer.
103 */
104 void Reset() { m_xHat.setZero(); }
105
106 /**
107 * Project the model into the future with a new control input u.
108 *
109 * @param u New control input from controller.
110 * @param dt Timestep for prediction.
111 */
112 void Predict(const InputVector& u, units::second_t dt);
113
114 /**
115 * Correct the state estimate x-hat using the measurements in y.
116 *
117 * @param u Same control input used in the last predict step.
118 * @param y Measurement vector.
119 */
120 void Correct(const InputVector& u, const OutputVector& y);
121
122 private:
123 LinearSystem<States, Inputs, Outputs>* m_plant;
124
125 /**
126 * The steady-state Kalman gain matrix.
127 */
128 Matrixd<States, Outputs> m_K;
129
130 /**
131 * The state estimate.
132 */
133 StateVector m_xHat;
134};
