41/// This class implements an extremely fast bulk output stream that can *only*
42/// output to a stream. It does not support seeking, reopening, rewinding, line
43/// buffered disciplines etc. It is a simple buffer that outputs
44/// a chunk at a time.
45class raw_ostream {
46public:
47 // Class kinds to support LLVM-style RTTI.
48 enum class OStreamKind {
49 OK_OStream,
50 OK_FDStream,
51 };
52
53private:
54 OStreamKind Kind;
55
56 /// The buffer is handled in such a way that the buffer is
57 /// uninitialized, unbuffered, or out of space when OutBufCur >=
58 /// OutBufEnd. Thus a single comparison suffices to determine if we
59 /// need to take the slow path to write a single character.
60 ///
61 /// The buffer is in one of three states:
62 /// 1. Unbuffered (BufferMode == Unbuffered)
63 /// 1. Uninitialized (BufferMode != Unbuffered && OutBufStart == 0).
64 /// 2. Buffered (BufferMode != Unbuffered && OutBufStart != 0 &&
65 /// OutBufEnd - OutBufStart >= 1).
66 ///
67 /// If buffered, then the raw_ostream owns the buffer if (BufferMode ==
68 /// InternalBuffer); otherwise the buffer has been set via SetBuffer and is
69 /// managed by the subclass.
70 ///
71 /// If a subclass installs an external buffer using SetBuffer then it can wait
72 /// for a \see write_impl() call to handle the data which has been put into
73 /// this buffer.
74 char *OutBufStart, *OutBufEnd, *OutBufCur;
75
76 /// Optional stream this stream is tied to. If this stream is written to, the
77 /// tied-to stream will be flushed first.
78 raw_ostream *TiedStream = nullptr;
79
80 enum class BufferKind {
81 Unbuffered = 0,
82 InternalBuffer,
83 ExternalBuffer
84 } BufferMode;
85
86public:

90 RED,
91 GREEN,
92 YELLOW,
93 BLUE,

96 WHITE,
97 SAVEDCOLOR,
98 RESET,
99 };

112 explicit raw_ostream(bool unbuffered = false,
113 OStreamKind K = OStreamKind::OK_OStream)
114 : Kind(K), BufferMode(unbuffered ? BufferKind::Unbuffered
115 : BufferKind::InternalBuffer) {

117 OutBufStart = OutBufEnd = OutBufCur = nullptr;

131 // Configuration Interface

133
134 /// If possible, pre-allocate \p ExtraSize bytes for stream data.
135 /// i.e. it extends internal buffers to keep additional ExtraSize bytes.
136 /// So that the stream could keep at least tell() + ExtraSize bytes
137 /// without re-allocations. reserveExtraSpace() does not change
138 /// the size/data of the stream.
139 virtual void reserveExtraSpace(uint64_t ExtraSize) {}
140
141 /// Set the stream to be buffered, with an automatically determined buffer
142 /// size.
143 void SetBuffered();
144
145 /// Set the stream to be buffered, using the specified buffer size.
146 void SetBufferSize(size_t Size) {
147 flush();
148 SetBufferAndMode(new char[Size], Size, BufferKind::InternalBuffer);
149 }
150
151 size_t GetBufferSize() const {
152 // If we're supposed to be buffered but haven't actually gotten around
153 // to allocating the buffer yet, return the value that would be used.
154 if (BufferMode != BufferKind::Unbuffered && OutBufStart == nullptr)
155 return preferred_buffer_size();
156
157 // Otherwise just return the size of the allocated buffer.
158 return OutBufEnd - OutBufStart;
159 }
160
161 /// Set the stream to be unbuffered. When unbuffered, the stream will flush
162 /// after every write. This routine will also flush the buffer immediately
163 /// when the stream is being set to unbuffered.
164 void SetUnbuffered() {
165 flush();
166 SetBufferAndMode(nullptr, 0, BufferKind::Unbuffered);
167 }
168
169 size_t GetNumBytesInBuffer() const {
170 return OutBufCur - OutBufStart;
171 }

177 void flush() {
178 if (OutBufCur != OutBufStart)
179 flush_nonempty();

182 raw_ostream &operator<<(char C) {
183 if (OutBufCur >= OutBufEnd)
184 return write(C);
185 *OutBufCur++ = C;
186 return *this;
187 }

189 raw_ostream &operator<<(unsigned char C) {
190 if (OutBufCur >= OutBufEnd)
191 return write(C);
192 *OutBufCur++ = C;
193 return *this;
194 }
195
196 raw_ostream &operator<<(signed char C) {
197 if (OutBufCur >= OutBufEnd)
198 return write(C);
199 *OutBufCur++ = C;
200 return *this;
201 }
202
203 raw_ostream &operator<<(span<const uint8_t> Arr) {
204 // Inline fast path, particularly for arrays with a known length.
205 size_t Size = Arr.size();
206
207 // Make sure we can use the fast path.
208 if (Size > (size_t)(OutBufEnd - OutBufCur))
209 return write(Arr.data(), Size);
210
211 if (Size) {
212 memcpy(OutBufCur, Arr.data(), Size);
213 OutBufCur += Size;
214 }
215 return *this;
216 }
217
218 raw_ostream &operator<<(std::string_view Str) {
219 // Inline fast path, particularly for strings with a known length.
220 size_t Size = Str.size();
221
222 // Make sure we can use the fast path.
223 if (Size > (size_t)(OutBufEnd - OutBufCur))
224 return write(Str.data(), Size);
225
226 if (Size) {
227 memcpy(OutBufCur, Str.data(), Size);
228 OutBufCur += Size;
229 }
230 return *this;
231 }

233 raw_ostream &operator<<(const char *Str) {
234 // Inline fast path, particularly for constant strings where a sufficiently

274 /// Changes the foreground color of text that will be output from this point
275 /// forward.
276 /// @param Color ANSI color to use, the special SAVEDCOLOR can be used to

297 /// "console" window. That is, the output would be displayed to the user

314 // Subclass Interface

317private:
318 /// The is the piece of the class that is implemented by subclasses. This
319 /// writes the \p Size bytes starting at
320 /// \p Ptr to the underlying stream.
321 ///
322 /// This function is guaranteed to only be called at a point at which it is
323 /// safe for the subclass to install a new buffer via SetBuffer.
324 ///
325 /// \param Ptr The start of the data to be written. For buffered streams this
326 /// is guaranteed to be the start of the buffer.
327 ///
328 /// \param Size The number of bytes to be written.
329 ///
330 /// \invariant { Size > 0 }
331 virtual void write_impl(const char *Ptr, size_t Size) = 0;
332
333 /// Return the current position within the stream, not counting the bytes
334 /// currently in the buffer.
335 virtual uint64_t current_pos() const = 0;
336
337protected:
338 /// Use the provided buffer as the raw_ostream buffer. This is intended for
339 /// use only by subclasses which can arrange for the output to go directly
340 /// into the desired output buffer, instead of being copied on each flush.
341 void SetBuffer(char *BufferStart, size_t Size) {
342 SetBufferAndMode(BufferStart, Size, BufferKind::ExternalBuffer);
343 }
344
345 /// Return an efficient buffer size for the underlying output mechanism.
346 virtual size_t preferred_buffer_size() const;
347
348 /// Return the beginning of the current stream buffer, or 0 if the stream is
349 /// unbuffered.
350 const char *getBufferStart() const { return OutBufStart; }
351
352 //===--------------------------------------------------------------------===//
353 // Private Interface
354 //===--------------------------------------------------------------------===//
355private:
356 /// Install the given buffer and mode.
357 void SetBufferAndMode(char *BufferStart, size_t Size, BufferKind Mode);
358
359 /// Flush the current buffer, which is known to be non-empty. This outputs the
360 /// currently buffered data and resets the buffer to empty.
361 void flush_nonempty();
362
363 /// Copy data into the buffer. Size must not be greater than the number of
364 /// unused bytes in the buffer.
365 void copy_to_buffer(const char *Ptr, size_t Size);
366
367 /// Flush the tied-to stream (if present) and then write the required data.
368 void flush_tied_then_write(const char *Ptr, size_t Size);
369
370 virtual void anchor();
371};

373/// Call the appropriate insertion operator, given an rvalue reference to a
374/// raw_ostream object and return a stream of the same type as the argument.
375template <typename OStream, typename T>
376std::enable_if_t<!std::is_reference<OStream>::value &&
377 std::is_base_of<raw_ostream, OStream>::value,
378 OStream &&>
379operator<<(OStream &&OS, const T &Value) {
380 OS << Value;
381 return std::move(OS);
382}

384/// An abstract base class for streams implementations that also support a
385/// pwrite operation. This is useful for code that can mostly stream out data,
386/// but needs to patch in a header that needs to know the output size.
387class raw_pwrite_stream : public raw_ostream {
388 virtual void pwrite_impl(const char *Ptr, size_t Size, uint64_t Offset) = 0;
389 void anchor() override;
390
391public:
392 explicit raw_pwrite_stream(bool Unbuffered = false,
393 OStreamKind K = OStreamKind::OK_OStream)
394 : raw_ostream(Unbuffered, K) {}
395 void pwrite(const char *Ptr, size_t Size, uint64_t Offset) {
396#ifndef NDEBUG
397 uint64_t Pos = tell();
398 // /dev/null always reports a pos of 0, so we cannot perform this check
399 // in that case.
400 if (Pos)
401 assert(Size + Offset <= Pos && "We don't support extending the stream");
402#endif
403 pwrite_impl(Ptr, Size, Offset);
404 }
405};

413class raw_fd_ostream : public raw_pwrite_stream {

430

432

440 void anchor() override;
441

469 fs::CreationDisposition Disp, fs::FileAccess Access,
470 fs::OpenFlags Flags);

472 /// FD is the file descriptor that this writes to. If ShouldClose is true,
473 /// this closes the file when the stream is destroyed. If FD is for stdout or
474 /// stderr, it will not be closed.
475 raw_fd_ostream(int fd, bool shouldClose, bool unbuffered = false,
476 OStreamKind K = OStreamKind::OK_OStream);

479

481 /// fsync.
482 void close();
483

485

487 /// to the offset specified from the beginning of the file.

489
490 std::error_code error() const { return EC; }
491
492 /// Return the value of the flag in this raw_fd_ostream indicating whether an
493 /// output error has been encountered.
494 /// This doesn't implicitly flush any pending output. Also, it doesn't
495 /// guarantee to detect all errors unless the stream has been closed.
496 bool has_error() const { return bool(EC); }
497
498 /// Set the flag read by has_error() to false. If the error flag is set at the
499 /// time when this raw_ostream's destructor is called, report_fatal_error is
500 /// called to report the error. Use clear_error() after handling the error to
501 /// avoid this behavior.
502 ///
503 /// "Errors should never pass silently.
504 /// Unless explicitly silenced."
505 /// - from The Zen of Python, by Tim Peters
506 ///

508};
