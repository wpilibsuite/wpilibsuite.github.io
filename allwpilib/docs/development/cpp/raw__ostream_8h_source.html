36/// This class implements an extremely fast bulk output stream that can *only*
37/// output to a stream. It does not support seeking, reopening, rewinding, line
38/// buffered disciplines etc. It is a simple buffer that outputs
39/// a chunk at a time.
40class raw_ostream {
41private:
42 /// The buffer is handled in such a way that the buffer is
43 /// uninitialized, unbuffered, or out of space when OutBufCur >=
44 /// OutBufEnd. Thus a single comparison suffices to determine if we
45 /// need to take the slow path to write a single character.
46 ///
47 /// The buffer is in one of three states:
48 /// 1. Unbuffered (BufferMode == Unbuffered)
49 /// 1. Uninitialized (BufferMode != Unbuffered && OutBufStart == 0).
50 /// 2. Buffered (BufferMode != Unbuffered && OutBufStart != 0 &&
51 /// OutBufEnd - OutBufStart >= 1).
52 ///
53 /// If buffered, then the raw_ostream owns the buffer if (BufferMode ==
54 /// InternalBuffer); otherwise the buffer has been set via SetBuffer and is
55 /// managed by the subclass.
56 ///
57 /// If a subclass installs an external buffer using SetBuffer then it can wait
58 /// for a \see write_impl() call to handle the data which has been put into
59 /// this buffer.
60 char *OutBufStart, *OutBufEnd, *OutBufCur;
61
62 enum BufferKind {
63 Unbuffered = 0,
64 InternalBuffer,
65 ExternalBuffer
66 } BufferMode;
67
68public:
69 // color order matches ANSI escape sequence, don't change
70 enum Colors {
71 BLACK = 0,
72 RED,
73 GREEN,
74 YELLOW,
75 BLUE,
76 MAGENTA,
77 CYAN,
78 WHITE,
79 SAVEDCOLOR
80 };
81
82 explicit raw_ostream(bool unbuffered = false)
83 : BufferMode(unbuffered ? Unbuffered : InternalBuffer) {
84 // Start out ready to flush.
85 OutBufStart = OutBufEnd = OutBufCur = nullptr;
86 }
87
88 raw_ostream(const raw_ostream &) = delete;
89 void operator=(const raw_ostream &) = delete;
90
91 virtual ~raw_ostream();
92
93 /// tell - Return the current offset with the file.
94 uint64_t tell() const { return current_pos() + GetNumBytesInBuffer(); }

100 /// Set the stream to be buffered, with an automatically determined buffer
101 /// size.
102 void SetBuffered();
103
104 /// Set the stream to be buffered, using the specified buffer size.
105 void SetBufferSize(size_t Size) {
106 flush();
107 SetBufferAndMode(new char[Size], Size, InternalBuffer);
108 }
109
110 size_t GetBufferSize() const {
111 // If we're supposed to be buffered but haven't actually gotten around
112 // to allocating the buffer yet, return the value that would be used.
113 if (BufferMode != Unbuffered && OutBufStart == nullptr)
114 return preferred_buffer_size();
115
116 // Otherwise just return the size of the allocated buffer.
117 return OutBufEnd - OutBufStart;
118 }
119
120 /// Set the stream to be unbuffered. When unbuffered, the stream will flush
121 /// after every write. This routine will also flush the buffer immediately
122 /// when the stream is being set to unbuffered.
123 void SetUnbuffered() {
124 flush();
125 SetBufferAndMode(nullptr, 0, Unbuffered);
126 }
127
128 size_t GetNumBytesInBuffer() const {
129 return OutBufCur - OutBufStart;
130 }

162 raw_ostream &operator<<(span<const uint8_t> Arr) {
163 // Inline fast path, particularly for arrays with a known length.
164 size_t Size = Arr.size();
165
166 // Make sure we can use the fast path.
167 if (Size > (size_t)(OutBufEnd - OutBufCur))
168 return write(Arr.data(), Size);
169
170 if (Size) {
171 memcpy(OutBufCur, Arr.data(), Size);
172 OutBufCur += Size;
173 }
174 return *this;
175 }

268private:
269 /// The is the piece of the class that is implemented by subclasses. This
270 /// writes the \p Size bytes starting at
271 /// \p Ptr to the underlying stream.
272 ///
273 /// This function is guaranteed to only be called at a point at which it is
274 /// safe for the subclass to install a new buffer via SetBuffer.
275 ///
276 /// \param Ptr The start of the data to be written. For buffered streams this
277 /// is guaranteed to be the start of the buffer.
278 ///
279 /// \param Size The number of bytes to be written.
280 ///
281 /// \invariant { Size > 0 }
282 virtual void write_impl(const char *Ptr, size_t Size) = 0;
283
284 // An out of line virtual method to provide a home for the class vtable.
285 virtual void handle();
286
287 /// Return the current position within the stream, not counting the bytes
288 /// currently in the buffer.
289 virtual uint64_t current_pos() const = 0;
290
291protected:
292 /// Use the provided buffer as the raw_ostream buffer. This is intended for
293 /// use only by subclasses which can arrange for the output to go directly
294 /// into the desired output buffer, instead of being copied on each flush.
295 void SetBuffer(char *BufferStart, size_t Size) {
296 SetBufferAndMode(BufferStart, Size, ExternalBuffer);
297 }
298
299 /// Return an efficient buffer size for the underlying output mechanism.
300 virtual size_t preferred_buffer_size() const;
301
302 /// Return the beginning of the current stream buffer, or 0 if the stream is
303 /// unbuffered.
304 const char *getBufferStart() const { return OutBufStart; }
305
306 //===--------------------------------------------------------------------===//
307 // Private Interface
308 //===--------------------------------------------------------------------===//
309private:
310 /// Install the given buffer and mode.
311 void SetBufferAndMode(char *BufferStart, size_t Size, BufferKind Mode);
312
313 /// Flush the current buffer, which is known to be non-empty. This outputs the
314 /// currently buffered data and resets the buffer to empty.
315 void flush_nonempty();
316
317 /// Copy data into the buffer. Size must not be greater than the number of
318 /// unused bytes in the buffer.
319 void copy_to_buffer(const char *Ptr, size_t Size);
320
321 virtual void anchor();
322};

324/// An abstract base class for streams implementations that also support a
325/// pwrite operation. This is useful for code that can mostly stream out data,
326/// but needs to patch in a header that needs to know the output size.
327class raw_pwrite_stream : public raw_ostream {
328 virtual void pwrite_impl(const char *Ptr, size_t Size, uint64_t Offset) = 0;
329 void anchor() override;
330
331public:
332 explicit raw_pwrite_stream(bool Unbuffered = false)
333 : raw_ostream(Unbuffered) {}
334 void pwrite(const char *Ptr, size_t Size, uint64_t Offset) {
335#ifndef NDBEBUG
336 uint64_t Pos = tell();
337 // /dev/null always reports a pos of 0, so we cannot perform this check
338 // in that case.
339 if (Pos)
340 assert(Size + Offset <= Pos && "We don't support extending the stream");
341#endif
342 pwrite_impl(Ptr, Size, Offset);
343 }
344};
