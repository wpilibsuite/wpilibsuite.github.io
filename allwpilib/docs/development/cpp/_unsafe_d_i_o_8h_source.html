54/**
55 * Unsafe digital output set function
56 * This function can be used to perform fast and determinstically set digital
57 * outputs. This function holds the DIO lock, so calling anyting other then
58 * functions on the Proxy object passed as a parameter can deadlock your
59 * program.
60 *
61 * @param[in] handle the HAL digital handle of the pin to toggle.
62 * @param[out] status status check
63 * @param[in] func A functor taking a ref to a DIOSetProxy object.
64 */
65template <typename Functor>
66void UnsafeManipulateDIO(HAL_DigitalHandle handle, int32_t* status,
67 Functor func) {
68 auto port = digitalChannelHandles->Get(handle, HAL_HandleEnum::DIO);
69 if (port == nullptr) {
70 *status = HAL_HANDLE_ERROR;
71 return;
72 }
73 wpi::mutex& dioMutex = detail::UnsafeGetDIOMutex();
74 tDIO* dSys = detail::UnsafeGetDigialSystem();
75 auto mask = detail::ComputeDigitalMask(handle, status);
76 if (*status != 0) {
77 return;
78 }
79 std::scoped_lock lock(dioMutex);
80
81 tDIO::tOutputEnable enableOE = dSys->readOutputEnable(status);
82 enableOE.value |= mask;
83 auto disableOE = enableOE;
84 disableOE.value &= ~mask;
85 tDIO::tDO enableDO = dSys->readDO(status);
86 enableDO.value |= mask;
87 auto disableDO = enableDO;
88 disableDO.value &= ~mask;
89
90 DIOSetProxy dioData{enableOE, disableOE, enableDO, disableDO, dSys};
91 func(dioData);
92}
