43 // the same as free_memory_list but optimized for small node sizes
44 // it is slower and does not support arrays
45 // but has very small overhead
46 // debug: allocate() and deallocate() mark memory as new and freed, respectively
47 // node_size is increased via two times fence size and fence is put in front and after
48 class small_free_memory_list
49 {
50 static constexpr std::size_t chunk_count(std::size_t number_of_nodes)
51 {
52 return number_of_nodes / chunk_max_nodes
53 + (number_of_nodes % chunk_max_nodes == 0 ? 0 : 1);
54 }
55
56 public:
57 // minimum element size
58 static constexpr std::size_t min_element_size = 1;
59 // alignment
60 static constexpr std::size_t min_element_alignment = 1;
61
62 // minimal size of the block that needs to be inserted
63 static constexpr std::size_t min_block_size(std::size_t node_size,
64 std::size_t number_of_nodes)
65 {
66 return chunk_count(number_of_nodes)
67 * (chunk_memory_offset + chunk_max_nodes * node_size);
68 }
