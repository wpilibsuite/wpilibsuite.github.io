17/**
18 * A wrapper around std::unique_ptr<Command> so commands have move-only
19 * semantics. Commands should only be stored and passed around when held in a
20 * CommandPtr instance. For more info, see
21 * https://github.com/wpilibsuite/allwpilib/issues/4303.
22 *
23 * Various classes in the command-based library accept a
24 * std::unique_ptr<Command>, use CommandPtr::Unwrap to convert.
25 * CommandPtr::UnwrapVector does the same for vectors.
26 */
27class CommandPtr final {
28 public:
29 explicit CommandPtr(std::unique_ptr<Command>&& command)
30 : m_ptr(std::move(command)) {}
31
32 template <class T, typename = std::enable_if_t<std::is_base_of_v<
33 Command, std::remove_reference_t<T>>>>
34 explicit CommandPtr(T&& command)
35 : CommandPtr(std::make_unique<std::remove_reference_t<T>>(
36 std::forward<T>(command))) {}
37
38 CommandPtr(CommandPtr&&) = default;
39 CommandPtr& operator=(CommandPtr&&) = default;
40

54 *

57 [[nodiscard]] CommandPtr AsProxy() &&;
