30 /// Tag type defining a memory pool optimized for arrays.
31 /// It keeps the nodes oredered inside the free list and searches the list for an appropriate memory block.
32 /// Array allocations are still pretty slow, if the array gets big enough it can get slower than \c new.
33 /// Node allocations are still fast, unless there is deallocation in random order.
34 /// \note Use this tag type only if you really need to have a memory pool!
35 /// \ingroup allocator
36 struct array_pool : WPI_EBO(std::true_type)
37 {
38 using type = detail::array_free_memory_list;
39 };

41 /// Tag type defining a memory pool optimized for small nodes.
42 /// The free list is intrusive and thus requires that each node has at least the size of a pointer.
43 /// This tag type does not have this requirement and thus allows zero-memory-overhead allocations of small nodes.
44 /// It is a little bit slower than \ref node_pool and does not support arrays.
45 /// \ingroup allocator
46 struct small_node_pool : WPI_EBO(std::false_type)
47 {
48 using type = detail::small_free_memory_list;
