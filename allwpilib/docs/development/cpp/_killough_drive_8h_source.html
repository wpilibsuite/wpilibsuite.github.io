31/**
32 * A class for driving Killough drive platforms.
33 *
34 * Killough drives are triangular with one omni wheel on each corner.
35 *
36 * Drive base diagram:
37 * <pre>
38 * /_____\
39 * / \ / \
40 * \ /
41 * ---
42 * </pre>
43 *
44 * Each Drive() function provides different inverse kinematic relations for a
45 * Killough drive. The default wheel vectors are parallel to their respective
46 * opposite sides, but can be overridden. See the constructor for more
47 * information.
48 *
49 * This library uses the NED axes convention (North-East-Down as external
50 * reference in the world frame):
51 * http://www.nuclearprojects.com/ins/images/axis_big.png.
52 *
53 * The positive X axis points ahead, the positive Y axis points right, and the
54 * and the positive Z axis points down. Rotations follow the right-hand rule, so
55 * clockwise rotation around the Z axis is positive.
56 *
57 * MotorSafety is enabled by default. The DriveCartesian or DrivePolar
58 * methods should be called periodically to avoid Motor Safety timeouts.
59 */
60class KilloughDrive : public RobotDriveBase,
61 public wpi::Sendable,
62 public wpi::SendableHelper<KilloughDrive> {
63 public:
64 static constexpr double kDefaultLeftMotorAngle = 60.0;
65 static constexpr double kDefaultRightMotorAngle = 120.0;
66 static constexpr double kDefaultBackMotorAngle = 270.0;
67
68 /**
69 * Wheel speeds for a Killough drive.
70 *
71 * Uses normalized voltage [-1.0..1.0].
72 */
73 struct WheelSpeeds {
74 double left = 0.0;
75 double right = 0.0;
76 double back = 0.0;
77 };

79 /**
80 * Construct a Killough drive with the given motors and default motor angles.
81 *
82 * The default motor angles make the wheels on each corner parallel to their
83 * respective opposite sides.
84 *
85 * If a motor needs to be inverted, do so before passing it in.
86 *
87 * @param leftMotor The motor on the left corner.
88 * @param rightMotor The motor on the right corner.
89 * @param backMotor The motor on the back corner.
90 */
91 KilloughDrive(SpeedController& leftMotor, SpeedController& rightMotor,
92 SpeedController& backMotor);
93
94 /**
95 * Construct a Killough drive with the given motors.
96 *
97 * Angles are measured in degrees clockwise from the positive X axis.
98 *
99 * @param leftMotor The motor on the left corner.
100 * @param rightMotor The motor on the right corner.
101 * @param backMotor The motor on the back corner.
102 * @param leftMotorAngle The angle of the left wheel's forward direction of
103 * travel.
104 * @param rightMotorAngle The angle of the right wheel's forward direction of
105 * travel.
106 * @param backMotorAngle The angle of the back wheel's forward direction of
107 * travel.
108 */
109 KilloughDrive(SpeedController& leftMotor, SpeedController& rightMotor,
110 SpeedController& backMotor, double leftMotorAngle,
111 double rightMotorAngle, double backMotorAngle);
112
113 ~KilloughDrive() override = default;
114
115 KilloughDrive(KilloughDrive&&) = default;
116 KilloughDrive& operator=(KilloughDrive&&) = default;
117
118 /**
119 * Drive method for Killough platform.
120 *
121 * Angles are measured clockwise from the positive X axis. The robot's speed
122 * is independent from its angle or rotation rate.
123 *
124 * @param ySpeed The robot's speed along the Y axis [-1.0..1.0]. Right is
125 * positive.
126 * @param xSpeed The robot's speed along the X axis [-1.0..1.0]. Forward is
127 * positive.
128 * @param zRotation The robot's rotation rate around the Z axis [-1.0..1.0].
129 * Clockwise is positive.
130 * @param gyroAngle The current angle reading from the gyro in degrees around
131 * the Z axis. Use this to implement field-oriented controls.
132 */
133 void DriveCartesian(double ySpeed, double xSpeed, double zRotation,
134 double gyroAngle = 0.0);
135
136 /**
137 * Drive method for Killough platform.
138 *
139 * Angles are measured clockwise from the positive X axis. The robot's speed
140 * is independent from its angle or rotation rate.
141 *
142 * @param magnitude The robot's speed at a given angle [-1.0..1.0]. Forward is
143 * positive.
144 * @param angle The angle around the Z axis at which the robot drives in
145 * degrees [-180..180].
146 * @param zRotation The robot's rotation rate around the Z axis [-1.0..1.0].
147 * Clockwise is positive.
148 */
149 void DrivePolar(double magnitude, double angle, double zRotation);
150
151 /**
152 * Cartesian inverse kinematics for Killough platform.
153 *
154 * Angles are measured clockwise from the positive X axis. The robot's speed
155 * is independent from its angle or rotation rate.
156 *
157 * @param ySpeed The robot's speed along the Y axis [-1.0..1.0]. Right is
158 * positive.
159 * @param xSpeed The robot's speed along the X axis [-1.0..1.0]. Forward is
160 * positive.
161 * @param zRotation The robot's rotation rate around the Z axis [-1.0..1.0].
162 * Clockwise is positive.
163 * @param gyroAngle The current angle reading from the gyro in degrees around
164 * the Z axis. Use this to implement field-oriented controls.
165 * @return Wheel speeds [-1.0..1.0].
166 */
167 WheelSpeeds DriveCartesianIK(double ySpeed, double xSpeed, double zRotation,
168 double gyroAngle = 0.0);
169
170 void StopMotor() override;
171 std::string GetDescription() const override;
172
173 void InitSendable(wpi::SendableBuilder& builder) override;
174
175 private:
176 SpeedController* m_leftMotor;
177 SpeedController* m_rightMotor;
178 SpeedController* m_backMotor;
179
180 Vector2d m_leftVec;
181 Vector2d m_rightVec;
182 Vector2d m_backVec;
183
184 bool reported = false;
185};
