11namespace frc {

13class DriverStation;
14
15/**
16 * Handle input from standard HID devices connected to the Driver Station.
17 *
18 * <p>This class handles standard input that comes from the Driver Station. Each
19 * time a value is requested the most recent value is returned. There is a
20 * single class instance for each device and the mapping of ports to hardware
21 * buttons depends on the code in the Driver Station.
22 */
23class GenericHID {
24 public:
25 enum RumbleType { kLeftRumble, kRightRumble };
26
27 enum HIDType {
28 kUnknown = -1,
29 kXInputUnknown = 0,
30 kXInputGamepad = 1,
31 kXInputWheel = 2,
32 kXInputArcadeStick = 3,
33 kXInputFlightStick = 4,
34 kXInputDancePad = 5,
35 kXInputGuitar = 6,
36 kXInputGuitar2 = 7,
37 kXInputDrumKit = 8,
38 kXInputGuitar3 = 11,
39 kXInputArcadePad = 19,
40 kHIDJoystick = 20,
41 kHIDGamepad = 21,
42 kHIDDriving = 22,
43 kHIDFlight = 23,
44 kHID1stPerson = 24
45 };
46
47 explicit GenericHID(int port);
48 virtual ~GenericHID() = default;
49
50 GenericHID(GenericHID&&) = default;
51 GenericHID& operator=(GenericHID&&) = default;
52
53 /**
54 * Get the button value (starting at button 1).
55 *
56 * The buttons are returned in a single 16 bit value with one bit representing
57 * the state of each button. The appropriate button is returned as a boolean
58 * value.
59 *
60 * This method returns true if the button is being held down at the time
61 * that this method is being called.
62 *
63 * @param button The button number to be read (starting at 1)
64 * @return The state of the button.
65 */
66 bool GetRawButton(int button) const;
67
68 /**
69 * Whether the button was pressed since the last check. %Button indexes begin
70 * at 1.
71 *
72 * This method returns true if the button went from not pressed to held down
73 * since the last time this method was called. This is useful if you only
74 * want to call a function once when you press the button.
75 *
76 * @param button The button index, beginning at 1.
77 * @return Whether the button was pressed since the last check.
78 */
79 bool GetRawButtonPressed(int button);
80
81 /**
82 * Whether the button was released since the last check. %Button indexes begin
83 * at 1.
84 *
85 * This method returns true if the button went from held down to not pressed
86 * since the last time this method was called. This is useful if you only
87 * want to call a function once when you release the button.
88 *
89 * @param button The button index, beginning at 1.
90 * @return Whether the button was released since the last check.
91 */
92 bool GetRawButtonReleased(int button);
93
94 /**
95 * Get the value of the axis.
96 *
97 * @param axis The axis to read, starting at 0.
98 * @return The value of the axis.
99 */
100 double GetRawAxis(int axis) const;
101
102 /**
103 * Get the angle in degrees of a POV on the HID.
104 *
105 * The POV angles start at 0 in the up direction, and increase clockwise
106 * (e.g. right is 90, upper-left is 315).
107 *
108 * @param pov The index of the POV to read (starting at 0)
109 * @return the angle of the POV in degrees, or -1 if the POV is not pressed.
110 */
111 int GetPOV(int pov = 0) const;
112
113 /**
114 * Get the number of axes for the HID.
115 *
116 * @return the number of axis for the current HID
117 */
118 int GetAxisCount() const;
119
120 /**
121 * Get the number of POVs for the HID.
122 *
123 * @return the number of POVs for the current HID
124 */
125 int GetPOVCount() const;
126
127 /**
128 * Get the number of buttons for the HID.
129 *
130 * @return the number of buttons on the current HID
131 */
132 int GetButtonCount() const;
133
134 /**
135 * Get if the HID is connected.
136 *
137 * @return true if the HID is connected
138 */
139 bool IsConnected() const;
140
141 /**
142 * Get the type of the HID.
143 *
144 * @return the type of the HID.
145 */
146 GenericHID::HIDType GetType() const;
147
148 /**
149 * Get the name of the HID.
150 *
151 * @return the name of the HID.
152 */
153 std::string GetName() const;
154
155 /**
156 * Get the axis type of a joystick axis.
157 *
158 * @return the axis type of a joystick axis.
159 */
160 int GetAxisType(int axis) const;
161
162 /**
163 * Get the port number of the HID.
164 *
165 * @return The port number of the HID.
166 */
167 int GetPort() const;
168
169 /**
170 * Set a single HID output value for the HID.
171 *
172 * @param outputNumber The index of the output to set (1-32)
173 * @param value The value to set the output to
174 */
175 void SetOutput(int outputNumber, bool value);
176
177 /**
178 * Set all output values for the HID.
179 *
180 * @param value The 32 bit output value (1 bit for each output)
181 */
182 void SetOutputs(int value);
183
184 /**
185 * Set the rumble output for the HID.
186 *
187 * The DS currently supports 2 rumble values, left rumble and right rumble.
188 *
189 * @param type Which rumble value to set
190 * @param value The normalized value (0 to 1) to set the rumble to
191 */
192 void SetRumble(RumbleType type, double value);
193
194 private:
195 int m_port;
196 int m_outputs = 0;
197 uint16_t m_leftRumble = 0;
198 uint16_t m_rightRumble = 0;
199};
