21/**
22 * Class for operating a compressor connected to a pneumatics module.
23 *
24 * The module will automatically run in closed loop mode by default whenever a
25 * Solenoid object is created. For most cases, a Compressor object does not need
26 * to be instantiated or used in a robot program. This class is only required in
27 * cases where the robot program needs a more detailed status of the compressor
28 * or to enable/disable closed loop control.
29 *
30 * Note: you cannot operate the compressor directly from this class as doing so
31 * would circumvent the safety provided by using the pressure switch and closed
32 * loop control. You can only turn off closed loop control, thereby stopping
33 * the compressor from operating.
34 */
35class Compressor : public wpi::Sendable,
36 public wpi::SendableHelper<Compressor> {
37 public:
38 /**
39 * Constructs a compressor for a specified module and type.
40 *
41 * @param module The module ID to use.
42 * @param moduleType The module type to use.
43 */
44 Compressor(int module, PneumaticsModuleType moduleType);
45
46 /**
47 * Constructs a compressor for a default module and specified type.
48 *
49 * @param moduleType The module type to use.
50 */
51 explicit Compressor(PneumaticsModuleType moduleType);
52
53 ~Compressor() override;
54
55 Compressor(const Compressor&) = delete;
56 Compressor& operator=(const Compressor&) = delete;
57
58 Compressor(Compressor&&) = default;
59 Compressor& operator=(Compressor&&) = default;
60
61 /**
62 * Starts closed-loop control. Note that closed loop control is enabled by
63 * default.
64 *
65 * @deprecated Use EnableDigital() instead.
66 */
67 WPI_DEPRECATED("Use EnableDigital() instead")
68 void Start();
69
70 /**
71 * Stops closed-loop control. Note that closed loop control is enabled by
72 * default.
73 *
74 * @deprecated Use Disable() instead.
75 */
76 WPI_DEPRECATED("Use Disable() instead")
77 void Stop();
78
79 /**
80 * Check if compressor output is active.
81 * To (re)enable the compressor use EnableDigital() or EnableAnalog(...).
82 *
83 * @return true if the compressor is on.
84 * @deprecated To avoid confusion in thinking this (re)enables the compressor
85 * use IsEnabled().
86 */
87 WPI_DEPRECATED(
88 "To avoid confusion in thinking this (re)enables the compressor use "
89 "IsEnabled()")
90 bool Enabled() const;
91
92 /**
93 * Returns whether the compressor is active or not.
94 *
95 * @return true if the compressor is on - otherwise false.
96 */
97 bool IsEnabled() const;
98
99 /**
100 * Check if the pressure switch is triggered.
101 *
102 * @return true if pressure is low
103 */
104 bool GetPressureSwitchValue() const;
105
106 /**
107 * Query how much current the compressor is drawing.
108 *
109 * @return The current through the compressor, in amps
110 */
111 units::ampere_t GetCurrent() const;
112
113 /**
114 * Query the analog input voltage (on channel 0) (if supported).
115 *
116 * @return The analog input voltage, in volts
117 */
118 units::volt_t GetAnalogVoltage() const;
119
120 /**
121 * Query the analog sensor pressure (on channel 0) (if supported). Note this
122 * is only for use with the REV Analog Pressure Sensor.
123 *
124 * @return The analog sensor pressure, in PSI
125 */
126 units::pounds_per_square_inch_t GetPressure() const;
127
128 /**
129 * Disable the compressor.
130 */
131 void Disable();
132
133 /**
134 * Enable compressor closed loop control using digital input.
135 */
136 void EnableDigital();
137
138 /**
139 * Enable compressor closed loop control using analog input. Note this is only
140 * for use with the REV Analog Pressure Sensor.
141 *
142 * <p>On CTRE PCM, this will enable digital control.
143 *
144 * @param minPressure The minimum pressure in PSI to enable compressor
145 * @param maxPressure The maximum pressure in PSI to disable compressor
146 */
147 void EnableAnalog(units::pounds_per_square_inch_t minPressure,
148 units::pounds_per_square_inch_t maxPressure);
149
150 /**
151 * Enable compressor closed loop control using hybrid input. Note this is only
152 * for use with the REV Analog Pressure Sensor.
153 *
154 * On CTRE PCM, this will enable digital control.
155 *
156 * @param minPressure The minimum pressure in PSI to enable compressor
157 * @param maxPressure The maximum pressure in PSI to disable compressor
158 */
159 void EnableHybrid(units::pounds_per_square_inch_t minPressure,
160 units::pounds_per_square_inch_t maxPressure);
161
162 CompressorConfigType GetConfigType() const;
163
164 void InitSendable(wpi::SendableBuilder& builder) override;
165
166 private:
167 std::shared_ptr<PneumaticsBase> m_module;
168};
169
170} // namespace frc
