36 /// A stateful \concept{concept_rawallocator,RawAllocator} that manages \concept{concept_node,nodes} of fixed size.
37 /// It uses a \ref memory_arena with a given \c BlockOrRawAllocator defaulting to \ref growing_block_allocator,
38 /// subdivides them in small nodes of given size and puts them onto a free list.
39 /// Allocation and deallocation simply remove or add nodes from this list and are thus fast.
40 /// The way the list is maintained can be controlled via the \c PoolType
41 /// which is either \ref node_pool, \ref array_pool or \ref small_node_pool.<br>
42 /// This kind of allocator is ideal for fixed size allocations and deallocations in any order,
43 /// for example in a node based container like \c std::list.
44 /// It is not so good for different allocation sizes and has some drawbacks for arrays
45 /// as described in \ref memory_pool_type.hpp.
46 /// \ingroup allocator
47 template <typename PoolType = node_pool, class BlockOrRawAllocator = default_allocator>
48 class memory_pool
49 : WPI_EBO(detail::default_leak_checker<detail::memory_pool_leak_handler>)
50 {
51 using free_list = typename PoolType::type;
52 using leak_checker = detail::default_leak_checker<detail::memory_pool_leak_handler>;
53
54 public:
55 using allocator_type = make_block_allocator_t<BlockOrRawAllocator>;
56 using pool_type = PoolType;
57
58 static constexpr std::size_t min_node_size =
59 WPI_IMPL_DEFINED(free_list::min_element_size);
60
61 /// \returns The minimum block size required for certain number of \concept{concept_node,node}.
62 /// \requires \c node_size must be a valid \concept{concept_node,node size}
63 /// and \c number_of_nodes must be a non-zero value.
64 /// \note MSVC's implementation of \c std::list for example is never empty and always allocates proxy nodes.
65 /// To get enough memory for \c N elements of a list, \c number_of_nodes needs to include the proxy count in addition to \c N.
66 static constexpr std::size_t min_block_size(std::size_t node_size,
67 std::size_t number_of_nodes) noexcept
68 {
69 return detail::memory_block_stack::implementation_offset()
70 + free_list::min_block_size(node_size, number_of_nodes);
71 }
72
73 /// \effects Creates it by specifying the size each \concept{concept_node,node} will have,
74 /// the initial block size for the arena and other constructor arguments for the \concept{concept_blockallocator,BlockAllocator}.
75 /// If the \c node_size is less than the \c min_node_size, the \c min_node_size will be the actual node size.
76 /// It will allocate an initial memory block with given size from the \concept{concept_blockallocator,BlockAllocator}
77 /// and puts it onto the free list.
78 /// \requires \c node_size must be a valid \concept{concept_node,node size}
79 /// and \c block_size must be at least \c min_block_size(node_size, 1).
80 template <typename... Args>
81 memory_pool(std::size_t node_size, std::size_t block_size, Args&&... args)
82 : arena_(block_size, detail::forward<Args>(args)...), free_list_(node_size)
83 {
84 allocate_block();
85 }
86
87 /// \effects Destroys the \ref memory_pool by returning all memory blocks,
88 /// regardless of properly deallocated back to the \concept{concept_blockallocator,BlockAllocator}.
89 ~memory_pool() noexcept {}
90
91 /// @{
92 /// \effects Moving a \ref memory_pool object transfers ownership over the free list,
93 /// i.e. the moved from pool is completely empty and the new one has all its memory.
94 /// That means that it is not allowed to call \ref deallocate_node() on a moved-from allocator
95 /// even when passing it memory that was previously allocated by this object.
96 memory_pool(memory_pool&& other) noexcept
97 : leak_checker(detail::move(other)),
98 arena_(detail::move(other.arena_)),
99 free_list_(detail::move(other.free_list_))
100 {
101 }
102
103 memory_pool& operator=(memory_pool&& other) noexcept
104 {
105 leak_checker::operator=(detail::move(other));
106 arena_ = detail::move(other.arena_);
107 free_list_ = detail::move(other.free_list_);
108 return *this;
109 }
110 /// @}
111
112 /// \effects Allocates a single \concept{concept_node,node} by removing it from the free list.
113 /// If the free list is empty, a new memory block will be allocated from the arena and put onto it.
114 /// The new block size will be \ref next_capacity() big.
115 /// \returns A node of size \ref node_size() suitable aligned,
116 /// i.e. suitable for any type where <tt>sizeof(T) < node_size()</tt>.
117 /// \throws Anything thrown by the used \concept{concept_blockallocator,BlockAllocator}'s allocation function if a growth is needed.
118 void* allocate_node()
119 {
120 if (free_list_.empty())
121 allocate_block();
122 WPI_MEMORY_ASSERT(!free_list_.empty());
123 return free_list_.allocate();
124 }
125
126 /// \effects Allocates a single \concept{concept_node,node} similar to \ref allocate_node().
127 /// But if the free list is empty, a new block will *not* be allocated.
128 /// \returns A suitable aligned node of size \ref node_size() or `nullptr`.
129 void* try_allocate_node() noexcept
130 {
131 return free_list_.empty() ? nullptr : free_list_.allocate();
132 }
133
134 /// \effects Allocates an \concept{concept_array,array} of nodes by searching for \c n continuous nodes on the list and removing them.
135 /// Depending on the \c PoolType this can be a slow operation or not allowed at all.
136 /// This can sometimes lead to a growth, even if technically there is enough continuous memory on the free list.
137 /// \returns An array of \c n nodes of size \ref node_size() suitable aligned.
138 /// \throws Anything thrown by the used \concept{concept_blockallocator,BlockAllocator}'s allocation function if a growth is needed,
139 /// or \ref bad_array_size if <tt>n * node_size()</tt> is too big.
140 /// \requires \c n must be valid \concept{concept_array,array count}.
141 void* allocate_array(std::size_t n)
142 {
143 detail::check_allocation_size<bad_array_size>(
144 n * node_size(), [&] { return pool_type::value ? next_capacity() : 0; },
145 info());
146 return allocate_array(n, node_size());
147 }
148
149 /// \effects Allocates an \concept{concept_array,array} of nodes similar to \ref allocate_array().
150 /// But it will never allocate a new memory block.
151 /// \returns An array of \c n nodes of size \ref node_size() suitable aligned
152 /// or `nullptr`.
153 void* try_allocate_array(std::size_t n) noexcept
154 {
155 return try_allocate_array(n, node_size());
156 }
157
158 /// \effects Deallocates a single \concept{concept_node,node} by putting it back onto the free list.
159 /// \requires \c ptr must be a result from a previous call to \ref allocate_node() on the same free list,
160 /// i.e. either this allocator object or a new object created by moving this to it.
161 void deallocate_node(void* ptr) noexcept
162 {
163 free_list_.deallocate(ptr);
164 }
165
166 /// \effects Deallocates a single \concept{concept_node,node} but it does not be a result of a previous call to \ref allocate_node().
167 /// \returns `true` if the node could be deallocated, `false` otherwise.
168 /// \note Some free list implementations can deallocate any memory,
169 /// doesn't matter where it is coming from.
170 bool try_deallocate_node(void* ptr) noexcept
171 {
172 if (!arena_.owns(ptr))
173 return false;
174 free_list_.deallocate(ptr);
175 return true;
176 }
177
178 /// \effects Deallocates an \concept{concept_array,array} by putting it back onto the free list.
179 /// \requires \c ptr must be a result from a previous call to \ref allocate_array() with the same \c n on the same free list,
180 /// i.e. either this allocator object or a new object created by moving this to it.
181 void deallocate_array(void* ptr, std::size_t n) noexcept
182 {
183 WPI_MEMORY_ASSERT_MSG(pool_type::value, "does not support array allocations");
184 free_list_.deallocate(ptr, n * node_size());
185 }
186
187 /// \effects Deallocates an \concept{concept_array,array} but it does not be a result of a previous call to \ref allocate_array().
188 /// \returns `true` if the node could be deallocated, `false` otherwise.
189 /// \note Some free list implementations can deallocate any memory,
190 /// doesn't matter where it is coming from.
191 bool try_deallocate_array(void* ptr, std::size_t n) noexcept
192 {
193 return try_deallocate_array(ptr, n, node_size());
194 }
195
196 /// \returns The size of each \concept{concept_node,node} in the pool,
197 /// this is either the same value as in the constructor or \c min_node_size if the value was too small.
198 std::size_t node_size() const noexcept
199 {
200 return free_list_.node_size();
201 }
202
203 /// \effects Returns the total amount of bytes remaining on the free list.
204 /// Divide it by \ref node_size() to get the number of nodes that can be allocated without growing the arena.
205 /// \note Array allocations may lead to a growth even if the capacity_left left is big enough.
206 std::size_t capacity_left() const noexcept
207 {
208 return free_list_.capacity() * node_size();
209 }
210
211 /// \returns The size of the next memory block after the free list gets empty and the arena grows.
212 /// \ref capacity_left() will increase by this amount.
213 /// \note Due to fence memory in debug mode this cannot be just divided by the \ref node_size() to get the number of nodes.
214 std::size_t next_capacity() const noexcept
215 {
216 return free_list_.usable_size(arena_.next_block_size());
217 }
218
219 /// \returns A reference to the \concept{concept_blockallocator,BlockAllocator} used for managing the arena.
220 /// \requires It is undefined behavior to move this allocator out into another object.
221 allocator_type& get_allocator() noexcept
222 {
223 return arena_.get_allocator();
224 }
225
226 private:
227 allocator_info info() const noexcept
228 {
229 return {WPI_MEMORY_LOG_PREFIX "::memory_pool", this};
230 }
231
232 void allocate_block()
233 {
234 auto mem = arena_.allocate_block();
235 free_list_.insert(static_cast<char*>(mem.memory), mem.size);
236 }
237
238 void* allocate_array(std::size_t n, std::size_t node_size)
239 {
240 auto mem = free_list_.empty() ? nullptr : free_list_.allocate(n * node_size);
241 if (!mem)
242 {
243 allocate_block();
244 mem = free_list_.allocate(n * node_size);
245 if (!mem)
246 WPI_THROW(bad_array_size(info(), n * node_size, capacity_left()));
247 }
248 return mem;
249 }
250
251 void* try_allocate_array(std::size_t n, std::size_t node_size) noexcept
252 {
253 return !pool_type::value || free_list_.empty() ? nullptr :
254 free_list_.allocate(n * node_size);
255 }
256
257 bool try_deallocate_array(void* ptr, std::size_t n, std::size_t node_size) noexcept
258 {
259 if (!pool_type::value || !arena_.owns(ptr))
260 return false;
261 free_list_.deallocate(ptr, n * node_size);
262 return true;
263 }

281 /// Specialization of the \ref allocator_traits for \ref memory_pool classes.
282 /// \note It is not allowed to mix calls through the specialization and through the member functions,
283 /// i.e. \ref memory_pool::allocate_node() and this \c allocate_node().
284 /// \ingroup allocator
285 template <typename PoolType, class ImplRawAllocator>
286 class allocator_traits<memory_pool<PoolType, ImplRawAllocator>>
287 {
288 public:
289 using allocator_type = memory_pool<PoolType, ImplRawAllocator>;
290 using is_stateful = std::true_type;
291
292 /// \returns The result of \ref memory_pool::allocate_node().
293 /// \throws Anything thrown by the pool allocation function
294 /// or a \ref bad_allocation_size exception.
295 static void* allocate_node(allocator_type& state, std::size_t size,
296 std::size_t alignment)
297 {
298 detail::check_allocation_size<bad_node_size>(size, max_node_size(state),
299 state.info());
300 detail::check_allocation_size<bad_alignment>(
301 alignment, [&] { return max_alignment(state); }, state.info());
302 auto mem = state.allocate_node();
303 state.on_allocate(size);
304 return mem;
305 }
306
307 /// \effects Forwards to \ref memory_pool::allocate_array()
308 /// with the number of nodes adjusted to be the minimum,
309 /// i.e. when the \c size is less than the \ref memory_pool::node_size().
310 /// \returns A \concept{concept_array,array} with specified properties.
311 /// \requires The \ref memory_pool has to support array allocations.
312 /// \throws Anything thrown by the pool allocation function.
313 static void* allocate_array(allocator_type& state, std::size_t count, std::size_t size,
314 std::size_t alignment)
315 {
316 detail::check_allocation_size<bad_node_size>(size, max_node_size(state),
317 state.info());
318 detail::check_allocation_size<bad_alignment>(
319 alignment, [&] { return max_alignment(state); }, state.info());
320 detail::check_allocation_size<bad_array_size>(count * size, max_array_size(state),
321 state.info());
322 auto mem = state.allocate_array(count, size);
323 state.on_allocate(count * size);
324 return mem;
325 }
326
327 /// \effects Just forwards to \ref memory_pool::deallocate_node().
328 static void deallocate_node(allocator_type& state, void* node, std::size_t size,
329 std::size_t) noexcept
330 {
331 state.deallocate_node(node);
332 state.on_deallocate(size);
333 }
334
335 /// \effects Forwards to \ref memory_pool::deallocate_array() with the same size adjustment.
336 static void deallocate_array(allocator_type& state, void* array, std::size_t count,
337 std::size_t size, std::size_t) noexcept
338 {
339 state.free_list_.deallocate(array, count * size);
340 state.on_deallocate(count * size);
341 }
342
343 /// \returns The maximum size of each node which is \ref memory_pool::node_size().
344 static std::size_t max_node_size(const allocator_type& state) noexcept
345 {
346 return state.node_size();
347 }
348
349 /// \returns An upper bound on the maximum array size which is \ref memory_pool::next_capacity().
350 static std::size_t max_array_size(const allocator_type& state) noexcept
351 {
352 return state.next_capacity();
353 }
354
355 /// \returns The maximum alignment which is the next bigger power of two if less than \c alignof(std::max_align_t)
356 /// or the maximum alignment itself otherwise.
357 static std::size_t max_alignment(const allocator_type& state) noexcept
358 {
359 return state.free_list_.alignment();
360 }
361 };

363 /// Specialization of the \ref composable_allocator_traits for \ref memory_pool classes.
364 /// \ingroup allocator
365 template <typename PoolType, class BlockOrRawAllocator>
366 class composable_allocator_traits<memory_pool<PoolType, BlockOrRawAllocator>>
367 {
368 using traits = allocator_traits<memory_pool<PoolType, BlockOrRawAllocator>>;
369
370 public:
371 using allocator_type = memory_pool<PoolType, BlockOrRawAllocator>;
372
373 /// \returns The result of \ref memory_pool::try_allocate_node()
374 /// or `nullptr` if the allocation size was too big.
375 static void* try_allocate_node(allocator_type& state, std::size_t size,
376 std::size_t alignment) noexcept
377 {
378 if (size > traits::max_node_size(state) || alignment > traits::max_alignment(state))
379 return nullptr;
380 return state.try_allocate_node();
381 }
382
383 /// \effects Forwards to \ref memory_pool::try_allocate_array()
384 /// with the number of nodes adjusted to be the minimum,
385 /// if the \c size is less than the \ref memory_pool::node_size().
386 /// \returns A \concept{concept_array,array} with specified properties
387 /// or `nullptr` if it was unable to allocate.
388 static void* try_allocate_array(allocator_type& state, std::size_t count,
389 std::size_t size, std::size_t alignment) noexcept
390 {
391 if (size > traits::max_node_size(state)
392 || count * size > traits::max_array_size(state)
393 || alignment > traits::max_alignment(state))
394 return nullptr;
395 return state.try_allocate_array(count, size);
396 }
397
398 /// \effects Just forwards to \ref memory_pool::try_deallocate_node().
399 /// \returns Whether the deallocation was successful.
400 static bool try_deallocate_node(allocator_type& state, void* node, std::size_t size,
401 std::size_t alignment) noexcept
402 {
403 if (size > traits::max_node_size(state) || alignment > traits::max_alignment(state))
404 return false;
405 return state.try_deallocate_node(node);
406 }
407
408 /// \effects Forwards to \ref memory_pool::deallocate_array() with the same size adjustment.
409 /// \returns Whether the deallocation was successful.
410 static bool try_deallocate_array(allocator_type& state, void* array, std::size_t count,
411 std::size_t size, std::size_t alignment) noexcept
412 {
413 if (size > traits::max_node_size(state)
414 || count * size > traits::max_array_size(state)
415 || alignment > traits::max_alignment(state))
416 return false;
417 return state.try_deallocate_array(array, count, size);
418 }
419 };
