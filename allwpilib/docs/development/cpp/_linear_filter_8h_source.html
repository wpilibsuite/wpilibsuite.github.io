24/**
25 * This class implements a linear, digital filter. All types of FIR and IIR
26 * filters are supported. Static factory methods are provided to create commonly
27 * used types of filters.
28 *
29 * Filters are of the form:<br>
30 * y[n] = (b0 x[n] + b1 x[n-1] + … + bP x[n-P]) -
31 * (a0 y[n-1] + a2 y[n-2] + … + aQ y[n-Q])
32 *
33 * Where:<br>
34 * y[n] is the output at time "n"<br>
35 * x[n] is the input at time "n"<br>
36 * y[n-1] is the output from the LAST time step ("n-1")<br>
37 * x[n-1] is the input from the LAST time step ("n-1")<br>
38 * b0 … bP are the "feedforward" (FIR) gains<br>
39 * a0 … aQ are the "feedback" (IIR) gains<br>
40 * IMPORTANT! Note the "-" sign in front of the feedback term! This is a common
41 * convention in signal processing.
42 *
43 * What can linear filters do? Basically, they can filter, or diminish, the
44 * effects of undesirable input frequencies. High frequencies, or rapid changes,
45 * can be indicative of sensor noise or be otherwise undesirable. A "low pass"
46 * filter smooths out the signal, reducing the impact of these high frequency
47 * components. Likewise, a "high pass" filter gets rid of slow-moving signal
48 * components, letting you detect large changes more easily.
49 *
50 * Example FRC applications of filters:
51 * - Getting rid of noise from an analog sensor input (note: the roboRIO's FPGA
52 * can do this faster in hardware)
53 * - Smoothing out joystick input to prevent the wheels from slipping or the
54 * robot from tipping
55 * - Smoothing motor commands so that unnecessary strain isn't put on
56 * electrical or mechanical components
57 * - If you use clever gains, you can make a PID controller out of this class!
58 *
59 * For more on filters, we highly recommend the following articles:<br>
60 * https://en.wikipedia.org/wiki/Linear_filter<br>
61 * https://en.wikipedia.org/wiki/Iir_filter<br>
62 * https://en.wikipedia.org/wiki/Fir_filter<br>
63 *
64 * Note 1: Calculate() should be called by the user on a known, regular period.
65 * You can use a Notifier for this or do it "inline" with code in a
66 * periodic function.
67 *
68 * Note 2: For ALL filters, gains are necessarily a function of frequency. If
69 * you make a filter that works well for you at, say, 100Hz, you will most
70 * definitely need to adjust the gains if you then want to run it at 200Hz!
71 * Combining this with Note 1 - the impetus is on YOU as a developer to make
72 * sure Calculate() gets called at the desired, constant frequency!
73 */
74template <class T>
75class LinearFilter {
76 public:
77 /**
78 * Create a linear FIR or IIR filter.
79 *
80 * @param ffGains The "feedforward" or FIR gains.
81 * @param fbGains The "feedback" or IIR gains.
82 */
83 LinearFilter(wpi::span<const double> ffGains, wpi::span<const double> fbGains)
84 : m_inputs(ffGains.size()),
85 m_outputs(fbGains.size()),
86 m_inputGains(ffGains.begin(), ffGains.end()),
87 m_outputGains(fbGains.begin(), fbGains.end()) {
88 for (size_t i = 0; i < ffGains.size(); ++i) {
89 m_inputs.emplace_front(0.0);
90 }
91 for (size_t i = 0; i < fbGains.size(); ++i) {
92 m_outputs.emplace_front(0.0);
93 }
94
95 static int instances = 0;
96 instances++;
97 wpi::math::MathSharedStore::ReportUsage(
98 wpi::math::MathUsageId::kFilter_Linear, 1);
99 }
100
101 /**
102 * Create a linear FIR or IIR filter.
103 *
104 * @param ffGains The "feedforward" or FIR gains.
105 * @param fbGains The "feedback" or IIR gains.
106 */
107 LinearFilter(std::initializer_list<double> ffGains,
108 std::initializer_list<double> fbGains)
109 : LinearFilter({ffGains.begin(), ffGains.end()},
110 {fbGains.begin(), fbGains.end()}) {}
111
112 // Static methods to create commonly used filters
113 /**
114 * Creates a one-pole IIR low-pass filter of the form:<br>
115 * y[n] = (1 - gain) x[n] + gain y[n-1]<br>
116 * where gain = e<sup>-dt / T</sup>, T is the time constant in seconds
117 *
118 * Note: T = 1 / (2 pi f) where f is the cutoff frequency in Hz, the frequency
119 * above which the input starts to attenuate.
120 *
121 * This filter is stable for time constants greater than zero.
122 *
123 * @param timeConstant The discrete-time time constant in seconds.
124 * @param period The period in seconds between samples taken by the
125 * user.
126 */
127 static LinearFilter<T> SinglePoleIIR(double timeConstant,
128 units::second_t period) {
129 double gain = std::exp(-period.value() / timeConstant);
130 return LinearFilter({1.0 - gain}, {-gain});
131 }
132
133 /**
134 * Creates a first-order high-pass filter of the form:<br>
135 * y[n] = gain x[n] + (-gain) x[n-1] + gain y[n-1]<br>
136 * where gain = e<sup>-dt / T</sup>, T is the time constant in seconds
137 *
138 * Note: T = 1 / (2 pi f) where f is the cutoff frequency in Hz, the frequency
139 * below which the input starts to attenuate.
140 *
141 * This filter is stable for time constants greater than zero.
142 *
143 * @param timeConstant The discrete-time time constant in seconds.
144 * @param period The period in seconds between samples taken by the
145 * user.
146 */
147 static LinearFilter<T> HighPass(double timeConstant, units::second_t period) {
148 double gain = std::exp(-period.value() / timeConstant);
149 return LinearFilter({gain, -gain}, {-gain});
150 }
151
152 /**
153 * Creates a K-tap FIR moving average filter of the form:<br>
154 * y[n] = 1/k (x[k] + x[k-1] + … + x[0])
155 *
156 * This filter is always stable.
157 *
158 * @param taps The number of samples to average over. Higher = smoother but
159 * slower
160 */
161 static LinearFilter<T> MovingAverage(int taps) {
162 if (taps <= 0) {
163 throw std::runtime_error("Number of taps must be greater than zero.");
164 }
165
166 std::vector<double> gains(taps, 1.0 / taps);
167 return LinearFilter(gains, {});
168 }
169
170 /**
171 * Creates a finite difference filter that computes the nth derivative of the
172 * input given the specified stencil points.
173 *
174 * Stencil points are the indices of the samples to use in the finite
175 * difference. 0 is the current sample, -1 is the previous sample, -2 is the
176 * sample before that, etc. Don't use positive stencil points (samples from
177 * the future) if the LinearFilter will be used for stream-based online
178 * filtering.
179 *
180 * @tparam Derivative The order of the derivative to compute.
181 * @tparam Samples The number of samples to use to compute the given
182 * derivative. This must be one more than the order of
183 * derivative or higher.
184 * @param stencil List of stencil points.
185 * @param period The period in seconds between samples taken by the user.
186 */
187 template <int Derivative, int Samples>
188 static LinearFilter<T> FiniteDifference(
189 const wpi::array<int, Samples> stencil, units::second_t period) {
190 // See
191 // https://en.wikipedia.org/wiki/Finite_difference_coefficient#Arbitrary_stencil_points
192 //
193 // For a given list of stencil points s of length n and the order of
194 // derivative d < n, the finite difference coefficients can be obtained by
195 // solving the following linear system for the vector a.
196 //
197 // [s₁⁰ ⋯ sₙ⁰ ][a₁] [ δ₀,d ]
198 // [ ⋮ ⋱ ⋮ ][⋮ ] = d! [ ⋮ ]
199 // [s₁ⁿ⁻¹ ⋯ sₙⁿ⁻¹][aₙ] [δₙ₋₁,d]
200 //
201 // where δᵢ,ⱼ are the Kronecker delta. The FIR gains are the elements of the
202 // vector a in reverse order divided by hᵈ.
203 //
204 // The order of accuracy of the approximation is of the form O(hⁿ⁻ᵈ).
205
206 static_assert(Derivative >= 1,
207 "Order of derivative must be greater than or equal to one.");
208 static_assert(Samples > 0, "Number of samples must be greater than zero.");
209 static_assert(Derivative < Samples,
210 "Order of derivative must be less than number of samples.");
211
212 Eigen::Matrix<double, Samples, Samples> S;
213 for (int row = 0; row < Samples; ++row) {
214 for (int col = 0; col < Samples; ++col) {
215 S(row, col) = std::pow(stencil[col], row);
216 }
217 }
218
219 // Fill in Kronecker deltas: https://en.wikipedia.org/wiki/Kronecker_delta
220 Eigen::Vector<double, Samples> d;
221 for (int i = 0; i < Samples; ++i) {
222 d(i) = (i == Derivative) ? Factorial(Derivative) : 0.0;
223 }
224
225 Eigen::Vector<double, Samples> a =
226 S.householderQr().solve(d) / std::pow(period.value(), Derivative);
227
228 // Reverse gains list
229 std::vector<double> ffGains;
230 for (int i = Samples - 1; i >= 0; --i) {
231 ffGains.push_back(a(i));
232 }
233
234 return LinearFilter(ffGains, {});
235 }
236
237 /**
238 * Creates a backward finite difference filter that computes the nth
239 * derivative of the input given the specified number of samples.
240 *
241 * For example, a first derivative filter that uses two samples and a sample
242 * period of 20 ms would be
243 *
244 * <pre><code>
245 * LinearFilter<double>::BackwardFiniteDifference<1, 2>(20_ms);
246 * </code></pre>
247 *
248 * @tparam Derivative The order of the derivative to compute.
249 * @tparam Samples The number of samples to use to compute the given
250 * derivative. This must be one more than the order of
251 * derivative or higher.
252 * @param period The period in seconds between samples taken by the user.
253 */
254 template <int Derivative, int Samples>
255 static LinearFilter<T> BackwardFiniteDifference(units::second_t period) {
256 // Generate stencil points from -(samples - 1) to 0
257 wpi::array<int, Samples> stencil{wpi::empty_array};
258 for (int i = 0; i < Samples; ++i) {
259 stencil[i] = -(Samples - 1) + i;
260 }
261
262 return FiniteDifference<Derivative, Samples>(stencil, period);
263 }
264
265 /**
266 * Reset the filter state.
267 */
268 void Reset() {
269 std::fill(m_inputs.begin(), m_inputs.end(), T{0.0});
270 std::fill(m_outputs.begin(), m_outputs.end(), T{0.0});
271 }
272
273 /**
274 * Calculates the next value of the filter.
275 *
276 * @param input Current input value.
277 *
278 * @return The filtered value at this step
279 */
280 T Calculate(T input) {
281 T retVal{0.0};
282
283 // Rotate the inputs
284 if (m_inputGains.size() > 0) {
285 m_inputs.push_front(input);
286 }
287
288 // Calculate the new value
289 for (size_t i = 0; i < m_inputGains.size(); ++i) {
290 retVal += m_inputs[i] * m_inputGains[i];
291 }
292 for (size_t i = 0; i < m_outputGains.size(); ++i) {
293 retVal -= m_outputs[i] * m_outputGains[i];
294 }
295
296 // Rotate the outputs
297 if (m_outputGains.size() > 0) {
298 m_outputs.push_front(retVal);
299 }
300
301 return retVal;
302 }
303
304 private:
305 wpi::circular_buffer<T> m_inputs;
306 wpi::circular_buffer<T> m_outputs;
307 std::vector<double> m_inputGains;
308 std::vector<double> m_outputGains;
309
310 /**
311 * Factorial of n.
312 *
313 * @param n Argument of which to take factorial.
314 */
315 static constexpr int Factorial(int n) {
316 if (n < 2) {
317 return 1;
318 } else {
319 return n * Factorial(n - 1);
320 }
321 }
322};
