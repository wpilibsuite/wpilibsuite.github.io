42 /**
43 * Sets the deadband applied to the drive inputs (e.g., joystick values).
44 *
45 * The default value is 0.02. Inputs smaller than the deadband are set to 0.0
46 * while inputs larger than the deadband are scaled from 0.0 to 1.0. See

49 * @param deadband The deadband to set.
50 */
51 void SetDeadband(double deadband);
52
53 /**
54 * Configure the scaling factor for using RobotDrive with motor controllers in

59 */
60 void SetMaxOutput(double maxOutput);

78 static void Desaturate(wpi::span<double> wheelSpeeds);

80 double m_deadband = 0.02;
