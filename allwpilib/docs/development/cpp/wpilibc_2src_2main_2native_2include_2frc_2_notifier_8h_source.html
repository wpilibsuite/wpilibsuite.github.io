39 /**
40 * Create a Notifier for timer event notification.
41 *
42 * This overload makes the underlying thread run with a real-time priority.
43 * This is useful for reducing scheduling jitter on processes which are
44 * sensitive to timing variance, like model-based control.
45 *
46 * @param priority The FIFO real-time scheduler priority ([1..99] where a
47 * higher number represents higher priority). See "man 7
48 * sched" for more details.
49 * @param handler The handler is called at the notification time which is set
50 * using StartSingle or StartPeriodic.
51 */
52 explicit Notifier(int priority, std::function<void()> handler);
53

55 Notifier(int priority, Callable&& f, Arg&& arg, Args&&... args)
56 : Notifier(priority,
57 std::bind(std::forward<Callable>(f), std::forward<Arg>(arg),
58 std::forward<Args>(args)...)) {}

82 /**
83 * Register for single event notification.
84 *
85 * A timer event is queued for a single event after the specified delay.
86 *
87 * @param delay Amount of time to wait before the handler is called.
88 */
89 void StartSingle(units::second_t delay);
90
91 /**
92 * Register for periodic event notification.
93 *
94 * A timer event is queued for periodic event notification. Each time the
95 * interrupt occurs, the event will be immediately requeued for the same time
96 * interval.
97 *
98 * @param period Period to call the handler starting one period
99 * after the call to this method.
100 */
101 void StartPeriodic(units::second_t period);
102
103 /**
104 * Stop timer events from occurring.
105 *
106 * Stop any repeating timer events from occurring. This will also remove any
107 * single notification events from the queue.
108 *
109 * If a timer-based call to the registered handler is in progress, this
110 * function will block until the handler call is complete.
111 */
112 void Stop();
113
114 /**
115 * Sets the HAL notifier thread priority.
116 *
117 * The HAL notifier thread is responsible for managing the FPGA's notifier
118 * interrupt and waking up user's Notifiers when it's their time to run.
119 * Giving the HAL notifier thread real-time priority helps ensure the user's
120 * real-time Notifiers, if any, are notified to run in a timely manner.
121 *
122 * @param realTime Set to true to set a real-time priority, false for standard
123 * priority.
124 * @param priority Priority to set the thread to. For real-time, this is 1-99
125 * with 99 being highest. For non-real-time, this is forced to
126 * 0. See "man 7 sched" for more details.
127 * @return True on success.
128 */
129 static bool SetHALThreadPriority(bool realTime, int32_t priority);
130
131 private:
132 /**
133 * Update the HAL alarm time.
134 *
135 * @param triggerTime the time at which the next alarm will be triggered
136 */
137 void UpdateAlarm(uint64_t triggerTime);
138
139 /**
140 * Update the HAL alarm time based on m_expirationTime.
141 */
142 void UpdateAlarm();
143

146

148 wpi::mutex m_processMutex;
149
150 // HAL handle, atomic for proper destruction

152
153 // Address of the handler

155
156 // The absolute expiration time

158
159 // The relative time (either periodic or single)
160 units::second_t m_period = 0_s;
161
162 // True if this is a periodic event
163 bool m_periodic = false;
164};
