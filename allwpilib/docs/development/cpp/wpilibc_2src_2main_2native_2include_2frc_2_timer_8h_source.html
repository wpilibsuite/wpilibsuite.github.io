12/**
13 * Pause the task for a specified time.
14 *
15 * Pause the execution of the program for a specified period of time given in
16 * seconds. Motors will continue to run at their last assigned values, and
17 * sensors will continue to update. Only the task containing the wait will pause
18 * until the wait time is expired.
19 *
20 * @param seconds Length of time to pause, in seconds.
21 */
22void Wait(units::second_t seconds);

31/**
32 * A timer class.
33 *
34 * Note that if the user calls frc::sim::RestartTiming(), they should also reset
35 * the timer so Get() won't return a negative duration.
36 */
37class Timer {
38 public:
39 /**
40 * Create a new timer object.
41 *
42 * Create a new timer object and reset the time to zero. The timer is
43 * initially not running and must be started.
44 */
45 Timer();
46
47 virtual ~Timer() = default;
48
49 Timer(const Timer&) = default;
50 Timer& operator=(const Timer&) = default;
51 Timer(Timer&&) = default;
52 Timer& operator=(Timer&&) = default;
53
54 /**
55 * Get the current time from the timer. If the clock is running it is derived
56 * from the current system clock the start time stored in the timer class. If
57 * the clock is not running, then return the time when it was last stopped.
58 *
59 * @return Current time value for this timer in seconds
60 */
61 units::second_t Get() const;
62
63 /**
64 * Reset the timer by setting the time to 0.
65 *
66 * Make the timer startTime the current time so new requests will be relative
67 * to now.
68 */
69 void Reset();
70
71 /**
72 * Start the timer running.
73 *
74 * Just set the running flag to true indicating that all time requests should
75 * be relative to the system clock. Note that this method is a no-op if the
76 * timer is already running.
77 */
78 void Start();
79
80 /**
81 * Stop the timer.
82 *
83 * This computes the time as of now and clears the running flag, causing all
84 * subsequent time requests to be read from the accumulated time rather than
85 * looking at the system clock.
86 */
87 void Stop();
88
89 /**
90 * Check if the period specified has passed.
91 *
92 * @param period The period to check.
93 * @return True if the period has passed.
94 */
95 bool HasElapsed(units::second_t period) const;
96
97 /**
98 * Check if the period specified has passed and if it has, advance the start
99 * time by that period. This is useful to decide if it's time to do periodic
100 * work without drifting later by the time it took to get around to checking.
101 *
102 * @param period The period to check for.
103 * @return True if the period has passed.
104 * @deprecated Use AdvanceIfElapsed() instead.
105 */
106 WPI_DEPRECATED("Use AdvanceIfElapsed() instead.")
107 bool HasPeriodPassed(units::second_t period);
108
109 /**
110 * Check if the period specified has passed and if it has, advance the start
111 * time by that period. This is useful to decide if it's time to do periodic
112 * work without drifting later by the time it took to get around to checking.
113 *
114 * @param period The period to check for.
115 * @return True if the period has passed.
116 */
117 bool AdvanceIfElapsed(units::second_t period);
118
119 /**
120 * Return the FPGA system clock time in seconds.
121 *
122 * Return the time from the FPGA hardware clock in seconds since the FPGA
123 * started. Rolls over after 71 minutes.
124 *
125 * @returns Robot running time in seconds.
126 */
127 static units::second_t GetFPGATimestamp();
128
129 /**
130 * Return the approximate match time.
131 *
132 * The FMS does not send an official match time to the robots, but does send
133 * an approximate match time. The value will count down the time remaining in
134 * the current period (auto or teleop).
135 *
136 * Warning: This is not an official time (so it cannot be used to dispute ref
137 * calls or guarantee that a function will trigger before the match ends).
138 *
139 * The Practice Match function of the DS approximates the behavior seen on the
140 * field.
141 *
142 * @return Time remaining in current match period (auto or teleop)
143 */
144 static units::second_t GetMatchTime();
145
146 private:
147 units::second_t m_startTime = 0_s;
148 units::second_t m_accumulatedTime = 0_s;
149 bool m_running = false;
150};
