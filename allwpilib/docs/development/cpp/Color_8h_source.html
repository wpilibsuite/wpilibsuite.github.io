750 constexpr Color(double r, double g, double b)
751 : red(roundAndClamp(r)),
752 green(roundAndClamp(g)),
753 blue(roundAndClamp(b)) {}
754
763 static constexpr Color FromHSV(int h, int s, int v) {
764 if (s == 0) {
765 return {v / 255.0, v / 255.0, v / 255.0};
766 }
767
768 int region = h / 30;
769 int remainder = (h - (region * 30)) * 6;
770
771 int p = (v * (255 - s)) >> 8;
772 int q = (v * (255 - ((s * remainder) >> 8))) >> 8;
773 int t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8;
774
775 switch (region) {
776 case 0:
777 return Color(v / 255.0, t / 255.0, p / 255.0);
778 case 1:
779 return Color(q / 255.0, v / 255.0, p / 255.0);
780 case 2:
781 return Color(p / 255.0, v / 255.0, t / 255.0);
782 case 3:
783 return Color(p / 255.0, q / 255.0, v / 255.0);
784 case 4:
785 return Color(t / 255.0, p / 255.0, v / 255.0);
786 default:
787 return Color(v / 255.0, p / 255.0, q / 255.0);
788 }
789 }
790
791 double red = 0.0;
792 double green = 0.0;
793 double blue = 0.0;
794
795 private:
796 static constexpr double kPrecision = 1.0 / (1 << 12);
797
798 static constexpr double roundAndClamp(double value) {
799 const auto rounded =
800 (static_cast<int>(value / kPrecision) + 0.5) * kPrecision;
801 return std::clamp(rounded, 0.0, 1.0);
802 }
803};
