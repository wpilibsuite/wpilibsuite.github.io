296 /**
297 * Same as Send(), but won't queue a send request if it can't be completed
298 * immediately.
299 *
300 * @param addr sockaddr_in or sockaddr_in6 with the address and port of the
301 * remote peer.
302 * @param bufs The buffers to be send.
303 * @return Number of bytes sent.
304 */
305 int TrySend(const sockaddr& addr, span<const Buffer> bufs) {
306 int val = uv_udp_try_send(GetRaw(), bufs.data(),
307 static_cast<unsigned>(bufs.size()), &addr);
308 if (val < 0) {
309 this->ReportError(val);
310 return 0;
311 }
312 return val;
313 }
