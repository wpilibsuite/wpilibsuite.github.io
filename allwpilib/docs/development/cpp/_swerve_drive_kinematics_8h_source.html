22/**
23 * Helper class that converts a chassis velocity (dx, dy, and dtheta components)
24 * into individual module states (speed and angle).
25 *
26 * The inverse kinematics (converting from a desired chassis velocity to
27 * individual module states) uses the relative locations of the modules with
28 * respect to the center of rotation. The center of rotation for inverse
29 * kinematics is also variable. This means that you can set your set your center
30 * of rotation in a corner of the robot to perform special evasion maneuvers.
31 *
32 * Forward kinematics (converting an array of module states into the overall
33 * chassis motion) is performs the exact opposite of what inverse kinematics
34 * does. Since this is an overdetermined system (more equations than variables),
35 * we use a least-squares approximation.
36 *
37 * The inverse kinematics: [moduleStates] = [moduleLocations] * [chassisSpeeds]
38 * We take the Moore-Penrose pseudoinverse of [moduleLocations] and then
39 * multiply by [moduleStates] to get our chassis speeds.
40 *
41 * Forward kinematics is also used for odometry -- determining the position of
42 * the robot on the field using encoders and a gyro.
43 */
44template <size_t NumModules>
45class SwerveDriveKinematics {
46 public:
47 /**
48 * Constructs a swerve drive kinematics object. This takes in a variable
49 * number of wheel locations as Translation2ds. The order in which you pass in
50 * the wheel locations is the same order that you will receive the module
51 * states when performing inverse kinematics. It is also expected that you
52 * pass in the module states in the same order when calling the forward
53 * kinematics methods.
54 *
55 * @param wheel The location of the first wheel relative to the physical
56 * center of the robot.
57 * @param wheels The locations of the other wheels relative to the physical
58 * center of the robot.
59 */
60 template <typename... Wheels>
61 explicit SwerveDriveKinematics(Translation2d wheel, Wheels&&... wheels)
62 : m_modules{wheel, wheels...}, m_moduleStates(wpi::empty_array) {
63 static_assert(sizeof...(wheels) >= 1,
64 "A swerve drive requires at least two modules");
65
66 for (size_t i = 0; i < NumModules; i++) {
67 // clang-format off
68 m_inverseKinematics.template block<2, 3>(i * 2, 0) <<
69 1, 0, (-m_modules[i].Y()).value(),
70 0, 1, (+m_modules[i].X()).value();
71 // clang-format on
72 }
73
74 m_forwardKinematics = m_inverseKinematics.householderQr();
75
76 wpi::math::MathSharedStore::ReportUsage(
77 wpi::math::MathUsageId::kKinematics_SwerveDrive, 1);
78 }
79
80 explicit SwerveDriveKinematics(
81 const wpi::array<Translation2d, NumModules>& wheels)
82 : m_modules{wheels}, m_moduleStates(wpi::empty_array) {
83 for (size_t i = 0; i < NumModules; i++) {
84 // clang-format off
85 m_inverseKinematics.template block<2, 3>(i * 2, 0) <<
86 1, 0, (-m_modules[i].Y()).value(),
87 0, 1, (+m_modules[i].X()).value();
88 // clang-format on
89 }
90
91 m_forwardKinematics = m_inverseKinematics.householderQr();
92
93 wpi::math::MathSharedStore::ReportUsage(
94 wpi::math::MathUsageId::kKinematics_SwerveDrive, 1);
95 }
96
97 SwerveDriveKinematics(const SwerveDriveKinematics&) = default;
98
99 /**
100 * Performs inverse kinematics to return the module states from a desired
101 * chassis velocity. This method is often used to convert joystick values into
102 * module speeds and angles.
103 *
104 * This function also supports variable centers of rotation. During normal
105 * operations, the center of rotation is usually the same as the physical
106 * center of the robot; therefore, the argument is defaulted to that use case.
107 * However, if you wish to change the center of rotation for evasive
108 * maneuvers, vision alignment, or for any other use case, you can do so.
109 *
110 * In the case that the desired chassis speeds are zero (i.e. the robot will
111 * be stationary), the previously calculated module angle will be maintained.
112 *
113 * @param chassisSpeeds The desired chassis speed.
114 * @param centerOfRotation The center of rotation. For example, if you set the
115 * center of rotation at one corner of the robot and provide a chassis speed
116 * that only has a dtheta component, the robot will rotate around that corner.
117 *
118 * @return An array containing the module states. Use caution because these
119 * module states are not normalized. Sometimes, a user input may cause one of
120 * the module speeds to go above the attainable max velocity. Use the
121 * DesaturateWheelSpeeds(wpi::array<SwerveModuleState, NumModules>*,
122 * units::meters_per_second_t) function to rectify this issue. In addition,
123 * you can leverage the power of C++17 to directly assign the module states to
124 * variables:
125 *
126 * @code{.cpp}
127 * auto [fl, fr, bl, br] = kinematics.ToSwerveModuleStates(chassisSpeeds);
128 * @endcode
129 */
130 wpi::array<SwerveModuleState, NumModules> ToSwerveModuleStates(
131 const ChassisSpeeds& chassisSpeeds,
132 const Translation2d& centerOfRotation = Translation2d{}) const;
133
134 /**
135 * Performs forward kinematics to return the resulting chassis state from the
136 * given module states. This method is often used for odometry -- determining
137 * the robot's position on the field using data from the real-world speed and
138 * angle of each module on the robot.
139 *
140 * @param wheelStates The state of the modules (as a SwerveModuleState type)
141 * as measured from respective encoders and gyros. The order of the swerve
142 * module states should be same as passed into the constructor of this class.
143 *
144 * @return The resulting chassis speed.
145 */
146 template <typename... ModuleStates>
147 ChassisSpeeds ToChassisSpeeds(ModuleStates&&... wheelStates) const;
148
149 /**
150 * Performs forward kinematics to return the resulting chassis state from the
151 * given module states. This method is often used for odometry -- determining
152 * the robot's position on the field using data from the real-world speed and
153 * angle of each module on the robot.
154 *
155 * @param moduleStates The state of the modules as an wpi::array of type
156 * SwerveModuleState, NumModules long as measured from respective encoders
157 * and gyros. The order of the swerve module states should be same as passed
158 * into the constructor of this class.
159 *
160 * @return The resulting chassis speed.
161 */
162 ChassisSpeeds ToChassisSpeeds(
163 wpi::array<SwerveModuleState, NumModules> moduleStates) const;
164
165 /**
166 * Renormalizes the wheel speeds if any individual speed is above the
167 * specified maximum.
168 *
169 * Sometimes, after inverse kinematics, the requested speed
170 * from one or more modules may be above the max attainable speed for the
171 * driving motor on that module. To fix this issue, one can reduce all the
172 * wheel speeds to make sure that all requested module speeds are at-or-below
173 * the absolute threshold, while maintaining the ratio of speeds between
174 * modules.
175 *
176 * @param moduleStates Reference to array of module states. The array will be
177 * mutated with the normalized speeds!
178 * @param attainableMaxSpeed The absolute max speed that a module can reach.
179 */
180 static void DesaturateWheelSpeeds(
181 wpi::array<SwerveModuleState, NumModules>* moduleStates,
182 units::meters_per_second_t attainableMaxSpeed);
183
184 private:
185 mutable Matrixd<NumModules * 2, 3> m_inverseKinematics;
186 Eigen::HouseholderQR<Matrixd<NumModules * 2, 3>> m_forwardKinematics;
187 wpi::array<Translation2d, NumModules> m_modules;
188 mutable wpi::array<SwerveModuleState, NumModules> m_moduleStates;
189
190 mutable Translation2d m_previousCoR;
191};
