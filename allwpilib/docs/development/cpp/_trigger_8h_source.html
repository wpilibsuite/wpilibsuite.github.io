23/**
24 * This class is a command-based wrapper around {@link BooleanEvent}, providing
25 * an easy way to link commands to inputs.

27 * This class is provided by the NewCommands VendorDep

32 public:
33 /**
34 * Creates a new trigger with the given condition determining whether it is
35 * active.
36 *
37 * <p>Polled by the default scheduler button loop.
38 *
39 * @param isActive returns whether or not the trigger should be active
40 */
41 explicit Trigger(std::function<bool()> isActive)
42 : BooleanEvent{CommandScheduler::GetInstance().GetDefaultButtonLoop(),
43 std::move(isActive)} {}

45 /**
46 * Create a new trigger that is active when the given condition is true.
47 *
48 * @param loop The loop instance that polls this trigger.
49 * @param isActive Whether the trigger is active.
50 */
51 Trigger(frc::EventLoop* loop, std::function<bool()> isActive)
52 : BooleanEvent{loop, std::move(isActive)} {}

62 /**
63 * Binds a command to start when the trigger becomes active. Takes a
64 * raw pointer, and so is non-owning; users are responsible for the lifespan

72 /**
73 * Binds a command to start when the trigger becomes active. Transfers
74 * command ownership to the button scheduler, so the user does not have to
75 * worry about lifespan - rvalue refs will be *moved*, lvalue refs will be
76 * *copied.*
77 *
78 * @param command The command to bind.
79 * @return The trigger, for chained calls.
80 */
81 template <class T, typename = std::enable_if_t<std::is_base_of_v<
82 Command, std::remove_reference_t<T>>>>
83 Trigger WhenActive(T&& command) {
84 this->Rising().IfHigh(
85 [command = std::make_unique<std::remove_reference_t<T>>(
86 std::forward<T>(command))] { command->Schedule(); });
87
88 return *this;
89 }

236 * Binds a command to start when the trigger becomes active, and be canceled
237 * when it again becomes active. Takes a raw pointer, and so is non-owning;
238 * users are responsible for the lifespan of the command.
239 *
240 * @param command The command to bind.
241 * @return The trigger, for chained calls.

244
245 /**
246 * Binds a command to start when the trigger becomes active, and be canceled
247 * when it again becomes active. Transfers command ownership to the button
248 * scheduler, so the user does not have to worry about lifespan - rvalue refs
249 * will be *moved*, lvalue refs will be *copied.*
250 *
251 * @param command The command to bind.
252 * @return The trigger, for chained calls.
253 */
254 template <class T, typename = std::enable_if_t<std::is_base_of_v<
255 Command, std::remove_reference_t<T>>>>
256 Trigger ToggleWhenActive(T&& command) {
257 this->Rising().IfHigh(
258 [command = std::make_unique<std::remove_reference_t<T>>(
259 std::forward<T>(command))] {
260 if (!command->IsScheduled()) {
261 command->Schedule();
262 } else {
263 command->Cancel();
264 }
265 });
266
267 return *this;
268 }

276 * @return The trigger, for chained calls.

280 /**
281 * Get a new event that events only when this one newly changes to true.

285 Trigger Rising() { return BooleanEvent::Rising().CastTo<Trigger>(); }
286
287 /**
288 * Get a new event that triggers only when this one newly changes to false.
289 *
290 * @return a new event representing when this one newly changes to false.

292 Trigger Falling() { return BooleanEvent::Falling().CastTo<Trigger>(); }

294 /**
295 * Composes two triggers with logical AND.
296 *
297 * @return A trigger which is active when both component triggers are active.
298 */
299 Trigger operator&&(std::function<bool()> rhs) {
300 return BooleanEvent::operator&&(rhs).CastTo<Trigger>();
301 }
302
303 /**
304 * Composes two triggers with logical OR.
305 *
306 * @return A trigger which is active when either component trigger is active.
307 */
308 Trigger operator||(std::function<bool()> rhs) {
309 return BooleanEvent::operator||(rhs).CastTo<Trigger>();
310 }
311
312 /**
313 * Composes a trigger with logical NOT.
314 *
315 * @return A trigger which is active when the component trigger is inactive,
316 * and vice-versa.
317 */
318 Trigger operator!() { return BooleanEvent::operator!().CastTo<Trigger>(); }
319
320 /**
321 * Creates a new debounced trigger from this trigger - it will become active
322 * when this trigger has been active for longer than the specified period.
323 *
324 * @param debounceTime The debounce period.
325 * @param type The debounce type.
326 * @return The debounced trigger.
327 */
328 Trigger Debounce(units::second_t debounceTime,
329 frc::Debouncer::DebounceType type =
330 frc::Debouncer::DebounceType::kRising) {
331 return BooleanEvent::Debounce(debounceTime, type).CastTo<Trigger>();
332 }
