118 // does leak checking per-object
119 // leak is detected upon destructor
120 template <class Handler>
121 class object_leak_checker : Handler
122 {
123 public:
124 object_leak_checker() noexcept : allocated_(0) {}
125
126 object_leak_checker(object_leak_checker&& other) noexcept
127 : allocated_(other.allocated_)
128 {
129 other.allocated_ = 0;
130 }
131
132 ~object_leak_checker() noexcept
133 {
134 if (allocated_ != 0)
135 this->operator()(allocated_);
136 }
137
138 object_leak_checker& operator=(object_leak_checker&& other) noexcept
139 {
140 allocated_ = other.allocated_;
141 other.allocated_ = 0;
142 return *this;
143 }
144
145 void on_allocate(std::size_t size) noexcept
146 {
147 allocated_ += std::ptrdiff_t(size);
148 }
149
150 void on_deallocate(std::size_t size) noexcept
151 {
152 allocated_ -= std::ptrdiff_t(size);
153 }
154
155 private:
156 std::ptrdiff_t allocated_;
157 };

159 // does leak checking on a global basis
160 // call macro WPI_MEMORY_GLOBAL_LEAK_CHECKER(handler, var_name) in the header
161 // when last counter gets destroyed, leak is detected
162 template <class Handler>
163 class global_leak_checker_impl
164 {
165 public:
166 struct counter : Handler
167 {
168 counter()
169 {
170 ++no_counter_objects_;
171 }
172
173 ~counter()
174 {
175 --no_counter_objects_;
176 if (no_counter_objects_ == 0u && allocated_ != 0u)
177 this->operator()(allocated_);
178 }
179 };
180
181 global_leak_checker_impl() noexcept {}
182 global_leak_checker_impl(global_leak_checker_impl&&) noexcept {}
183 ~global_leak_checker_impl() noexcept {}
184
185 global_leak_checker_impl& operator=(global_leak_checker_impl&&) noexcept
186 {
187 return *this;
188 }
189
190 void on_allocate(std::size_t size) noexcept
191 {
192 allocated_ += std::ptrdiff_t(size);
193 }
194
195 void on_deallocate(std::size_t size) noexcept
196 {
197 allocated_ -= std::ptrdiff_t(size);
198 }
199
200 private:
201 static std::atomic<std::size_t> no_counter_objects_;
202 static std::atomic<std::ptrdiff_t> allocated_;
203 };
