19/**
20 * Idle handle.
21 *
22 * Idle handles will generate a signal once per loop iteration, right
23 * before the Prepare handles.
24 *
25 * The notable difference with Prepare handles is that when there are active
26 * idle handles, the loop will perform a zero timeout poll instead of blocking
27 * for I/O.
28 *
29 * @warning Despite the name, idle handles will signal every loop iteration,
30 * not when the loop is actually "idle". This also means they can easly become
31 * CPU hogs.
32 */
33class Idle final : public HandleImpl<Idle, uv_idle_t> {
34 struct private_init {};
35
36 public:
37 explicit Idle(const private_init&) {}
38 ~Idle() noexcept override = default;
39
40 /**
41 * Create an idle handle.
42 *
43 * @param loop Loop object where this handle runs.
44 */
45 static std::shared_ptr<Idle> Create(Loop& loop);
46
47 /**
48 * Create an idle handle.
49 *
50 * @param loop Loop object where this handle runs.
51 */
52 static std::shared_ptr<Idle> Create(const std::shared_ptr<Loop>& loop) {
53 return Create(*loop);
54 }
55
56 /**
57 * Start the handle.
58 */
59 void Start();
60
61 /**
62 * Stop the handle. The signal will no longer be generated.
63 */
64 void Stop() { Invoke(&uv_idle_stop, GetRaw()); }
65
66 /**
67 * Signal generated once per loop iteration prior to Prepare signals.
68 */
69 sig::Signal<> idle;
70};
