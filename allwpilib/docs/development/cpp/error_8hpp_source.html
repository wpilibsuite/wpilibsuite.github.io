20 /// Contains information about an allocator.
21 /// It can be used for logging in the various handler functions.
22 /// \ingroup core
23 struct allocator_info
24 {
25 /// The name of the allocator.
26 /// It is a NTBS whose lifetime is not managed by this object,
27 /// it must be stored elsewhere or be a string literal.
28 const char* name;
29
30 /// A pointer representing an allocator.
31 /// It does not necessarily point to the beginning of the allocator object,
32 /// the only guarantee is that different allocator objects result in a different pointer value.
33 /// For stateless allocators it is sometimes \c nullptr.
34 /// \note The pointer must not be cast back to any allocator type.
35 const void* allocator;
36
37 /// \effects Creates it by giving it the name of the allocator and a pointer.
38 constexpr allocator_info(const char* n, const void* alloc) noexcept
39 : name(n), allocator(alloc)
40 {
41 }
42
43 /// @{
44 /// \effects Compares two \ref allocator_info objects, they are equal, if the \ref allocator is the same.
45 /// \returns The result of the comparision.
46 friend constexpr bool operator==(const allocator_info& a,
47 const allocator_info& b) noexcept
48 {
49 return a.allocator == b.allocator;
50 }
51
52 friend constexpr bool operator!=(const allocator_info& a,
53 const allocator_info& b) noexcept
54 {
55 return a.allocator != b.allocator;
56 }
57 /// @}
58 };

60 /// The exception class thrown when a low level allocator runs out of memory.
61 /// It is derived from \c std::bad_alloc.
62 /// This can happen if a low level allocation function like \c std::malloc() runs out of memory.
63 /// Throwing can be prohibited by the handler function.
64 /// \ingroup core
65 class out_of_memory : public std::bad_alloc
66 {
67 public:
68 /// The type of the handler called in the constructor of \ref out_of_memory.
69 /// When an out of memory situation is encountered and the exception class created,
70 /// this handler gets called.
71 /// It is especially useful if exception support is disabled.
72 /// It gets the \ref allocator_info and the amount of memory that was tried to be allocated.
73 /// \requiredbe It can log the error, throw a different exception derived from \c std::bad_alloc or abort the program.
74 /// If it returns, this exception object will be created and thrown.
75 /// \defaultbe On a hosted implementation it logs the error on \c stderr and continues execution,
76 /// leading to this exception being thrown.
77 /// On a freestanding implementation it does nothing.
78 /// \note It is different from \c std::new_handler; it will not be called in a loop trying to allocate memory
79 /// or something like that. Its only job is to report the error.
80 using handler = void (*)(const allocator_info& info, std::size_t amount);
81
82 /// \effects Sets \c h as the new \ref handler in an atomic operation.
83 /// A \c nullptr sets the default \ref handler.
84 /// \returns The previous \ref handler. This is never \c nullptr.
85 static handler set_handler(handler h);
86
87 /// \returns The current \ref handler. This is never \c nullptr.
88 static handler get_handler();
89
90 /// \effects Creates it by passing it the \ref allocator_info and the amount of memory failed to be allocated.
91 /// It also calls the \ref handler to control whether or not it will be thrown.
92 out_of_memory(const allocator_info& info, std::size_t amount);
93
94 /// \returns A static NTBS that describes the error.
95 /// It does not contain any specific information since there is no memory for formatting.
96 const char* what() const noexcept override;
97
98 /// \returns The \ref allocator_info passed to it in the constructor.
99 const allocator_info& allocator() const noexcept
100 {
101 return info_;
102 }
103
104 /// \returns The amount of memory that was tried to be allocated.
105 /// This is the value passed in the constructor.
106 std::size_t failed_allocation_size() const noexcept
107 {
108 return amount_;
109 }
110
111 private:
112 allocator_info info_;
113 std::size_t amount_;
114 };

135 /// The exception class thrown when an allocation size is bigger than the supported maximum.
136 /// This size is either the node, array or alignment parameter in a call to an allocation function.
137 /// If those exceed the supported maximum returned by \c max_node_size(), \c max_array_size() or \c max_alignment(),
138 /// one of its derived classes will be thrown or this class if in a situation where the type is unknown.
139 /// It is derived from \c std::bad_alloc.
140 /// Throwing can be prohibited by the handler function.
141 /// \note Even if all parameters are less than the maximum, \ref out_of_memory or a similar exception can be thrown,
142 /// because the maximum functions return an upper bound and not the actual supported maximum size,
143 /// since it always depends on fence memory, alignment buffer and the like.
144 /// \note A user should only \c catch for \c bad_allocation_size, not the derived classes.
145 /// \note Most checks will only be done if \ref WPI_MEMORY_CHECK_ALLOCATION_SIZE is \c true.
146 /// \ingroup core
147 class bad_allocation_size : public std::bad_alloc
148 {
149 public:
150 /// The type of the handler called in the constructor of \ref bad_allocation_size.
151 /// When a bad allocation size is detected and the exception object created,
152 /// this handler gets called.
153 /// It is especially useful if exception support is disabled.
154 /// It gets the \ref allocator_info, the size passed to the function and the supported size
155 /// (the latter is still an upper bound).
156 /// \requiredbe It can log the error, throw a different exception derived from \c std::bad_alloc or abort the program.
157 /// If it returns, this exception object will be created and thrown.
158 /// \defaultbe On a hosted implementation it logs the error on \c stderr and continues execution,
159 /// leading to this exception being thrown.
160 /// On a freestanding implementation it does nothing.
161 using handler = void (*)(const allocator_info& info, std::size_t passed,
162 std::size_t supported);
163
164 /// \effects Sets \c h as the new \ref handler in an atomic operation.
165 /// A \c nullptr sets the default \ref handler.
166 /// \returns The previous \ref handler. This is never \c nullptr.
167 static handler set_handler(handler h);
168
169 /// \returns The current \ref handler. This is never \c nullptr.
170 static handler get_handler();
171
172 /// \effects Creates it by passing it the \ref allocator_info, the size passed to the allocation function
173 /// and an upper bound on the supported size.
174 /// It also calls the \ref handler to control whether or not it will be thrown.
175 bad_allocation_size(const allocator_info& info, std::size_t passed,
176 std::size_t supported);
177
178 /// \returns A static NTBS that describes the error.
179 /// It does not contain any specific information since there is no memory for formatting.
180 const char* what() const noexcept override;
181
182 /// \returns The \ref allocator_info passed to it in the constructor.
183 const allocator_info& allocator() const noexcept
184 {
185 return info_;
186 }
187
188 /// \returns The size or alignment value that was passed to the allocation function
189 /// which was too big. This is the same value passed to the constructor.
190 std::size_t passed_value() const noexcept
191 {
192 return passed_;
193 }
194
195 /// \returns An upper bound on the maximum supported size/alignment.
196 /// It is only an upper bound, values below can fail, but values above will always fail.
197 std::size_t supported_value() const noexcept
198 {
199 return supported_;
200 }
201
202 private:
203 allocator_info info_;
204 std::size_t passed_, supported_;
205 };

265 void check_allocation_size(std::size_t passed, Func f, const allocator_info& info)
266 {
267#if WPI_MEMORY_CHECK_ALLOCATION_SIZE
268 auto supported = f();
269 if (passed > supported)
270 WPI_THROW(Ex(info, passed, supported));
271#else
272 (void)passed;
273 (void)f;
274 (void)info;
275#endif
276 }

278 template <class Ex>
279 void check_allocation_size(std::size_t passed, std::size_t supported,
280 const allocator_info& info)
281 {
282 check_allocation_size<Ex>(
283 passed, [&] { return supported; }, info);
284 }
