13/**
14 * I2C bus interface class.
15 *
16 * This class is intended to be used by sensor (and other I2C device) drivers.
17 * It probably should not be used directly.
18 */
19class I2C {
20 public:
21 enum Port { kOnboard = 0, kMXP };
22
23 /**
24 * Constructor.
25 *
26 * @param port The I2C port to which the device is connected.
27 * @param deviceAddress The address of the device on the I2C bus.
28 */
29 I2C(Port port, int deviceAddress);
30
31 ~I2C();
32
33 I2C(I2C&&) = default;
34 I2C& operator=(I2C&&) = default;
35
36 Port GetPort() const;
37 int GetDeviceAddress() const;
38
39 /**
40 * Generic transaction.
41 *
42 * This is a lower-level interface to the I2C hardware giving you more control
43 * over each transaction. If you intend to write multiple bytes in the same
44 * transaction and do not plan to receive anything back, use writeBulk()
45 * instead. Calling this with a receiveSize of 0 will result in an error.
46 *
47 * @param dataToSend Buffer of data to send as part of the transaction.
48 * @param sendSize Number of bytes to send as part of the transaction.
49 * @param dataReceived Buffer to read data into.
50 * @param receiveSize Number of bytes to read from the device.
51 * @return Transfer Aborted... false for success, true for aborted.
52 */
53 bool Transaction(uint8_t* dataToSend, int sendSize, uint8_t* dataReceived,
54 int receiveSize);
55
56 /**
57 * Attempt to address a device on the I2C bus.
58 *
59 * This allows you to figure out if there is a device on the I2C bus that
60 * responds to the address specified in the constructor.
61 *
62 * @return Transfer Aborted... false for success, true for aborted.
63 */
64 bool AddressOnly();
65
66 /**
67 * Execute a write transaction with the device.
68 *
69 * Write a single byte to a register on a device and wait until the
70 * transaction is complete.
71 *
72 * @param registerAddress The address of the register on the device to be
73 * written.
74 * @param data The byte to write to the register on the device.
75 * @return Transfer Aborted... false for success, true for aborted.
76 */
77 bool Write(int registerAddress, uint8_t data);
78
79 /**
80 * Execute a bulk write transaction with the device.
81 *
82 * Write multiple bytes to a device and wait until the
83 * transaction is complete.
84 *
85 * @param data The data to write to the register on the device.
86 * @param count The number of bytes to be written.
87 * @return Transfer Aborted... false for success, true for aborted.
88 */
89 bool WriteBulk(uint8_t* data, int count);
90
91 /**
92 * Execute a read transaction with the device.
93 *
94 * Read bytes from a device.
95 * Most I2C devices will auto-increment the register pointer internally
96 * allowing you to read consecutive registers on a device in a single
97 * transaction.
98 *
99 * @param registerAddress The register to read first in the transaction.
100 * @param count The number of bytes to read in the transaction.
101 * @param data A pointer to the array of bytes to store the data
102 * read from the device.
103 * @return Transfer Aborted... false for success, true for aborted.
104 */
105 bool Read(int registerAddress, int count, uint8_t* data);
106
107 /**
108 * Execute a read only transaction with the device.
109 *
110 * Read bytes from a device. This method does not write any data to prompt the
111 * device.
112 *
113 * @param buffer A pointer to the array of bytes to store the data read from
114 * the device.
115 * @param count The number of bytes to read in the transaction.
116 * @return Transfer Aborted... false for success, true for aborted.
117 */
118 bool ReadOnly(int count, uint8_t* buffer);
119
120 /**
121 * Verify that a device's registers contain expected values.
122 *
123 * Most devices will have a set of registers that contain a known value that
124 * can be used to identify them. This allows an I2C device driver to easily
125 * verify that the device contains the expected value.
126 *
127 * @pre The device must support and be configured to use register
128 * auto-increment.
129 *
130 * @param registerAddress The base register to start reading from the device.
131 * @param count The size of the field to be verified.
132 * @param expected A buffer containing the values expected from the
133 * device.
134 */
135 bool VerifySensor(int registerAddress, int count, const uint8_t* expected);
136
137 private:
138 hal::I2CPort m_port;
139 int m_deviceAddress;
140};
