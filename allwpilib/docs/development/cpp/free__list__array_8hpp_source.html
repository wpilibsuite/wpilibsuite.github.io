19 // an array of free_memory_list types
20 // indexed via size, AccessPolicy does necessary conversions
21 // requires trivial destructible FreeList type
22 template <class FreeList, class AccessPolicy>
23 class free_list_array
24 {
25 // not supported on GCC 4.7
26 //static_assert(std::is_trivially_destructible<FreeList>::value,
27 // "free list must be trivially destructible");
28 public:
29 // creates sufficient elements to support up to given maximum node size
30 // all lists are initially empty
31 // actual number is calculated via policy
32 // memory is taken from fixed_memory_stack, it must be sufficient
33 free_list_array(fixed_memory_stack& stack, const char* end,
34 std::size_t max_node_size) noexcept
35 : no_elements_(AccessPolicy::index_from_size(max_node_size) - min_size_index + 1)
36 {
37 array_ = static_cast<FreeList*>(
38 stack.allocate(end, no_elements_ * sizeof(FreeList), alignof(FreeList)));
39 WPI_MEMORY_ASSERT_MSG(array_, "insufficient memory for free lists");
40 for (std::size_t i = 0u; i != no_elements_; ++i)
41 {
42 auto node_size = AccessPolicy::size_from_index(i + min_size_index);
43 ::new (static_cast<void*>(array_ + i)) FreeList(node_size);
44 }
45 }
46
47 // move constructor, does not actually move the elements, just the pointer
48 free_list_array(free_list_array&& other) noexcept
49 : array_(other.array_), no_elements_(other.no_elements_)
50 {
51 other.array_ = nullptr;
52 other.no_elements_ = 0u;
53 }
54
55 // destructor, does nothing, list must be trivially destructible!
56 ~free_list_array() noexcept = default;
57
58 free_list_array& operator=(free_list_array&& other) noexcept
59 {
60 array_ = other.array_;
61 no_elements_ = other.no_elements_;
62
63 other.array_ = nullptr;
64 other.no_elements_ = 0u;
65 return *this;
66 }
67
68 // access free list for given size
69 FreeList& get(std::size_t node_size) const noexcept
70 {
71 auto i = AccessPolicy::index_from_size(node_size);
72 if (i < min_size_index)
73 i = min_size_index;
74 return array_[i - min_size_index];
75 }
76
77 // number of free lists
78 std::size_t size() const noexcept
79 {
80 return no_elements_;
81 }
82
83 // maximum supported node size
84 std::size_t max_node_size() const noexcept
85 {
86 return AccessPolicy::size_from_index(no_elements_ + min_size_index - 1);
87 }
88
89 private:
90 static const std::size_t min_size_index;
91
92 FreeList* array_;
93 std::size_t no_elements_;
94 };
