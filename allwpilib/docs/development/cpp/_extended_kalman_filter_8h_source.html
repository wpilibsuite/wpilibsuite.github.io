16/**
17 * A Kalman filter combines predictions from a model and measurements to give an
18 * estimate of the true system state. This is useful because many states cannot
19 * be measured directly as a result of sensor noise, or because the state is
20 * "hidden".
21 *
22 * Kalman filters use a K gain matrix to determine whether to trust the model or
23 * measurements more. Kalman filter theory uses statistics to compute an optimal
24 * K gain which minimizes the sum of squares error in the state estimate. This K
25 * gain is used to correct the state estimate by some amount of the difference
26 * between the actual measurements and the measurements predicted by the model.
27 *
28 * An extended Kalman filter supports nonlinear state and measurement models. It
29 * propagates the error covariance by linearizing the models around the state
30 * estimate, then applying the linear Kalman filter equations.
31 *
32 * For more on the underlying math, read
33 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf chapter 9
34 * "Stochastic control theory".
35 *
36 * @tparam States The number of states.
37 * @tparam Inputs The number of inputs.
38 * @tparam Outputs The number of outputs.
39 */
40template <int States, int Inputs, int Outputs>
41class ExtendedKalmanFilter {
42 public:
43 using StateVector = Vectord<States>;
44 using InputVector = Vectord<Inputs>;
45 using OutputVector = Vectord<Outputs>;
46
47 using StateArray = wpi::array<double, States>;
48 using OutputArray = wpi::array<double, Outputs>;
49
50 using StateMatrix = Matrixd<States, States>;

52 /**

58 * the measurement vector.
59 * @param stateStdDevs Standard deviations of model states.
60 * @param measurementStdDevs Standard deviations of measurements.
61 * @param dt Nominal discretization timestep.
62 */
63 ExtendedKalmanFilter(
64 std::function<StateVector(const StateVector&, const InputVector&)> f,

75 * the measurement vector.
76 * @param stateStdDevs Standard deviations of model states.
77 * @param measurementStdDevs Standard deviations of measurements.
78 * @param residualFuncY A function that computes the residual of two

83 ExtendedKalmanFilter(

158 *
159 * @param u Same control input used in the predict step.

166 template <int Rows>
167 void Correct(
168 const InputVector& u, const Vectord<Rows>& y,

172 /**
173 * Correct the state estimate x-hat using the measurements in y.
174 *
175 * This is useful for when the measurements available during a timestep's
176 * Correct() call vary. The h(x, u) passed to the constructor is used if one
177 * is not provided (the two-argument version of this function).
178 *
179 * @param u Same control input used in the predict step.
180 * @param y Measurement vector.
181 * @param h A vector-valued function of x and u that returns
182 * the measurement vector.
183 * @param R Discrete measurement noise covariance matrix.
184 * @param residualFuncY A function that computes the residual of two
185 * measurement vectors (i.e. it subtracts them.)
186 * @param addFuncX A function that adds two state vectors.
187 */
188 template <int Rows>
189 void Correct(
190 const InputVector& u, const Vectord<Rows>& y,
191 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
192 const Matrixd<Rows, Rows>& R,
193 std::function<Vectord<Rows>(const Vectord<Rows>&, const Vectord<Rows>&)>
194 residualFuncY,
195 std::function<StateVector(const StateVector&, const StateVector&)>
196 addFuncX);
197
198 private:
199 std::function<StateVector(const StateVector&, const InputVector&)> m_f;
200 std::function<OutputVector(const StateVector&, const InputVector&)> m_h;
201 std::function<OutputVector(const OutputVector&, const OutputVector&)>
202 m_residualFuncY;
203 std::function<StateVector(const StateVector&, const StateVector&)> m_addFuncX;
204 StateVector m_xHat;
205 StateMatrix m_P;
206 StateMatrix m_contQ;
207 Matrixd<Outputs, Outputs> m_contR;
208 units::second_t m_dt;
209
210 StateMatrix m_initP;
211};
