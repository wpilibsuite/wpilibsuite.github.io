22/**
23 * A Kalman filter combines predictions from a model and measurements to give an
24 * estimate of the true system state. This is useful because many states cannot
25 * be measured directly as a result of sensor noise, or because the state is
26 * "hidden".
27 *
28 * Kalman filters use a K gain matrix to determine whether to trust the model or
29 * measurements more. Kalman filter theory uses statistics to compute an optimal
30 * K gain which minimizes the sum of squares error in the state estimate. This K
31 * gain is used to correct the state estimate by some amount of the difference
32 * between the actual measurements and the measurements predicted by the model.
33 *
34 * An extended Kalman filter supports nonlinear state and measurement models. It
35 * propagates the error covariance by linearizing the models around the state
36 * estimate, then applying the linear Kalman filter equations.
37 *
38 * For more on the underlying math, read
39 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf chapter 9
40 * "Stochastic control theory".
41 *
42 * @tparam States The number of states.
43 * @tparam Inputs The number of inputs.
44 * @tparam Outputs The number of outputs.
45 */
46template <int States, int Inputs, int Outputs>
47class ExtendedKalmanFilter {
48 public:
49 /**
50 * Constructs an extended Kalman filter.
51 *
52 * @param f A vector-valued function of x and u that returns
53 * the derivative of the state vector.
54 * @param h A vector-valued function of x and u that returns
55 * the measurement vector.
56 * @param stateStdDevs Standard deviations of model states.
57 * @param measurementStdDevs Standard deviations of measurements.
58 * @param dt Nominal discretization timestep.
59 */
60 ExtendedKalmanFilter(std::function<Eigen::Vector<double, States>(
61 const Eigen::Vector<double, States>&,
62 const Eigen::Vector<double, Inputs>&)>
63 f,
64 std::function<Eigen::Vector<double, Outputs>(
65 const Eigen::Vector<double, States>&,
66 const Eigen::Vector<double, Inputs>&)>
67 h,
68 const wpi::array<double, States>& stateStdDevs,
69 const wpi::array<double, Outputs>& measurementStdDevs,
70 units::second_t dt)
71 : m_f(f), m_h(h) {
72 m_contQ = MakeCovMatrix(stateStdDevs);
73 m_contR = MakeCovMatrix(measurementStdDevs);
74 m_residualFuncY = [](auto a, auto b) -> Eigen::Vector<double, Outputs> {
75 return a - b;
76 };
77 m_addFuncX = [](auto a, auto b) -> Eigen::Vector<double, States> {
78 return a + b;
79 };
80 m_dt = dt;
81
82 Reset();
83
84 Eigen::Matrix<double, States, States> contA =
85 NumericalJacobianX<States, States, Inputs>(
86 m_f, m_xHat, Eigen::Vector<double, Inputs>::Zero());
87 Eigen::Matrix<double, Outputs, States> C =
88 NumericalJacobianX<Outputs, States, Inputs>(
89 m_h, m_xHat, Eigen::Vector<double, Inputs>::Zero());
90
91 Eigen::Matrix<double, States, States> discA;
92 Eigen::Matrix<double, States, States> discQ;
93 DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
94
95 Eigen::Matrix<double, Outputs, Outputs> discR =
96 DiscretizeR<Outputs>(m_contR, dt);
97
98 if (IsDetectable<States, Outputs>(discA, C) && Outputs <= States) {
99 m_initP = drake::math::DiscreteAlgebraicRiccatiEquation(
100 discA.transpose(), C.transpose(), discQ, discR);
101 } else {
102 m_initP = Eigen::Matrix<double, States, States>::Zero();
103 }
104 m_P = m_initP;
105 }
106
107 /**
108 * Constructs an extended Kalman filter.
109 *
110 * @param f A vector-valued function of x and u that returns
111 * the derivative of the state vector.
112 * @param h A vector-valued function of x and u that returns
113 * the measurement vector.
114 * @param stateStdDevs Standard deviations of model states.
115 * @param measurementStdDevs Standard deviations of measurements.
116 * @param residualFuncY A function that computes the residual of two
117 * measurement vectors (i.e. it subtracts them.)
118 * @param addFuncX A function that adds two state vectors.
119 * @param dt Nominal discretization timestep.
120 */
121 ExtendedKalmanFilter(std::function<Eigen::Vector<double, States>(
122 const Eigen::Vector<double, States>&,
123 const Eigen::Vector<double, Inputs>&)>
124 f,
125 std::function<Eigen::Vector<double, Outputs>(
126 const Eigen::Vector<double, States>&,
127 const Eigen::Vector<double, Inputs>&)>
128 h,
129 const wpi::array<double, States>& stateStdDevs,
130 const wpi::array<double, Outputs>& measurementStdDevs,
131 std::function<Eigen::Vector<double, Outputs>(
132 const Eigen::Vector<double, Outputs>&,
133 const Eigen::Vector<double, Outputs>&)>
134 residualFuncY,
135 std::function<Eigen::Vector<double, States>(
136 const Eigen::Vector<double, States>&,
137 const Eigen::Vector<double, States>&)>
138 addFuncX,
139 units::second_t dt)
140 : m_f(f), m_h(h), m_residualFuncY(residualFuncY), m_addFuncX(addFuncX) {
141 m_contQ = MakeCovMatrix(stateStdDevs);
142 m_contR = MakeCovMatrix(measurementStdDevs);
143 m_dt = dt;
144
145 Reset();
146
147 Eigen::Matrix<double, States, States> contA =
148 NumericalJacobianX<States, States, Inputs>(
149 m_f, m_xHat, Eigen::Vector<double, Inputs>::Zero());
150 Eigen::Matrix<double, Outputs, States> C =
151 NumericalJacobianX<Outputs, States, Inputs>(
152 m_h, m_xHat, Eigen::Vector<double, Inputs>::Zero());
153
154 Eigen::Matrix<double, States, States> discA;
155 Eigen::Matrix<double, States, States> discQ;
156 DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
157
158 Eigen::Matrix<double, Outputs, Outputs> discR =
159 DiscretizeR<Outputs>(m_contR, dt);
160
161 if (IsDetectable<States, Outputs>(discA, C) && Outputs <= States) {
162 m_initP = drake::math::DiscreteAlgebraicRiccatiEquation(
163 discA.transpose(), C.transpose(), discQ, discR);
164 } else {
165 m_initP = Eigen::Matrix<double, States, States>::Zero();
166 }
167 m_P = m_initP;
168 }

225 /**
226 * Project the model into the future with a new control input u.
227 *
228 * @param u New control input from controller.
229 * @param dt Timestep for prediction.
230 */
231 void Predict(const Eigen::Vector<double, Inputs>& u, units::second_t dt) {
232 // Find continuous A
233 Eigen::Matrix<double, States, States> contA =
234 NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
235
236 // Find discrete A and Q
237 Eigen::Matrix<double, States, States> discA;
238 Eigen::Matrix<double, States, States> discQ;
239 DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
240
241 m_xHat = RK4(m_f, m_xHat, u, dt);
242
243 // Pₖ₊₁⁻ = APₖ⁻Aᵀ + Q
244 m_P = discA * m_P * discA.transpose() + discQ;
245
246 m_dt = dt;
247 }

249 /**
250 * Correct the state estimate x-hat using the measurements in y.
251 *
252 * @param u Same control input used in the predict step.
253 * @param y Measurement vector.
254 */
255 void Correct(const Eigen::Vector<double, Inputs>& u,
256 const Eigen::Vector<double, Outputs>& y) {
257 Correct<Outputs>(u, y, m_h, m_contR, m_residualFuncY, m_addFuncX);
258 }

260 template <int Rows>
261 void Correct(const Eigen::Vector<double, Inputs>& u,
262 const Eigen::Vector<double, Rows>& y,
263 std::function<Eigen::Vector<double, Rows>(
264 const Eigen::Vector<double, States>&,
265 const Eigen::Vector<double, Inputs>&)>
266 h,
267 const Eigen::Matrix<double, Rows, Rows>& R) {
268 auto residualFuncY = [](auto a, auto b) -> Eigen::Vector<double, Rows> {
269 return a - b;
270 };
271 auto addFuncX = [](auto a, auto b) -> Eigen::Vector<double, States> {
272 return a + b;
273 };
274 Correct<Rows>(u, y, h, R, residualFuncY, addFuncX);
275 }

277 /**
278 * Correct the state estimate x-hat using the measurements in y.
279 *
280 * This is useful for when the measurements available during a timestep's
281 * Correct() call vary. The h(x, u) passed to the constructor is used if one
282 * is not provided (the two-argument version of this function).
283 *
284 * @param u Same control input used in the predict step.
285 * @param y Measurement vector.
286 * @param h A vector-valued function of x and u that returns
287 * the measurement vector.
288 * @param R Discrete measurement noise covariance matrix.
289 * @param residualFuncY A function that computes the residual of two
290 * measurement vectors (i.e. it subtracts them.)
291 * @param addFuncX A function that adds two state vectors.
292 */
293 template <int Rows>
294 void Correct(const Eigen::Vector<double, Inputs>& u,
295 const Eigen::Vector<double, Rows>& y,
296 std::function<Eigen::Vector<double, Rows>(
297 const Eigen::Vector<double, States>&,
298 const Eigen::Vector<double, Inputs>&)>
299 h,
300 const Eigen::Matrix<double, Rows, Rows>& R,
301 std::function<Eigen::Vector<double, Rows>(
302 const Eigen::Vector<double, Rows>&,
303 const Eigen::Vector<double, Rows>&)>
304 residualFuncY,
305 std::function<Eigen::Vector<double, States>(
306 const Eigen::Vector<double, States>&,
307 const Eigen::Vector<double, States>)>
308 addFuncX) {
309 const Eigen::Matrix<double, Rows, States> C =
310 NumericalJacobianX<Rows, States, Inputs>(h, m_xHat, u);
311 const Eigen::Matrix<double, Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
312
313 Eigen::Matrix<double, Rows, Rows> S = C * m_P * C.transpose() + discR;
314
315 // We want to put K = PCᵀS⁻¹ into Ax = b form so we can solve it more
316 // efficiently.
317 //
318 // K = PCᵀS⁻¹
319 // KS = PCᵀ
320 // (KS)ᵀ = (PCᵀ)ᵀ
321 // SᵀKᵀ = CPᵀ
322 //
323 // The solution of Ax = b can be found via x = A.solve(b).
324 //
325 // Kᵀ = Sᵀ.solve(CPᵀ)
326 // K = (Sᵀ.solve(CPᵀ))ᵀ
327 Eigen::Matrix<double, States, Rows> K =
328 S.transpose().ldlt().solve(C * m_P.transpose()).transpose();
329
330 // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − h(x̂ₖ₊₁⁻, uₖ₊₁))
331 m_xHat = addFuncX(m_xHat, K * residualFuncY(y, h(m_xHat, u)));
332
333 // Pₖ₊₁⁺ = (I − KC)Pₖ₊₁⁻
334 m_P = (Eigen::Matrix<double, States, States>::Identity() - K * C) * m_P;
335 }
336
337 private:
338 std::function<Eigen::Vector<double, States>(
339 const Eigen::Vector<double, States>&,
340 const Eigen::Vector<double, Inputs>&)>
341 m_f;
342 std::function<Eigen::Vector<double, Outputs>(
343 const Eigen::Vector<double, States>&,
344 const Eigen::Vector<double, Inputs>&)>
345 m_h;
346 std::function<Eigen::Vector<double, Outputs>(
347 const Eigen::Vector<double, Outputs>&,
348 const Eigen::Vector<double, Outputs>)>
349 m_residualFuncY;
350 std::function<Eigen::Vector<double, States>(
351 const Eigen::Vector<double, States>&,
352 const Eigen::Vector<double, States>)>
353 m_addFuncX;
354 Eigen::Vector<double, States> m_xHat;
355 Eigen::Matrix<double, States, States> m_P;
356 Eigen::Matrix<double, States, States> m_contQ;
357 Eigen::Matrix<double, Outputs, Outputs> m_contR;
358 units::second_t m_dt;
359
360 Eigen::Matrix<double, States, States> m_initP;
361};
