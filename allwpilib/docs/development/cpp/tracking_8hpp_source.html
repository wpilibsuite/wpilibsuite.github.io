73 /// A \concept{concept_blockallocator,BlockAllocator} adapter that tracks another allocator using a \concept{concept_tracker,tracker}.
74 /// It wraps another \concept{concept_blockallocator,BlockAllocator} and calls the tracker function before forwarding to it.
75 /// The class can then be used anywhere a \concept{concept_blockallocator,BlockAllocator} is required and the memory usage will be tracked.<br>
76 /// It will only call the <tt>on_allocator_growth()</tt> and <tt>on_allocator_shrinking()</tt> tracking functions,
77 /// since a \concept{concept_blockallocator,BlockAllocator} is normally used inside higher allocators only.
78 /// \ingroup adapter
79 template <class Tracker, class BlockOrRawAllocator>
80 class tracked_block_allocator
81 : WPI_EBO(Tracker, make_block_allocator_t<BlockOrRawAllocator>)
82 {
83 public:
84 using allocator_type = make_block_allocator_t<BlockOrRawAllocator>;
85 using tracker = Tracker;
86

88 /// \effects Creates it by giving it a \concept{concept_tracker,tracker} and the tracked \concept{concept_rawallocator,RawAllocator}.
89 /// It will embed both objects.
90 explicit tracked_block_allocator(tracker t = {}) noexcept : tracker(detail::move(t)) {}
91
92 tracked_block_allocator(tracker t, allocator_type&& alloc) noexcept
93 : tracker(detail::move(t)), allocator_type(detail::move(alloc))
94 {
95 }
96 /// @}

99 /// The allocator will be constructed using \c block_size and \c args.
100 template <typename... Args>

102 : tracker(detail::move(t)), allocator_type(block_size, detail::forward<Args>(args)...)
103 {
104 }

106 /// \effects Calls <tt>Tracker::on_allocator_growth()</tt> after forwarding to the allocator.
