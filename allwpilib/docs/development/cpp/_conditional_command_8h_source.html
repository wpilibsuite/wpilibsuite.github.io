16/**
17 * Runs one of two commands, depending on the value of the given condition when
18 * this command is initialized. Does not actually schedule the selected command
19 * - rather, the command is run through this command; this ensures that the
20 * command will behave as expected if used as part of a CommandGroup. Requires
21 * the requirements of both commands, again to ensure proper functioning when
22 * used in a CommandGroup. If this is undesired, consider using
23 * ScheduleCommand.
24 *
25 * <p>As this command contains multiple component commands within it, it is
26 * technically a command group; the command instances that are passed to it
27 * cannot be added to any other groups, or scheduled individually.
28 *
29 * <p>As a rule, CommandGroups require the union of the requirements of their
30 * component commands.
31 *
32 * This class is provided by the NewCommands VendorDep
33 *
34 * @see ScheduleCommand
35 */
36class ConditionalCommand
37 : public CommandHelper<CommandBase, ConditionalCommand> {
38 public:
39 /**
40 * Creates a new ConditionalCommand.
41 *
42 * @param onTrue the command to run if the condition is true
43 * @param onFalse the command to run if the condition is false
44 * @param condition the condition to determine which command to run
45 */
46 template <class T1, class T2,
47 typename = std::enable_if_t<
48 std::is_base_of_v<Command, std::remove_reference_t<T1>>>,
49 typename = std::enable_if_t<
50 std::is_base_of_v<Command, std::remove_reference_t<T2>>>>
51 ConditionalCommand(T1&& onTrue, T2&& onFalse, std::function<bool()> condition)
52 : ConditionalCommand(std::make_unique<std::remove_reference_t<T1>>(
53 std::forward<T1>(onTrue)),
54 std::make_unique<std::remove_reference_t<T2>>(
55 std::forward<T2>(onFalse)),
56 condition) {}
57
58 /**
59 * Creates a new ConditionalCommand.
60 *
61 * @param onTrue the command to run if the condition is true
62 * @param onFalse the command to run if the condition is false
63 * @param condition the condition to determine which command to run
64 */
65 ConditionalCommand(std::unique_ptr<Command>&& onTrue,
66 std::unique_ptr<Command>&& onFalse,
67 std::function<bool()> condition);
68
69 ConditionalCommand(ConditionalCommand&& other) = default;
70
71 // No copy constructors for command groups
72 ConditionalCommand(const ConditionalCommand& other) = delete;
73
74 void Initialize() override;
75
76 void Execute() override;
77
78 void End(bool interrupted) override;
79
80 bool IsFinished() override;
81
82 bool RunsWhenDisabled() const override;
83
84 private:
85 std::unique_ptr<Command> m_onTrue;
86 std::unique_ptr<Command> m_onFalse;
87 std::function<bool()> m_condition;
88 Command* m_selectedCommand{nullptr};
89 bool m_runsWhenDisabled = true;
90};
91} // namespace frc2
