78  LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
79  const Eigen::Matrix<double, States, Inputs>& B,
80  const Eigen::Matrix<double, States, States>& Q,
81  const Eigen::Matrix<double, Inputs, Inputs>& R,
82  units::second_t dt) {
83  Eigen::Matrix<double, States, States> discA;
84  Eigen::Matrix<double, States, Inputs> discB;
85  DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
86 
87  Eigen::Matrix<double, States, States> S =
88  drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R);
89  Eigen::Matrix<double, Inputs, Inputs> tmp =
90  discB.transpose() * S * discB + R;
91  m_K = tmp.llt().solve(discB.transpose() * S * discA);
92 
93  Reset();
94  }

193  template <int Outputs>
194  void LatencyCompensate(const LinearSystem<States, Inputs, Outputs>& plant,
195  units::second_t dt, units::second_t inputDelay) {
196  Eigen::Matrix<double, States, States> discA;
197  Eigen::Matrix<double, States, Inputs> discB;
198  DiscretizeAB<States, Inputs>(plant.A(), plant.B(), dt, &discA, &discB);
199  m_K = m_K * (discA - discB * m_K).pow(inputDelay / dt);
200  }
