78  LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
79  const Eigen::Matrix<double, States, Inputs>& B,
80  const Eigen::Matrix<double, States, States>& Q,
81  const Eigen::Matrix<double, Inputs, Inputs>& R,
82  units::second_t dt) {
83  Eigen::Matrix<double, States, States> discA;
84  Eigen::Matrix<double, States, Inputs> discB;
85  DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
86 
87  Eigen::Matrix<double, States, States> S =
88  drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R);
89 
90  // K = (BᵀSB + R)⁻¹BᵀSA
91  m_K = (discB.transpose() * S * discB + R)
92  .llt()
93  .solve(discB.transpose() * S * discA);
94 
95  Reset();
96  }

108  LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
109  const Eigen::Matrix<double, States, Inputs>& B,
110  const Eigen::Matrix<double, States, States>& Q,
111  const Eigen::Matrix<double, Inputs, Inputs>& R,
112  const Eigen::Matrix<double, States, Inputs>& N,
113  units::second_t dt) {
114  Eigen::Matrix<double, States, States> discA;
115  Eigen::Matrix<double, States, Inputs> discB;
116  DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
117 
118  Eigen::Matrix<double, States, States> S =
119  drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R, N);
120 
121  // K = (BᵀSB + R)⁻¹(BᵀSA + Nᵀ)
122  m_K = (B.transpose() * S * B + R)
123  .llt()
124  .solve(discB.transpose() * S * discA + N.transpose());
125 
126  Reset();
127  }

226  template <int Outputs>
227  void LatencyCompensate(const LinearSystem<States, Inputs, Outputs>& plant,
228  units::second_t dt, units::second_t inputDelay) {
229  Eigen::Matrix<double, States, States> discA;
230  Eigen::Matrix<double, States, Inputs> discB;
231  DiscretizeAB<States, Inputs>(plant.A(), plant.B(), dt, &discA, &discB);
232 
233  m_K = m_K * (discA - discB * m_K).pow(inputDelay / dt);
234  }

239 
240  // Computed controller output
241  Eigen::Matrix<double, Inputs, 1> m_u;

243  // Controller gain
