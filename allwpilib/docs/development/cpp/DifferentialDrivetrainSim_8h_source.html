250  static DifferentialDrivetrainSim CreateKitbotSim(
251  frc::DCMotor motor, double gearing, units::meter_t wheelSize,
252  const std::array<double, 7>& measurementStdDevs = {}) {
253  // MOI estimation -- note that I = m r^2 for point masses
254  units::kilogram_square_meter_t batteryMoi = 12.5_lb * 10_in * 10_in;
255  units::kilogram_square_meter_t gearboxMoi = (2.8_lb + 2.0_lb) *
256  2 // CIM plus toughbox per side
257  * (26_in / 2) * (26_in / 2);
258 
259  return DifferentialDrivetrainSim{
260  motor, gearing, batteryMoi + gearboxMoi, 60_lb,
261  wheelSize / 2.0, 26_in, measurementStdDevs};
262  }
