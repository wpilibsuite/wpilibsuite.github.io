23/**
24 * Handle.
25 * Handles are not moveable or copyable and cannot be directly constructed.
26 * This class provides shared_ptr ownership and shared_from_this.
27 * Use the specific handle type Create() functions to create handles.
28 */
29class Handle : public std::enable_shared_from_this<Handle> {
30 public:
31 using Type = uv_handle_type;
32
33 Handle(const Handle&) = delete;
34 Handle(Handle&&) = delete;
35 Handle& operator=(const Handle&) = delete;
36 Handle& operator=(Handle&&) = delete;
37 virtual ~Handle() noexcept;
38
39 /**
40 * Get the type of the handle.
41 *
42 * A base handle offers no functionality to promote it to the actual handle
43 * type. By means of this function, the type of the underlying handle as
44 * specified by Type is made available.
45 *
46 * @return The actual type of the handle.
47 */
48 Type GetType() const noexcept { return m_uv_handle->type; }
49
50 /**
51 * Get the name of the type of the handle. E.g. "pipe" for pipe handles.
52 */
53 std::string_view GetTypeName() const noexcept {
54 return uv_handle_type_name(m_uv_handle->type);
55 }
56
57 /**
58 * Get the loop where this handle runs.
59 *
60 * @return The loop.
61 */
62 std::shared_ptr<Loop> GetLoop() const noexcept {
63 return GetLoopRef().shared_from_this();
64 }
65
66 /**
67 * Get the loop where this handle runs.
68 *
69 * @return The loop.
70 */
71 Loop& GetLoopRef() const noexcept {
72 return *static_cast<Loop*>(m_uv_handle->loop->data);
73 }
74
75 /**
76 * Check if the handle is active.
77 *
78 * What _active_ means depends on the type of handle:
79 *
80 * * An AsyncHandle handle is always active and cannot be deactivated,
81 * except by closing it with uv_close().
82 * * A PipeHandle, TcpHandle, UDPHandle, etc. handle - basically any handle
83 * that deals with I/O - is active when it is doing something that involves
84 * I/O, like reading, writing, connecting, accepting new connections, etc.
85 * * A CheckHandle, IdleHandle, TimerHandle, etc. handle is active when it
86 * has been started with a call to `Start()`.
87 *
88 * Rule of thumb: if a handle of type `FooHandle` has a `Start()` member
89 * method, then itâ€™s active from the moment that method is called. Likewise,
90 * `Stop()` deactivates the handle again.
91 *
92 * @return True if the handle is active, false otherwise.
93 */
94 bool IsActive() const noexcept { return uv_is_active(m_uv_handle) != 0; }
95
96 /**
97 * Check if a handle is closing or closed.
98 *
99 * This function should only be used between the initialization of the
100 * handle and the arrival of the close callback.
101 *
102 * @return True if the handle is closing or closed, false otherwise.
103 */
104 bool IsClosing() const noexcept {
105 return m_closed || uv_is_closing(m_uv_handle) != 0;
106 }
107
108 /**
109 * Request handle to be closed.
110 *
111 * This **must** be called on each handle before memory is released.
112 * In-progress requests are cancelled and this can result in error() being
113 * emitted.
114 *
115 * The handle will emit closed() when finished.
116 */
117 void Close() noexcept;
118
119 /**
120 * Set if the loop is closing.
121 *
122 * This is set during EventLoopRunner.Stop(), and can be used for other cases
123 * to indicate the loop should be closing. For instance for a uv_walk loop can
124 * use this to close existing handles.
125 *
126 * @param loopClosing true to set the loop currently in closing stages.
127 */
128 void SetLoopClosing(bool loopClosing) noexcept {
129 m_loopClosing = loopClosing;
130 }
131
132 /**
133 * Get the loop closing status.
134 *
135 * This can be used from closed() in order to tell if a closing loop is the
136 * reason for the close, or another reason.
137 *
138 * @return true if the loop is closing, otherwise false.
139 */
140 bool IsLoopClosing() const noexcept { return m_loopClosing; }
141
142 /**
143 * Reference the given handle.
144 *
145 * References are idempotent, that is, if a handle is already referenced
146 * calling this function again will have no effect.
147 */
148 void Reference() noexcept { uv_ref(m_uv_handle); }
149
150 /**
151 * Unreference the given handle.
152 *
153 * References are idempotent, that is, if a handle is not referenced calling
154 * this function again will have no effect.
155 */
156 void Unreference() noexcept { uv_unref(m_uv_handle); }
157
158 /**
159 * Check if the given handle is referenced.
160 * @return True if the handle is referenced, false otherwise.
161 */
162 bool HasReference() const noexcept { return uv_has_ref(m_uv_handle) != 0; }
163
164 /**
165 * Return the size of the underlying handle type.
166 * @return The size of the underlying handle type.
167 */
168 size_t RawSize() const noexcept { return uv_handle_size(m_uv_handle->type); }
169
170 /**
171 * Get the underlying handle data structure.
172 *
173 * @return The underlying handle data structure.
174 */
175 uv_handle_t* GetRawHandle() const noexcept { return m_uv_handle; }
176
177 /**
178 * Set the functions used for allocating and releasing buffers. The size
179 * passed to the allocator function is a "suggested" size--it's just an
180 * indication, not related in any way to the pending data to be read. The
181 * user is free to allocate the amount of memory they decide. For example,
182 * applications with custom allocation schemes may decide to use a different
183 * size which matches the memory chunks they already have for other purposes.
184 *
185 * @warning Be very careful changing the allocator after the loop has started
186 * running; there are no interlocks between this and buffers currently in
187 * flight.
188 *
189 * @param alloc Allocation function
190 * @param dealloc Deallocation function
191 */
192 void SetBufferAllocator(std::function<Buffer(size_t)> alloc,
193 std::function<void(Buffer&)> dealloc) {
194 m_allocBuf = std::move(alloc);
195 m_freeBuf = std::move(dealloc);
196 }
197
198 /**
199 * Free a buffer. Uses the function provided to SetBufFree() or
200 * Buffer::Deallocate by default.
201 *
202 * @param buf The buffer
203 */
204 void FreeBuf(Buffer& buf) const noexcept { m_freeBuf(buf); }
205
206 /**
207 * Gets user-defined data.
208 * @return User-defined data if any, nullptr otherwise.
209 */
210 template <typename T = void>
211 std::shared_ptr<T> GetData() const {
212 return std::static_pointer_cast<T>(m_data);
213 }
214
215 /**
216 * Sets user-defined data.
217 * @param data User-defined arbitrary data.
218 */
219 void SetData(std::shared_ptr<void> data) { m_data = std::move(data); }
220
221 /**
222 * Error signal
223 */
224 sig::Signal<Error> error;
225
226 /**
227 * Closed signal
228 */
229 sig::Signal<> closed;
230
231 /**
232 * Report an error.
233 * @param err Error code
234 */
235 void ReportError(int err) const { error(Error(err)); }
236
237 protected:
238 explicit Handle(uv_handle_t* uv_handle) : m_uv_handle{uv_handle} {
239 m_uv_handle->data = this;
240 }
241
242 void Keep() noexcept { m_self = shared_from_this(); }
243 void Release() noexcept { m_self.reset(); }
244 void ForceClosed() noexcept { m_closed = true; }
245
246 static void AllocBuf(uv_handle_t* handle, size_t size, uv_buf_t* buf);
247 static void DefaultFreeBuf(Buffer& buf);
248
249 template <typename F, typename... Args>
250 bool Invoke(F&& f, Args&&... args) const {
251 auto err = std::forward<F>(f)(std::forward<Args>(args)...);
252 if (err < 0) {
253 ReportError(err);
254 }
255 return err == 0;
256 }
257
258 private:
259 std::shared_ptr<Handle> m_self;
260 uv_handle_t* m_uv_handle;
261 bool m_closed = false;
262 bool m_loopClosing = false;
263 std::function<Buffer(size_t)> m_allocBuf{&Buffer::Allocate};
264 std::function<void(Buffer&)> m_freeBuf{&DefaultFreeBuf};
265 std::shared_ptr<void> m_data;
266};
267
268/**
269 * Handle.
270 */
271template <typename T, typename U>
272class HandleImpl : public Handle {
273 public:
274 std::shared_ptr<T> shared_from_this() {
275 return std::static_pointer_cast<T>(Handle::shared_from_this());
276 }
277
278 std::shared_ptr<const T> shared_from_this() const {
279 return std::static_pointer_cast<const T>(Handle::shared_from_this());
280 }
281
282 /**
283 * Get the underlying handle data structure.
284 *
285 * @return The underlying handle data structure.
286 */
287 U* GetRaw() const noexcept {
288 return reinterpret_cast<U*>(this->GetRawHandle());
289 }
290
291 protected:
292 HandleImpl() : Handle{static_cast<uv_handle_t*>(std::malloc(sizeof(U)))} {}
293};
