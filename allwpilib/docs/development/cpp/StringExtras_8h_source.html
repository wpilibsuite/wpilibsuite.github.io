34constexpr char hexdigit(unsigned X, bool LowerCase = false) noexcept {
35 const char HexChar = LowerCase ? 'a' : 'A';
36 return X < 10 ? '0' + X : HexChar + X - 10;
37}

43constexpr unsigned hexDigitValue(char C) noexcept {
44 if (C >= '0' && C <= '9') {
45 return C - '0';
46 }
47 if (C >= 'a' && C <= 'f') {
48 return C - 'a' + 10U;
49 }
50 if (C >= 'A' && C <= 'F') {
51 return C - 'A' + 10U;
52 }
53 return (std::numeric_limits<unsigned>::max)();
54}

107inline std::string utohexstr(unsigned long long val, // NOLINT(runtime/int)
108 bool lowerCase = false) {
109 char buf[17];
110 char* bufptr = std::end(buf);
111
112 if (val == 0) {
113 *--bufptr = '0';
114 }
115
116 while (val) {
117 unsigned char mod = static_cast<unsigned char>(val) & 15;
118 *--bufptr = hexdigit(mod, lowerCase);
119 val >>= 4;
120 }
121
122 return std::string(bufptr, std::end(buf));
123}

370constexpr std::string_view slice(std::string_view str,
371 std::string_view::size_type start,
372 std::string_view::size_type end) noexcept {
373 auto length = str.size();
374 start = (std::min)(start, length);
375 end = (std::min)((std::max)(start, end), length);
376 return {str.data() + start, end - start};
377}

582template <typename T,
583 std::enable_if_t<std::numeric_limits<T>::is_signed, bool> = true>
584inline std::optional<T> parse_integer(std::string_view str,
585 unsigned radix) noexcept {
586 long long val; // NOLINT(runtime/int)
587 if (detail::GetAsSignedInteger(str, radix, val) ||
588 static_cast<T>(val) != val) {
589 return std::nullopt;
590 }
591 return val;
592}

594template <typename T,
595 std::enable_if_t<!std::numeric_limits<T>::is_signed, bool> = true>
596inline std::optional<T> parse_integer(std::string_view str,
597 unsigned radix) noexcept {
598 using Int = unsigned long long; // NOLINT(runtime/int)
599 Int val;
600 // The additional cast to unsigned long long is required to avoid the
601 // Visual C++ warning C4805: '!=' : unsafe mix of type 'bool' and type
602 // 'unsigned __int64' when instantiating getAsInteger with T = bool.
603 if (detail::GetAsUnsignedInteger(str, radix, val) ||
604 static_cast<Int>(static_cast<T>(val)) != val) {
605 return std::nullopt;
606 }
607 return val;
608}
