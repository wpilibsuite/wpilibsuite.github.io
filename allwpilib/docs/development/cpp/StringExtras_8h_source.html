115 inline std::string utohexstr(uint64_t X, bool LowerCase = false) {
116  char Buffer[17];
117  char *BufPtr = std::end(Buffer);
118 
119  if (X == 0) *--BufPtr = '0';
120 
121  while (X) {
122  unsigned char Mod = static_cast<unsigned char>(X) & 15;
123  *--BufPtr = hexdigit(Mod, LowerCase);
124  X >>= 4;
125  }
126 
127  return std::string(BufPtr, std::end(Buffer));
128 }

132 inline std::string toHex(StringRef Input, bool LowerCase = false) {
133  static const char *const LUT = "0123456789ABCDEF";
134  const uint8_t Offset = LowerCase ? 32 : 0;
135  size_t Length = Input.size();
136 
137  std::string Output;
138  Output.reserve(2 * Length);
139  for (size_t i = 0; i < Length; ++i) {
140  const unsigned char c = Input[i];
141  Output.push_back(LUT[c >> 4] | Offset);
142  Output.push_back(LUT[c & 15] | Offset);
143  }
144  return Output;
145 }

151 inline uint8_t hexFromNibbles(char MSB, char LSB) {
152  unsigned U1 = hexDigitValue(MSB);
153  unsigned U2 = hexDigitValue(LSB);
154  assert(U1 != (std::numeric_limits<unsigned>::max)() && U2 != (std::numeric_limits<unsigned>::max)());
155 
156  return static_cast<uint8_t>((U1 << 4) | U2);
157 }

161 inline std::string fromHex(StringRef Input) {
162  if (Input.empty())
163  return std::string();
164 
165  std::string Output;
166  Output.reserve((Input.size() + 1) / 2);
167  if (Input.size() % 2 == 1) {
168  Output.push_back(hexFromNibbles('0', Input.front()));
169  Input = Input.drop_front();
170  }
171 
172  assert(Input.size() % 2 == 0);
173  while (!Input.empty()) {
174  uint8_t Hex = hexFromNibbles(Input[0], Input[1]);
175  Output.push_back(Hex);
176  Input = Input.drop_front(2);
177  }
178  return Output;
179 }

302 inline std::string join_impl(IteratorT Begin, IteratorT End,
303  StringRef Separator, std::input_iterator_tag) {
304  std::string S;
305  if (Begin == End)
306  return S;
307 
308  S += (*Begin);
309  while (++Begin != End) {
310  S += Separator;
311  S += (*Begin);
312  }
313  return S;
314 }

316 template <typename IteratorT>
317 inline std::string join_impl(IteratorT Begin, IteratorT End,
318  StringRef Separator, std::forward_iterator_tag) {
319  std::string S;
320  if (Begin == End)
321  return S;
322 
323  size_t Len = (std::distance(Begin, End) - 1) * Separator.size();
324  for (IteratorT I = Begin; I != End; ++I)
325  Len += (*Begin).size();
326  S.reserve(Len);
327  S += (*Begin);
328  while (++Begin != End) {
329  S += Separator;
330  S += (*Begin);
331  }
332  return S;
333 }

390 template <typename Sep, typename... Args>
391 inline std::string join_items(Sep Separator, Args &&... Items) {
392  std::string Result;
393  if (sizeof...(Items) == 0)
394  return Result;
395 
396  size_t NS = detail::join_one_item_size(Separator);
397  size_t NI = detail::join_items_size(std::forward<Args>(Items)...);
398  Result.reserve(NI + (sizeof...(Items) - 1) * NS + 1);
399  detail::join_items_impl(Result, Separator, std::forward<Args>(Items)...);
400  return Result;
401 }
