15namespace frc {
16/**
17 * Helper class that converts a chassis velocity (dx and dtheta components) to
18 * left and right wheel velocities for a differential drive.
19 *
20 * Inverse kinematics converts a desired chassis speed into left and right
21 * velocity components whereas forward kinematics converts left and right
22 * component velocities into a linear and angular chassis speed.
23 */
24class WPILIB_DLLEXPORT DifferentialDriveKinematics {
25 public:
26 /**
27 * Constructs a differential drive kinematics object.
28 *
29 * @param trackWidth The track width of the drivetrain. Theoretically, this is
30 * the distance between the left wheels and right wheels. However, the
31 * empirical value may be larger than the physical measured value due to
32 * scrubbing effects.
33 */
34 explicit DifferentialDriveKinematics(units::meter_t trackWidth)
35 : trackWidth(trackWidth) {
36 wpi::math::MathSharedStore::ReportUsage(
37 wpi::math::MathUsageId::kKinematics_DifferentialDrive, 1);
38 }
39
40 /**
41 * Returns a chassis speed from left and right component velocities using
42 * forward kinematics.
43 *
44 * @param wheelSpeeds The left and right velocities.
45 * @return The chassis speed.
46 */
47 constexpr ChassisSpeeds ToChassisSpeeds(
48 const DifferentialDriveWheelSpeeds& wheelSpeeds) const {
49 return {(wheelSpeeds.left + wheelSpeeds.right) / 2.0, 0_mps,
50 (wheelSpeeds.right - wheelSpeeds.left) / trackWidth * 1_rad};
51 }
52
53 /**
54 * Returns left and right component velocities from a chassis speed using
55 * inverse kinematics.
56 *
57 * @param chassisSpeeds The linear and angular (dx and dtheta) components that
58 * represent the chassis' speed.
59 * @return The left and right velocities.
60 */
61 constexpr DifferentialDriveWheelSpeeds ToWheelSpeeds(
62 const ChassisSpeeds& chassisSpeeds) const {
63 return {chassisSpeeds.vx - trackWidth / 2 * chassisSpeeds.omega / 1_rad,
64 chassisSpeeds.vx + trackWidth / 2 * chassisSpeeds.omega / 1_rad};
65 }
66
67 units::meter_t trackWidth;
68};
69} // namespace frc
