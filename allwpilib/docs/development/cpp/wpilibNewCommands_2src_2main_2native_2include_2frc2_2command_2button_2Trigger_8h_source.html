287 template <class T, typename = std::enable_if_t<std::is_base_of_v<
288 Command, std::remove_reference_t<T>>>>
289 Trigger ToggleWhenActive(T&& command, bool interruptible = true) {
290 CommandScheduler::GetInstance().AddButton(
291 [pressedLast = m_isActive(), *this,
292 command = std::make_unique<std::remove_reference_t<T>>(
293 std::forward<T>(command)),
294 interruptible]() mutable {
295 bool pressed = m_isActive();
296
297 if (!pressedLast && pressed) {
298 if (command->IsScheduled()) {
299 command->Cancel();
300 } else {
301 command->Schedule(interruptible);
302 }
303 }
304
305 pressedLast = pressed;
306 });
307 return *this;
308 }
