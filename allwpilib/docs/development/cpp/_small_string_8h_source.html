23/// SmallString - A SmallString is just a SmallVector with methods and accessors
24/// that make it work better as a string (e.g. operator+ etc).
25template<unsigned InternalLen>
26class SmallString : public SmallVector<char, InternalLen> {
27public:
28 /// Default ctor - Initialize to empty.
29 SmallString() = default;
30
31 /// Initialize from a std::string_view.
32 SmallString(std::string_view S)
33 : SmallVector<char, InternalLen>(S.begin(), S.end()) {}
34
35 /// Initialize with a range.
36 template<typename ItTy>
37 SmallString(ItTy S, ItTy E) : SmallVector<char, InternalLen>(S, E) {}

185 // Extra methods.
186
187 /// Explicit conversion to std::string_view.
188 std::string_view str() const { return {this->begin(), this->size()}; }
189
190 /// Explicit conversion to std::string.

206 // Extra operators.
207 const SmallString &operator=(std::string_view RHS) {
208 this->clear();

210 }
211
212 SmallString &operator+=(std::string_view RHS) {
213 this->append(RHS.begin(), RHS.end());
214 return *this;
215 }
216 SmallString &operator+=(char C) {
217 this->push_back(C);
218 return *this;
219 }
