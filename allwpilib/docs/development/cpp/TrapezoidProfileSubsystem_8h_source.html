18template <class Distance>

20 using Distance_t = units::unit_t<Distance>;

24 using State = typename frc::TrapezoidProfile<Distance>::State;
25 using Constraints = typename frc::TrapezoidProfile<Distance>::Constraints;
26
27 public:
37 explicit TrapezoidProfileSubsystem(Constraints constraints,
38 Distance_t initialPosition = Distance_t{0},
39 units::second_t period = 20_ms)
40 : m_constraints(constraints),
41 m_state{initialPosition, Velocity_t(0)},
42 m_goal{initialPosition, Velocity_t{0}},
43 m_period(period) {}
44
45 void Periodic() override {
46 auto profile =
47 frc::TrapezoidProfile<Distance>(m_constraints, m_goal, m_state);
48 m_state = profile.Calculate(m_period);
49 if (m_enabled) {
50 UseState(m_state);
51 }
52 }
53
59 void SetGoal(State goal) { m_goal = goal; }
60
66 void SetGoal(Distance_t goal) { m_goal = State{goal, Velocity_t(0)}; }
67
68 protected:
75 virtual void UseState(State state) = 0;
76
80 void Enable() { m_enabled = true; }
81
85 void Disable() { m_enabled = false; }
86
87 private:
88 Constraints m_constraints;
89 State m_state;
90 State m_goal;
91 units::second_t m_period;
92 bool m_enabled{false};
