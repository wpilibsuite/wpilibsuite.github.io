21/**
22 * Visual 2D representation of arms, elevators, and general mechanisms through
23 * a node-based API.
24 *
25 * A Mechanism2d object is published and contains at least one root node. A root
26 * is the anchor point of other nodes (such as ligaments). Other nodes are
27 * recursively appended based on other nodes.
28 *
29 * Except for the Mechanism2d container object, none of the objects should be
30 * passed or interacted with by value! Obtain pointers from factory methods such
31 * as Mechanism2d.GetRoot() and MechanismObject2d.Append<>(). The Mechanism2d
32 * container object owns the root nodes, and each node internally owns the nodes
33 * based on it. Beware not to let the Mechanism2d object out of scope - all
34 * nodes will be recursively destructed!
35 *
36 * @see MechanismObject2d
37 * @see MechanismLigament2d
38 * @see MechanismRoot2d
39 */
40class Mechanism2d : public nt::NTSendable,
41 public wpi::SendableHelper<Mechanism2d> {
42 public:
43 /**
44 * Create a new Mechanism2d with the given dimensions and background color.
45 *
46 * The dimensions represent the canvas that all the nodes are drawn on. The
47 * default color is dark blue.
48 *
49 * @param width the width
50 * @param height the height
51 * @param backgroundColor the background color
52 */
53 Mechanism2d(double width, double height,
54 const Color8Bit& backgroundColor = {0, 0, 32});
55
56 /**
57 * Get or create a root in this Mechanism2d with the given name and
58 * position.
59 *
60 * <p>If a root with the given name already exists, the given x and y
61 * coordinates are not used.
62 *
63 * @param name the root name
64 * @param x the root x coordinate
65 * @param y the root y coordinate
66 * @return a new root object, or the existing one with the given name.
67 */
68 MechanismRoot2d* GetRoot(std::string_view name, double x, double y);
69
70 /**
71 * Set the Mechanism2d background color.
72 *
73 * @param color the new background color
74 */
75 void SetBackgroundColor(const Color8Bit& color);
76
77 void InitSendable(nt::NTSendableBuilder& builder) override;
78
79 private:
80 double m_width;
81 double m_height;
82 char m_color[10];
83 mutable wpi::mutex m_mutex;
84 std::shared_ptr<nt::NetworkTable> m_table;
85 wpi::StringMap<std::unique_ptr<MechanismRoot2d>> m_roots;
86};
87} // namespace frc
