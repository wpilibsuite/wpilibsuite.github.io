64struct VideoMode : public CS_VideoMode {
65 enum PixelFormat {
66 kUnknown = CS_PIXFMT_UNKNOWN,
67 kMJPEG = CS_PIXFMT_MJPEG,
68 kYUYV = CS_PIXFMT_YUYV,
69 kRGB565 = CS_PIXFMT_RGB565,
70 kBGR = CS_PIXFMT_BGR,
71 kGray = CS_PIXFMT_GRAY
72 };
73 VideoMode() {
74 pixelFormat = 0;
75 width = 0;
76 height = 0;
77 fps = 0;
78 }
79 VideoMode(PixelFormat pixelFormat_, int width_, int height_, int fps_) {
80 pixelFormat = pixelFormat_;
81 width = width_;
82 height = height_;
83 fps = fps_;
84 }
85 explicit operator bool() const { return pixelFormat == kUnknown; }
86
87 bool operator==(const VideoMode& other) const {
88 return pixelFormat == other.pixelFormat && width == other.width &&
89 height == other.height && fps == other.fps;
90 }
91
92 bool operator!=(const VideoMode& other) const { return !(*this == other); }
93};

122 RawEvent() = default;
123 explicit RawEvent(RawEvent::Kind kind_) : kind{kind_} {}
124 RawEvent(std::string_view name_, CS_Handle handle_, RawEvent::Kind kind_)
125 : kind{kind_}, name{name_} {
126 if (kind_ == kSinkCreated || kind_ == kSinkDestroyed ||
127 kind_ == kSinkEnabled || kind_ == kSinkDisabled) {
128 sinkHandle = handle_;
129 } else {
130 sourceHandle = handle_;
131 }
132 }
133 RawEvent(std::string_view name_, CS_Source source_, const VideoMode& mode_)
134 : kind{kSourceVideoModeChanged},
135 sourceHandle{source_},
136 name{name_},
137 mode{mode_} {}
138 RawEvent(std::string_view name_, CS_Source source_, RawEvent::Kind kind_,
139 CS_Property property_, CS_PropertyKind propertyKind_, int value_,
140 std::string_view valueStr_)
141 : kind{kind_},
142 sourceHandle{source_},
143 name{name_},
144 propertyHandle{property_},
145 propertyKind{propertyKind_},
146 value{value_},
147 valueStr{valueStr_} {}
148
149 Kind kind;
150
151 // Valid for kSource* and kSink* respectively
152 CS_Source sourceHandle = CS_INVALID_HANDLE;
153 CS_Sink sinkHandle = CS_INVALID_HANDLE;

155 // Source/sink/property name
156 std::string name;
157
158 // Fields for kSourceVideoModeChanged event
159 VideoMode mode;
160
161 // Fields for kSourceProperty* events
162 CS_Property propertyHandle;
163 CS_PropertyKind propertyKind;
164 int value;
165 std::string valueStr;
166
167 // Listener that was triggered
168 CS_Listener listener{0};
169};
