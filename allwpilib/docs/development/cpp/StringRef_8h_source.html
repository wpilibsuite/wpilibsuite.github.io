189  int compare(StringRef RHS) const noexcept {
190  // Check the prefix for a mismatch.
191  if (int Res = compareMemory(Data, RHS.Data, (std::min)(Length, RHS.Length)))
192  return Res < 0 ? -1 : 1;
193 
194  // Otherwise the prefixes match, so we only need to check the lengths.
195  if (Length == RHS.Length)
196  return 0;
197  return Length < RHS.Length ? -1 : 1;
198  }

284  LLVM_ATTRIBUTE_ALWAYS_INLINE
285  size_t find(char C, size_t From = 0) const noexcept {
286  size_t FindBegin = (std::min)(From, Length);
287  if (FindBegin < Length) { // Avoid calling memchr with nullptr.
288  // Just forward to memchr, which is faster than a hand-rolled loop.
289  if (const void *P = ::memchr(Data + FindBegin, C, Length - FindBegin))
290  return static_cast<const char *>(P) - Data;
291  }
292  return npos;
293  }

483  typename std::enable_if<std::numeric_limits<T>::is_signed, bool>::type
484  getAsInteger(unsigned Radix, T &Result) const noexcept {
485  long long LLVal;
486  if (getAsSignedInteger(*this, Radix, LLVal) ||

493  template <typename T>
494  typename std::enable_if<!std::numeric_limits<T>::is_signed, bool>::type
495  getAsInteger(unsigned Radix, T &Result) const noexcept {
496  unsigned long long ULLVal;
497  // The additional cast to unsigned long long is required to avoid the
498  // Visual C++ warning C4805: '!=' : unsafe mix of type 'bool' and type
499  // 'unsigned __int64' when instantiating getAsInteger with T = bool.
500  if (getAsUnsignedInteger(*this, Radix, ULLVal) ||
501  static_cast<unsigned long long>(static_cast<T>(ULLVal)) != ULLVal)
502  return true;
503  Result = ULLVal;
504  return false;
505  }
