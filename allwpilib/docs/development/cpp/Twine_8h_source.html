215  bool isValid() const {
216  // Nullary twines always have Empty on the RHS.
217  if (isNullary() && getRHSKind() != EmptyKind)
218  return false;
219 
220  // Null should never appear on the RHS.
221  if (getRHSKind() == NullKind)
222  return false;
223 
224  // The RHS cannot be non-empty if the LHS is empty.
225  if (getRHSKind() != EmptyKind && getLHSKind() == EmptyKind)
226  return false;
227 #if 0 // causes spurious warnings
228  // A twine child should always be binary.
229  if (getLHSKind() == TwineKind &&
230  !LHS.twine->isBinary())
231  return false;
232  if (getRHSKind() == TwineKind &&
233  !RHS.twine->isBinary())
234  return false;
235 #endif
236  return true;
237  }

459  StringRef getSingleStringRef() const {
460  assert(isSingleStringRef() &&"This cannot be had as a single stringref!");

512  inline Twine Twine::concat(const Twine &Suffix) const {
513  // Concatenation with null is null.
514  if (isNull() || Suffix.isNull())
515  return Twine(NullKind);
516 
517  // Concatenation with empty yields the other side.
518  if (isEmpty())
519  return Suffix;
520  if (Suffix.isEmpty())
521  return *this;
522 
523  // Otherwise we need to create a new node, taking care to fold in unary
524  // twines.
525  Child NewLHS, NewRHS;
526  NewLHS.twine = this;
527  NewRHS.twine = &Suffix;
528  NodeKind NewLHSKind = TwineKind, NewRHSKind = TwineKind;
529  if (isUnary()) {
530  NewLHS = LHS;
531  NewLHSKind = getLHSKind();
532  }
533  if (Suffix.isUnary()) {
534  NewRHS = Suffix.LHS;
535  NewRHSKind = Suffix.getLHSKind();
536  }
537 
538  return Twine(NewLHS, NewLHSKind, NewRHS, NewRHSKind);
539  }
