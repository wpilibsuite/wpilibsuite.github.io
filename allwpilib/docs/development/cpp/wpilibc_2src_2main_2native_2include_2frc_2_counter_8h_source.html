23/**
24 * Class for counting the number of ticks on a digital input channel.
25 *
26 * This is a general purpose class for counting repetitive events. It can return
27 * the number of counts, the period of the most recent cycle, and detect when
28 * the signal being counted has stopped by supplying a maximum cycle time.
29 *
30 * All counters will immediately start counting - Reset() them if you need them
31 * to be zeroed before use.
32 */
33class Counter : public CounterBase,
34 public wpi::Sendable,
35 public wpi::SendableHelper<Counter> {
36 friend class DMA;
37 friend class DMASample;

373 /**
374 * Set the maximum period where the device is still considered "moving".
375 *
376 * Sets the maximum period where the device is considered moving. This value
377 * is used to determine the "stopped" state of the counter using the
378 * GetStopped method.
379 *
380 * @param maxPeriod The maximum period where the counted device is considered
381 * moving in seconds.
382 */
383 void SetMaxPeriod(units::second_t maxPeriod) final;
384
385 /**
386 * Select whether you want to continue updating the event timer output when
387 * there are no samples captured.
388 *
389 * The output of the event timer has a buffer of periods that are averaged and
390 * posted to a register on the FPGA. When the timer detects that the event
391 * source has stopped (based on the MaxPeriod) the buffer of samples to be
392 * averaged is emptied. If you enable the update when empty, you will be
393 * notified of the stopped source and the event time will report 0 samples.
394 * If you disable update when empty, the most recent average will remain on
395 * the output until a new sample is acquired. You will never see 0 samples
396 * output (except when there have been no events since an FPGA reset) and you
397 * will likely not see the stopped bit become true (since it is updated at the
398 * end of an average and there are no samples to average).
399 *
400 * @param enabled True to enable update when empty
401 */
402 void SetUpdateWhenEmpty(bool enabled);
403
404 /**
405 * Determine if the clock is stopped.
406 *
407 * Determine if the clocked input is stopped based on the MaxPeriod value set
408 * using the SetMaxPeriod method. If the clock exceeds the MaxPeriod, then the
409 * device (and counter) are assumed to be stopped and it returns true.
410 *
411 * @return Returns true if the most recent counter period exceeds the
412 * MaxPeriod value set by SetMaxPeriod.
413 */
414 bool GetStopped() const override;
415

421 bool GetDirection() const override;
