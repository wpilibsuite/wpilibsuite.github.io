13/**
14 * Computes unscented transform of a set of sigma points and weights. CovDim
15 * returns the mean and covariance in a tuple.
16 *
17 * This works in conjunction with the UnscentedKalmanFilter class.
18 *
19 * @tparam CovDim Dimension of covariance of sigma points after passing
20 * through the transform.
21 * @tparam States Number of states.
22 * @param sigmas List of sigma points.
23 * @param Wm Weights for the mean.
24 * @param Wc Weights for the covariance.
25 * @param meanFunc A function that computes the mean of 2 * States + 1 state
26 * vectors using a given set of weights.
27 * @param residualFunc A function that computes the residual of two state
28 * vectors (i.e. it subtracts them.)
29 *
30 * @return Tuple of x, mean of sigma points; P, covariance of sigma points after
31 * passing through the transform.
32 */
33template <int CovDim, int States>
34std::tuple<Eigen::Vector<double, CovDim>, Eigen::Matrix<double, CovDim, CovDim>>
35UnscentedTransform(const Eigen::Matrix<double, CovDim, 2 * States + 1>& sigmas,
36 const Eigen::Vector<double, 2 * States + 1>& Wm,
37 const Eigen::Vector<double, 2 * States + 1>& Wc,
38 std::function<Eigen::Vector<double, CovDim>(
39 const Eigen::Matrix<double, CovDim, 2 * States + 1>&,
40 const Eigen::Vector<double, 2 * States + 1>&)>
41 meanFunc,
42 std::function<Eigen::Vector<double, CovDim>(
43 const Eigen::Vector<double, CovDim>&,
44 const Eigen::Vector<double, CovDim>&)>
45 residualFunc) {
46 // New mean is usually just the sum of the sigmas * weight:
47 // n
48 // dot = Σ W[k] Xᵢ[k]
49 // k=1
50 Eigen::Vector<double, CovDim> x = meanFunc(sigmas, Wm);
51
52 // New covariance is the sum of the outer product of the residuals times the
53 // weights
54 Eigen::Matrix<double, CovDim, 2 * States + 1> y;
55 for (int i = 0; i < 2 * States + 1; ++i) {
56 // y[:, i] = sigmas[:, i] - x
57 y.template block<CovDim, 1>(0, i) =
58 residualFunc(sigmas.template block<CovDim, 1>(0, i), x);
59 }
60 Eigen::Matrix<double, CovDim, CovDim> P =
61 y * Eigen::DiagonalMatrix<double, 2 * States + 1>(Wc) * y.transpose();
62
63 return std::make_tuple(x, P);
64}
