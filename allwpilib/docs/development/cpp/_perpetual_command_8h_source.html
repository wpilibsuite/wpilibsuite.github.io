20/**
21 * A command that runs another command in perpetuity, ignoring that command's
22 * end conditions. While this class does not extend {@link CommandGroupBase},
23 * it is still considered a CommandGroup, as it allows one to compose another
24 * command within it; the command instances that are passed to it cannot be
25 * added to any other groups, or scheduled individually.
26 *
27 * <p>As a rule, CommandGroups require the union of the requirements of their
28 * component commands.
29 *
30 * This class is provided by the NewCommands VendorDep
31 *
32 * @deprecated PerpetualCommand violates the assumption that execute() doesn't
33get called after isFinished() returns true -- an assumption that should be
34valid. This was unsafe/undefined behavior from the start, and RepeatCommand
35provides an easy way to achieve similar end results with slightly different (and
36safe) semantics.
37 */
38class PerpetualCommand : public CommandHelper<CommandBase, PerpetualCommand> {
39 public:
40 /**
41 * Creates a new PerpetualCommand. Will run another command in perpetuity,
42 * ignoring that command's end conditions, unless this command itself is
43 * interrupted.
44 *
45 * @param command the command to run perpetually
46 */
47 WPI_DEPRECATED(
48 "PerpetualCommand violates the assumption that execute() doesn't get "
49 "called after isFinished() returns true -- an assumption that should be "
50 "valid."
51 "This was unsafe/undefined behavior from the start, and RepeatCommand "
52 "provides an easy way to achieve similar end results with slightly "
53 "different (and safe) semantics.")
54 explicit PerpetualCommand(std::unique_ptr<Command>&& command);
55 WPI_IGNORE_DEPRECATED
56
57 /**
58 * Creates a new PerpetualCommand. Will run another command in perpetuity,
59 * ignoring that command's end conditions, unless this command itself is
60 * interrupted.
61 *
62 * @param command the command to run perpetually
63 */
64 template <class T, typename = std::enable_if_t<std::is_base_of_v<
65 Command, std::remove_reference_t<T>>>>
66 WPI_DEPRECATED(
67 "PerpetualCommand violates the assumption that execute() doesn't get "
68 "called after isFinished() returns true -- an assumption that should be "
69 "valid."
70 "This was unsafe/undefined behavior from the start, and RepeatCommand "
71 "provides an easy way to achieve similar end results with slightly "
72 "different (and safe) semantics.")
73 explicit PerpetualCommand(T&& command)
74 : PerpetualCommand(std::make_unique<std::remove_reference_t<T>>(
75 std::forward<T>(command))) {}
76 WPI_UNIGNORE_DEPRECATED
77
78 PerpetualCommand(PerpetualCommand&& other) = default;
79
80 // No copy constructors for command groups
81 PerpetualCommand(const PerpetualCommand& other) = delete;
82
83 // Prevent template expansion from emulating copy ctor
84 PerpetualCommand(PerpetualCommand&) = delete;
85
86 void Initialize() override;
87
88 void Execute() override;
89
90 void End(bool interrupted) override;
91
92 private:
93 std::unique_ptr<Command> m_command;
94};
95} // namespace frc2
