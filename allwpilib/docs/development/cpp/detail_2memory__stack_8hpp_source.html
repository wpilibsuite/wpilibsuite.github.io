62 debug_magic m = debug_magic::new_memory) noexcept
63 {
64 auto memory = cur_;
65 detail::debug_fill(memory, offset, m);
66 cur_ += offset;
67 return memory;
68 }

70 // allocates memory by advancing the stack, returns nullptr if insufficient
71 // debug: mark memory as new_memory, put fence in front and back
72 void* allocate(const char* end, std::size_t size, std::size_t alignment,
73 std::size_t fence_size = debug_fence_size) noexcept
74 {
75 if (cur_ == nullptr)
76 return nullptr;
77
78 auto remaining = std::size_t(end - cur_);
79 auto offset = align_offset(cur_ + fence_size, alignment);
80 if (fence_size + offset + size + fence_size > remaining)
81 return nullptr;
82
83 return allocate_unchecked(size, offset, fence_size);
84 }
85
86 // same as allocate() but does not check the size
87 // note: pass it the align OFFSET, not the alignment
88 void* allocate_unchecked(std::size_t size, std::size_t align_offset,
89 std::size_t fence_size = debug_fence_size) noexcept
90 {
91 bump(fence_size, debug_magic::fence_memory);
92 bump(align_offset, debug_magic::alignment_memory);
93 auto mem = bump_return(size);
94 bump(fence_size, debug_magic::fence_memory);
95 return mem;
96 }
97
98 // unwindws the stack to a certain older position
99 // debug: marks memory from new top to old top as freed
100 // doesn't check for invalid pointer
101 void unwind(char* top) noexcept
102 {
103 debug_fill(top, std::size_t(cur_ - top), debug_magic::freed_memory);
104 cur_ = top;
105 }
106
107 // returns the current top
108 char* top() const noexcept
109 {
110 return cur_;
111 }
112
113 private:
114 char* cur_;
115 };
