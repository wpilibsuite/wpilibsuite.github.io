23/**
24 * Runs one of a selection of commands, either using a selector and a key to
25 * command mapping, or a supplier that returns the command directly at runtime.
26 * Does not actually schedule the selected command - rather, the command is run
27 * through this command; this ensures that the command will behave as expected
28 * if used as part of a CommandGroup. Requires the requirements of all included
29 * commands, again to ensure proper functioning when used in a CommandGroup. If
30 * this is undesired, consider using ScheduleCommand.
31 *
32 * <p>As this command contains multiple component commands within it, it is
33 * technically a command group; the command instances that are passed to it
34 * cannot be added to any other groups, or scheduled individually.
35 *
36 * <p>As a rule, CommandGroups require the union of the requirements of their
37 * component commands.
38 */
39template <typename Key>
40class SelectCommand : public CommandHelper<CommandBase, SelectCommand<Key>> {
41 public:
42 /**
43 * Creates a new selectcommand.
44 *
45 * @param commands the map of commands to choose from
46 * @param selector the selector to determine which command to run
47 */
48 template <class... Types,
49 typename = std::enable_if_t<std::conjunction_v<
50 std::is_base_of<Command, std::remove_reference_t<Types>>...>>>
51 explicit SelectCommand(std::function<Key()> selector,
52 std::pair<Key, Types>... commands)
53 : m_selector{std::move(selector)} {
54 std::vector<std::pair<Key, std::unique_ptr<Command>>> foo;
55
56 ((void)foo.emplace_back(commands.first,
57 std::make_unique<std::remove_reference_t<Types>>(
58 std::move(commands.second))),
59 ...);
60
61 for (auto&& command : foo) {
62 if (!CommandGroupBase::RequireUngrouped(*command.second)) {
63 return;
64 }
65 }
66
67 for (auto&& command : foo) {
68 this->AddRequirements(command.second->GetRequirements());
69 m_runsWhenDisabled &= command.second->RunsWhenDisabled();
70 m_commands.emplace(std::move(command.first), std::move(command.second));
71 }
72 }
73
74 SelectCommand(
75 std::function<Key()> selector,
76 std::vector<std::pair<Key, std::unique_ptr<Command>>>&& commands)
77 : m_selector{std::move(selector)} {
78 for (auto&& command : commands) {
79 if (!CommandGroupBase::RequireUngrouped(*command.second)) {
80 return;
81 }
82 }
83
84 for (auto&& command : commands) {
85 this->AddRequirements(command.second->GetRequirements());
86 m_runsWhenDisabled &= command.second->RunsWhenDisabled();
87 m_commands.emplace(std::move(command.first), std::move(command.second));
88 }
89 }

91 // No copy constructors for command groups
92 SelectCommand(const SelectCommand& other) = delete;
93
94 // Prevent template expansion from emulating copy ctor
95 SelectCommand(SelectCommand&) = delete;

97 /**
98 * Creates a new selectcommand.
99 *
100 * @param toRun a supplier providing the command to run
101 */
102 explicit SelectCommand(std::function<Command*()> toRun)
103 : m_toRun{std::move(toRun)} {}
104
105 SelectCommand(SelectCommand&& other) = default;
106
107 void Initialize() override;
108
109 void Execute() override { m_selectedCommand->Execute(); }
110
111 void End(bool interrupted) override {
112 return m_selectedCommand->End(interrupted);
113 }
114
115 bool IsFinished() override { return m_selectedCommand->IsFinished(); }
116
117 bool RunsWhenDisabled() const override { return m_runsWhenDisabled; }
118
119 protected:
120 std::unique_ptr<Command> TransferOwnership() && override {
121 return std::make_unique<SelectCommand>(std::move(*this));
122 }
123
124 private:
125 std::unordered_map<Key, std::unique_ptr<Command>> m_commands;
126 std::function<Key()> m_selector;
127 std::function<Command*()> m_toRun;
128 Command* m_selectedCommand;
129 bool m_runsWhenDisabled = true;
130};
131
132template <typename T>
133void SelectCommand<T>::Initialize() {
134 if (m_selector) {
135 auto find = m_commands.find(m_selector());
136 if (find == m_commands.end()) {
137 m_selectedCommand = new PrintCommand(
138 "SelectCommand selector value does not correspond to any command!");
139 return;
140 }
141 m_selectedCommand = find->second.get();
142 } else {
143 m_selectedCommand = m_toRun();
144 }
145 m_selectedCommand->Initialize();
146}
