22/**
23 * Flag values for use with topic listeners.
24 *
25 * The flags are a bitmask and must be OR'ed together to indicate the
26 * combination of events desired to be received.
27 *
28 * The constants kPublish, kUnpublish, and kProperties represent different
29 * events that can occur to entries.
30 *
31 * @ingroup ntcore_cpp_api
32 */
33struct TopicListenerFlags {
34 TopicListenerFlags() = delete;
35
36 /**
37 * Initial listener addition.
38 * Set this flag to receive immediate notification of entries matching the
39 * flag criteria (generally only useful when combined with kPublish).
40 */
41 static constexpr unsigned int kImmediate = NT_TOPIC_NOTIFY_IMMEDIATE;
42
43 /**
44 * Newly published topic.
45 *
46 * Set this flag to receive a notification when a topic is initially
47 * published.
48 */
49 static constexpr unsigned int kPublish = NT_TOPIC_NOTIFY_PUBLISH;
50
51 /**
52 * Topic has no more publishers.
53 *
54 * Set this flag to receive a notification when a topic has no more
55 * publishers.
56 */
57 static constexpr unsigned int kUnpublish = NT_TOPIC_NOTIFY_UNPUBLISH;
58
59 /**
60 * Topic's properties changed.
61 *
62 * Set this flag to receive a notification when an topic's properties change.
63 */
64 static constexpr unsigned int kProperties = NT_TOPIC_NOTIFY_PROPERTIES;
65};
