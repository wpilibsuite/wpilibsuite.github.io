180  ~Property() { StopListener(); }

182  void StartListener() {
183  if (entry && listener == 0 && createListener) {
184  listener = createListener(entry);
185  }
186  }
187 
188  void StopListener() {
189  if (entry && listener != 0) {
190  entry.RemoveListener(listener);
191  listener = 0;
192  }
193  }
