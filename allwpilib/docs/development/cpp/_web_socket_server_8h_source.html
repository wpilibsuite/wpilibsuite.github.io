28/**
29 * WebSocket HTTP server helper. Handles websocket-specific headers. User
30 * must provide the HttpParser.
31 */
32class WebSocketServerHelper {
33 public:
34 /**
35 * Constructor.
36 * @param req HttpParser for request
37 */
38 explicit WebSocketServerHelper(HttpParser& req);
39
40 /**
41 * Get whether or not this was a websocket upgrade.
42 * Only valid during and after the upgrade event.
43 */
44 bool IsWebsocket() const { return m_websocket; }
45
46 /**
47 * Try to find a match to the list of sub-protocols provided by the client.
48 * The list is priority ordered, so the first match wins.
49 * Only valid during and after the upgrade event.
50 * @param protocols Acceptable protocols
51 * @return Pair; first item is true if a match was made, false if not.
52 * Second item is the matched protocol if a match was made, otherwise
53 * is empty.
54 */
55 std::pair<bool, std::string_view> MatchProtocol(
56 span<const std::string_view> protocols);
57
58 /**
59 * Try to find a match to the list of sub-protocols provided by the client.
60 * The list is priority ordered, so the first match wins.
61 * Only valid during and after the upgrade event.
62 * @param protocols Acceptable protocols
63 * @return Pair; first item is true if a match was made, false if not.
64 * Second item is the matched protocol if a match was made, otherwise
65 * is empty.
66 */
67 std::pair<bool, std::string_view> MatchProtocol(
68 std::initializer_list<std::string_view> protocols) {
69 return MatchProtocol({protocols.begin(), protocols.end()});
70 }
71
72 /**
73 * Accept the upgrade. Disconnect other readers (such as the HttpParser
74 * reader) before calling this. See also WebSocket::CreateServer().
75 * @param stream Connection stream
76 * @param protocol The subprotocol to send to the client
77 */
78 std::shared_ptr<WebSocket> Accept(uv::Stream& stream,
79 std::string_view protocol = {}) {
80 return WebSocket::CreateServer(stream, m_key, m_version, protocol);
81 }
82
83 bool IsUpgrade() const { return m_gotHost && m_websocket; }
84
85 /**
86 * Upgrade event. Call Accept() to accept the upgrade.
87 */
88 sig::Signal<> upgrade;
89
90 private:
91 bool m_gotHost = false;
92 bool m_websocket = false;
93 SmallVector<std::string, 2> m_protocols;
94 SmallString<64> m_key;
95 SmallString<16> m_version;
96};

98/**
99 * Dedicated WebSocket server.
100 */
101class WebSocketServer : public std::enable_shared_from_this<WebSocketServer> {
102 struct private_init {};
103
104 public:
105 /**
106 * Server options.
107 */
108 struct ServerOptions {
109 /**
110 * Checker for URL. Return true if URL should be accepted. By default all
111 * URLs are accepted.
112 */
113 std::function<bool(std::string_view)> checkUrl;
114
115 /**
116 * Checker for Host header. Return true if Host should be accepted. By
117 * default all hosts are accepted.
118 */
119 std::function<bool(std::string_view)> checkHost;
120 };
121
122 /**
123 * Private constructor.
124 */
125 WebSocketServer(uv::Stream& stream, span<const std::string_view> protocols,
126 ServerOptions options, const private_init&);
127
128 /**
129 * Starts a dedicated WebSocket server on the provided connection. The
130 * connection should be an accepted client stream.
131 * This also sets the stream user data to the socket server.
132 * A connected event is emitted when the connection is opened.
133 * @param stream Connection stream
134 * @param protocols Acceptable subprotocols
135 * @param options Handshake options
136 */
137 static std::shared_ptr<WebSocketServer> Create(
138 uv::Stream& stream, span<const std::string_view> protocols = {},
139 const ServerOptions& options = {});
140
141 /**
142 * Starts a dedicated WebSocket server on the provided connection. The
143 * connection should be an accepted client stream.
144 * This also sets the stream user data to the socket server.
145 * A connected event is emitted when the connection is opened.
146 * @param stream Connection stream
147 * @param protocols Acceptable subprotocols
148 * @param options Handshake options
149 */
150 static std::shared_ptr<WebSocketServer> Create(
151 uv::Stream& stream, std::initializer_list<std::string_view> protocols,
152 const ServerOptions& options = {}) {
153 return Create(stream, {protocols.begin(), protocols.end()}, options);
154 }
155
156 /**
157 * Connected event. First parameter is the URL, second is the websocket.
158 */
159 sig::Signal<std::string_view, WebSocket&> connected;
160
161 private:
162 uv::Stream& m_stream;
163 HttpParser m_req{HttpParser::kRequest};
164 WebSocketServerHelper m_helper;
165 SmallVector<std::string, 2> m_protocols;
166 ServerOptions m_options;
167 bool m_aborted = false;
168 sig::ScopedConnection m_dataConn;
169 sig::ScopedConnection m_errorConn;
170 sig::ScopedConnection m_endConn;
171
172 void Abort(uint16_t code, std::string_view reason);
173};
