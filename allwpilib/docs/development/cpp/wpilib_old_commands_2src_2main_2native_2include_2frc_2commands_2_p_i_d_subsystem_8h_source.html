17/**
18 * This class is designed to handle the case where there is a Subsystem which
19 * uses a single PIDController almost constantly (for instance, an elevator
20 * which attempts to stay at a constant height).
21 *
22 * It provides some convenience methods to run an internal PIDController. It
23 * also allows access to the internal PIDController in order to give total
24 * control to the programmer.
25 */
26class PIDSubsystem : public Subsystem, public PIDOutput, public PIDSource {
27 public:
28 /**
29 * Instantiates a PIDSubsystem that will use the given P, I, and D values.
30 *
31 * @param name the name
32 * @param p the proportional value
33 * @param i the integral value
34 * @param d the derivative value
35 */
36 PIDSubsystem(std::string_view name, double p, double i, double d);
37
38 /**
39 * Instantiates a PIDSubsystem that will use the given P, I, D, and F values.
40 *
41 * @param name the name
42 * @param p the proportional value
43 * @param i the integral value
44 * @param d the derivative value
45 * @param f the feedforward value
46 */
47 PIDSubsystem(std::string_view name, double p, double i, double d, double f);
48
49 /**
50 * Instantiates a PIDSubsystem that will use the given P, I, D, and F values.
51 *
52 * It will also space the time between PID loop calculations to be equal to
53 * the given period.
54 *
55 * @param name the name
56 * @param p the proportional value
57 * @param i the integral value
58 * @param d the derivative value
59 * @param f the feedfoward value
60 * @param period the time (in seconds) between calculations
61 */
62 PIDSubsystem(std::string_view name, double p, double i, double d, double f,
63 double period);
64
65 /**
66 * Instantiates a PIDSubsystem that will use the given P, I, and D values.
67 *
68 * It will use the class name as its name.
69 *
70 * @param p the proportional value
71 * @param i the integral value
72 * @param d the derivative value
73 */
74 PIDSubsystem(double p, double i, double d);
75
76 /**
77 * Instantiates a PIDSubsystem that will use the given P, I, D, and F values.
78 *
79 * It will use the class name as its name.
80 *
81 * @param p the proportional value
82 * @param i the integral value
83 * @param d the derivative value
84 * @param f the feedforward value
85 */
86 PIDSubsystem(double p, double i, double d, double f);
87
88 /**
89 * Instantiates a PIDSubsystem that will use the given P, I, D, and F values.
90 *
91 * It will use the class name as its name. It will also space the time
92 * between PID loop calculations to be equal to the given period.
93 *
94 * @param p the proportional value
95 * @param i the integral value
96 * @param d the derivative value
97 * @param f the feedforward value
98 * @param period the time (in seconds) between calculations
99 */
100 PIDSubsystem(double p, double i, double d, double f, double period);
101
102 ~PIDSubsystem() override = default;
103
104 PIDSubsystem(PIDSubsystem&&) = default;
105 PIDSubsystem& operator=(PIDSubsystem&&) = default;
106
107 /**
108 * Enables the internal PIDController.
109 */
110 void Enable();
111
112 /**
113 * Disables the internal PIDController.
114 */
115 void Disable();
116
117 // PIDOutput interface
118 void PIDWrite(double output) override;
119
120 // PIDSource interface
121
122 double PIDGet() override;
123
124 /**
125 * Sets the setpoint to the given value.
126 *
127 * If SetRange() was called, then the given setpoint will be trimmed to fit
128 * within the range.
129 *
130 * @param setpoint the new setpoint
131 */
132 void SetSetpoint(double setpoint);
133
134 /**
135 * Adds the given value to the setpoint.
136 *
137 * If SetRange() was used, then the bounds will still be honored by this
138 * method.
139 *
140 * @param deltaSetpoint the change in the setpoint
141 */
142 void SetSetpointRelative(double deltaSetpoint);
143
144 /**
145 * Sets the maximum and minimum values expected from the input.
146 *
147 * @param minimumInput the minimum value expected from the input
148 * @param maximumInput the maximum value expected from the output
149 */
150 void SetInputRange(double minimumInput, double maximumInput);
151
152 /**
153 * Sets the maximum and minimum values to write.
154 *
155 * @param minimumOutput the minimum value to write to the output
156 * @param maximumOutput the maximum value to write to the output
157 */
158 void SetOutputRange(double minimumOutput, double maximumOutput);
159
160 /**
161 * Return the current setpoint.
162 *
163 * @return The current setpoint
164 */
165 double GetSetpoint();
166
167 /**
168 * Returns the current position.
169 *
170 * @return the current position
171 */
172 double GetPosition();
173
174 /**
175 * Returns the current rate.
176 *
177 * @return the current rate
178 */
179 double GetRate();
180
181 /**
182 * Set the absolute error which is considered tolerable for use with
183 * OnTarget.
184 *
185 * @param absValue absolute error which is tolerable
186 */
187 virtual void SetAbsoluteTolerance(double absValue);
188
189 /**
190 * Set the percentage error which is considered tolerable for use with
191 * OnTarget().
192 *
193 * @param percent percentage error which is tolerable
194 */
195 virtual void SetPercentTolerance(double percent);
196
197 /**
198 * Return true if the error is within the percentage of the total input range,
199 * determined by SetTolerance().
200 *
201 * This asssumes that the maximum and minimum input were set using SetInput().
202 * Use OnTarget() in the IsFinished() method of commands that use this
203 * subsystem.
204 *
205 * Currently this just reports on target as the actual value passes through
206 * the setpoint. Ideally it should be based on being within the tolerance for
207 * some period of time.
208 *
209 * @return True if the error is within the percentage tolerance of the input
210 * range
211 */
212 virtual bool OnTarget() const;
213
214 protected:
215 /**
216 * Returns the PIDController used by this PIDSubsystem.
217 *
218 * Use this if you would like to fine tune the PID loop.
219 *
220 * @return The PIDController used by this PIDSubsystem
221 */
222 std::shared_ptr<PIDController> GetPIDController();
223
224 virtual double ReturnPIDInput() = 0;
225 virtual void UsePIDOutput(double output) = 0;
226
227 private:
228 // The internal PIDController
229 std::shared_ptr<PIDController> m_controller;
230};
