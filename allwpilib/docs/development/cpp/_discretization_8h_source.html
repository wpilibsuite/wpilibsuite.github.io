13/**
14 * Discretizes the given continuous A matrix.
15 *
16 * @tparam States Number of states.
17 * @param contA Continuous system matrix.
18 * @param dt Discretization timestep.
19 * @param discA Storage for discrete system matrix.
20 */
21template <int States>
22void DiscretizeA(const Matrixd<States, States>& contA, units::second_t dt,
23 Matrixd<States, States>* discA) {
24 *discA = (contA * dt.value()).exp();
25}

27/**
28 * Discretizes the given continuous A and B matrices.
29 *
30 * @tparam States Number of states.
31 * @tparam Inputs Number of inputs.
32 * @param contA Continuous system matrix.
33 * @param contB Continuous input matrix.
34 * @param dt Discretization timestep.
35 * @param discA Storage for discrete system matrix.
36 * @param discB Storage for discrete input matrix.
37 */
38template <int States, int Inputs>
39void DiscretizeAB(const Matrixd<States, States>& contA,
40 const Matrixd<States, Inputs>& contB, units::second_t dt,
41 Matrixd<States, States>* discA,
42 Matrixd<States, Inputs>* discB) {
43 // Matrices are blocked here to minimize matrix exponentiation calculations
44 Matrixd<States + Inputs, States + Inputs> Mcont;
45 Mcont.setZero();
46 Mcont.template block<States, States>(0, 0) = contA * dt.value();
47 Mcont.template block<States, Inputs>(0, States) = contB * dt.value();
48
49 // Discretize A and B with the given timestep
50 Matrixd<States + Inputs, States + Inputs> Mdisc = Mcont.exp();
51 *discA = Mdisc.template block<States, States>(0, 0);
52 *discB = Mdisc.template block<States, Inputs>(0, States);
53}

55/**
56 * Discretizes the given continuous A and Q matrices.
57 *
58 * @tparam States Number of states.
59 * @param contA Continuous system matrix.
60 * @param contQ Continuous process noise covariance matrix.
61 * @param dt Discretization timestep.
62 * @param discA Storage for discrete system matrix.
63 * @param discQ Storage for discrete process noise covariance matrix.
64 */
65template <int States>
66void DiscretizeAQ(const Matrixd<States, States>& contA,
67 const Matrixd<States, States>& contQ, units::second_t dt,
68 Matrixd<States, States>* discA,
69 Matrixd<States, States>* discQ) {
70 // Make continuous Q symmetric if it isn't already
71 Matrixd<States, States> Q = (contQ + contQ.transpose()) / 2.0;
72
73 // Set up the matrix M = [[-A, Q], [0, A.T]]
74 Matrixd<2 * States, 2 * States> M;
75 M.template block<States, States>(0, 0) = -contA;
76 M.template block<States, States>(0, States) = Q;
77 M.template block<States, States>(States, 0).setZero();
78 M.template block<States, States>(States, States) = contA.transpose();
79
80 Matrixd<2 * States, 2 * States> phi = (M * dt.value()).exp();
81
82 // Phi12 = phi[0:States, States:2*States]
83 // Phi22 = phi[States:2*States, States:2*States]
84 Matrixd<States, States> phi12 = phi.block(0, States, States, States);
85 Matrixd<States, States> phi22 = phi.block(States, States, States, States);
86
87 *discA = phi22.transpose();
88
89 Q = *discA * phi12;
90
91 // Make discrete Q symmetric if it isn't already
92 *discQ = (Q + Q.transpose()) / 2.0;
93}

95/**
96 * Discretizes the given continuous A and Q matrices.
97 *
98 * Rather than solving a 2N x 2N matrix exponential like in DiscretizeAQ()
99 * (which is expensive), we take advantage of the structure of the block matrix
100 * of A and Q.
101 *
102 * 1) The exponential of A*t, which is only N x N, is relatively cheap.
103 * 2) The upper-right quarter of the 2N x 2N matrix, which we can approximate
104 * using a taylor series to several terms and still be substantially cheaper
105 * than taking the big exponential.
106 *
107 * @tparam States Number of states.
108 * @param contA Continuous system matrix.
109 * @param contQ Continuous process noise covariance matrix.
110 * @param dt Discretization timestep.
111 * @param discA Storage for discrete system matrix.
112 * @param discQ Storage for discrete process noise covariance matrix.
113 */
114template <int States>
115void DiscretizeAQTaylor(const Matrixd<States, States>& contA,
116 const Matrixd<States, States>& contQ,
117 units::second_t dt, Matrixd<States, States>* discA,
118 Matrixd<States, States>* discQ) {
119 // Make continuous Q symmetric if it isn't already
120 Matrixd<States, States> Q = (contQ + contQ.transpose()) / 2.0;
121
122 Matrixd<States, States> lastTerm = Q;
123 double lastCoeff = dt.value();
124
125 // Aᵀⁿ
126 Matrixd<States, States> Atn = contA.transpose();
127
128 Matrixd<States, States> phi12 = lastTerm * lastCoeff;
129
130 // i = 6 i.e. 5th order should be enough precision
131 for (int i = 2; i < 6; ++i) {
132 lastTerm = -contA * lastTerm + Q * Atn;
133 lastCoeff *= dt.value() / static_cast<double>(i);
134
135 phi12 += lastTerm * lastCoeff;
136
137 Atn *= contA.transpose();
138 }
139
140 DiscretizeA<States>(contA, dt, discA);
141 Q = *discA * phi12;
142
143 // Make discrete Q symmetric if it isn't already
144 *discQ = (Q + Q.transpose()) / 2.0;
145}

147/**
148 * Returns a discretized version of the provided continuous measurement noise
149 * covariance matrix.
150 *
151 * @tparam Outputs Number of outputs.
152 * @param R Continuous measurement noise covariance matrix.
153 * @param dt Discretization timestep.
154 */
155template <int Outputs>
156Matrixd<Outputs, Outputs> DiscretizeR(const Matrixd<Outputs, Outputs>& R,
157 units::second_t dt) {
158 return R / dt.value();
