374class HttpMultipartScanner {
375 public:

378 Reset(saveSkipped);

381
382 // Change the boundary. This is only safe to do when IsDone() is true (or
383 // immediately after construction).
384 void SetBoundary(std::string_view boundary);
385
386 // Reset the scanner. This allows reuse of internal buffers.
387 void Reset(bool saveSkipped = false);
388
389 // Execute the scanner. Will automatically call Reset() on entry if IsDone()
390 // is true.
391 // @param in input data
392 // @return the input not consumed; empty if all input consumed
393 std::string_view Execute(std::string_view in);
394
395 // Returns true when the boundary has been found.
396 bool IsDone() const { return m_state == kDone; }
397
398 // Get the skipped data. Will be empty if saveSkipped was false.
399 std::string_view GetSkipped() const {
400 return m_saveSkipped ? std::string_view{m_buf} : std::string_view{};
401 }
402
403 private:
404 SmallString<64> m_boundaryWith, m_boundaryWithout;
405
406 // Internal state
407 enum State { kBoundary, kPadding, kDone };
408 State m_state;
409 size_t m_posWith, m_posWithout;
410 enum Dashes { kUnknown, kWith, kWithout };
411 Dashes m_dashes;
412
413 // Buffer
414 bool m_saveSkipped;
415 std::string m_buf;
416};
