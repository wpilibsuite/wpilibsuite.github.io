157#define FRC_CheckErrorStatus(status, format, ...) \
158 do { \
159 if ((status) < 0) { \
160 throw ::frc::MakeError(status, __FILE__, __LINE__, __FUNCTION__, \
161 FMT_STRING(format), __VA_ARGS__); \
162 } else if ((status) > 0) { \

164 FMT_STRING(format), __VA_ARGS__); \
