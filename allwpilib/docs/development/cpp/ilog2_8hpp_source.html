26 inline std::size_t ilog2_base(std::uint64_t x)
27 {
28#if defined(__GNUC__)
29 unsigned long long value = x;
30 return sizeof(value) * CHAR_BIT - static_cast<unsigned>(__builtin_clzll(value));
31#else
32 // Adapted from https://stackoverflow.com/a/40943402
33 std::size_t clz = 64;
34 std::size_t c = 32;
35 do
36 {
37 auto tmp = x >> c;
38 if (tmp != 0)
39 {
40 clz -= c;
41 x = tmp;
42 }
43 c = c >> 1;
44 } while (c != 0);
45 clz -= x ? 1 : 0;
46
47 return 64 - clz;
48#endif
49 }

51 // ilog2() implementation, cuts part after the comma
52 // e.g. 1 -> 0, 2 -> 1, 3 -> 1, 4 -> 2, 5 -> 2
53 inline std::size_t ilog2(std::uint64_t x)

55 return ilog2_base(x) - 1;
