19 template <int States, int Inputs, int Outputs, typename KalmanFilterType>
20 class KalmanFilterLatencyCompensator {
21  public:
22  struct ObserverSnapshot {
23  Eigen::Matrix<double, States, 1> xHat;
24  Eigen::Matrix<double, States, States> errorCovariances;
25  Eigen::Matrix<double, Inputs, 1> inputs;
26  Eigen::Matrix<double, Outputs, 1> localMeasurements;
27 
28  ObserverSnapshot(const KalmanFilterType& observer,
29  const Eigen::Matrix<double, Inputs, 1>& u,
30  const Eigen::Matrix<double, Outputs, 1>& localY)
31  : xHat(observer.Xhat()),
32  errorCovariances(observer.P()),
33  inputs(u),
34  localMeasurements(localY) {}
35  };
36 
37  void AddObserverState(const KalmanFilterType& observer,
38  Eigen::Matrix<double, Inputs, 1> u,
39  Eigen::Matrix<double, Outputs, 1> localY,
40  units::second_t timestamp) {
41  // Add the new state into the vector.
42  m_pastObserverSnapshots.emplace_back(timestamp,
43  ObserverSnapshot{observer, u, localY});
44 
45  // Remove the oldest snapshot if the vector exceeds our maximum size.
46  if (m_pastObserverSnapshots.size() > kMaxPastObserverStates) {
47  m_pastObserverSnapshots.erase(m_pastObserverSnapshots.begin());
48  }
49  }
50 
51  template <int Rows>
52  void ApplyPastMeasurement(
53  KalmanFilterType* observer, units::second_t nominalDt,
54  Eigen::Matrix<double, Rows, 1> y,
55  std::function<void(const Eigen::Matrix<double, Inputs, 1>& u,
56  const Eigen::Matrix<double, Rows, 1>& y)>
57  globalMeasurementCorrect,
58  units::second_t timestamp) {
59  if (m_pastObserverSnapshots.size() == 0) {
60  // State map was empty, which means that we got a measurement right at
61  // startup. The only thing we can do is ignore the measurement.
62  return;
63  }
64 
65  // We will perform a binary search to find the index of the element in the
66  // vector that has a timestamp that is equal to or greater than the vision
67  // measurement timestamp.
68  auto lowerBoundIter = std::lower_bound(
69  m_pastObserverSnapshots.cbegin(), m_pastObserverSnapshots.cend(),
70  timestamp,
71  [](const auto& entry, const auto& ts) { return entry.first < ts; });
72  int index = std::distance(m_pastObserverSnapshots.cbegin(), lowerBoundIter);
73 
74  // High and Low should be the same. The sampled timestamp is greater than or
75  // equal to the vision pose timestamp. We will now find the entry which is
76  // closest in time to the requested timestamp.
77 
78  size_t indexOfClosestEntry =
79  units::math::abs(
80  timestamp - m_pastObserverSnapshots[std::max(index - 1, 0)].first) <
81  units::math::abs(timestamp -
82  m_pastObserverSnapshots[index].first)
83  ? index - 1
84  : index;
85 
86  units::second_t lastTimestamp =
87  m_pastObserverSnapshots[indexOfClosestEntry].first - nominalDt;
88 
89  // We will now go back in time to the state of the system at the time when
90  // the measurement was captured. We will reset the observer to that state,
91  // and apply correction based on the measurement. Then, we will go back
92  // through all observer states until the present and apply past inputs to
93  // get the present estimated state.
94  for (size_t i = indexOfClosestEntry; i < m_pastObserverSnapshots.size();
95  ++i) {
96  auto& [key, snapshot] = m_pastObserverSnapshots[i];
97 
98  if (i == indexOfClosestEntry) {
99  observer->SetP(snapshot.errorCovariances);
100  observer->SetXhat(snapshot.xHat);
101  }
102 
103  observer->Predict(snapshot.inputs, key - lastTimestamp);
104  observer->Correct(snapshot.inputs, snapshot.localMeasurements);
105 
106  if (i == indexOfClosestEntry) {
107  // Note that the measurement is at a timestep close but probably not
108  // exactly equal to the timestep for which we called predict. This makes
109  // the assumption that the dt is small enough that the difference
110  // between the measurement time and the time that the inputs were
111  // captured at is very small.
112  globalMeasurementCorrect(snapshot.inputs, y);
113  }
114 
115  lastTimestamp = key;
116  snapshot = ObserverSnapshot{*observer, snapshot.inputs,
117  snapshot.localMeasurements};
118  }
119  }
120 
121  private:
122  static constexpr size_t kMaxPastObserverStates = 300;
123  std::vector<std::pair<units::second_t, ObserverSnapshot>>
124  m_pastObserverSnapshots;
125 };
126 } // namespace frc
