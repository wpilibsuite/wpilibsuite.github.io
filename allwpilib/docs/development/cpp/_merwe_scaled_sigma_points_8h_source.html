14/**
15 * Generates sigma points and weights according to Van der Merwe's 2004
16 * dissertation[1] for the UnscentedKalmanFilter class.
17 *
18 * It parametrizes the sigma points using alpha, beta, kappa terms, and is the
19 * version seen in most publications. Unless you know better, this should be
20 * your default choice.
21 *
22 * [1] R. Van der Merwe "Sigma-Point Kalman Filters for Probabilitic
23 * Inference in Dynamic State-Space Models" (Doctoral dissertation)
24 *
25 * @tparam States The dimensionality of the state. 2*States+1 weights will be
26 * generated.
27 */
28template <int States>
29class MerweScaledSigmaPoints {
30 public:
31 /**
32 * Constructs a generator for Van der Merwe scaled sigma points.
33 *
34 * @param alpha Determines the spread of the sigma points around the mean.
35 * Usually a small positive value (1e-3).
36 * @param beta Incorporates prior knowledge of the distribution of the mean.
37 * For Gaussian distributions, beta = 2 is optimal.
38 * @param kappa Secondary scaling parameter usually set to 0 or 3 - States.
39 */
40 explicit MerweScaledSigmaPoints(double alpha = 1e-3, double beta = 2,
41 int kappa = 3 - States) {
42 m_alpha = alpha;
43 m_kappa = kappa;
44
45 ComputeWeights(beta);
46 }
47
48 /**
49 * Returns number of sigma points for each variable in the state x.
50 */
51 int NumSigmas() { return 2 * States + 1; }
52
53 /**
54 * Computes the sigma points for an unscented Kalman filter given the mean
55 * (x) and covariance(P) of the filter.
56 *
57 * @param x An array of the means.
58 * @param P Covariance of the filter.
59 *
60 * @return Two dimensional array of sigma points. Each column contains all of
61 * the sigmas for one dimension in the problem space. Ordered by
62 * Xi_0, Xi_{1..n}, Xi_{n+1..2n}.
63 *
64 */
65 Eigen::Matrix<double, States, 2 * States + 1> SigmaPoints(
66 const Eigen::Vector<double, States>& x,
67 const Eigen::Matrix<double, States, States>& P) {
68 double lambda = std::pow(m_alpha, 2) * (States + m_kappa) - States;
69 Eigen::Matrix<double, States, States> U =
70 ((lambda + States) * P).llt().matrixL();
71
72 Eigen::Matrix<double, States, 2 * States + 1> sigmas;
73 sigmas.template block<States, 1>(0, 0) = x;
74 for (int k = 0; k < States; ++k) {
75 sigmas.template block<States, 1>(0, k + 1) =
76 x + U.template block<States, 1>(0, k);
77 sigmas.template block<States, 1>(0, States + k + 1) =
78 x - U.template block<States, 1>(0, k);
79 }
80
81 return sigmas;
82 }
83
84 /**
85 * Returns the weight for each sigma point for the mean.
86 */
87 const Eigen::Vector<double, 2 * States + 1>& Wm() const { return m_Wm; }
88
89 /**
90 * Returns an element of the weight for each sigma point for the mean.
91 *
92 * @param i Element of vector to return.
93 */
94 double Wm(int i) const { return m_Wm(i, 0); }
95
96 /**
97 * Returns the weight for each sigma point for the covariance.
98 */
99 const Eigen::Vector<double, 2 * States + 1>& Wc() const { return m_Wc; }
100
101 /**
102 * Returns an element of the weight for each sigma point for the covariance.
103 *
104 * @param i Element of vector to return.
105 */
106 double Wc(int i) const { return m_Wc(i, 0); }
107
108 private:
109 Eigen::Vector<double, 2 * States + 1> m_Wm;
110 Eigen::Vector<double, 2 * States + 1> m_Wc;
111 double m_alpha;
112 int m_kappa;
113
114 /**
115 * Computes the weights for the scaled unscented Kalman filter.
116 *
117 * @param beta Incorporates prior knowledge of the distribution of the mean.
118 */
119 void ComputeWeights(double beta) {
120 double lambda = std::pow(m_alpha, 2) * (States + m_kappa) - States;
121
122 double c = 0.5 / (States + lambda);
123 m_Wm = Eigen::Vector<double, 2 * States + 1>::Constant(c);
124 m_Wc = Eigen::Vector<double, 2 * States + 1>::Constant(c);
125
126 m_Wm(0) = lambda / (States + lambda);
127 m_Wc(0) = lambda / (States + lambda) + (1 - std::pow(m_alpha, 2) + beta);
128 }
129};
