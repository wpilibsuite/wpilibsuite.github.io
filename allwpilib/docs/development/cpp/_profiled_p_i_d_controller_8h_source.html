268 /**
269 * Returns the next output of the PID controller.
270 *
271 * @param measurement The current measurement of the process variable.
272 */
273 double Calculate(Distance_t measurement) {
274 if (m_controller.IsContinuousInputEnabled()) {
275 // Get error which is smallest distance between goal and measurement
276 auto errorBound = (m_maximumInput - m_minimumInput) / 2.0;
277 auto goalMinDistance = frc::InputModulus<Distance_t>(
278 m_goal.position - measurement, -errorBound, errorBound);
279 auto setpointMinDistance = frc::InputModulus<Distance_t>(
280 m_setpoint.position - measurement, -errorBound, errorBound);
281
282 // Recompute the profile goal with the smallest error, thus giving the
283 // shortest path. The goal may be outside the input range after this
284 // operation, but that's OK because the controller will still go there and
285 // report an error of zero. In other words, the setpoint only needs to be
286 // offset from the measurement by the input range modulus; they don't need
287 // to be equal.
288 m_goal.position = goalMinDistance + measurement;
289 m_setpoint.position = setpointMinDistance + measurement;
290 }
291
292 frc::TrapezoidProfile<Distance> profile{m_constraints, m_goal, m_setpoint};
293 m_setpoint = profile.Calculate(GetPeriod());
294 return m_controller.Calculate(measurement.value(),
295 m_setpoint.position.value());
296 }
