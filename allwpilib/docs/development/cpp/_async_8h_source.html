24/**
25 * Async handle.
26 * Async handles allow the user to "wakeup" the event loop and have a signal
27 * generated from another thread.
28 *
29 * Data may be passed into the callback called on the event loop by using
30 * template parameters. If data parameters are used, the async callback will
31 * be called once for every call to Send(). If no data parameters are used,
32 * the async callback may or may not be called for every call to Send() (e.g.
33 * the calls may be coaleasced).
34 */
35template <typename... T>
36class Async final : public HandleImpl<Async<T...>, uv_async_t> {
37 struct private_init {};
38
39 public:
40 Async(const std::shared_ptr<Loop>& loop, const private_init&)
41 : m_loop{loop} {}
42 ~Async() noexcept override {
43 if (auto loop = m_loop.lock()) {
44 this->Close();
45 } else {
46 this->ForceClosed();
47 }
48 }
49
50 /**
51 * Create an async handle.
52 *
53 * @param loop Loop object where this handle runs.
54 */
55 static std::shared_ptr<Async> Create(Loop& loop) {
56 return Create(loop.shared_from_this());
57 }
58
59 /**
60 * Create an async handle.
61 *
62 * @param loop Loop object where this handle runs.
63 */
64 static std::shared_ptr<Async> Create(const std::shared_ptr<Loop>& loop) {
65 auto h = std::make_shared<Async>(loop, private_init{});
66 int err =
67 uv_async_init(loop->GetRaw(), h->GetRaw(), [](uv_async_t* handle) {
68 auto& h = *static_cast<Async*>(handle->data);
69 std::scoped_lock lock(h.m_mutex);
70 for (auto&& v : h.m_data) {
71 std::apply(h.wakeup, v);
72 }
73 h.m_data.clear();
74 });
75 if (err < 0) {
76 loop->ReportError(err);
77 return nullptr;
78 }
79 h->Keep();
80 return h;
81 }
82
83 /**
84 * Wakeup the event loop and emit the event.
85 *
86 * It’s safe to call this function from any thread including the loop thread.
87 * An async event will be emitted on the loop thread.
88 */
89 template <typename... U>
90 void Send(U&&... u) {
91 auto loop = m_loop.lock();
92 if (loop && loop->GetThreadId() == std::this_thread::get_id()) {
93 // called from within the loop, just call the function directly
94 wakeup(std::forward<U>(u)...);
95 return;
96 }
97
98 {
99 std::scoped_lock lock(m_mutex);
100 m_data.emplace_back(std::forward_as_tuple(std::forward<U>(u)...));
101 }
102 if (loop) {
103 this->Invoke(&uv_async_send, this->GetRaw());
104 }
105 }
106
107 /**
108 * Wakeup the event loop and emit the event.
109 * This function assumes the loop still exists, which makes it a bit faster.
110 *
111 * It’s safe to call this function from any thread.
112 * An async event will be emitted on the loop thread.
113 */
114 void UnsafeSend() { Invoke(&uv_async_send, this->GetRaw()); }
115
116 /**
117 * Signal generated (on event loop thread) when the async event occurs.
118 */
119 sig::Signal<T...> wakeup;
120
121 private:
122 wpi::mutex m_mutex;
123 std::vector<std::tuple<T...>> m_data;
124 std::weak_ptr<Loop> m_loop;
125};

156 /**
157 * Wakeup the event loop and emit the event.
158 *
159 * It’s safe to call this function from any thread.
160 * An async event will be emitted on the loop thread.
161 */
162 void Send() {
163 if (auto loop = m_loop.lock()) {
164 if (loop->GetThreadId() == std::this_thread::get_id()) {
165 // called from within the loop, just call the function directly
166 wakeup();
167 } else {
168 Invoke(&uv_async_send, GetRaw());
169 }
170 }
171 }

176 *
177 * It’s safe to call this function from any thread.
178 * An async event will be emitted on the loop thread.
179 */
180 void UnsafeSend() { Invoke(&uv_async_send, GetRaw()); }
181
182 /**
183 * Signal generated (on event loop thread) when the async event occurs.
184 */
185 sig::Signal<> wakeup;
186
187 private:
188 std::weak_ptr<Loop> m_loop;
