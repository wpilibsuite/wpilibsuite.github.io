18/**
19 * A Kalman filter combines predictions from a model and measurements to give an
20 * estimate of the true system state. This is useful because many states cannot
21 * be measured directly as a result of sensor noise, or because the state is
22 * "hidden".
23 *
24 * Kalman filters use a K gain matrix to determine whether to trust the model or
25 * measurements more. Kalman filter theory uses statistics to compute an optimal
26 * K gain which minimizes the sum of squares error in the state estimate. This K
27 * gain is used to correct the state estimate by some amount of the difference
28 * between the actual measurements and the measurements predicted by the model.
29 *
30 * An unscented Kalman filter uses nonlinear state and measurement models. It
31 * propagates the error covariance using sigma points chosen to approximate the
32 * true probability distribution.
33 *
34 * For more on the underlying math, read
35 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf chapter 9
36 * "Stochastic control theory".
37 *
38 * <p> This class implements a square-root-form unscented Kalman filter
39 * (SR-UKF). For more information about the SR-UKF, see
40 * https://www.researchgate.net/publication/3908304.
41 *
42 * @tparam States The number of states.
43 * @tparam Inputs The number of inputs.
44 * @tparam Outputs The number of outputs.
45 */
46template <int States, int Inputs, int Outputs>
47class UnscentedKalmanFilter {
48 public:
49 using StateVector = Vectord<States>;
50 using InputVector = Vectord<Inputs>;
51 using OutputVector = Vectord<Outputs>;
52
53 using StateArray = wpi::array<double, States>;
54 using OutputArray = wpi::array<double, Outputs>;
55
56 using StateMatrix = Matrixd<States, States>;
57
58 /**
59 * Constructs an unscented Kalman filter.
60 *
61 * @param f A vector-valued function of x and u that returns
62 * the derivative of the state vector.
63 * @param h A vector-valued function of x and u that returns
64 * the measurement vector.
65 * @param stateStdDevs Standard deviations of model states.
66 * @param measurementStdDevs Standard deviations of measurements.
67 * @param dt Nominal discretization timestep.
68 */
69 UnscentedKalmanFilter(
70 std::function<StateVector(const StateVector&, const InputVector&)> f,
71 std::function<OutputVector(const StateVector&, const InputVector&)> h,
72 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
73 units::second_t dt);
74
75 /**
76 * Constructs an unscented Kalman filter with custom mean, residual, and
77 * addition functions. Using custom functions for arithmetic can be useful if
78 * you have angles in the state or measurements, because they allow you to
79 * correctly account for the modular nature of angle arithmetic.
80 *
81 * @param f A vector-valued function of x and u that returns
82 * the derivative of the state vector.
83 * @param h A vector-valued function of x and u that returns
84 * the measurement vector.
85 * @param stateStdDevs Standard deviations of model states.
86 * @param measurementStdDevs Standard deviations of measurements.
87 * @param meanFuncX A function that computes the mean of 2 * States +
88 * 1 state vectors using a given set of weights.
89 * @param meanFuncY A function that computes the mean of 2 * States +
90 * 1 measurement vectors using a given set of
91 * weights.
92 * @param residualFuncX A function that computes the residual of two
93 * state vectors (i.e. it subtracts them.)
94 * @param residualFuncY A function that computes the residual of two
95 * measurement vectors (i.e. it subtracts them.)
96 * @param addFuncX A function that adds two state vectors.
97 * @param dt Nominal discretization timestep.
98 */
99 UnscentedKalmanFilter(
100 std::function<StateVector(const StateVector&, const InputVector&)> f,
101 std::function<OutputVector(const StateVector&, const InputVector&)> h,
102 const StateArray& stateStdDevs, const OutputArray& measurementStdDevs,
103 std::function<StateVector(const Matrixd<States, 2 * States + 1>&,
104 const Vectord<2 * States + 1>&)>
105 meanFuncX,
106 std::function<OutputVector(const Matrixd<Outputs, 2 * States + 1>&,
107 const Vectord<2 * States + 1>&)>
108 meanFuncY,
109 std::function<StateVector(const StateVector&, const StateVector&)>
110 residualFuncX,
111 std::function<OutputVector(const OutputVector&, const OutputVector&)>
112 residualFuncY,
113 std::function<StateVector(const StateVector&, const StateVector&)>
114 addFuncX,
115 units::second_t dt);
116
117 /**
118 * Returns the square-root error covariance matrix S.
119 */
120 const StateMatrix& S() const { return m_S; }
121
122 /**
123 * Returns an element of the square-root error covariance matrix S.
124 *
125 * @param i Row of S.
126 * @param j Column of S.
127 */
128 double S(int i, int j) const { return m_S(i, j); }
129
130 /**
131 * Set the current square-root error covariance matrix S.
132 *
133 * @param S The square-root error covariance matrix S.
134 */
135 void SetS(const StateMatrix& S) { m_S = S; }
136
137 /**
138 * Returns the reconstructed error covariance matrix P.
139 */
140 StateMatrix P() const { return m_S.transpose() * m_S; }
141
142 /**
143 * Set the current square-root error covariance matrix S by taking the square
144 * root of P.
145 *
146 * @param P The error covariance matrix P.
147 */
148 void SetP(const StateMatrix& P) { m_S = P.llt().matrixU(); }
149
150 /**
151 * Returns the state estimate x-hat.
152 */
153 const StateVector& Xhat() const { return m_xHat; }
154
155 /**
156 * Returns an element of the state estimate x-hat.
157 *
158 * @param i Row of x-hat.
159 */
160 double Xhat(int i) const { return m_xHat(i); }
161
162 /**
163 * Set initial state estimate x-hat.
164 *
165 * @param xHat The state estimate x-hat.
166 */
167 void SetXhat(const StateVector& xHat) { m_xHat = xHat; }
168
169 /**
170 * Set an element of the initial state estimate x-hat.
171 *
172 * @param i Row of x-hat.
173 * @param value Value for element of x-hat.
174 */
175 void SetXhat(int i, double value) { m_xHat(i) = value; }
176
177 /**
178 * Resets the observer.
179 */
180 void Reset() {
181 m_xHat.setZero();
182 m_S.setZero();
183 m_sigmasF.setZero();
184 }
185
186 /**
187 * Project the model into the future with a new control input u.
188 *
189 * @param u New control input from controller.
190 * @param dt Timestep for prediction.
191 */
192 void Predict(const InputVector& u, units::second_t dt);
193
194 /**
195 * Correct the state estimate x-hat using the measurements in y.
196 *
197 * @param u Same control input used in the predict step.
198 * @param y Measurement vector.
199 */
200 void Correct(const InputVector& u, const OutputVector& y) {
201 Correct<Outputs>(u, y, m_h, m_contR, m_meanFuncY, m_residualFuncY,
202 m_residualFuncX, m_addFuncX);
203 }
204
205 /**
206 * Correct the state estimate x-hat using the measurements in y.
207 *
208 * This is useful for when the measurements available during a timestep's
209 * Correct() call vary. The h(x, u) passed to the constructor is used if one
210 * is not provided (the two-argument version of this function).
211 *
212 * @param u Same control input used in the predict step.
213 * @param y Measurement vector.
214 * @param h A vector-valued function of x and u that returns the measurement
215 * vector.
216 * @param R Measurement noise covariance matrix (continuous-time).
217 */
218 template <int Rows>
219 void Correct(
220 const InputVector& u, const Vectord<Rows>& y,
221 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
222 const Matrixd<Rows, Rows>& R);
223
224 /**
225 * Correct the state estimate x-hat using the measurements in y.
226 *
227 * This is useful for when the measurements available during a timestep's
228 * Correct() call vary. The h(x, u) passed to the constructor is used if one
229 * is not provided (the two-argument version of this function).
230 *
231 * @param u Same control input used in the predict step.
232 * @param y Measurement vector.
233 * @param h A vector-valued function of x and u that returns the
234 * measurement vector.
235 * @param R Measurement noise covariance matrix (continuous-time).
236 * @param meanFuncY A function that computes the mean of 2 * States + 1
237 * measurement vectors using a given set of weights.
238 * @param residualFuncY A function that computes the residual of two
239 * measurement vectors (i.e. it subtracts them.)
240 * @param residualFuncX A function that computes the residual of two state
241 * vectors (i.e. it subtracts them.)
242 * @param addFuncX A function that adds two state vectors.
243 */
244 template <int Rows>
245 void Correct(
246 const InputVector& u, const Vectord<Rows>& y,
247 std::function<Vectord<Rows>(const StateVector&, const InputVector&)> h,
248 const Matrixd<Rows, Rows>& R,
249 std::function<Vectord<Rows>(const Matrixd<Rows, 2 * States + 1>&,
250 const Vectord<2 * States + 1>&)>
251 meanFuncY,
252 std::function<Vectord<Rows>(const Vectord<Rows>&, const Vectord<Rows>&)>
253 residualFuncY,
254 std::function<StateVector(const StateVector&, const StateVector&)>
255 residualFuncX,
256 std::function<StateVector(const StateVector&, const StateVector&)>
257 addFuncX);
258
259 private:
260 std::function<StateVector(const StateVector&, const InputVector&)> m_f;
261 std::function<OutputVector(const StateVector&, const InputVector&)> m_h;
262 std::function<StateVector(const Matrixd<States, 2 * States + 1>&,
263 const Vectord<2 * States + 1>&)>
264 m_meanFuncX;
265 std::function<OutputVector(const Matrixd<Outputs, 2 * States + 1>&,
266 const Vectord<2 * States + 1>&)>
267 m_meanFuncY;
268 std::function<StateVector(const StateVector&, const StateVector&)>
269 m_residualFuncX;
270 std::function<OutputVector(const OutputVector&, const OutputVector&)>
271 m_residualFuncY;
272 std::function<StateVector(const StateVector&, const StateVector&)> m_addFuncX;
273 StateVector m_xHat;
274 StateMatrix m_S;
275 StateMatrix m_contQ;
276 Matrixd<Outputs, Outputs> m_contR;
277 Matrixd<States, 2 * States + 1> m_sigmasF;
278 units::second_t m_dt;
279
280 MerweScaledSigmaPoints<States> m_pts;
281};
