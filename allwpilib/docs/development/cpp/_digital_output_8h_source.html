15/**
16 * Class to write to digital outputs.
17 *
18 * Write values to the digital output channels. Other devices implemented
19 * elsewhere will allocate channels automatically so for those devices it
20 * shouldn't be done here.
21 */
22class DigitalOutput : public DigitalSource,
23 public wpi::Sendable,
24 public wpi::SendableHelper<DigitalOutput> {
25 public:
26 /**
27 * Create an instance of a digital output.
28 *
29 * Create a digital output given a channel.
30 *
31 * @param channel The digital channel 0-9 are on-board, 10-25 are on the MXP
32 * port
33 */
34 explicit DigitalOutput(int channel);
35
36 ~DigitalOutput() override;
37
38 DigitalOutput(DigitalOutput&&) = default;
39 DigitalOutput& operator=(DigitalOutput&&) = default;
40
41 /**
42 * Set the value of a digital output.
43 *
44 * Set the value of a digital output to either one (true) or zero (false).
45 *
46 * @param value 1 (true) for high, 0 (false) for disabled
47 */
48 void Set(bool value);
49
50 /**
51 * Gets the value being output from the Digital Output.
52 *
53 * @return the state of the digital output.
54 */
55 bool Get() const;
56
57 // Digital Source Interface
58 /**
59 * @return The HAL Handle to the specified source.
60 */
61 HAL_Handle GetPortHandleForRouting() const override;
62
63 /**
64 * @return The type of analog trigger output to be used. 0 for Digitals
65 */
66 AnalogTriggerType GetAnalogTriggerTypeForRouting() const override;
67
68 /**
69 * Is source an AnalogTrigger
70 */
71 bool IsAnalogTrigger() const override;
72
73 /**
74 * @return The GPIO channel number that this object represents.
75 */
76 int GetChannel() const override;
77
78 /**
79 * Output a single pulse on the digital output line.
80 *
81 * Send a single pulse on the digital output line where the pulse duration is
82 * specified in seconds. Maximum pulse length is 0.0016 seconds.
83 *
84 * @param length The pulse length in seconds
85 */
86 void Pulse(double length);
87
88 /**
89 * Determine if the pulse is still going.
90 *
91 * Determine if a previously started pulse is still going.
92 */
93 bool IsPulsing() const;
94
95 /**
96 * Change the PWM frequency of the PWM output on a Digital Output line.
97 *
98 * The valid range is from 0.6 Hz to 19 kHz. The frequency resolution is
99 * logarithmic.
100 *
101 * There is only one PWM frequency for all digital channels.
102 *
103 * @param rate The frequency to output all digital output PWM signals.
104 */
105 void SetPWMRate(double rate);
106
107 /**
108 * Enable a PWM Output on this line.
109 *
110 * Allocate one of the 6 DO PWM generator resources from this module.
111 *
112 * Supply the initial duty-cycle to output so as to avoid a glitch when first
113 * starting.
114 *
115 * The resolution of the duty cycle is 8-bit for low frequencies (1kHz or
116 * less) but is reduced the higher the frequency of the PWM signal is.
117 *
118 * @param initialDutyCycle The duty-cycle to start generating. [0..1]
119 */
120 void EnablePWM(double initialDutyCycle);
121
122 /**
123 * Change this line from a PWM output back to a static Digital Output line.
124 *
125 * Free up one of the 6 DO PWM generator resources that were in use.
126 */
127 void DisablePWM();
128
129 /**
130 * Change the duty-cycle that is being generated on the line.
131 *
132 * The resolution of the duty cycle is 8-bit for low frequencies (1kHz or
133 * less) but is reduced the higher the frequency of the PWM signal is.
134 *
135 * @param dutyCycle The duty-cycle to change to. [0..1]
136 */
137 void UpdateDutyCycle(double dutyCycle);
138
139 /**
140 * Indicates this output is used by a simulated device.
141 *
142 * @param device simulated device handle
143 */
144 void SetSimDevice(HAL_SimDeviceHandle device);
145
146 void InitSendable(wpi::SendableBuilder& builder) override;
147
148 private:
149 int m_channel;
150 hal::Handle<HAL_DigitalHandle> m_handle;
151 hal::Handle<HAL_DigitalPWMHandle> m_pwmGenerator;
152};
