22template <typename R>
23struct WorkerThreadAsync {
24 using AfterWorkFunction = std::function<void(R)>;
25
26 ~WorkerThreadAsync() { UnsetLoop(); }
27
28 void SetLoop(uv::Loop& loop) {
29 auto async = uv::Async<AfterWorkFunction, R>::Create(loop);
30 async->wakeup.connect(
31 [](AfterWorkFunction func, R result) { func(result); });
32 m_async = async;
33 }
34
35 void UnsetLoop() {
36 if (auto async = m_async.lock()) {
37 async->Close();
38 m_async.reset();
39 }
40 }
41
42 std::weak_ptr<uv::Async<AfterWorkFunction, R>> m_async;
43};

103template <typename R, typename... T>
104void RunWorkerThreadRequest(WorkerThreadThread<R, T...>& thr,
105 WorkerThreadRequest<R, T...>& req) {
106 R result = std::apply(req.work, std::move(req.params));
107 if (req.afterWork) {
108 if (auto async = thr.m_async.m_async.lock()) {
109 async->Send(std::move(req.afterWork), std::move(result));
110 }
111 } else {
112 thr.m_promises.SetValue(req.promiseId, std::move(result));
113 }
114}
115
116template <typename... T>
117void RunWorkerThreadRequest(WorkerThreadThread<void, T...>& thr,
118 WorkerThreadRequest<void, T...>& req) {
119 std::apply(req.work, req.params);
120 if (req.afterWork) {
121 if (auto async = thr.m_async.m_async.lock()) {
122 async->Send(std::move(req.afterWork));
123 }
124 } else {
125 thr.m_promises.SetValue(req.promiseId);
126 }
127}

129template <typename R, typename... T>
130void WorkerThreadThread<R, T...>::Main() {
131 std::vector<Request> requests;
132 while (m_active) {
133 std::unique_lock lock(m_mutex);
134 m_cond.wait(lock, [&] { return !m_active || !m_requests.empty(); });
135 if (!m_active) {
136 break;
137 }
138
139 // don't want to hold the lock while executing the callbacks
140 requests.swap(m_requests);
141 lock.unlock();
142
143 for (auto&& req : requests) {
144 if (!m_active) {
145 break; // requests may be long-running
146 }
147 RunWorkerThreadRequest(*this, req);
148 }
149 requests.clear();
150 m_promises.Notify();
151 }
152}

159template <typename R, typename... T>
160class WorkerThread<R(T...)> final {
161 using Thread = detail::WorkerThreadThread<R, T...>;
162
163 public:
164 using WorkFunction = std::function<R(T...)>;
165 using AfterWorkFunction =
166 typename detail::WorkerThreadAsync<R>::AfterWorkFunction;
167
168 WorkerThread() { m_owner.Start(); }
169
170 /**
171 * Set the loop. This must be called from the loop thread.
172 * Subsequent calls to QueueWorkThen will run afterWork on the provided
173 * loop (via an async handle).
174 *
175 * @param loop the loop to use for running afterWork routines
176 */
177 void SetLoop(uv::Loop& loop) {
178 if (auto thr = m_owner.GetThread()) {
179 thr->m_async.SetLoop(loop);
180 }
181 }
182
183 /**
184 * Set the loop. This must be called from the loop thread.
185 * Subsequent calls to QueueWorkThen will run afterWork on the provided
186 * loop (via an async handle).
187 *
188 * @param loop the loop to use for running afterWork routines
189 */
190 void SetLoop(std::shared_ptr<uv::Loop> loop) { SetLoop(*loop); }
191
192 /**
193 * Unset the loop. This must be called from the loop thread.
194 * Subsequent calls to QueueWorkThen will no longer run afterWork.
195 */
196 void UnsetLoop() {
197 if (auto thr = m_owner.GetThread()) {
198 thr->m_async.UnsetLoop();
199 }
200 }
201
202 /**
203 * Get the handle used by QueueWorkThen() to run afterWork.
204 * This handle is set by SetLoop().
205 * Calling Close() on this handle is the same as calling UnsetLoop().
206 *
207 * @return The handle (if nullptr, no handle is set)
208 */
209 std::shared_ptr<uv::Handle> GetHandle() const {
210 if (auto thr = m_owner.GetThread()) {
211 return thr->m_async.m_async.lock();
212 } else {
213 return nullptr;
214 }
215 }
216
217 /**
218 * Wakeup the worker thread, call the work function, and return a future for
219 * the result.
220 *
221 * It’s safe to call this function from any thread.
222 * The work function will be called on the worker thread.
223 *
224 * The future will return a default-constructed result if this class is
225 * destroyed while waiting for a result.
226 *
227 * @param work Work function (called on worker thread)
228 * @param u Arguments to work function
229 */
230 template <typename... U>
231 future<R> QueueWork(WorkFunction work, U&&... u) {
232 if (auto thr = m_owner.GetThread()) {
233 // create the future
234 uint64_t req = thr->m_promises.CreateRequest();
235
236 // add the parameters to the input queue
237 thr->m_requests.emplace_back(
238 req, std::move(work), std::forward_as_tuple(std::forward<U>(u)...));
239
240 // signal the thread
241 thr->m_cond.notify_one();
242
243 // return future
244 return thr->m_promises.CreateFuture(req);
245 }
246
247 // XXX: is this the right thing to do?
248 return future<R>();
249 }
250
251 /**
252 * Wakeup the worker thread, call the work function, and call the afterWork
253 * function with the result on the loop set by SetLoop().
254 *
255 * It’s safe to call this function from any thread.
256 * The work function will be called on the worker thread, and the afterWork
257 * function will be called on the loop thread.
258 *
259 * SetLoop() must be called prior to calling this function for afterWork to
260 * be called.
261 *
262 * @param work Work function (called on worker thread)
263 * @param afterWork After work function (called on loop thread)
264 * @param u Arguments to work function
265 */
266 template <typename... U>
267 void QueueWorkThen(WorkFunction work, AfterWorkFunction afterWork, U&&... u) {
268 if (auto thr = m_owner.GetThread()) {
269 // add the parameters to the input queue
270 thr->m_requests.emplace_back(
271 std::move(work), std::move(afterWork),
272 std::forward_as_tuple(std::forward<U>(u)...));
273
274 // signal the thread
275 thr->m_cond.notify_one();
276 }
277 }
278
279 private:
280 SafeThreadOwner<Thread> m_owner;
281};
