93/**
94 * A source for video that provides a sequence of frames.

96class VideoSource {

101 enum Kind {

105 kCv = CS_SOURCE_CV
106 };
107
108 /** Connection strategy. Used for SetConnectionStrategy(). */
109 enum ConnectionStrategy {
110 /**
111 * Automatically connect or disconnect based on whether any sinks are
112 * connected to this source. This is the default behavior.
113 */
114 kConnectionAutoManage = CS_CONNECTION_AUTO_MANAGE,
115
116 /**
117 * Try to keep the connection open regardless of whether any sinks are
118 * connected.
119 */
120 kConnectionKeepOpen = CS_CONNECTION_KEEP_OPEN,
121
122 /**
123 * Never open the connection. If this is set when the connection is open,
124 * close the connection.
125 */
126 kConnectionForceClose = CS_CONNECTION_FORCE_CLOSE
127 };
128
129 VideoSource() noexcept = default;
130 VideoSource(const VideoSource& source);
131 VideoSource(VideoSource&& other) noexcept;
132 VideoSource& operator=(VideoSource other) noexcept;
133 ~VideoSource();
134
135 explicit operator bool() const { return m_handle != 0; }
136
137 int GetHandle() const { return m_handle; }
138
139 bool operator==(const VideoSource& other) const {
140 return m_handle == other.m_handle;
141 }
142
143 bool operator!=(const VideoSource& other) const { return !(*this == other); }
144
145 /**
146 * Get the kind of the source.
147 */
148 Kind GetKind() const;
149

151 * Get the name of the source. The name is an arbitrary identifier

354 CS_Source m_handle{0};

869 * Create a MJPEG-over-HTTP server sink.

886 * Get the listen address of the server.
887 */
888 std::string GetListenAddress() const;

891 * Get the port number of the server.

893 int GetPort() const;

895 /**
896 * Set the stream resolution for clients that don't specify it.
897 *
898 * <p>It is not necessary to set this if it is the same as the source
899 * resolution.
900 *
901 * <p>Setting this different than the source resolution will result in

907 * @param height height, 0 for unspecified

910
911 /**
