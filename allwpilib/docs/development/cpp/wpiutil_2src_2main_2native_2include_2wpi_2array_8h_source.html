17/**
18 * This class is a wrapper around std::array that does compile time size
19 * checking.
20 *
21 * std::array's implicit constructor can result in uninitialized elements if the
22 * number of arguments doesn't match the std::array size.
23 */
24template <typename T, size_t N>
25class array : public std::array<T, N> {
26 public:
27 constexpr explicit array(empty_array_t) {}
28
29 template <typename... Ts>
30 constexpr array(T arg, Ts&&... args) // NOLINT
31 : std::array<T, N>{std::forward<T>(arg), std::forward<Ts>(args)...} {
32 static_assert(1 + sizeof...(args) == N, "Dimension mismatch");
33 }
34
35 constexpr array(const array<T, N>&) = default;
36 constexpr array& operator=(const array<T, N>&) = default;
37 constexpr array(array<T, N>&&) = default;
38 constexpr array& operator=(array<T, N>&&) = default;

40 constexpr array(const std::array<T, N>& rhs) { // NOLINT
41 *static_cast<std::array<T, N>*>(this) = rhs;
