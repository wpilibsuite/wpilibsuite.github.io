13namespace frc {
14/**
15 * A class that limits the rate of change of an input value. Useful for
16 * implementing voltage, setpoint, and/or output ramps. A slew-rate limit
17 * is most appropriate when the quantity being controlled is a velocity or
18 * a voltage; when controlling a position, consider using a TrapezoidProfile
19 * instead.
20 *
21 * @see TrapezoidProfile
22 */
23template <class Unit>
24class SlewRateLimiter {
25 public:
26 using Unit_t = units::unit_t<Unit>;
27 using Rate = units::compound_unit<Unit, units::inverse<units::seconds>>;
28 using Rate_t = units::unit_t<Rate>;
29
30 /**
31 * Creates a new SlewRateLimiter with the given rate limit and initial value.
32 *
33 * @param rateLimit The rate-of-change limit.
34 * @param initialValue The initial value of the input.
35 */
36 explicit SlewRateLimiter(Rate_t rateLimit, Unit_t initialValue = Unit_t{0})
37 : m_rateLimit{rateLimit},
38 m_prevVal{initialValue},
39 m_prevTime{units::microsecond_t(wpi::Now())} {}
40
41 /**
42 * Filters the input to limit its slew rate.
43 *
44 * @param input The input value whose slew rate is to be limited.
45 * @return The filtered value, which will not change faster than the slew
46 * rate.
47 */
48 Unit_t Calculate(Unit_t input) {
49 units::second_t currentTime = units::microsecond_t(wpi::Now());
50 units::second_t elapsedTime = currentTime - m_prevTime;
51 m_prevVal += std::clamp(input - m_prevVal, -m_rateLimit * elapsedTime,
52 m_rateLimit * elapsedTime);
53 m_prevTime = currentTime;
54 return m_prevVal;
55 }
56
57 /**
58 * Resets the slew rate limiter to the specified value; ignores the rate limit
59 * when doing so.
60 *
61 * @param value The value to reset to.
62 */
63 void Reset(Unit_t value) {
64 m_prevVal = value;
65 m_prevTime = units::microsecond_t(wpi::Now());
66 }
67
68 private:
69 Rate_t m_rateLimit;
70 Unit_t m_prevVal;
71 units::second_t m_prevTime;
72};
73} // namespace frc
