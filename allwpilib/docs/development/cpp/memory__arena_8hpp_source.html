25 /// A memory block.
26 /// It is defined by its starting address and size.
27 /// \ingroup core
28 struct memory_block
29 {
30 void* memory; ///< The address of the memory block (might be \c nullptr).
31 std::size_t size; ///< The size of the memory block (might be \c 0).
32
33 /// \effects Creates an invalid memory block with starting address \c nullptr and size \c 0.
34 memory_block() noexcept : memory_block(nullptr, std::size_t(0)) {}
35
36 /// \effects Creates a memory block from a given starting address and size.
37 memory_block(void* mem, std::size_t s) noexcept : memory(mem), size(s) {}
38
39 /// \effects Creates a memory block from a [begin,end) range.
40 memory_block(void* begin, void* end) noexcept
41 : memory_block(begin, static_cast<std::size_t>(static_cast<char*>(end)
42 - static_cast<char*>(begin)))
43 {
44 }
45
46 /// \returns Whether or not a pointer is inside the memory.
47 bool contains(const void* address) const noexcept
48 {
49 auto mem = static_cast<const char*>(memory);
50 auto addr = static_cast<const char*>(address);
51 return addr >= mem && addr < mem + size;
52 }
53 };

96 // stores memory block in an intrusive linked list and allows LIFO access
97 class memory_block_stack
98 {
99 public:
100 memory_block_stack() noexcept : head_(nullptr) {}
101
102 ~memory_block_stack() noexcept {}
103
104 memory_block_stack(memory_block_stack&& other) noexcept : head_(other.head_)
105 {
106 other.head_ = nullptr;
107 }
108
109 memory_block_stack& operator=(memory_block_stack&& other) noexcept
110 {
111 memory_block_stack tmp(detail::move(other));
112 swap(*this, tmp);
113 return *this;
114 }
115
116 friend void swap(memory_block_stack& a, memory_block_stack& b) noexcept
117 {
118 detail::adl_swap(a.head_, b.head_);
119 }

121 // the raw allocated block returned from an allocator
122 using allocated_mb = memory_block;
123
124 // the inserted block slightly smaller to allow for the fixup value
125 using inserted_mb = memory_block;
126
127 // how much an inserted block is smaller
128 static constexpr std::size_t implementation_offset() noexcept
129 {
130 // node size rounded up to the next multiple of max_alignment.
131 return (sizeof(node) / max_alignment + (sizeof(node) % max_alignment != 0))
132 * max_alignment;
133 }
134
135 // pushes a memory block
136 void push(allocated_mb block) noexcept;
137
138 // pops a memory block and returns the original block
139 allocated_mb pop() noexcept;
140
141 // steals the top block from another stack
142 void steal_top(memory_block_stack& other) noexcept;
143
144 // returns the last pushed() inserted memory block
145 inserted_mb top() const noexcept
146 {
147 WPI_MEMORY_ASSERT(head_);
148 auto mem = static_cast<void*>(head_);
149 return {static_cast<char*>(mem) + implementation_offset(), head_->usable_size};
150 }
151
152 bool empty() const noexcept
153 {
154 return head_ == nullptr;
155 }
156
157 bool owns(const void* ptr) const noexcept;
158
159 // O(n) size
160 std::size_t size() const noexcept;
161
162 private:
163 struct node
164 {
165 node* prev;
166 std::size_t usable_size;
167
168 node(node* p, std::size_t size) noexcept : prev(p), usable_size(size) {}
169 };
170
171 node* head_;
172 };

216 while (!cached_.empty())

265 /// A memory arena that manages huge memory blocks for a higher-level allocator.
266 /// Some allocators like \ref memory_stack work on huge memory blocks,
267 /// this class manages them fro those allocators.
268 /// It uses a \concept{concept_blockallocator,BlockAllocator} for the allocation of those blocks.
269 /// The memory blocks in use are put onto a stack like structure, deallocation will pop from the top,
270 /// so it is only possible to deallocate the last allocated block of the arena.
271 /// By default, blocks are not really deallocated but stored in a cache.
272 /// This can be disabled with the second template parameter,
273 /// passing it \ref uncached_arena (or \c false) disables it,
274 /// \ref cached_arena (or \c true) enables it explicitly.
275 /// \ingroup core
276 template <class BlockAllocator, bool Cached /* = true */>
277 class memory_arena : WPI_EBO(BlockAllocator),
278 WPI_EBO(detail::memory_arena_cache<Cached>)
279 {
280 static_assert(is_block_allocator<BlockAllocator>::value,
281 "BlockAllocator is not a BlockAllocator!");
282 using cache = detail::memory_arena_cache<Cached>;
283
284 public:
285 using allocator_type = BlockAllocator;
286 using is_cached = std::integral_constant<bool, Cached>;
287
288 /// \returns The minimum block size required for an arena containing the given amount of memory.
289 /// If an arena is created with the result of `min_block_size(n)`, the resulting capacity will be exactly `n`.
290 /// \requires `byte_size` must be a positive number.
291 static constexpr std::size_t min_block_size(std::size_t byte_size) noexcept
292 {
293 return detail::memory_block_stack::implementation_offset() + byte_size;
294 }
295
296 /// \effects Creates it by giving it the size and other arguments for the \concept{concept_blockallocator,BlockAllocator}.
297 /// It forwards these arguments to its constructor.
298 /// \requires \c block_size must be greater than \c min_block_size(0) and other requirements depending on the \concept{concept_blockallocator,BlockAllocator}.
299 /// \throws Anything thrown by the constructor of the \c BlockAllocator.
300 template <typename... Args>
301 explicit memory_arena(std::size_t block_size, Args&&... args)
302 : allocator_type(block_size, detail::forward<Args>(args)...)
303 {
304 WPI_MEMORY_ASSERT(block_size > min_block_size(0));
305 }
306
307 /// \effects Deallocates all memory blocks that where requested back to the \concept{concept_blockallocator,BlockAllocator}.
308 ~memory_arena() noexcept
309 {
310 // clear cache
311 shrink_to_fit();
312 // now deallocate everything
313 while (!used_.empty())
314 allocator_type::deallocate_block(used_.pop());
315 }
316
317 /// @{
318 /// \effects Moves the arena.
319 /// The new arena takes ownership over all the memory blocks from the other arena object,
320 /// which is empty after that.
321 /// This does not invalidate any memory blocks.
322 memory_arena(memory_arena&& other) noexcept
323 : allocator_type(detail::move(other)),
324 cache(detail::move(other)),
325 used_(detail::move(other.used_))
326 {
327 }
328
329 memory_arena& operator=(memory_arena&& other) noexcept
330 {
331 memory_arena tmp(detail::move(other));
332 swap(*this, tmp);
333 return *this;
334 }
335 /// @}
336
337 /// \effects Swaps to memory arena objects.
338 /// This does not invalidate any memory blocks.
339 friend void swap(memory_arena& a, memory_arena& b) noexcept
340 {
341 detail::adl_swap(static_cast<allocator_type&>(a), static_cast<allocator_type&>(b));
342 detail::adl_swap(static_cast<cache&>(a), static_cast<cache&>(b));
343 detail::adl_swap(a.used_, b.used_);
344 }
345
346 /// \effects Allocates a new memory block.
347 /// It first uses a cache of previously deallocated blocks, if caching is enabled,
348 /// if it is empty, allocates a new one.
349 /// \returns The new \ref memory_block.
350 /// \throws Anything thrown by the \concept{concept_blockallocator,BlockAllocator} allocation function.
351 memory_block allocate_block()
352 {
353 if (!this->take_from_cache(used_))
354 used_.push(allocator_type::allocate_block());
355
356 auto block = used_.top();
357 detail::debug_fill_internal(block.memory, block.size, false);
358 return block;
359 }
360
361 /// \returns The current memory block.
362 /// This is the memory block that will be deallocated by the next call to \ref deallocate_block().
363 memory_block current_block() const noexcept
364 {
365 return used_.top();
366 }
367
368 /// \effects Deallocates the current memory block.
369 /// The current memory block is the block on top of the stack of blocks.
370 /// If caching is enabled, it does not really deallocate it but puts it onto a cache for later use,
371 /// use \ref shrink_to_fit() to purge that cache.
372 void deallocate_block() noexcept
373 {
374 auto block = used_.top();
375 detail::debug_fill_internal(block.memory, block.size, true);
376 this->do_deallocate_block(get_allocator(), used_);
377 }
378
379 /// \returns If `ptr` is in memory owned by the arena.
380 bool owns(const void* ptr) const noexcept
381 {
382 return used_.owns(ptr);
383 }
384
385 /// \effects Purges the cache of unused memory blocks by returning them.
386 /// The memory blocks will be deallocated in reversed order of allocation.
387 /// Does nothing if caching is disabled.
388 void shrink_to_fit() noexcept
389 {
390 this->do_shrink_to_fit(get_allocator());
391 }
392
393 /// \returns The capacity of the arena, i.e. how many blocks are used and cached.
394 std::size_t capacity() const noexcept
395 {
396 return size() + cache_size();
397 }
398
399 /// \returns The size of the cache, i.e. how many blocks can be allocated without allocation.
400 std::size_t cache_size() const noexcept
401 {
402 return cache::cache_size();
403 }
404
405 /// \returns The size of the arena, i.e. how many blocks are in use.
406 /// It is always smaller or equal to the \ref capacity().
407 std::size_t size() const noexcept
408 {
409 return used_.size();
410 }
411
412 /// \returns The size of the next memory block,
413 /// i.e. of the next call to \ref allocate_block().
414 /// If there are blocks in the cache, returns size of the next one.
415 /// Otherwise forwards to the \concept{concept_blockallocator,BlockAllocator} and subtracts an implementation offset.
416 std::size_t next_block_size() const noexcept
417 {
418 return this->cache_empty() ?
419 allocator_type::next_block_size()
420 - detail::memory_block_stack::implementation_offset() :
421 this->cached_block_size();
422 }
423
424 /// \returns A reference of the \concept{concept_blockallocator,BlockAllocator} object.
425 /// \requires It is undefined behavior to move this allocator out into another object.
426 allocator_type& get_allocator() noexcept
427 {
428 return *this;
429 }
430
431 private:
432 detail::memory_block_stack used_;
433 };

442 /// A \concept{concept_blockallocator,BlockAllocator} that uses a given \concept{concept_rawallocator,RawAllocator} for allocating the blocks.
443 /// It calls the \c allocate_array() function with a node of size \c 1 and maximum alignment on the used allocator for the block allocation.
444 /// The size of the next memory block will grow by a given factor after each allocation,
445 /// allowing an amortized constant allocation time in the higher level allocator.
446 /// The factor can be given as rational in the template parameter, default is \c 2.
447 /// \ingroup adapter
448 template <class RawAllocator = default_allocator, unsigned Num = 2, unsigned Den = 1>
449 class growing_block_allocator
450 : WPI_EBO(allocator_traits<RawAllocator>::allocator_type)
451 {
452 static_assert(float(Num) / Den >= 1.0, "invalid growth factor");
453
454 using traits = allocator_traits<RawAllocator>;
455
456 public:
457 using allocator_type = typename traits::allocator_type;
458
459 /// \effects Creates it by giving it the initial block size, the allocator object and the growth factor.
460 /// By default, it uses a default-constructed allocator object and a growth factor of \c 2.
461 /// \requires \c block_size must be greater than 0.
462 explicit growing_block_allocator(std::size_t block_size,
463 allocator_type alloc = allocator_type()) noexcept
464 : allocator_type(detail::move(alloc)), block_size_(block_size)
465 {
466 }
467
468 /// \effects Allocates a new memory block and increases the block size for the next allocation.
469 /// \returns The new \ref memory_block.
470 /// \throws Anything thrown by the \c allocate_array() function of the \concept{concept_rawallocator,RawAllocator}.
471 memory_block allocate_block()
472 {
473 auto memory =
474 traits::allocate_array(get_allocator(), block_size_, 1, detail::max_alignment);
475 memory_block block(memory, block_size_);
476 block_size_ = grow_block_size(block_size_);
477 return block;
478 }
479
480 /// \effects Deallocates a previously allocated memory block.
481 /// This does not decrease the block size.
482 /// \requires \c block must be previously returned by a call to \ref allocate_block().
483 void deallocate_block(memory_block block) noexcept
484 {
485 traits::deallocate_array(get_allocator(), block.memory, block.size, 1,
486 detail::max_alignment);
487 }
488
489 /// \returns The size of the memory block returned by the next call to \ref allocate_block().
490 std::size_t next_block_size() const noexcept
491 {
492 return block_size_;
493 }
494
495 /// \returns A reference to the used \concept{concept_rawallocator,RawAllocator} object.
496 allocator_type& get_allocator() noexcept
497 {
498 return *this;
499 }
500
501 /// \returns The growth factor.
502 static float growth_factor() noexcept
503 {
504 static constexpr auto factor = float(Num) / Den;
505 return factor;
506 }
507
508 static std::size_t grow_block_size(std::size_t block_size) noexcept
509 {
510 return block_size * Num / Den;
511 }
512
513 private:
514 std::size_t block_size_;
515 };

523 /// A \concept{concept_blockallocator,BlockAllocator} that allows only one block allocation.
524 /// It can be used to prevent higher-level allocators from expanding.
525 /// The one block allocation is performed through the \c allocate_array() function of the given \concept{concept_rawallocator,RawAllocator}.
526 /// \ingroup adapter
527 template <class RawAllocator = default_allocator>
528 class fixed_block_allocator : WPI_EBO(allocator_traits<RawAllocator>::allocator_type)
529 {
530 using traits = allocator_traits<RawAllocator>;
531
532 public:
533 using allocator_type = typename traits::allocator_type;
534
535 /// \effects Creates it by passing it the size of the block and the allocator object.
536 /// \requires \c block_size must be greater than 0,
537 explicit fixed_block_allocator(std::size_t block_size,
538 allocator_type alloc = allocator_type()) noexcept
539 : allocator_type(detail::move(alloc)), block_size_(block_size)
540 {
541 }
542
543 /// \effects Allocates a new memory block or throws an exception if there was already one allocation.
544 /// \returns The new \ref memory_block.
545 /// \throws Anything thrown by the \c allocate_array() function of the \concept{concept_rawallocator,RawAllocator} or \ref out_of_memory if this is not the first call.
546 memory_block allocate_block()
547 {
548 if (block_size_)
549 {
550 auto mem = traits::allocate_array(get_allocator(), block_size_, 1,
551 detail::max_alignment);
552 memory_block block(mem, block_size_);
553 block_size_ = 0u;
554 return block;
555 }
556 WPI_THROW(out_of_fixed_memory(info(), block_size_));
557 }
558
559 /// \effects Deallocates the previously allocated memory block.
560 /// It also resets and allows a new call again.
561 void deallocate_block(memory_block block) noexcept
562 {
563 detail::debug_check_pointer([&] { return block_size_ == 0u; }, info(),
564 block.memory);
565 traits::deallocate_array(get_allocator(), block.memory, block.size, 1,
566 detail::max_alignment);
567 block_size_ = block.size;
568 }
569
570 /// \returns The size of the next block which is either the initial size or \c 0.
571 std::size_t next_block_size() const noexcept
572 {
573 return block_size_;
574 }
575
576 /// \returns A reference to the used \concept{concept_rawallocator,RawAllocator} object.
577 allocator_type& get_allocator() noexcept
578 {
579 return *this;
580 }
581
582 private:
583 allocator_info info() noexcept
584 {
585 return {WPI_MEMORY_LOG_PREFIX "::fixed_block_allocator", this};
586 }
587
588 std::size_t block_size_;
589 };
