13namespace frc {
14/**
15 * Class for differential drive odometry. Odometry allows you to track the
16 * robot's position on the field over the course of a match using readings from
17 * 2 encoders and a gyroscope.
18 *
19 * Teams can use odometry during the autonomous period for complex tasks like
20 * path following. Furthermore, odometry can be used for latency compensation
21 * when using computer-vision systems.
22 *
23 * It is important that you reset your encoders to zero before using this class.
24 * Any subsequent pose resets also require the encoders to be reset to zero.
25 */
26class WPILIB_DLLEXPORT DifferentialDriveOdometry {
27 public:
28 /**
29 * Constructs a DifferentialDriveOdometry object.
30 *
31 * @param gyroAngle The angle reported by the gyroscope.
32 * @param initialPose The starting position of the robot on the field.
33 */
34 explicit DifferentialDriveOdometry(const Rotation2d& gyroAngle,
35 const Pose2d& initialPose = Pose2d());
36
37 /**
38 * Resets the robot's position on the field.
39 *
40 * You NEED to reset your encoders (to zero) when calling this method.
41 *
42 * The gyroscope angle does not need to be reset here on the user's robot
43 * code. The library automatically takes care of offsetting the gyro angle.
44 *
45 * @param pose The position on the field that your robot is at.
46 * @param gyroAngle The angle reported by the gyroscope.
47 */
48 void ResetPosition(const Pose2d& pose, const Rotation2d& gyroAngle) {
49 m_pose = pose;
50 m_previousAngle = pose.Rotation();
51 m_gyroOffset = m_pose.Rotation() - gyroAngle;
52
53 m_prevLeftDistance = 0_m;
54 m_prevRightDistance = 0_m;
55 }
56
57 /**
58 * Returns the position of the robot on the field.
59 * @return The pose of the robot.
60 */
61 const Pose2d& GetPose() const { return m_pose; }
62
63 /**
64 * Updates the robot position on the field using distance measurements from
65 * encoders. This method is more numerically accurate than using velocities to
66 * integrate the pose and is also advantageous for teams that are using lower
67 * CPR encoders.
68 *
69 * @param gyroAngle The angle reported by the gyroscope.
70 * @param leftDistance The distance traveled by the left encoder.
71 * @param rightDistance The distance traveled by the right encoder.
72 * @return The new pose of the robot.
73 */
74 const Pose2d& Update(const Rotation2d& gyroAngle, units::meter_t leftDistance,
75 units::meter_t rightDistance);
76
77 private:
78 Pose2d m_pose;
79
80 Rotation2d m_gyroOffset;
81 Rotation2d m_previousAngle;
82
83 units::meter_t m_prevLeftDistance = 0_m;
84 units::meter_t m_prevRightDistance = 0_m;
85};
86} // namespace frc
