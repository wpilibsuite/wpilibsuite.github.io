26/**
27 * Event loop.
28 *
29 * The event loop is the central part of uv functionality. It takes care of
30 * polling for I/O and scheduling signals to be generated based on different
31 * sources of events.
32 *
33 * The event loop is not moveable, copyable, or directly constructible. Use
34 * Create() to create an event loop, or GetDefault() to get the default loop
35 * if you know your program will only have a single one.
36 */
37class Loop final : public std::enable_shared_from_this<Loop> {
38 struct private_init {};
39
40 public:
41 using Time = std::chrono::duration<uint64_t, std::milli>;
42
43 enum Mode {
44 kDefault = UV_RUN_DEFAULT,
45 kOnce = UV_RUN_ONCE,
46 kNoWait = UV_RUN_NOWAIT
47 };
48
49 explicit Loop(const private_init&) noexcept;
50
51 Loop(const Loop&) = delete;
52 Loop& operator=(const Loop&) = delete;
53 Loop(Loop&& oth) = delete;
54 Loop& operator=(Loop&& oth) = delete;
55
56 ~Loop() noexcept;
57
58 /**
59 * Create a new event loop. The created loop is not the default event loop.
60 *
61 * @return The newly created loop. May return nullptr if a failure occurs.
62 */
63 static std::shared_ptr<Loop> Create();
64
65 /**
66 * Create the default event loop. Only use this event loop if a single loop
67 * is needed for the entire application.
68 *
69 * @return The newly created loop. May return nullptr if a failure occurs.
70 */
71 static std::shared_ptr<Loop> GetDefault();
72
73 /**
74 * Release all internal loop resources.
75 *
76 * Call this function only when the loop has finished executing and all open
77 * handles and requests have been closed, or the loop will emit an error.
78 *
79 * error() will be emitted in case of errors.
80 */
81 void Close();
82
83 /**
84 * Run the event loop.
85 *
86 * Available modes are:
87 *
88 * * `Loop::kDefault`: Run the event loop until there are no
89 * active and referenced handles or requests.
90 * * `Loop::kOnce`: Run a single event loop iteration. Note that this
91 * function blocksif there are no pending callbacks.
92 * * `Loop::kNoWait`: Run a single event loop iteration, but don't block
93 * if there are no pending callbacks.
94 *
95 * @return True when done, false in all other cases.
96 */
97 bool Run(Mode mode = kDefault) {
98 m_tid = std::this_thread::get_id();
99 int rv = uv_run(m_loop, static_cast<uv_run_mode>(static_cast<int>(mode)));
100 m_tid = std::thread::id{};
101 return rv == 0;
102 }
103
104 /**
105 * Check if there are active resources.
106 *
107 * @return True if there are active resources in the loop.
108 */
109 bool IsAlive() const noexcept { return uv_loop_alive(m_loop) != 0; }
110
111 /**
112 * Stop the event loop.
113 *
114 * This will cause Run() to end as soon as possible.
115 * This will happen not sooner than the next loop iteration.
116 * If this function was called before blocking for I/O, the loop won’t block
117 * for I/O on this iteration.
118 */
119 void Stop() noexcept { uv_stop(m_loop); }
120
121 /**
122 * Get backend file descriptor.
123 *
124 * Only kqueue, epoll and event ports are supported.
125 * This can be used in conjunction with `run(Loop::kNoWait)` to poll
126 * in one thread and run the event loop’s callbacks in another.
127 *
128 * @return The backend file descriptor.
129 */
130 int GetDescriptor() const noexcept { return uv_backend_fd(m_loop); }
131
132 /**
133 * Get the poll timeout.
134 *
135 * @return A `std::pair` composed of a boolean value that is true in case of
136 * valid timeout, false otherwise, and the timeout
137 * (`std::chrono::duration<uint64_t, std::milli>`).
138 */
139 std::pair<bool, Time> GetTimeout() const noexcept {
140 auto to = uv_backend_timeout(m_loop);
141 return std::make_pair(to == -1, Time{to});
142 }
143
144 /**
145 * Return the current timestamp in milliseconds.
146 *
147 * The timestamp is cached at the start of the event loop tick.
148 * The timestamp increases monotonically from some arbitrary point in
149 * time.
150 * Don’t make assumptions about the starting point, you will only get
151 * disappointed.
152 *
153 * @return The current timestamp in milliseconds (actual type is
154 * `std::chrono::duration<uint64_t, std::milli>`).
155 */
156 Time Now() const noexcept { return Time{uv_now(m_loop)}; }
157
158 /**
159 * Update the event loop’s concept of _now_.
160 *
161 * The current time is cached at the start of the event loop tick in order
162 * to reduce the number of time-related system calls.
163 * You won’t normally need to call this function unless you have callbacks
164 * that block the event loop for longer periods of time, where _longer_ is
165 * somewhat subjective but probably on the order of a millisecond or more.
166 */
167 void UpdateTime() noexcept { uv_update_time(m_loop); }
168
169 /**
170 * Walk the list of handles.
171 *
172 * The callback will be executed once for each handle that is still active.
173 *
174 * @param callback A function to be invoked once for each active handle.
175 */
176 void Walk(function_ref<void(Handle&)> callback);
177
178 /**
179 * Reinitialize any kernel state necessary in the child process after
180 * a fork(2) system call.
181 *
182 * Previously started watchers will continue to be started in the child
183 * process.
184 *
185 * It is necessary to explicitly call this function on every event loop
186 * created in the parent process that you plan to continue to use in the
187 * child, including the default loop (even if you don’t continue to use it
188 * in the parent). This function must be called before calling any API
189 * function using the loop in the child. Failure to do so will result in
190 * undefined behaviour, possibly including duplicate events delivered to
191 * both parent and child or aborting the child process.
192 *
193 * When possible, it is preferred to create a new loop in the child process
194 * instead of reusing a loop created in the parent. New loops created in the
195 * child process after the fork should not use this function.
196 *
197 * Note that this function is not implemented on Windows.
198 * Note also that this function is experimental in `libuv`. It may contain
199 * bugs, and is subject to change or removal. API and ABI stability is not
200 * guaranteed.
201 *
202 * error() will be emitted in case of errors.
203 */
204 void Fork();
205
206 /**
207 * Get the underlying event loop data structure.
208 *
209 * @return The underlying event loop data structure.
210 */
211 uv_loop_t* GetRaw() const noexcept { return m_loop; }
212
213 /**
214 * Gets user-defined data.
215 * @return User-defined data if any, nullptr otherwise.
216 */
217 template <typename T = void>
218 std::shared_ptr<T> GetData() const {
219 return std::static_pointer_cast<T>(m_data);
220 }
221
222 /**
223 * Sets user-defined data.
224 * @param data User-defined arbitrary data.
225 */
226 void SetData(std::shared_ptr<void> data) { m_data = std::move(data); }
227
228 /**
229 * Get the thread id of the loop thread. If the loop is not currently
230 * running, returns default-constructed thread id.
231 */
232 std::thread::id GetThreadId() const { return m_tid; }
233
234 /**
235 * Error signal
236 */
237 sig::Signal<Error> error;
238
239 /**
240 * Reports error.
241 * @param err Error code
242 */
243 void ReportError(int err) { error(Error(err)); }
244
245 private:
246 std::shared_ptr<void> m_data;
247 uv_loop_t* m_loop;
248 uv_loop_t m_loopStruct;
249 std::atomic<std::thread::id> m_tid;
250};
