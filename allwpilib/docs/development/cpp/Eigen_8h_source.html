11 template <int Rows, int Cols, int... Args>
12 struct fmt::formatter<Eigen::Matrix<double, Rows, Cols, Args...>> {
13  char presentation = 'f';
14 
15  constexpr auto parse(fmt::format_parse_context& ctx) {
16  auto it = ctx.begin(), end = ctx.end();
17  if (it != end && (*it == 'f' || *it == 'e')) {
18  presentation = *it++;
19  }
20 
21  if (it != end && *it != '}') {
22  throw fmt::format_error("invalid format");
23  }
24 
25  return it;
26  }
27 
28  template <typename FormatContext>
29  auto format(const Eigen::Matrix<double, Rows, Cols, Args...>& mat,
30  FormatContext& ctx) {
31  auto out = ctx.out();
32  for (int i = 0; i < Rows; ++i) {
33  for (int j = 0; j < Cols; ++j) {
34  out = fmt::format_to(out, " {:f}", mat(i, j));
35  }
36 
37  if (i < Rows - 1) {
38  out = fmt::format_to(out, "\n");
39  }
40  }
41 
42  return out;
43  }
44 };
