123 /**
124 * Calculates a rough estimate of the moment of inertia of an arm given its
125 * length and mass.
126 *
127 * @param length The length of the arm.
128 * @param mass The mass of the arm.
129 *
130 * @return The calculated moment of inertia.
131 */
132 static constexpr units::kilogram_square_meter_t EstimateMOI(
133 units::meter_t length, units::kilogram_t mass) {
