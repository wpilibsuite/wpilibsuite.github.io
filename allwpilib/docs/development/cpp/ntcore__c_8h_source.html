100/** A NetworkTables string. */
101struct NT_String {
102 /**
103 * String contents (UTF-8).
104 * The string is NOT required to be zero-terminated.
105 * When returned by the library, this is zero-terminated and allocated with
106 * std::malloc().
107 */
108 char* str;
109
110 /**
111 * Length of the string in bytes. If the string happens to be zero
112 * terminated, this does not include the zero-termination.
113 */
114 size_t len;
115};

117/** NetworkTables Entry Value. Note this is a typed union. */
118struct NT_Value {
119 enum NT_Type type;
120 uint64_t last_change;
121 union {
122 NT_Bool v_boolean;
123 double v_double;
124 struct NT_String v_string;
125 struct NT_String v_raw;
126 struct {
127 NT_Bool* arr;
128 size_t size;
129 } arr_boolean;
130 struct {
131 double* arr;
132 size_t size;
133 } arr_double;
134 struct {
135 struct NT_String* arr;
136 size_t size;
137 } arr_string;
138 } data;
139};
