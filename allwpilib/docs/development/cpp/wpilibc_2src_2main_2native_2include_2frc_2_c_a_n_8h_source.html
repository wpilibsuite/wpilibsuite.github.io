12struct CANData {
13 uint8_t data[8];
14 int32_t length;
15 uint64_t timestamp;

18/**
19 * High level class for interfacing with CAN devices conforming to
20 * the standard CAN spec.
21 *
22 * No packets that can be sent gets blocked by the RoboRIO, so all methods
23 * work identically in all robot modes.
24 *
25 * All methods are thread save, however the buffer objects passed in
26 * by the user need to not be modified for the duration of their calls.
27 */
28class CAN {
29 public:
30 /**
31 * Create a new CAN communication interface with the specific device ID.
32 * This uses the team manufacturer and device types.
33 * The device ID is 6 bits (0-63)
34 *
35 * @param deviceId The device id
36 */
37 explicit CAN(int deviceId);
38
39 /**
40 * Create a new CAN communication interface with a specific device ID,
41 * manufacturer and device type. The device ID is 6 bits, the
42 * manufacturer is 8 bits, and the device type is 5 bits.
43 *
44 * @param deviceId The device ID
45 * @param deviceManufacturer The device manufacturer
46 * @param deviceType The device type
47 */
48 CAN(int deviceId, int deviceManufacturer, int deviceType);
49
50 /**
51 * Closes the CAN communication.
52 */
53 ~CAN();
54
55 CAN(CAN&&) = default;
56 CAN& operator=(CAN&&) = default;
57
58 /**
59 * Write a packet to the CAN device with a specific ID. This ID is 10 bits.
60 *
61 * @param data The data to write (8 bytes max)
62 * @param length The data length to write
63 * @param apiId The API ID to write.
64 */
65 void WritePacket(const uint8_t* data, int length, int apiId);
66
67 /**
68 * Write a repeating packet to the CAN device with a specific ID. This ID is
69 * 10 bits. The RoboRIO will automatically repeat the packet at the specified
70 * interval
71 *
72 * @param data The data to write (8 bytes max)
73 * @param length The data length to write
74 * @param apiId The API ID to write.
75 * @param repeatMs The period to repeat the packet at.
76 */
77 void WritePacketRepeating(const uint8_t* data, int length, int apiId,
78 int repeatMs);
79
80 /**
81 * Write an RTR frame to the CAN device with a specific ID. This ID is 10
82 * bits. The length by spec must match what is returned by the responding
83 * device
84 *
85 * @param length The length to request (0 to 8)
86 * @param apiId The API ID to write.
87 */
88 void WriteRTRFrame(int length, int apiId);
89
90 /**
91 * Write a packet to the CAN device with a specific ID. This ID is 10 bits.
92 *
93 * @param data The data to write (8 bytes max)
94 * @param length The data length to write
95 * @param apiId The API ID to write.
96 */
97 int WritePacketNoError(const uint8_t* data, int length, int apiId);
98
99 /**
100 * Write a repeating packet to the CAN device with a specific ID. This ID is
101 * 10 bits. The RoboRIO will automatically repeat the packet at the specified
102 * interval
103 *
104 * @param data The data to write (8 bytes max)
105 * @param length The data length to write
106 * @param apiId The API ID to write.
107 * @param repeatMs The period to repeat the packet at.
108 */
109 int WritePacketRepeatingNoError(const uint8_t* data, int length, int apiId,
110 int repeatMs);
111
112 /**
113 * Write an RTR frame to the CAN device with a specific ID. This ID is 10
114 * bits. The length by spec must match what is returned by the responding
115 * device
116 *
117 * @param length The length to request (0 to 8)
118 * @param apiId The API ID to write.
119 */
120 int WriteRTRFrameNoError(int length, int apiId);
121
122 /**
123 * Stop a repeating packet with a specific ID. This ID is 10 bits.
124 *
125 * @param apiId The API ID to stop repeating
126 */
127 void StopPacketRepeating(int apiId);
128
129 /**
130 * Read a new CAN packet. This will only return properly once per packet
131 * received. Multiple calls without receiving another packet will return
132 * false.
133 *
134 * @param apiId The API ID to read.
135 * @param data Storage for the received data.
136 * @return True if the data is valid, otherwise false.
137 */
138 bool ReadPacketNew(int apiId, CANData* data);
139
140 /**
141 * Read a CAN packet. The will continuously return the last packet received,
142 * without accounting for packet age.
143 *
144 * @param apiId The API ID to read.
145 * @param data Storage for the received data.
146 * @return True if the data is valid, otherwise false.
147 */
148 bool ReadPacketLatest(int apiId, CANData* data);
149
150 /**
151 * Read a CAN packet. The will return the last packet received until the
152 * packet is older then the requested timeout. Then it will return false.
153 *
154 * @param apiId The API ID to read.
155 * @param timeoutMs The timeout time for the packet
156 * @param data Storage for the received data.
157 * @return True if the data is valid, otherwise false.
158 */
159 bool ReadPacketTimeout(int apiId, int timeoutMs, CANData* data);
160
161 static constexpr HAL_CANManufacturer kTeamManufacturer = HAL_CAN_Man_kTeamUse;
162 static constexpr HAL_CANDeviceType kTeamDeviceType =
163 HAL_CAN_Dev_kMiscellaneous;
164
165 private:
166 hal::Handle<HAL_CANHandle> m_handle;
167};
168} // namespace frc
