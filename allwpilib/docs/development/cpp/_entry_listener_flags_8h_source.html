10/** Entry listener flags */
11namespace nt::EntryListenerFlags {

13/**
14 * Flag values for use with entry listeners.
15 *
16 * The flags are a bitmask and must be OR'ed together to indicate the
17 * combination of events desired to be received.
18 *
19 * The constants kNew, kDelete, kUpdate, and kFlags represent different events
20 * that can occur to entries.
21 *
22 * By default, notifications are only generated for remote changes occurring
23 * after the listener is created. The constants kImmediate and kLocal are
24 * modifiers that cause notifications to be generated at other times.
25 *
26 * @ingroup ntcore_cpp_api
27 */
28enum {
29 /**
30 * Initial listener addition.
31 * Set this flag to receive immediate notification of entries matching the
32 * flag criteria (generally only useful when combined with kNew).
33 */
34 kImmediate = NT_NOTIFY_IMMEDIATE,
35
36 /**
37 * Changed locally.
38 * Set this flag to receive notification of both local changes and changes
39 * coming from remote nodes. By default, notifications are only generated
40 * for remote changes. Must be combined with some combination of kNew,
41 * kDelete, kUpdate, and kFlags to receive notifications of those respective
42 * events.
43 */
44 kLocal = NT_NOTIFY_LOCAL,
45
46 /**
47 * Newly created entry.
48 * Set this flag to receive a notification when an entry is created.
49 */
50 kNew = NT_NOTIFY_NEW,
51
52 /**
53 * Entry was deleted.
54 * Set this flag to receive a notification when an entry is deleted.
55 */
56 kDelete = NT_NOTIFY_DELETE,
57
58 /**
59 * Entry's value changed.
60 * Set this flag to receive a notification when an entry's value (or type)
61 * changes.
62 */
63 kUpdate = NT_NOTIFY_UPDATE,
64
65 /**
66 * Entry's flags changed.
67 * Set this flag to receive a notification when an entry's flags value
68 * changes.
69 */
70 kFlags = NT_NOTIFY_FLAGS
71};
72
73} // namespace nt::EntryListenerFlags
