237template<class RealType, class CompatibleObjectType>
238struct is_compatible_object_type_impl<true, RealType, CompatibleObjectType>
239{
240 static constexpr auto value =

247{
248 static auto constexpr value = is_compatible_object_type_impl <
249 std::conjunction<std::negation<std::is_same<void, CompatibleObjectType>>,
250 has_mapped_type<CompatibleObjectType>,
251 has_key_type<CompatibleObjectType>>::value,
252 typename BasicJsonType::object_t, CompatibleObjectType >::value;
253};

267 static auto constexpr value =

702enum class value_t : std::uint8_t
703{
704 null,
705 object,
706 array,
707 string,
708 boolean,
709 number_integer,
710 number_unsigned,
711 number_float,
712 discarded
713};
714
725inline bool operator<(const value_t lhs, const value_t rhs) noexcept
726{
727 static constexpr std::array<std::uint8_t, 8> order = {{
728 0 /* null */, 3 /* object */, 4 /* array */, 5 /* string */,
729 1 /* boolean */, 2 /* integer */, 2 /* unsigned */, 2 /* float */
730 }
731 };
732
733 const auto l_index = static_cast<std::size_t>(lhs);
734 const auto r_index = static_cast<std::size_t>(rhs);
735 return l_index < order.size() and r_index < order.size() and order[l_index] < order[r_index];
736}

739template<typename BasicJsonType, typename ArithmeticType,
740 enable_if_t<std::is_arithmetic<ArithmeticType>::value and
741 not std::is_same<ArithmeticType, bool>::value,
742 int> = 0>
743void get_arithmetic_value(const BasicJsonType& j, ArithmeticType& val)
744{
745 switch (static_cast<value_t>(j))
746 {
747 case value_t::number_unsigned:
748 {
749 val = static_cast<ArithmeticType>(*j.template get_ptr<const uint64_t*>());
750 break;
751 }
752 case value_t::number_integer:
753 {
754 val = static_cast<ArithmeticType>(*j.template get_ptr<const int64_t*>());
755 break;
756 }
757 case value_t::number_float:
758 {
759 val = static_cast<ArithmeticType>(*j.template get_ptr<const double*>());
760 break;
761 }
762
763 default:
764 JSON_THROW(type_error::create(302, "type must be number, but is", j.type_name()));
765 }
766}

826void from_json_array_impl(const BasicJsonType& j, CompatibleArrayType& arr, priority_tag<0> /*unused*/)

885template<typename BasicJsonType>
886inline
887void from_json(const BasicJsonType& j, typename BasicJsonType::object_t& obj)
888{
889 if (!j.is_object())
890 {
891 JSON_THROW(type_error::create(302, "type must be object, but is", j.type_name()));
892 }
893
894 auto inner_object = j.template get_ptr<const typename BasicJsonType::object_t*>();
895 for (const auto& i : *inner_object) {
896 obj.try_emplace(i.first(), i.second);
897 }
898}

900template<typename BasicJsonType, typename CompatibleObjectType,
901 enable_if_t<is_compatible_object_type<BasicJsonType, CompatibleObjectType>::value and
902 not std::is_same<typename BasicJsonType::object_t, CompatibleObjectType>::value, int> = 0>
903void from_json(const BasicJsonType& j, CompatibleObjectType& obj)
904{
905 if (JSON_UNLIKELY(not j.is_object()))
906 {
907 JSON_THROW(type_error::create(302, "type must be object, but is", j.type_name()));
908 }
909
910 auto inner_object = j.template get_ptr<const typename BasicJsonType::object_t*>();
911 using std::begin;
912 using std::end;
913 using value_type = typename CompatibleObjectType::value_type;
914 std::vector<value_type> v;
915 v.reserve(j.size());
916 for (const auto& p : *inner_object)
917 {
918 v.emplace_back(
919 p.first(),
920 p.second
921 .template get<typename CompatibleObjectType::mapped_type>());
922 }
923 // we could avoid the assignment, but this might require a for loop, which
924 // might be less efficient than the container constructor for some
925 // containers (would it?)
926 obj = CompatibleObjectType(std::make_move_iterator(begin(v)),
927 std::make_move_iterator(end(v)));
928}
