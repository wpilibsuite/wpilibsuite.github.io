18namespace frc {
19/**
20 * This holonomic drive controller can be used to follow trajectories using a
21 * holonomic drivetrain (i.e. swerve or mecanum). Holonomic trajectory following
22 * is a much simpler problem to solve compared to skid-steer style drivetrains
23 * because it is possible to individually control forward, sideways, and angular
24 * velocity.
25 *
26 * The holonomic drive controller takes in one PID controller for each
27 * direction, forward and sideways, and one profiled PID controller for the
28 * angular direction. Because the heading dynamics are decoupled from
29 * translations, users can specify a custom heading that the drivetrain should
30 * point toward. This heading reference is profiled for smoothness.
31 */
32class WPILIB_DLLEXPORT HolonomicDriveController {
33 public:
34 /**
35 * Constructs a holonomic drive controller.
36 *
37 * @param xController A PID Controller to respond to error in the
38 * field-relative x direction.
39 * @param yController A PID Controller to respond to error in the
40 * field-relative y direction.
41 * @param thetaController A profiled PID controller to respond to error in
42 * angle.
43 */
44 HolonomicDriveController(
45 frc2::PIDController xController, frc2::PIDController yController,
46 ProfiledPIDController<units::radian> thetaController);
47
48 /**
49 * Returns true if the pose error is within tolerance of the reference.
50 */
51 bool AtReference() const;
52
53 /**
54 * Sets the pose error which is considered tolerable for use with
55 * AtReference().
56 *
57 * @param tolerance Pose error which is tolerable.
58 */
59 void SetTolerance(const Pose2d& tolerance);
60
61 /**
62 * Returns the next output of the holonomic drive controller.
63 *
64 * The reference pose, linear velocity, and angular velocity should come from
65 * a drivetrain trajectory.
66 *
67 * @param currentPose The current pose.
68 * @param poseRef The desired pose.
69 * @param linearVelocityRef The desired linear velocity.
70 * @param angleRef The desired ending angle.
71 */
72 ChassisSpeeds Calculate(const Pose2d& currentPose, const Pose2d& poseRef,
73 units::meters_per_second_t linearVelocityRef,
74 const Rotation2d& angleRef);
75
76 /**
77 * Returns the next output of the holonomic drive controller.
78 *
79 * The reference pose, linear velocity, and angular velocity should come from
80 * a drivetrain trajectory.
81 *
82 * @param currentPose The current pose.
83 * @param desiredState The desired pose, linear velocity, and angular velocity
84 * from a trajectory.
85 * @param angleRef The desired ending angle.
86 */
87 ChassisSpeeds Calculate(const Pose2d& currentPose,
88 const Trajectory::State& desiredState,
89 const Rotation2d& angleRef);
90
91 /**
92 * Enables and disables the controller for troubleshooting purposes. When
93 * Calculate() is called on a disabled controller, only feedforward values
94 * are returned.
95 *
96 * @param enabled If the controller is enabled or not.
97 */
98 void SetEnabled(bool enabled);
99
100 private:
101 Pose2d m_poseError;
102 Rotation2d m_rotationError;
103 Pose2d m_poseTolerance;
104 bool m_enabled = true;
105
106 frc2::PIDController m_xController;
107 frc2::PIDController m_yController;
108 ProfiledPIDController<units::radian> m_thetaController;
109
110 bool m_firstRun = true;
111};
112} // namespace frc
