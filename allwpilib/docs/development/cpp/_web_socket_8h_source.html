55 /**
56 * Connection states.
57 */
58 enum State {
59 /** The connection is not yet open. */
60 CONNECTING = 0,
61 /** The connection is open and ready to communicate. */
62 OPEN,
63 /** The connection is in the process of closing. */
64 CLOSING,
65 /** The connection failed. */
66 FAILED,
67 /** The connection is closed. */
68 CLOSED
69 };

84 /**
85 * Starts a client connection by performing the initial client handshake.
86 * An open event is emitted when the handshake completes.
87 * This sets the stream user data to the websocket.
88 * @param stream Connection stream
89 * @param uri The Request-URI to send
90 * @param host The host or host:port to send
91 * @param protocols The list of subprotocols
92 * @param options Handshake options
93 */
94 static std::shared_ptr<WebSocket> CreateClient(
95 uv::Stream& stream, std::string_view uri, std::string_view host,
96 span<const std::string_view> protocols = {},
97 const ClientOptions& options = {});

99 /**
100 * Starts a client connection by performing the initial client handshake.
101 * An open event is emitted when the handshake completes.
102 * This sets the stream user data to the websocket.
103 * @param stream Connection stream
104 * @param uri The Request-URI to send
105 * @param host The host or host:port to send
106 * @param protocols The list of subprotocols
107 * @param options Handshake options
108 */
109 static std::shared_ptr<WebSocket> CreateClient(
110 uv::Stream& stream, std::string_view uri, std::string_view host,
111 std::initializer_list<std::string_view> protocols,
112 const ClientOptions& options = {}) {
113 return CreateClient(stream, uri, host, {protocols.begin(), protocols.end()},
114 options);
115 }

117 /**
118 * Starts a server connection by performing the initial server side handshake.
119 * This should be called after the HTTP headers have been received.
120 * An open event is emitted when the handshake completes.
121 * This sets the stream user data to the websocket.
122 * @param stream Connection stream
123 * @param key The value of the Sec-WebSocket-Key header field in the client
124 * request
125 * @param version The value of the Sec-WebSocket-Version header field in the
126 * client request
127 * @param protocol The subprotocol to send to the client (in the
128 * Sec-WebSocket-Protocol header field).
129 */
130 static std::shared_ptr<WebSocket> CreateServer(
131 uv::Stream& stream, std::string_view key, std::string_view version,
132 std::string_view protocol = {});

271 /**
272 * Send a continuation frame. This is used to send additional parts of a
273 * message started with SendTextFragment() or SendBinaryFragment().
274 * @param data Data to send

276 * @param callback Callback which is invoked when the write completes.
277 */
278 void SendFragment(span<const uv::Buffer> data, bool fin,
279 std::function<void(span<uv::Buffer>, uv::Error)> callback) {
280 Send(kOpCont | (fin ? kFlagFin : 0), data, std::move(callback));
