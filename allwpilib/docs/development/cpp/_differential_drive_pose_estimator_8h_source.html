19/**
20 * This class wraps an Unscented Kalman Filter to fuse latency-compensated
21 * vision measurements with differential drive encoder measurements. It will
22 * correct for noisy vision measurements and encoder drift. It is intended to be
23 * an easy drop-in for DifferentialDriveOdometry. In fact, if you never call
24 * AddVisionMeasurement(), and only call Update(), this will behave exactly the
25 * same as DifferentialDriveOdometry.
26 *
27 * Update() should be called every robot loop (if your robot loops are faster or
28 * slower than the default, then you should change the nominal delta time via
29 * the constructor).
30 *
31 * AddVisionMeasurement() can be called as infrequently as you want; if you
32 * never call it, then this class will behave like regular encoder odometry.
33 *
34 * The state-space system used internally has the following states (x), inputs
35 * (u), and outputs (y):
36 *
37 * <strong> x = [x, y, theta, dist_l, dist_r]ᵀ </strong> in the field coordinate
38 * system containing x position, y position, heading, left encoder distance,
39 * and right encoder distance.
40 *
41 * <strong> u = [v_x, v_y, omega]ᵀ </strong> containing x velocity, y velocity,
42 * and angular velocity in the field coordinate system.
43 *
44 * NB: Using velocities make things considerably easier, because it means that
45 * teams don't have to worry about getting an accurate model. Basically, we
46 * suspect that it's easier for teams to get good encoder data than it is for
47 * them to perform system identification well enough to get a good model.
48 *
49 * <strong> y = [x, y, theta]ᵀ </strong> from vision containing x position, y
50 * position, and heading; or <strong>y = [dist_l, dist_r, theta] </strong>
51 * containing left encoder position, right encoder position, and gyro heading.
52 */
53class WPILIB_DLLEXPORT DifferentialDrivePoseEstimator {
54 public:
55 /**
56 * Constructs a DifferentialDrivePoseEstimator.
57 *
58 * @param gyroAngle The gyro angle of the robot.
59 * @param initialPose The estimated initial pose.
60 * @param stateStdDevs Standard deviations of model states.
61 * Increase these numbers to trust your
62 * model's state estimates less. This matrix
63 * is in the form
64 * [x, y, theta, dist_l, dist_r]ᵀ,
65 * with units in meters and radians.
66 * @param localMeasurementStdDevs Standard deviations of the encoder and gyro
67 * measurements. Increase these numbers to
68 * trust sensor readings from
69 * encoders and gyros less.
70 * This matrix is in the form
71 * [dist_l, dist_r, theta]ᵀ, with units in
72 * meters and radians.
73 * @param visionMeasurementStdDevs Standard deviations of the vision
74 * measurements. Increase these numbers to
75 * trust global measurements from
76 * vision less. This matrix is in the form
77 * [x, y, theta]ᵀ, with units in meters and
78 * radians.
79 * @param nominalDt The period of the loop calling Update().
80 */
81 DifferentialDrivePoseEstimator(
82 const Rotation2d& gyroAngle, const Pose2d& initialPose,
83 const wpi::array<double, 5>& stateStdDevs,
84 const wpi::array<double, 3>& localMeasurementStdDevs,
85 const wpi::array<double, 3>& visionMeasurementStdDevs,
86 units::second_t nominalDt = 0.02_s);
87
88 /**
89 * Sets the pose estimator's trust of global measurements. This might be used
90 * to change trust in vision measurements after the autonomous period, or to
91 * change trust as distance to a vision target increases.
92 *
93 * @param visionMeasurementStdDevs Standard deviations of the vision
94 * measurements. Increase these numbers to
95 * trust global measurements from vision
96 * less. This matrix is in the form
97 * [x, y, theta]ᵀ, with units in meters and
98 * radians.
99 */
100 void SetVisionMeasurementStdDevs(
101 const wpi::array<double, 3>& visionMeasurementStdDevs);
102
103 /**
104 * Resets the robot's position on the field.
105 *
106 * You NEED to reset your encoders to zero when calling this method. The
107 * gyroscope angle does not need to be reset here on the user's robot code.
108 * The library automatically takes care of offsetting the gyro angle.
109 *
110 * @param pose The estimated pose of the robot on the field.
111 * @param gyroAngle The current gyro angle.
112 */
113 void ResetPosition(const Pose2d& pose, const Rotation2d& gyroAngle);
114
115 /**
116 * Returns the pose of the robot at the current time as estimated by the
117 * Unscented Kalman Filter.
118 *
119 * @return The estimated robot pose.
120 */
121 Pose2d GetEstimatedPosition() const;
122
123 /**
124 * Adds a vision measurement to the Unscented Kalman Filter. This will correct
125 * the odometry pose estimate while still accounting for measurement noise.
126 *
127 * This method can be called as infrequently as you want, as long as you are
128 * calling Update() every loop.
129 *
130 * To promote stability of the pose estimate and make it robust to bad vision
131 * data, we recommend only adding vision measurements that are already within
132 * one meter or so of the current pose estimate.
133 *
134 * @param visionRobotPose The pose of the robot as measured by the vision
135 * camera.
136 * @param timestamp The timestamp of the vision measurement in seconds.
137 * Note that if you don't use your own time source by
138 * calling UpdateWithTime(), then you must use a
139 * timestamp with an epoch since FPGA startup (i.e. the
140 * epoch of this timestamp is the same epoch as
141 * frc::Timer::GetFPGATimestamp(). This means that
142 * you should use frc::Timer::GetFPGATimestamp() as
143 * your time source in this case.
144 */
145 void AddVisionMeasurement(const Pose2d& visionRobotPose,
146 units::second_t timestamp);
147
148 /**
149 * Adds a vision measurement to the Unscented Kalman Filter. This will correct
150 * the odometry pose estimate while still accounting for measurement noise.
151 *
152 * This method can be called as infrequently as you want, as long as you are
153 * calling Update() every loop.
154 *
155 * To promote stability of the pose estimate and make it robust to bad vision
156 * data, we recommend only adding vision measurements that are already within
157 * one meter or so of the current pose estimate.
158 *
159 * Note that the vision measurement standard deviations passed into this
160 * method will continue to apply to future measurements until a subsequent
161 * call to SetVisionMeasurementStdDevs() or this method.
162 *
163 * @param visionRobotPose The pose of the robot as measured by the
164 * vision camera.
165 * @param timestamp The timestamp of the vision measurement in
166 * seconds. Note that if you don't use your
167 * own time source by calling
168 * UpdateWithTime(), then you must use a
169 * timestamp with an epoch since FPGA startup
170 * (i.e. the epoch of this timestamp is the
171 * same epoch as
172 * frc::Timer::GetFPGATimestamp(). This means
173 * that you should use
174 * frc::Timer::GetFPGATimestamp() as your
175 * time source in this case.
176 * @param visionMeasurementStdDevs Standard deviations of the vision
177 * measurements. Increase these numbers to
178 * trust global measurements from vision
179 * less. This matrix is in the form
180 * [x, y, theta]ᵀ, with units in meters and
181 * radians.
182 */
183 void AddVisionMeasurement(
184 const Pose2d& visionRobotPose, units::second_t timestamp,
185 const wpi::array<double, 3>& visionMeasurementStdDevs) {
186 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
187 AddVisionMeasurement(visionRobotPose, timestamp);
188 }
189
190 /**
191 * Updates the Unscented Kalman Filter using only wheel encoder information.
192 * Note that this should be called every loop iteration.
193 *
194 * @param gyroAngle The current gyro angle.
195 * @param wheelSpeeds The velocities of the wheels in meters per second.
196 * @param leftDistance The distance traveled by the left encoder.
197 * @param rightDistance The distance traveled by the right encoder.
198 *
199 * @return The estimated pose of the robot.
200 */
201 Pose2d Update(const Rotation2d& gyroAngle,
202 const DifferentialDriveWheelSpeeds& wheelSpeeds,
203 units::meter_t leftDistance, units::meter_t rightDistance);
204
205 /**
206 * Updates the Unscented Kalman Filter using only wheel encoder information.
207 * Note that this should be called every loop iteration.
208 *
209 * @param currentTime The time at which this method was called.
210 * @param gyroAngle The current gyro angle.
211 * @param wheelSpeeds The velocities of the wheels in meters per second.
212 * @param leftDistance The distance traveled by the left encoder.
213 * @param rightDistance The distance traveled by the right encoder.
214 *
215 * @return The estimated pose of the robot.
216 */
217 Pose2d UpdateWithTime(units::second_t currentTime,
218 const Rotation2d& gyroAngle,
219 const DifferentialDriveWheelSpeeds& wheelSpeeds,
220 units::meter_t leftDistance,
221 units::meter_t rightDistance);
222
223 private:
224 UnscentedKalmanFilter<5, 3, 3> m_observer;
225 TimeInterpolatableBuffer<Pose2d> m_poseBuffer{1.5_s};
226 std::function<void(const Vectord<3>& u, const Vectord<3>& y)> m_visionCorrect;
227
228 Matrixd<3, 3> m_visionContR;
229
230 units::second_t m_nominalDt;
231 units::second_t m_prevTime = -1_s;
232
233 Rotation2d m_gyroOffset;
234 Rotation2d m_previousAngle;
235
236 template <int Dim>
237 static wpi::array<double, Dim> StdDevMatrixToArray(
238 const Vectord<Dim>& stdDevs);
239
240 static Vectord<5> F(const Vectord<5>& x, const Vectord<3>& u);
241 static Vectord<5> FillStateVector(const Pose2d& pose,
242 units::meter_t leftDistance,
243 units::meter_t rightDistance);
244};
