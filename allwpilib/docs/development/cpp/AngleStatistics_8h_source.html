22template <int States>
23Eigen::Matrix<double, States, 1> AngleResidual(
24 const Eigen::Matrix<double, States, 1>& a,
25 const Eigen::Matrix<double, States, 1>& b, int angleStateIdx) {
26 Eigen::Matrix<double, States, 1> ret = a - b;
27 ret[angleStateIdx] =
28 AngleModulus(units::radian_t{ret[angleStateIdx]}).to<double>();
29 return ret;
30}

56template <int States>
57Eigen::Matrix<double, States, 1> AngleAdd(
58 const Eigen::Matrix<double, States, 1>& a,
59 const Eigen::Matrix<double, States, 1>& b, int angleStateIdx) {
60 Eigen::Matrix<double, States, 1> ret = a + b;
61 ret[angleStateIdx] =
62 InputModulus(ret[angleStateIdx], -wpi::numbers::pi, wpi::numbers::pi);
63 return ret;
64}

88template <int CovDim, int States>
89Eigen::Matrix<double, CovDim, 1> AngleMean(
90 const Eigen::Matrix<double, CovDim, 2 * States + 1>& sigmas,
91 const Eigen::Matrix<double, 2 * States + 1, 1>& Wm, int angleStatesIdx) {
92 double sumSin = sigmas.row(angleStatesIdx)
93 .unaryExpr([](auto it) { return std::sin(it); })
94 .sum();
95 double sumCos = sigmas.row(angleStatesIdx)
96 .unaryExpr([](auto it) { return std::cos(it); })
97 .sum();
98
99 Eigen::Matrix<double, CovDim, 1> ret = sigmas * Wm;
100 ret[angleStatesIdx] = std::atan2(sumSin, sumCos);
101 return ret;
102}
