88 template <int CovDim, int States>
89 Eigen::Matrix<double, CovDim, 1> AngleMean(
90  const Eigen::Matrix<double, CovDim, 2 * States + 1>& sigmas,
91  const Eigen::Matrix<double, 2 * States + 1, 1>& Wm, int angleStatesIdx) {
92  double sumSin = sigmas.row(angleStatesIdx)
93  .unaryExpr([](auto it) { return std::sin(it); })
94  .sum();
95  double sumCos = sigmas.row(angleStatesIdx)
96  .unaryExpr([](auto it) { return std::cos(it); })
97  .sum();
98 
99  Eigen::Matrix<double, CovDim, 1> ret = sigmas * Wm;
100  ret[angleStatesIdx] = std::atan2(sumSin, sumCos);
101  return ret;
102 }
