19/**
20 * Flag values for use with value listeners.
21 *
22 * The flags are a bitmask and must be OR'ed together to indicate the
23 * combination of events desired to be received.
24 *
25 * By default, notifications are only generated for remote changes occurring
26 * after the listener is created. The constants kImmediate and kLocal are
27 * modifiers that cause notifications to be generated at other times.
28 */
29struct ValueListenerFlags {
30 ValueListenerFlags() = delete;
31
32 /**
33 * Initial listener addition.
34 *
35 * Set this flag to receive immediate notification of the current value.
36 */
37 static constexpr unsigned int kImmediate = NT_VALUE_NOTIFY_IMMEDIATE;
38
39 /**
40 * Changed locally.
41 *
42 * Set this flag to receive notification of both local changes and changes
43 * coming from remote nodes. By default, notifications are only generated for
44 * remote changes.
45 */
46 static constexpr unsigned int kLocal = NT_VALUE_NOTIFY_LOCAL;
47};

49/**
50 * Value change listener. This calls back to a callback function when a value
51 * change matching the specified mask occurs. The callback function is called
52 * asynchronously on a separate thread, so it's important to use synchronization
53 * or atomics when accessing any shared state from the callback function.
54 */
55class ValueListener final {
56 public:
57 ValueListener() = default;
58
59 /**
60 * Create a listener for value changes on a subscriber. This does NOT keep the
61 * subscriber active.
62 *
63 * @param subscriber Subscriber
64 * @param mask Bitmask of ValueListenerFlags values
65 * @param listener Listener function
66 */
67 ValueListener(Subscriber& subscriber, unsigned int mask,
68 std::function<void(const ValueNotification&)> listener);

70 /**
71 * Create a listener for value changes on a subscriber. This does NOT keep the

76 * @param listener Listener function
77 */
78 ValueListener(MultiSubscriber& subscriber, unsigned int mask,
79 std::function<void(const ValueNotification&)> listener);

81 /**

83 *
84 * @param entry Entry
85 * @param mask Bitmask of ValueListenerFlags values
86 * @param listener Listener function

89 std::function<void(const ValueNotification&)> listener);

106 /**
107 * Wait for the value listener queue to be empty. This is primarily useful for
108 * deterministic testing. This blocks until either the value listener queue is
109 * empty (e.g. there are no more events that need to be passed along to
110 * callbacks or poll queues) or the timeout expires.
111 *
112 * @param timeout timeout, in seconds. Set to 0 for non-blocking behavior, or
113 * a negative value to block indefinitely
114 * @return False if timed out, otherwise true.
115 */
116 bool WaitForQueue(double timeout);

118 private:
119 NT_ValueListener m_handle{0};
