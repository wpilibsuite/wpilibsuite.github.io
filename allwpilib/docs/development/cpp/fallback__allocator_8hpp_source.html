20 /// A \concept{raw_allocator,RawAllocator} with a fallback.
21 /// Allocation first tries `Default`, if it fails,
22 /// it uses `Fallback`.
23 /// \requires `Default` must be a composable \concept{concept_rawallocator,RawAllocator},
24 /// `Fallback` must be a \concept{concept_rawallocator,RawAllocator}.
25 /// \ingroup adapter
26 template <class Default, class Fallback>
27 class fallback_allocator
28 : WPI_EBO(detail::ebo_storage<0, typename allocator_traits<Default>::allocator_type>),
29 WPI_EBO(detail::ebo_storage<1, typename allocator_traits<Fallback>::allocator_type>)
30 {
31 using default_traits = allocator_traits<Default>;
32 using default_composable_traits = composable_allocator_traits<Default>;
33 using fallback_traits = allocator_traits<Fallback>;
34 using fallback_composable_traits = composable_allocator_traits<Fallback>;
35 using fallback_composable =
36 is_composable_allocator<typename fallback_traits::allocator_type>;
37
38 public:
39 using default_allocator_type = typename allocator_traits<Default>::allocator_type;
40 using fallback_allocator_type = typename allocator_traits<Fallback>::allocator_type;
41
42 using is_stateful =
43 std::integral_constant<bool, default_traits::is_stateful::value
44 || fallback_traits::is_stateful::value>;

46 /// \effects Default constructs both allocators.
47 /// \notes This function only participates in overload resolution, if both allocators are not stateful.

51 detail::ebo_storage<1, fallback_allocator_type>({})

63 /// @{
64 /// \effects First calls the compositioning (de)allocation function on the `default_allocator_type`.
65 /// If that fails, uses the non-compositioning function of the `fallback_allocator_type`.
66 void* allocate_node(std::size_t size, std::size_t alignment)
67 {
68 auto ptr = default_composable_traits::try_allocate_node(get_default_allocator(),
69 size, alignment);

72 return ptr;
