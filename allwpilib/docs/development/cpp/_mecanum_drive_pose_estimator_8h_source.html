20namespace frc {
21/**
22 * This class wraps an Unscented Kalman Filter to fuse latency-compensated
23 * vision measurements with mecanum drive encoder velocity measurements. It will
24 * correct for noisy measurements and encoder drift. It is intended to be an
25 * easy but more accurate drop-in for MecanumDriveOdometry.
26 *
27 * Update() should be called every robot loop. If your loops are faster or
28 * slower than the default of 0.02s, then you should change the nominal delta
29 * time by specifying it in the constructor.
30 *
31 * AddVisionMeasurement() can be called as infrequently as you want; if you
32 * never call it, then this class will behave mostly like regular encoder
33 * odometry.
34 *
35 * The state-space system used internally has the following states (x), inputs
36 * (u), and outputs (y):
37 *
38 * <strong> x = [x, y, theta]ᵀ </strong> in the field coordinate system
39 * containing x position, y position, and heading.
40 *
41 * <strong> u = [v_l, v_r, dtheta]ᵀ </strong> containing left wheel velocity,
42 * right wheel velocity, and change in gyro heading.
43 *
44 * <strong> y = [x, y, theta]ᵀ </strong> from vision containing x position, y
45 * position, and heading; or <strong> y = [theta]ᵀ </strong> containing gyro
46 * heading.
47 */
48class WPILIB_DLLEXPORT MecanumDrivePoseEstimator {
49 public:
50 /**
51 * Constructs a MecanumDrivePoseEstimator.
52 *
53 * @param gyroAngle The current gyro angle.
54 * @param initialPose The starting pose estimate.
55 * @param kinematics A correctly-configured kinematics object
56 * for your drivetrain.
57 * @param stateStdDevs Standard deviations of model states.
58 * Increase these numbers to trust your
59 * model's state estimates less. This matrix
60 * is in the form [x, y, theta]ᵀ, with units
61 * in meters and radians.
62 * @param localMeasurementStdDevs Standard deviations of the encoder and gyro
63 * measurements. Increase these numbers to
64 * trust sensor readings from encoders
65 * and gyros less. This matrix is in the form
66 * [theta], with units in radians.
67 * @param visionMeasurementStdDevs Standard deviations of the vision
68 * measurements. Increase these numbers to
69 * trust global measurements from vision
70 * less. This matrix is in the form
71 * [x, y, theta]ᵀ, with units in meters and
72 * radians.
73 * @param nominalDt The time in seconds between each robot
74 * loop.
75 */
76 MecanumDrivePoseEstimator(
77 const Rotation2d& gyroAngle, const Pose2d& initialPose,
78 MecanumDriveKinematics kinematics,
79 const wpi::array<double, 3>& stateStdDevs,
80 const wpi::array<double, 1>& localMeasurementStdDevs,
81 const wpi::array<double, 3>& visionMeasurementStdDevs,
82 units::second_t nominalDt = 0.02_s);
83
84 /**
85 * Sets the pose estimator's trust of global measurements. This might be used
86 * to change trust in vision measurements after the autonomous period, or to
87 * change trust as distance to a vision target increases.
88 *
89 * @param visionMeasurementStdDevs Standard deviations of the vision
90 * measurements. Increase these numbers to
91 * trust global measurements from vision
92 * less. This matrix is in the form
93 * [x, y, theta]ᵀ, with units in meters and
94 * radians.
95 */
96 void SetVisionMeasurementStdDevs(
97 const wpi::array<double, 3>& visionMeasurementStdDevs);
98
99 /**
100 * Resets the robot's position on the field.
101 *
102 * <p>You NEED to reset your encoders (to zero) when calling this method.
103 *
104 * <p>The gyroscope angle does not need to be reset in the user's robot code.
105 * The library automatically takes care of offsetting the gyro angle.
106 *
107 * @param pose The position on the field that your robot is at.
108 * @param gyroAngle The angle reported by the gyroscope.
109 */
110 void ResetPosition(const Pose2d& pose, const Rotation2d& gyroAngle);
111
112 /**
113 * Gets the pose of the robot at the current time as estimated by the Extended
114 * Kalman Filter.
115 *
116 * @return The estimated robot pose in meters.
117 */
118 Pose2d GetEstimatedPosition() const;
119
120 /**
121 * Add a vision measurement to the Unscented Kalman Filter. This will correct
122 * the odometry pose estimate while still accounting for measurement noise.
123 *
124 * This method can be called as infrequently as you want, as long as you are
125 * calling Update() every loop.
126 *
127 * @param visionRobotPose The pose of the robot as measured by the vision
128 * camera.
129 * @param timestamp The timestamp of the vision measurement in seconds.
130 * Note that if you don't use your own time source by
131 * calling UpdateWithTime() then you must use a
132 * timestamp with an epoch since FPGA startup
133 * (i.e. the epoch of this timestamp is the same
134 * epoch as Timer#GetFPGATimestamp.) This means
135 * that you should use Timer#GetFPGATimestamp as your
136 * time source or sync the epochs.
137 */
138 void AddVisionMeasurement(const Pose2d& visionRobotPose,
139 units::second_t timestamp);
140
141 /**
142 * Adds a vision measurement to the Unscented Kalman Filter. This will correct
143 * the odometry pose estimate while still accounting for measurement noise.
144 *
145 * This method can be called as infrequently as you want, as long as you are
146 * calling Update() every loop.
147 *
148 * Note that the vision measurement standard deviations passed into this
149 * method will continue to apply to future measurements until a subsequent
150 * call to SetVisionMeasurementStdDevs() or this method.
151 *
152 * @param visionRobotPose The pose of the robot as measured by the
153 * vision camera.
154 * @param timestamp The timestamp of the vision measurement in
155 * seconds. Note that if you don't use your
156 * own time source by calling
157 * UpdateWithTime(), then you must use a
158 * timestamp with an epoch since FPGA startup
159 * (i.e. the epoch of this timestamp is the
160 * same epoch as
161 * frc::Timer::GetFPGATimestamp(). This means
162 * that you should use
163 * frc::Timer::GetFPGATimestamp() as your
164 * time source in this case.
165 * @param visionMeasurementStdDevs Standard deviations of the vision
166 * measurements. Increase these numbers to
167 * trust global measurements from vision
168 * less. This matrix is in the form
169 * [x, y, theta]ᵀ, with units in meters and
170 * radians.
171 */
172 void AddVisionMeasurement(
173 const Pose2d& visionRobotPose, units::second_t timestamp,
174 const wpi::array<double, 3>& visionMeasurementStdDevs) {
175 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
176 AddVisionMeasurement(visionRobotPose, timestamp);
177 }
178
179 /**
180 * Updates the the Unscented Kalman Filter using only wheel encoder
181 * information. This should be called every loop, and the correct loop period
182 * must be passed into the constructor of this class.
183 *
184 * @param gyroAngle The current gyro angle.
185 * @param wheelSpeeds The current speeds of the mecanum drive wheels.
186 * @return The estimated pose of the robot in meters.
187 */
188 Pose2d Update(const Rotation2d& gyroAngle,
189 const MecanumDriveWheelSpeeds& wheelSpeeds);
190
191 /**
192 * Updates the the Unscented Kalman Filter using only wheel encoder
193 * information. This should be called every loop, and the correct loop period
194 * must be passed into the constructor of this class.
195 *
196 * @param currentTime Time at which this method was called, in seconds.
197 * @param gyroAngle The current gyroscope angle.
198 * @param wheelSpeeds The current speeds of the mecanum drive wheels.
199 * @return The estimated pose of the robot in meters.
200 */
201 Pose2d UpdateWithTime(units::second_t currentTime,
202 const Rotation2d& gyroAngle,
203 const MecanumDriveWheelSpeeds& wheelSpeeds);
204
205 private:
206 UnscentedKalmanFilter<3, 3, 1> m_observer;
207 MecanumDriveKinematics m_kinematics;
208 KalmanFilterLatencyCompensator<3, 3, 1, UnscentedKalmanFilter<3, 3, 1>>
209 m_latencyCompensator;
210 std::function<void(const Eigen::Vector<double, 3>& u,
211 const Eigen::Vector<double, 3>& y)>
212 m_visionCorrect;
213
214 Eigen::Matrix3d m_visionContR;
215
216 units::second_t m_nominalDt;
217 units::second_t m_prevTime = -1_s;
218
219 Rotation2d m_gyroOffset;
220 Rotation2d m_previousAngle;
221
222 template <int Dim>
223 static wpi::array<double, Dim> StdDevMatrixToArray(
224 const Eigen::Vector<double, Dim>& vector) {
225 wpi::array<double, Dim> array;
226 for (size_t i = 0; i < Dim; ++i) {
227 array[i] = vector(i);
228 }
229 return array;
230 }
231};
