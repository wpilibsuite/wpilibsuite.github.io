20template <int Rows, int Cols, typename Matrix, typename T, typename... Ts>
21void MatrixImpl(Matrix& result, T elem, Ts... elems) {
22 constexpr int count = Rows * Cols - (sizeof...(Ts) + 1);
23
24 result(count / Cols, count % Cols) = elem;
25 if constexpr (sizeof...(Ts) > 0) {
26 MatrixImpl<Rows, Cols>(result, elems...);
27 }
28}

46template <typename Matrix, typename T, typename... Ts>
47void WhiteNoiseVectorImpl(Matrix& result, T elem, Ts... elems) {
48 std::random_device rd;
49 std::mt19937 gen{rd()};
50 std::normal_distribution<> distr{0.0, elem};
51
52 result(result.rows() - (sizeof...(Ts) + 1)) = distr(gen);
53 if constexpr (sizeof...(Ts) > 0) {
54 WhiteNoiseVectorImpl(result, elems...);
55 }

58template <int States, int Inputs>
59bool IsStabilizableImpl(const Eigen::Matrix<double, States, States>& A,
60 const Eigen::Matrix<double, States, Inputs>& B) {
61 Eigen::EigenSolver<Eigen::Matrix<double, States, States>> es(A);
62
63 for (int i = 0; i < States; ++i) {
64 if (es.eigenvalues()[i].real() * es.eigenvalues()[i].real() +
65 es.eigenvalues()[i].imag() * es.eigenvalues()[i].imag() <
66 1) {
67 continue;
68 }
69
70 Eigen::Matrix<std::complex<double>, States, States + Inputs> E;
71 E << es.eigenvalues()[i] * Eigen::Matrix<std::complex<double>, States,
72 States>::Identity() -
73 A,
74 B;
75
76 Eigen::ColPivHouseholderQR<
77 Eigen::Matrix<std::complex<double>, States, States + Inputs>>
78 qr(E);
79 if (qr.rank() < States) {
80 return false;
81 }
82 }
83 return true;
84}

163template <size_t N>
164Eigen::Matrix<double, N, N> MakeCostMatrix(const std::array<double, N>& costs) {
165 Eigen::DiagonalMatrix<double, N> result;
166 auto& diag = result.diagonal();
167 for (size_t i = 0; i < N; ++i) {
168 diag(i) = 1.0 / std::pow(costs[i], 2);
169 }
170 return result;
171}

212template <int N>
213Eigen::Matrix<double, N, 1> MakeWhiteNoiseVector(
214 const std::array<double, N>& stdDevs) {
215 std::random_device rd;
216 std::mt19937 gen{rd()};
217
218 Eigen::Matrix<double, N, 1> result;
219 for (int i = 0; i < N; ++i) {
220 // Passing a standard deviation of 0.0 to std::normal_distribution is
221 // undefined behavior
222 if (stdDevs[i] == 0.0) {
223 result(i) = 0.0;
224 } else {
225 std::normal_distribution distr{0.0, stdDevs[i]};
226 result(i) = distr(gen);
227 }
228 }
229 return result;
230}

293template <int Inputs>
294Eigen::Matrix<double, Inputs, 1> ClampInputMaxMagnitude(
295 const Eigen::Matrix<double, Inputs, 1>& u,
296 const Eigen::Matrix<double, Inputs, 1>& umin,
297 const Eigen::Matrix<double, Inputs, 1>& umax) {
298 Eigen::Matrix<double, Inputs, 1> result;
299 for (int i = 0; i < Inputs; ++i) {
300 result(i) = std::clamp(u(i), umin(i), umax(i));
301 }
302 return result;
303}

314template <int Inputs>
315Eigen::Matrix<double, Inputs, 1> NormalizeInputVector(
316 const Eigen::Matrix<double, Inputs, 1>& u, double maxMagnitude) {
317 double maxValue = u.template lpNorm<Eigen::Infinity>();
318
319 if (maxValue > maxMagnitude) {
320 return u * maxMagnitude / maxValue;
321 }
322 return u;
323}
