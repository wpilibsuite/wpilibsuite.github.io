57 template <typename... Wheels>
58 explicit SwerveDriveKinematics(Translation2d wheel, Wheels&&... wheels)
59 : m_modules{wheel, wheels...} {
60 static_assert(sizeof...(wheels) >= 1,
61 "A swerve drive requires at least two modules");
62
63 for (size_t i = 0; i < NumModules; i++) {
64 // clang-format off
65 m_inverseKinematics.template block<2, 3>(i * 2, 0) <<
66 1, 0, (-m_modules[i].Y()).template to<double>(),
67 0, 1, (+m_modules[i].X()).template to<double>();
68 // clang-format on
69 }
70
71 m_forwardKinematics = m_inverseKinematics.householderQr();
72
73 wpi::math::MathSharedStore::ReportUsage(
74 wpi::math::MathUsageId::kKinematics_SwerveDrive, 1);
75 }
