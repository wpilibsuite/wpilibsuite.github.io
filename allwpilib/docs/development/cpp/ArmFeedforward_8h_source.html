20class WPILIB_DLLEXPORT ArmFeedforward {
21 public:
22 using Angle = units::radians;
23 using Velocity = units::radians_per_second;
24 using Acceleration = units::compound_unit<units::radians_per_second,
25 units::inverse<units::second>>;
26 using kv_unit =
27 units::compound_unit<units::volts,
28 units::inverse<units::radians_per_second>>;
29 using ka_unit =
30 units::compound_unit<units::volts, units::inverse<Acceleration>>;
31
32 constexpr ArmFeedforward() = default;
33
42 constexpr ArmFeedforward(
43 units::volt_t kS, units::volt_t kCos, units::unit_t<kv_unit> kV,
44 units::unit_t<ka_unit> kA = units::unit_t<ka_unit>(0))
45 : kS(kS), kCos(kCos), kV(kV), kA(kA) {}

55 units::volt_t Calculate(units::unit_t<Angle> angle,
56 units::unit_t<Velocity> velocity,
57 units::unit_t<Acceleration> acceleration =
58 units::unit_t<Acceleration>(0)) const {
59 return kS * wpi::sgn(velocity) + kCos * units::math::cos(angle) +
60 kV * velocity + kA * acceleration;
61 }
62
63 // Rearranging the main equation from the calculate() method yields the
64 // formulas for the methods below:
65
78 units::unit_t<Velocity> MaxAchievableVelocity(
79 units::volt_t maxVoltage, units::unit_t<Angle> angle,
80 units::unit_t<Acceleration> acceleration) {
81 // Assume max velocity is positive
82 return (maxVoltage - kS - kCos * units::math::cos(angle) -
83 kA * acceleration) /
84 kV;
85 }
86
99 units::unit_t<Velocity> MinAchievableVelocity(
100 units::volt_t maxVoltage, units::unit_t<Angle> angle,
101 units::unit_t<Acceleration> acceleration) {
102 // Assume min velocity is negative, ks flips sign
103 return (-maxVoltage + kS - kCos * units::math::cos(angle) -
104 kA * acceleration) /
105 kV;
106 }
107
120 units::unit_t<Acceleration> MaxAchievableAcceleration(
121 units::volt_t maxVoltage, units::unit_t<Angle> angle,
122 units::unit_t<Velocity> velocity) {
123 return (maxVoltage - kS * wpi::sgn(velocity) -
124 kCos * units::math::cos(angle) - kV * velocity) /
125 kA;
126 }
127
140 units::unit_t<Acceleration> MinAchievableAcceleration(
141 units::volt_t maxVoltage, units::unit_t<Angle> angle,
142 units::unit_t<Velocity> velocity) {
143 return MaxAchievableAcceleration(-maxVoltage, angle, velocity);
144 }
145
146 units::volt_t kS{0};
147 units::volt_t kCos{0};
148 units::unit_t<kv_unit> kV{0};
149 units::unit_t<ka_unit> kA{0};
150};
151} // namespace frc
