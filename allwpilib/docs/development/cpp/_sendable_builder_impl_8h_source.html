152 private:
153 struct Property {
154 Property(nt::NetworkTable& table, std::string_view key)
155 : entry(table.GetEntry(key)) {}
156
157 Property(const Property&) = delete;
158 Property& operator=(const Property&) = delete;
159
160 Property(Property&& other) noexcept
161 : entry(other.entry),
162 listener(other.listener),
163 update(std::move(other.update)),
164 createListener(std::move(other.createListener)) {
165 other.entry = nt::NetworkTableEntry();
166 other.listener = 0;
167 }
168
169 Property& operator=(Property&& other) noexcept {
170 entry = other.entry;
171 listener = other.listener;
172 other.entry = nt::NetworkTableEntry();
173 other.listener = 0;
174 update = std::move(other.update);
175 createListener = std::move(other.createListener);
176 return *this;
177 }
178
179 ~Property() { StopListener(); }
180
181 void StartListener() {
182 if (entry && listener == 0 && createListener) {
183 listener = createListener(entry);
184 }
185 }
186
187 void StopListener() {
188 if (entry && listener != 0) {
189 entry.RemoveListener(listener);
190 listener = 0;
191 }
192 }
193
194 nt::NetworkTableEntry entry;
195 NT_EntryListener listener = 0;
196 std::function<void(nt::NetworkTableEntry entry, uint64_t time)> update;
197 std::function<NT_EntryListener(nt::NetworkTableEntry entry)> createListener;
198 };
