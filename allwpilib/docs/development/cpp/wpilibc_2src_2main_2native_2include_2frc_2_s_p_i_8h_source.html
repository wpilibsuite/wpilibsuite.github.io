121 /**
122 * Write data to the peripheral device. Blocks until there is space in the
123 * output FIFO.
124 *
125 * If not running in output only mode, also saves the data received
126 * on the CIPO input during the transfer into the receive FIFO.
127 */
128 virtual int Write(uint8_t* data, int size);
129
130 /**
131 * Read a word from the receive FIFO.
132 *
133 * Waits for the current transfer to complete if the receive FIFO is empty.
134 *
135 * If the receive FIFO is empty, there is no active transfer, and initiate
136 * is false, errors.
137 *
138 * @param initiate If true, this function pushes "0" into the transmit
139 * buffer and initiates a transfer. If false, this
140 * function assumes that data is already in the receive
141 * FIFO from a previous write.
142 * @param dataReceived Buffer to receive data from the device
143 * @param size The length of the transaction, in bytes
144 */
145 virtual int Read(bool initiate, uint8_t* dataReceived, int size);
146
147 /**
148 * Perform a simultaneous read/write transaction with the device
149 *
150 * @param dataToSend The data to be written out to the device
151 * @param dataReceived Buffer to receive data from the device
152 * @param size The length of the transaction, in bytes
153 */
154 virtual int Transaction(uint8_t* dataToSend, uint8_t* dataReceived, int size);
155
156 /**
157 * Initialize automatic SPI transfer engine.
158 *
159 * Only a single engine is available, and use of it blocks use of all other
160 * chip select usage on the same physical SPI port while it is running.
161 *
162 * @param bufferSize buffer size in bytes
163 */
164 void InitAuto(int bufferSize);
165
166 /**
167 * Frees the automatic SPI transfer engine.
168 */
169 void FreeAuto();

171 /**
172 * Set the data to be transmitted by the engine.
173 *
174 * Up to 16 bytes are configurable, and may be followed by up to 127 zero
175 * bytes.
176 *
177 * @param dataToSend data to send (maximum 16 bytes)
178 * @param zeroSize number of zeros to send after the data
179 */
180 void SetAutoTransmitData(wpi::span<const uint8_t> dataToSend, int zeroSize);
181
182 /**
183 * Start running the automatic SPI transfer engine at a periodic rate.
184 *

186 * function.

188 * @param period period between transfers (us resolution)

190 void StartAutoRate(units::second_t period);
191
192 /**
193 * Start running the automatic SPI transfer engine at a periodic rate.
194 *
195 * InitAuto() and SetAutoTransmitData() must be called before calling this
196 * function.
197 *
198 * @deprecated use unit-safe StartAutoRate(units::second_t period) instead.
199 *
200 * @param period period between transfers, in seconds (us resolution)
201 */
202 WPI_DEPRECATED("Use StartAutoRate with unit-safety instead")
203 void StartAutoRate(double period);
204
205 /**
206 * Start running the automatic SPI transfer engine when a trigger occurs.
207 *
208 * InitAuto() and SetAutoTransmitData() must be called before calling this
209 * function.
210 *
211 * @param source digital source for the trigger (may be an analog trigger)
212 * @param rising trigger on the rising edge
213 * @param falling trigger on the falling edge
214 */
215 void StartAutoTrigger(DigitalSource& source, bool rising, bool falling);
216
217 /**
218 * Stop running the automatic SPI transfer engine.

220 void StopAuto();

222 /**
223 * Force the engine to make a single transfer.
224 */
225 void ForceAutoRead();

227 /**
228 * Read data that has been transferred by the automatic SPI transfer engine.
229 *
230 * Transfers may be made a byte at a time, so it's necessary for the caller
231 * to handle cases where an entire transfer has not been completed.
232 *
233 * Each received data sequence consists of a timestamp followed by the
234 * received data bytes, one byte per word (in the least significant byte).
235 * The length of each received data sequence is the same as the combined
236 * size of the data and zeroSize set in SetAutoTransmitData().
237 *
238 * Blocks until numToRead words have been read or timeout expires.
239 * May be called with numToRead=0 to retrieve how many words are available.
240 *
241 * @param buffer buffer where read words are stored
242 * @param numToRead number of words to read
243 * @param timeout timeout (ms resolution)
244 * @return Number of words remaining to be read
245 */
246 int ReadAutoReceivedData(uint32_t* buffer, int numToRead,
247 units::second_t timeout);
248
249 /**
250 * Get the number of bytes dropped by the automatic SPI transfer engine due
251 * to the receive buffer being full.
252 *
253 * @return Number of bytes dropped
254 */
255 int GetAutoDroppedCount();
256
257 /**
258 * Configure the Auto SPI Stall time between reads.
259 *
260 * @param port The number of the port to use. 0-3 for Onboard CS0-CS2, 4 for
261 * MXP.
262 * @param csToSclkTicks the number of ticks to wait before asserting the cs
263 * pin
264 * @param stallTicks the number of ticks to stall for
265 * @param pow2BytesPerRead the number of bytes to read before stalling
266 */
267 void ConfigureAutoStall(HAL_SPIPort port, int csToSclkTicks, int stallTicks,
268 int pow2BytesPerRead);

271 * Initialize the accumulator.

324 /**
325 * Set the center value of the accumulator.
326 *
327 * The center value is subtracted from each value before it is added to the

329 * accelerometers to make integration work and to take the device offset into

348 */
349 int64_t GetAccumulatorValue() const;
350
351 /**
352 * Read the number of accumulated values.
353 *
354 * Read the count of the accumulated values since the accumulator was last
355 * Reset().
356 *
357 * @return The number of times samples from the channel were accumulated.
358 */
359 int64_t GetAccumulatorCount() const;
360
361 /**
362 * Read the average of the accumulated value.
363 *
364 * @return The accumulated average value (value / count).
365 */
366 double GetAccumulatorAverage() const;
367
368 /**
369 * Read the accumulated value and the number of accumulated values atomically.
370 *
371 * This function reads the value and count atomically.
372 * This can be used for averaging.
373 *
374 * @param value Pointer to the 64-bit accumulated output.
375 * @param count Pointer to the number of accumulation cycles.
376 */
377 void GetAccumulatorOutput(int64_t& value, int64_t& count) const;
378
379 /**
380 * Set the center value of the accumulator integrator.
381 *
382 * The center value is subtracted from each value*dt before it is added to the
383 * integrated value. This is used for the center value of devices like gyros
384 * and accelerometers to take the device offset into account when integrating.
385 */
386 void SetAccumulatorIntegratedCenter(double center);
387
388 /**
389 * Read the integrated value. This is the sum of (each value * time between
390 * values).
391 *
392 * @return The integrated value accumulated since the last Reset().
393 */
394 double GetAccumulatorIntegratedValue() const;
395
396 /**
397 * Read the average of the integrated value. This is the sum of (each value
398 * times the time between values), divided by the count.
399 *
400 * @return The average of the integrated value accumulated since the last
401 * Reset().
402 */
403 double GetAccumulatorIntegratedAverage() const;
404
