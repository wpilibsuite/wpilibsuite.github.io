12/**
13 * This class provides an easy way to link commands to OI inputs.
14 *
15 * It is very easy to link a button to a command. For instance, you could link
16 * the trigger button of a joystick to a "score" command.
17 *
18 * This class represents a subclass of Trigger that is specifically aimed at
19 * buttons on an operator interface as a common use case of the more generalized
20 * Trigger objects. This is a simple wrapper around Trigger with the method
21 * names renamed to fit the Button object use.
22 *
23 * This class is provided by the OldCommands VendorDep
24 */
25class Button : public Trigger {
26 public:
27 Button() = default;
28 Button(Button&&) = default;
29 Button& operator=(Button&&) = default;
30
31 /**
32 * Specifies the command to run when a button is first pressed.
33 *
34 * @param command The pointer to the command to run
35 */
36 virtual void WhenPressed(Command* command);
37
38 /**
39 * Specifies the command to be scheduled while the button is pressed.
40 *
41 * The command will be scheduled repeatedly while the button is pressed and
42 * will be canceled when the button is released.
43 *
44 * @param command The pointer to the command to run
45 */
46 virtual void WhileHeld(Command* command);
47
48 /**
49 * Specifies the command to run when the button is released.
50 *
51 * The command will be scheduled a single time.
52 *
53 * @param command The pointer to the command to run
54 */
55 virtual void WhenReleased(Command* command);
56
57 /**
58 * Cancels the specificed command when the button is pressed.
59 *
60 * @param command The command to be canceled
61 */
62 virtual void CancelWhenPressed(Command* command);
63
64 /**
65 * Toggle the specified command when the button is pressed.
66 *
67 * @param command The command to be toggled
68 */
69 virtual void ToggleWhenPressed(Command* command);
70};
