25/**
26 * The scheduler responsible for running Commands. A Command-based robot should
27 * call Run() on the singleton instance in its periodic block in order to run
28 * commands synchronously from the main loop. Subsystems should be registered
29 * with the scheduler using RegisterSubsystem() in order for their Periodic()
30 * methods to be called and for their default commands to be scheduled.

32 * This class is provided by the NewCommands VendorDep

112 /**
113 * Runs a single iteration of the scheduler. The execution occurs in the
114 * following order:
115 *
116 * <p>Subsystem periodic methods are called.
117 *
118 * <p>Button bindings are polled, and new commands are scheduled from them.
119 *
120 * <p>Currently-scheduled commands are executed.
121 *
122 * <p>End conditions are checked on currently-scheduled commands, and commands
123 * that are finished have their end methods called and are removed.
124 *
125 * <p>Any subsystems not being used as requirements have their default methods
126 * started.
127 */
128 void Run();

155 /**
156 * Sets the default command for a subsystem. Registers that subsystem if it
157 * is not already registered. Default commands will run whenever there is no
158 * other command currently scheduled that requires the subsystem. Default
159 * commands should be written to never end (i.e. their IsFinished() method
160 * should return false), as they would simply be re-scheduled if they do.
161 * Default commands must also require their subsystem.
162 *
163 * @param subsystem the subsystem whose default command will be set
164 * @param defaultCommand the default command to associate with the subsystem
165 */
166 template <class T, typename = std::enable_if_t<std::is_base_of_v<
167 Command, std::remove_reference_t<T>>>>
168 void SetDefaultCommand(Subsystem* subsystem, T&& defaultCommand) {
169 if (!defaultCommand.HasRequirement(subsystem)) {
170 throw FRC_MakeError(frc::err::CommandIllegalUse, "{}",
171 "Default commands must require their subsystem!");
172 }
173 if (defaultCommand.IsFinished()) {
174 throw FRC_MakeError(frc::err::CommandIllegalUse, "{}",
175 "Default commands should not end!");
176 }
177 SetDefaultCommandImpl(subsystem,
178 std::make_unique<std::remove_reference_t<T>>(
179 std::forward<T>(defaultCommand)));
180 }
