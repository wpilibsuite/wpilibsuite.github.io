23/**
24 * Class for representing a layout of AprilTags on a field and reading them from
25 * a JSON format.
26 *
27 * The JSON format contains two top-level objects, "tags" and "field".
28 * The "tags" object is a list of all AprilTags contained within a layout. Each
29 * AprilTag serializes to a JSON object containing an ID and a Pose3d. The
30 * "field" object is a descriptor of the size of the field in meters with
31 * "width" and "length" values. This is to account for arbitrary field sizes
32 * when transforming the poses.
33 *
34 * Pose3ds are assumed to be measured from the bottom-left corner of the field,
35 * when the blue alliance is at the left. By default, Pose3ds will be returned
36 * as declared when calling GetTagPose(int).
37 * SetOrigin(AprilTagFieldLayout::OriginPosition) can be used to transform the
38 * poses returned by GetTagPose(int) to be correct relative to a different
39 * coordinate frame.
40 */
41class WPILIB_DLLEXPORT AprilTagFieldLayout {
42 public:
43 enum class OriginPosition {
44 kBlueAllianceWallRightSide,
45 kRedAllianceWallRightSide,
46 };
47
48 AprilTagFieldLayout() = default;
49
50 /**
51 * Construct a new AprilTagFieldLayout with values imported from a JSON file.
52 *
53 * @param path Path of the JSON file to import from.
54 */
55 explicit AprilTagFieldLayout(std::string_view path);
56
57 /**
58 * Construct a new AprilTagFieldLayout from a vector of AprilTag objects.
59 *
60 * @param apriltags Vector of AprilTags.
61 * @param fieldLength Length of field the layout is representing.
62 * @param fieldWidth Width of field the layout is representing.
63 */
64 AprilTagFieldLayout(std::vector<AprilTag> apriltags,
65 units::meter_t fieldLength, units::meter_t fieldWidth);
66
67 /**
68 * Sets the origin based on a predefined enumeration of coordinate frame
69 * origins. The origins are calculated from the field dimensions.
70 *
71 * This transforms the Pose3ds returned by GetTagPose(int) to return the
72 * correct pose relative to a predefined coordinate frame.
73 *
74 * @param origin The predefined origin
75 */
76 void SetOrigin(OriginPosition origin);
77
78 /**
79 * Sets the origin for tag pose transformation.
80 *
81 * This tranforms the Pose3ds returned by GetTagPose(int) to return the
82 * correct pose relative to the provided origin.
83 *
84 * @param origin The new origin for tag transformations
85 */
86 void SetOrigin(const Pose3d& origin);
87
88 /**
89 * Gets an AprilTag pose by its ID.
90 *
91 * @param ID The ID of the tag.
92 * @return The pose corresponding to the ID that was passed in or an empty
93 * optional if a tag with that ID is not found.
94 */
95 std::optional<Pose3d> GetTagPose(int ID) const;
96
97 /**
98 * Serializes an AprilTagFieldLayout to a JSON file.
99 *
100 * @param path The path to write the JSON file to.
101 */
102 void Serialize(std::string_view path);
103
104 /*
105 * Checks equality between this AprilTagFieldLayout and another object.
106 */
107 bool operator==(const AprilTagFieldLayout&) const = default;
108
109 private:
110 std::unordered_map<int, AprilTag> m_apriltags;
111 units::meter_t m_fieldLength;
112 units::meter_t m_fieldWidth;
113 Pose3d m_origin;
114
115 friend WPILIB_DLLEXPORT void to_json(wpi::json& json,
116 const AprilTagFieldLayout& layout);
117
118 friend WPILIB_DLLEXPORT void from_json(const wpi::json& json,
119 AprilTagFieldLayout& layout);
120};
