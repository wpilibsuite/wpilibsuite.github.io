12/**
13 * A trapezoid-shaped velocity profile.
14 *
15 * While this class can be used for a profiled movement from start to finish,
16 * the intended usage is to filter a reference's dynamics based on trapezoidal
17 * velocity constraints. To compute the reference obeying this constraint, do
18 * the following.
19 *
20 * Initialization:
21 * @code{.cpp}
22 * TrapezoidProfile::Constraints constraints{kMaxV, kMaxA};
23 * double previousProfiledReference = initialReference;
24 * @endcode
25 *
26 * Run on update:
27 * @code{.cpp}
28 * TrapezoidProfile profile{constraints, unprofiledReference,
29 * previousProfiledReference};
30 * previousProfiledReference = profile.Calculate(timeSincePreviousUpdate);
31 * @endcode
32 *
33 * where `unprofiledReference` is free to change between calls. Note that when
34 * the unprofiled reference is within the constraints, `Calculate()` returns the
35 * unprofiled reference unchanged.
36 *
37 * Otherwise, a timer can be started to provide monotonic values for
38 * `Calculate()` and to determine when the profile has completed via
39 * `IsFinished()`.
40 */
41template <class Distance>
42class TrapezoidProfile {
43 public:
44 using Distance_t = units::unit_t<Distance>;
45 using Velocity =
46 units::compound_unit<Distance, units::inverse<units::seconds>>;
47 using Velocity_t = units::unit_t<Velocity>;
48 using Acceleration =
49 units::compound_unit<Velocity, units::inverse<units::seconds>>;
50 using Acceleration_t = units::unit_t<Acceleration>;
51
52 class Constraints {
53 public:
54 Constraints() {
55 wpi::math::MathSharedStore::ReportUsage(
56 wpi::math::MathUsageId::kTrajectory_TrapezoidProfile, 1);
57 }
58 Constraints(Velocity_t maxVelocity_, Acceleration_t maxAcceleration_)
59 : maxVelocity{maxVelocity_}, maxAcceleration{maxAcceleration_} {
60 wpi::math::MathSharedStore::ReportUsage(
61 wpi::math::MathUsageId::kTrajectory_TrapezoidProfile, 1);
62 }
63 Velocity_t maxVelocity{0};
64 Acceleration_t maxAcceleration{0};
65 };
66
67 class State {
68 public:
69 Distance_t position{0};
70 Velocity_t velocity{0};
71 bool operator==(const State& rhs) const {
72 return position == rhs.position && velocity == rhs.velocity;
73 }
74 bool operator!=(const State& rhs) const { return !(*this == rhs); }
75 };
76
77 /**
78 * Construct a TrapezoidProfile.
79 *
80 * @param constraints The constraints on the profile, like maximum velocity.
81 * @param goal The desired state when the profile is complete.
82 * @param initial The initial state (usually the current state).
83 */
84 TrapezoidProfile(Constraints constraints, State goal,
85 State initial = State{Distance_t{0}, Velocity_t{0}});
86
87 TrapezoidProfile(const TrapezoidProfile&) = default;
88 TrapezoidProfile& operator=(const TrapezoidProfile&) = default;
89 TrapezoidProfile(TrapezoidProfile&&) = default;
90 TrapezoidProfile& operator=(TrapezoidProfile&&) = default;
91
92 /**
93 * Calculate the correct position and velocity for the profile at a time t
94 * where the beginning of the profile was at time t = 0.
95 *
96 * @param t The time since the beginning of the profile.
97 */
98 State Calculate(units::second_t t) const;
99
100 /**
101 * Returns the time left until a target distance in the profile is reached.
102 *
103 * @param target The target distance.
104 */
105 units::second_t TimeLeftUntil(Distance_t target) const;
106
107 /**
108 * Returns the total time the profile takes to reach the goal.
109 */
110 units::second_t TotalTime() const { return m_endDeccel; }
111
112 /**
113 * Returns true if the profile has reached the goal.
114 *
115 * The profile has reached the goal if the time since the profile started
116 * has exceeded the profile's total time.
117 *
118 * @param t The time since the beginning of the profile.
119 */
120 bool IsFinished(units::second_t t) const { return t >= TotalTime(); }
121
122 private:
123 /**
124 * Returns true if the profile inverted.
125 *
126 * The profile is inverted if goal position is less than the initial position.
127 *
128 * @param initial The initial state (usually the current state).
129 * @param goal The desired state when the profile is complete.
130 */
131 static bool ShouldFlipAcceleration(const State& initial, const State& goal) {
132 return initial.position > goal.position;
133 }
134
135 // Flip the sign of the velocity and position if the profile is inverted
136 State Direct(const State& in) const {
137 State result = in;
138 result.position *= m_direction;
139 result.velocity *= m_direction;
140 return result;
141 }
142
143 // The direction of the profile, either 1 for forwards or -1 for inverted
144 int m_direction;
145
146 Constraints m_constraints;
147 State m_initial;
148 State m_goal;
149
150 units::second_t m_endAccel;
151 units::second_t m_endFullSpeed;
152 units::second_t m_endDeccel;
153};
154} // namespace frc
