22/**
23 * Timer handle.
24 * Timer handles are used to schedule signals to be called in the future.
25 */
26class Timer final : public HandleImpl<Timer, uv_timer_t> {
27 struct private_init {};

29 public:

32 explicit Timer(const private_init&) {}
33 ~Timer() noexcept override = default;

42 /**
43 * Create a timer handle.
44 *
45 * @param loop Loop object where this handle runs.
46 */

48 return Create(*loop);

51 /**
52 * Create a timer that calls a functor after a given time interval.
53 *
54 * @param loop Loop object where the timer should run.
55 * @param timeout Time interval
56 * @param func Functor
57 */
58 static void SingleShot(Loop& loop, Time timeout, std::function<void()> func);

60 /**
61 * Create a timer that calls a functor after a given time interval.
62 *
63 * @param loop Loop object where the timer should run.
64 * @param timeout Time interval
65 * @param func Functor
66 */
67 static void SingleShot(const std::shared_ptr<Loop>& loop, Time timeout,
68 std::function<void()> func) {
69 return SingleShot(*loop, timeout, std::move(func));
70 }
71
72 /**
73 * Start the timer.
74 *
75 * If timeout is zero, an event is emitted on the next event loop
76 * iteration. If repeat is non-zero, an event is emitted first
77 * after timeout milliseconds and then repeatedly after repeat milliseconds.
78 *
79 * @param timeout Milliseconds before to emit an event (use
80 * `std::chrono::duration<uint64_t, std::milli>`).
81 * @param repeat Milliseconds between successive events (use
82 * `std::chrono::duration<uint64_t, std::milli>`).
83 */
84 void Start(Time timeout, Time repeat = Time{0});
85
86 /**
87 * Stop the timer.
88 */
89 void Stop() { Invoke(&uv_timer_stop, GetRaw()); }
90
91 /**
92 * Stop the timer and restart it if it was repeating.
93 *
94 * Stop the timer, and if it is repeating restart it using the repeat value
95 * as the timeout.
96 * If the timer has never been started before it emits sigError.
97 */
98 void Again() { Invoke(&uv_timer_again, GetRaw()); }
99
100 /**
101 * Set the repeat interval value.
102 *
103 * The timer will be scheduled to run on the given interval and will follow
104 * normal timer semantics in the case of a time-slice overrun.
105 * For example, if a 50ms repeating timer first runs for 17ms, it will be
106 * scheduled to run again 33ms later. If other tasks consume more than the
107 * 33ms following the first timer event, then another event will be emitted
108 * as soon as possible.
109 *
110 * If the repeat value is set from a listener bound to an event, it does
111 * not immediately take effect. If the timer was non-repeating before, it
112 * will have been stopped. If it was repeating, then the old repeat value
113 * will have been used to schedule the next timeout.
114 *
115 * @param repeat Repeat interval in milliseconds (use
116 * `std::chrono::duration<uint64_t, std::milli>`).
117 */
118 void SetRepeat(Time repeat) { uv_timer_set_repeat(GetRaw(), repeat.count()); }
119
120 /**
121 * Get the timer repeat value.
122 * @return Timer repeat value in milliseconds (as a
123 * `std::chrono::duration<uint64_t, std::milli>`).
124 */
125 Time GetRepeat() const { return Time{uv_timer_get_repeat(GetRaw())}; }
126
127 /**
128 * Signal generated when the timeout event occurs.
129 */
130 sig::Signal<> timeout;
131};
