89 /// A \concept{concept_rawallocator,RawAllocator} that stores another allocator.
90 /// The \concept{concept_storagepolicy,StoragePolicy} defines the allocator type being stored and how it is stored.
91 /// The \c Mutex controls synchronization of the access.
92 /// \ingroup storage
93 template <class StoragePolicy, class Mutex>
94 class allocator_storage
95 : WPI_EBO(StoragePolicy,
96 detail::mutex_storage<
97 detail::mutex_for<typename StoragePolicy::allocator_type, Mutex>>)

100 using composable_traits =

112 /// \effects Creates it by default-constructing the \c StoragePolicy.
113 /// \requires The \c StoragePolicy must be default-constructible.
114 /// \notes The default constructor may create an invalid allocator storage not associated with any allocator.
115 /// If that is the case, it must not be used.
116 allocator_storage() = default;

118 /// \effects Creates it by passing it an allocator.
119 /// The allocator will be forwarded to the \c StoragePolicy, it decides whether it will be moved, its address stored or something else.
120 /// \requires The expression <tt>new storage_policy(std::forward<Alloc>(alloc))</tt> must be well-formed,
121 /// otherwise this constructor does not participate in overload resolution.

123 class Alloc,
124 // MSVC seems to ignore access rights in SFINAE below
125 // use this to prevent this constructor being chosen instead of move for types inheriting from it
126 WPI_REQUIRES(
127 (!std::is_base_of<allocator_storage, typename std::decay<Alloc>::type>::value))>
128 allocator_storage(Alloc&& alloc,
129 WPI_SFINAE(new storage_policy(detail::forward<Alloc>(alloc))))
130 : storage_policy(detail::forward<Alloc>(alloc))

132 }

137 /// otherwise this constructor does not participate in overload resolution.

290 /// \returns A proxy object that acts like a pointer to the stored allocator.

292 /// As long as the proxy object lives and is not moved from, the \c Mutex will be kept locked.

295 {
296 return detail::lock_allocator(get_allocator(), static_cast<actual_mutex&>(*this));
297 }

301 std::declval<actual_mutex&>())))
302 {
303 return detail::lock_allocator(get_allocator(), static_cast<actual_mutex&>(*this));
304 }
305 /// @}.

308 /// that is you can use the compositioning functions.

523 /// Specifies whether or not a \concept{concept_rawallocator,RawAllocator} has shared semantics.
524 /// It is shared, if - like \ref allocator_reference - if multiple objects refer to the same internal allocator and if it can be copied.
525 /// This sharing is stateful, however, stateless allocators are not considered shared in the meaning of this traits. <br>
526 /// If a \c RawAllocator is shared, it will be directly embedded inside \ref reference_storage since it already provides \ref allocator_reference like semantics, so there is no need to add them manually,<br>
527 /// Specialize it for your own types, if they provide sharing semantics and can be copied.
528 /// They also must provide an `operator==` to check whether two allocators refer to the same shared one.
529 /// \note This makes no guarantees about the lifetime of the shared object, the sharing allocators can either own or refer to a shared object.
530 /// \ingroup storage
531 template <class RawAllocator>
532 struct is_shared_allocator : std::false_type
533 {
534 };

536 /// A \concept{concept_storagepolicy,StoragePolicy} that stores a reference to an allocator.
537 /// For stateful allocators it only stores a pointer to an allocator object and copying/moving only copies the pointer.
538 /// For stateless allocators it does not store anything, an allocator will be constructed as needed.
539 /// For allocators that are already shared (determined through \ref is_shared_allocator) it will store the allocator type directly.
540 /// \note It does not take ownership over the allocator in the stateful case, the user has to ensure that the allocator object stays valid.
541 /// In the other cases the lifetime does not matter.
542 /// \ingroup storage
543 template <class RawAllocator>
544 class reference_storage
545#ifndef DOXYGEN
546 : WPI_EBO(detail::reference_storage_impl<
547 typename allocator_traits<RawAllocator>::allocator_type,
548 decltype(detail::reference_type(
549 typename allocator_traits<RawAllocator>::is_stateful{},
550 is_shared_allocator<RawAllocator>{}))>)
551#endif
552 {
553 using storage = detail::reference_storage_impl<
554 typename allocator_traits<RawAllocator>::allocator_type,
555 decltype(
556 detail::reference_type(typename allocator_traits<RawAllocator>::is_stateful{},
557 is_shared_allocator<RawAllocator>{}))>;
558
559 public:
560 using allocator_type = typename allocator_traits<RawAllocator>::allocator_type;
561
562 /// Default constructor.
563 /// \effects If the allocator is stateless, this has no effect and the object is usable as an allocator.
564 /// If the allocator is stateful, creates an invalid reference without any associated allocator.
565 /// Then it must not be used.
566 /// If the allocator is shared, default constructs the shared allocator.
567 /// If the shared allocator does not have a default constructor, this constructor is ill-formed.
568 reference_storage() noexcept = default;
569
570 /// \effects Creates it from a stateless or shared allocator.
571 /// It will not store anything, only creates the allocator as needed.
572 /// \requires The \c RawAllocator is stateless or shared.
573 reference_storage(const allocator_type& alloc) noexcept : storage(alloc) {}
574
575 /// \effects Creates it from a reference to a stateful allocator.
576 /// It will store a pointer to this allocator object.
577 /// \note The user has to take care that the lifetime of the reference does not exceed the allocator lifetime.
578 reference_storage(allocator_type& alloc) noexcept : storage(alloc) {}
579
580 /// @{
581 /// \effects Copies the \c allocator_reference object.
582 /// Only copies the pointer to it in the stateful case.
583 reference_storage(const reference_storage&) noexcept = default;
584 reference_storage& operator=(const reference_storage&) noexcept = default;
585 /// @}
586
587 /// \returns Whether or not the reference is valid.
588 /// It is only invalid, if it was created by the default constructor and the allocator is stateful.
589 explicit operator bool() const noexcept
590 {
591 return storage::is_valid();
592 }
593
594 /// \returns Returns a reference to the allocator.
595 /// \requires The reference must be valid.
596 allocator_type& get_allocator() const noexcept
597 {
598 return storage::get_allocator();
599 }
600
601 protected:
602 ~reference_storage() noexcept = default;
603
604 bool is_composable() const noexcept
605 {
606 return is_composable_allocator<allocator_type>::value;
607 }
608 };
