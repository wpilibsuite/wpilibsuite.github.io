24/**
25 * This class wraps Swerve Drive Odometry to fuse latency-compensated
26 * vision measurements with swerve drive encoder distance measurements. It is
27 * intended to be a drop-in for SwerveDriveOdometry.
28 *
29 * Update() should be called every robot loop.
30 *
31 * AddVisionMeasurement() can be called as infrequently as you want; if you
32 * never call it, then this class will behave as regular encoder
33 * odometry.
34 *
35 * The state-space system used internally has the following states (x) and
36 * outputs (y):
37 *
38 * <strong> x = [x, y, theta]ᵀ </strong> in the field coordinate
39 * system containing x position, y position, and heading.
40 *
41 * <strong> y = [x, y, theta]ᵀ </strong> from vision containing x position, y
42 * position, and heading.
43 */
44template <size_t NumModules>
45class SwerveDrivePoseEstimator {
46 public:
47 /**
48 * Constructs a SwerveDrivePoseEstimator with default standard deviations
49 * for the model and vision measurements.
50 *
51 * The default standard deviations of the model states are
52 * 0.1 meters for x, 0.1 meters for y, and 0.1 radians for heading.
53 * The default standard deviations of the vision measurements are
54 * 0.9 meters for x, 0.9 meters for y, and 0.9 radians for heading.
55 *
56 * @param kinematics A correctly-configured kinematics object
57 * for your drivetrain.
58 * @param gyroAngle The current gyro angle.
59 * @param modulePositions The current distance and rotation
60 * measurements of the swerve modules.
61 * @param initialPose The starting pose estimate.
62 */
63 SwerveDrivePoseEstimator(
64 SwerveDriveKinematics<NumModules>& kinematics,
65 const Rotation2d& gyroAngle,
66 const wpi::array<SwerveModulePosition, NumModules>& modulePositions,
67 const Pose2d& initialPose)
68 : SwerveDrivePoseEstimator{kinematics, gyroAngle,
69 modulePositions, initialPose,
70 {0.1, 0.1, 0.1}, {0.9, 0.9, 0.9}} {}
71
72 /**
73 * Constructs a SwerveDrivePoseEstimator.
74 *
75 * @param kinematics A correctly-configured kinematics object
76 * for your drivetrain.
77 * @param gyroAngle The current gyro angle.
78 * @param modulePositions The current distance and rotation
79 * measurements of the swerve modules.
80 * @param initialPose The starting pose estimate.
81 * @param stateStdDevs Standard deviations of model states.
82 * Increase these numbers to trust your
83 * model's state estimates less. This matrix
84 * is in the form [x, y, theta]ᵀ, with units
85 * in meters and radians.
86 * @param visionMeasurementStdDevs Standard deviations of the vision
87 * measurements. Increase these numbers to
88 * trust global measurements from vision
89 * less. This matrix is in the form
90 * [x, y, theta]ᵀ, with units in meters and
91 * radians.
92 */
93 SwerveDrivePoseEstimator(
94 SwerveDriveKinematics<NumModules>& kinematics,
95 const Rotation2d& gyroAngle,
96 const wpi::array<SwerveModulePosition, NumModules>& modulePositions,
97 const Pose2d& initialPose, const wpi::array<double, 3>& stateStdDevs,
98 const wpi::array<double, 3>& visionMeasurementStdDevs)
99 : m_kinematics{kinematics},
100 m_odometry{kinematics, gyroAngle, modulePositions, initialPose} {
101 for (size_t i = 0; i < 3; ++i) {
102 m_q[i] = stateStdDevs[i] * stateStdDevs[i];
103 }
104
105 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
106 }
107
108 /**
109 * Resets the robot's position on the field.
110 *
111 * The gyroscope angle does not need to be reset in the user's robot code.
112 * The library automatically takes care of offsetting the gyro angle.
113 *
114 * @param gyroAngle The angle reported by the gyroscope.
115 * @param modulePositions The current distance and rotation measurements of
116 * the swerve modules.
117 * @param pose The position on the field that your robot is at.
118 */
119 void ResetPosition(
120 const Rotation2d& gyroAngle,
121 const wpi::array<SwerveModulePosition, NumModules>& modulePositions,
122 const Pose2d& pose) {
123 // Reset state estimate and error covariance
124 m_odometry.ResetPosition(gyroAngle, modulePositions, pose);
125 m_poseBuffer.Clear();
126 }
127
128 /**
129 * Gets the estimated robot pose.
130 *
131 * @return The estimated robot pose in meters.
132 */
133 Pose2d GetEstimatedPosition() const { return m_odometry.GetPose(); }
134
135 /**
136 * Sets the pose estimator's trust of global measurements. This might be used
137 * to change trust in vision measurements after the autonomous period, or to
138 * change trust as distance to a vision target increases.
139 *
140 * @param visionMeasurementStdDevs Standard deviations of the vision
141 * measurements. Increase these numbers to
142 * trust global measurements from vision
143 * less. This matrix is in the form
144 * [x, y, theta]ᵀ, with units in meters and
145 * radians.
146 */
147 void SetVisionMeasurementStdDevs(
148 const wpi::array<double, 3>& visionMeasurementStdDevs) {
149 wpi::array<double, 3> r{wpi::empty_array};
150 for (size_t i = 0; i < 3; ++i) {
151 r[i] = visionMeasurementStdDevs[i] * visionMeasurementStdDevs[i];
152 }
153
154 // Solve for closed form Kalman gain for continuous Kalman filter with A = 0
155 // and C = I. See wpimath/algorithms.md.
156 for (size_t row = 0; row < 3; ++row) {
157 if (m_q[row] == 0.0) {
158 m_visionK(row, row) = 0.0;
159 } else {
160 m_visionK(row, row) =
161 m_q[row] / (m_q[row] + std::sqrt(m_q[row] * r[row]));
162 }
163 }
164 }
165
166 /**
167 * Adds a vision measurement to the Kalman Filter. This will correct the
168 * odometry pose estimate while still accounting for measurement noise.
169 *
170 * This method can be called as infrequently as you want, as long as you are
171 * calling Update() every loop.
172 *
173 * To promote stability of the pose estimate and make it robust to bad vision
174 * data, we recommend only adding vision measurements that are already within
175 * one meter or so of the current pose estimate.
176 *
177 * @param visionRobotPose The pose of the robot as measured by the vision
178 * camera.
179 * @param timestamp The timestamp of the vision measurement in seconds.
180 * Note that if you don't use your own time source by
181 * calling UpdateWithTime() then you must use a
182 * timestamp with an epoch since FPGA startup
183 * (i.e. the epoch of this timestamp is the same
184 * epoch as Timer#GetFPGATimestamp.) This means
185 * that you should use Timer#GetFPGATimestamp as your
186 * time source or sync the epochs.
187 */
188 void AddVisionMeasurement(const Pose2d& visionRobotPose,
189 units::second_t timestamp) {
190 // Step 1: Get the estimated pose from when the vision measurement was made.
191 auto sample = m_poseBuffer.Sample(timestamp);
192
193 if (!sample.has_value()) {
194 return;
195 }
196
197 // Step 2: Measure the twist between the odometry pose and the vision pose
198 auto twist = sample.value().pose.Log(visionRobotPose);
199
200 // Step 3: We should not trust the twist entirely, so instead we scale this
201 // twist by a Kalman gain matrix representing how much we trust vision
202 // measurements compared to our current pose.
203 frc::Vectord<3> k_times_twist =
204 m_visionK * frc::Vectord<3>{twist.dx.value(), twist.dy.value(),
205 twist.dtheta.value()};
206
207 // Step 4: Convert back to Twist2d
208 Twist2d scaledTwist{units::meter_t{k_times_twist(0)},
209 units::meter_t{k_times_twist(1)},
210 units::radian_t{k_times_twist(2)}};
211
212 // Step 5: Reset Odometry to state at sample with vision adjustment.
213 m_odometry.ResetPosition(sample.value().gyroAngle,
214 sample.value().modulePostions,
215 sample.value().pose.Exp(scaledTwist));
216
217 // Step 6: Replay odometry inputs between sample time and latest recorded
218 // sample to update the pose buffer and correct odometry.
219 auto internal_buf = m_poseBuffer.GetInternalBuffer();
220
221 auto upper_bound = std::lower_bound(
222 internal_buf.begin(), internal_buf.end(), timestamp,
223 [](const auto& pair, auto t) { return t > pair.first; });
224
225 for (auto entry = upper_bound; entry != internal_buf.end(); entry++) {
226 UpdateWithTime(entry->first, entry->second.gyroAngle,
227 entry->second.modulePostions);
228 }
229 }

231 /**
232 * Adds a vision measurement to the Kalman Filter. This will correct the
233 * odometry pose estimate while still accounting for measurement noise.
234 *
235 * This method can be called as infrequently as you want, as long as you are
236 * calling Update() every loop.
237 *
238 * To promote stability of the pose estimate and make it robust to bad vision
239 * data, we recommend only adding vision measurements that are already within
240 * one meter or so of the current pose estimate.
241 *
242 * Note that the vision measurement standard deviations passed into this
243 * method will continue to apply to future measurements until a subsequent
244 * call to SetVisionMeasurementStdDevs() or this method.
245 *
246 * @param visionRobotPose The pose of the robot as measured by the
247 * vision camera.
248 * @param timestamp The timestamp of the vision measurement in
249 * seconds. Note that if you don't use your
250 * own time source by calling
251 * UpdateWithTime(), then you must use a
252 * timestamp with an epoch since FPGA startup
253 * (i.e. the epoch of this timestamp is the
254 * same epoch as
255 * frc::Timer::GetFPGATimestamp(). This means
256 * that you should use
257 * frc::Timer::GetFPGATimestamp() as your
258 * time source in this case.
259 * @param visionMeasurementStdDevs Standard deviations of the vision
260 * measurements. Increase these numbers to
261 * trust global measurements from vision
262 * less. This matrix is in the form
263 * [x, y, theta]ᵀ, with units in meters and
264 * radians.
265 */
266 void AddVisionMeasurement(
267 const Pose2d& visionRobotPose, units::second_t timestamp,
268 const wpi::array<double, 3>& visionMeasurementStdDevs) {
269 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
270 AddVisionMeasurement(visionRobotPose, timestamp);
271 }

273 /**
274 * Updates the Kalman Filter using only wheel encoder information. This should
275 * be called every loop.
276 *
277 * @param gyroAngle The current gyro angle.
278 * @param modulePositions The current distance and rotation measurements of
279 * the swerve modules.
280 * @return The estimated robot pose in meters.
281 */
282 Pose2d Update(
283 const Rotation2d& gyroAngle,
284 const wpi::array<SwerveModulePosition, NumModules>& modulePositions) {
285 return UpdateWithTime(units::microsecond_t(wpi::Now()), gyroAngle,
286 modulePositions);
287 }
288
289 /**
290 * Updates the Kalman Filter using only wheel encoder information. This should
291 * be called every loop.
292 *
293 * @param currentTime Time at which this method was called, in seconds.
294 * @param gyroAngle The current gyro angle.
295 * @param modulePositions The current distance traveled and rotations of
296 * the swerve modules.
297 * @return The estimated robot pose in meters.
298 */
299 Pose2d UpdateWithTime(
300 units::second_t currentTime, const Rotation2d& gyroAngle,
301 const wpi::array<SwerveModulePosition, NumModules>& modulePositions) {
302 m_odometry.Update(gyroAngle, modulePositions);
303
304 wpi::array<SwerveModulePosition, NumModules> internalModulePositions{
305 wpi::empty_array};
306
307 for (size_t i = 0; i < NumModules; i++) {
308 internalModulePositions[i].distance = modulePositions[i].distance;
309 internalModulePositions[i].angle = modulePositions[i].angle;
310 }
311
312 m_poseBuffer.AddSample(currentTime, {GetEstimatedPosition(), gyroAngle,
313 internalModulePositions});
314
315 return GetEstimatedPosition();
316 }

318 private:
319 struct InterpolationRecord {
320 // The pose observed given the current sensor inputs and the previous pose.
321 Pose2d pose;
322
323 // The current gyroscope angle.
324 Rotation2d gyroAngle;
325
326 // The distances traveled and rotations meaured at each module.
327 wpi::array<SwerveModulePosition, NumModules> modulePostions;
328
329 /**
330 * Checks equality between this InterpolationRecord and another object.
331 *
332 * @param other The other object.
333 * @return Whether the two objects are equal.
334 */
335 bool operator==(const InterpolationRecord& other) const = default;
336
337 /**
338 * Checks inequality between this InterpolationRecord and another object.
339 *
340 * @param other The other object.
341 * @return Whether the two objects are not equal.
342 */
343 bool operator!=(const InterpolationRecord& other) const = default;
344
345 /**
346 * Interpolates between two InterpolationRecords.
347 *
348 * @param endValue The end value for the interpolation.
349 * @param i The interpolant (fraction).
350 *
351 * @return The interpolated state.
352 */
353 InterpolationRecord Interpolate(
354 SwerveDriveKinematics<NumModules>& kinematics,
355 InterpolationRecord endValue, double i) const {
356 if (i < 0) {
357 return *this;
358 } else if (i > 1) {
359 return endValue;
360 } else {
361 // Find the new module distances.
362 wpi::array<SwerveModulePosition, NumModules> modulePositions{
363 wpi::empty_array};
364 // Find the distance between this measurement and the
365 // interpolated measurement.
366 wpi::array<SwerveModulePosition, NumModules> modulesDelta{
367 wpi::empty_array};
368
369 for (size_t i = 0; i < NumModules; i++) {
370 modulePositions[i].distance =
371 wpi::Lerp(this->modulePostions[i].distance,
372 endValue.modulePostions[i].distance, i);
373 modulePositions[i].angle =
374 wpi::Lerp(this->modulePostions[i].angle,
375 endValue.modulePostions[i].angle, i);
376
377 modulesDelta[i].distance =
378 modulePositions[i].distance - this->modulePostions[i].distance;
379 modulesDelta[i].angle = modulePositions[i].angle;
380 }
381
382 // Find the new gyro angle.
383 auto gyro = wpi::Lerp(this->gyroAngle, endValue.gyroAngle, i);
384
385 // Create a twist to represent this changed based on the interpolated
386 // sensor inputs.
387 auto twist = kinematics.ToTwist2d(modulesDelta);
388 twist.dtheta = (gyro - gyroAngle).Radians();
389
390 return {pose.Exp(twist), gyro, modulePositions};
391 }
392 }
393 };
394
395 SwerveDriveKinematics<NumModules>& m_kinematics;
396 SwerveDriveOdometry<NumModules> m_odometry;
397 wpi::array<double, 3> m_q{wpi::empty_array};
398 Eigen::Matrix3d m_visionK = Eigen::Matrix3d::Zero();
399
400 TimeInterpolatableBuffer<InterpolationRecord> m_poseBuffer{
401 1.5_s, [this](const InterpolationRecord& start,
402 const InterpolationRecord& end, double t) {
403 return start.Interpolate(this->m_kinematics, end, t);
404 }};
405};
