24/**
25 * This class wraps an Unscented Kalman Filter to fuse latency-compensated
26 * vision measurements with swerve drive encoder velocity measurements. It will
27 * correct for noisy measurements and encoder drift. It is intended to be an
28 * easy but more accurate drop-in for SwerveDriveOdometry.
29 *
30 * Update() should be called every robot loop. If your loops are faster or
31 * slower than the default of 20 ms, then you should change the nominal delta
32 * time by specifying it in the constructor.
33 *
34 * AddVisionMeasurement() can be called as infrequently as you want; if you
35 * never call it, then this class will behave mostly like regular encoder
36 * odometry.
37 *
38 * The state-space system used internally has the following states (x), inputs
39 * (u), and outputs (y):
40 *
41 * <strong> x = [x, y, theta, s_0, ..., s_n]ᵀ </strong> in the field coordinate
42 * system containing x position, y position, and heading, followed by the
43 * distance travelled by each wheel.
44 *
45 * <strong> u = [v_x, v_y, omega, v_0, ... v_n]ᵀ </strong> containing x
46 * velocity, y velocity, and angular velocity in the field coordinate system,
47 * followed by the velocity measured at each wheel.
48 *
49 * <strong> y = [x, y, theta]ᵀ </strong> from vision containing x position, y
50 * position, and heading; or <strong> y = [theta, s_0, ..., s_n]ᵀ </strong>
51 * containing gyro heading, followed by the distance travelled by each wheel.
52 */
53template <size_t NumModules>
54class SwerveDrivePoseEstimator {
55 public:
56 static constexpr size_t States = 3 + NumModules;
57 static constexpr size_t Inputs = 3 + NumModules;
58 static constexpr size_t Outputs = 1 + NumModules;
59
60 /**
61 * Constructs a SwerveDrivePoseEstimator.
62 *
63 * @param gyroAngle The current gyro angle.
64 * @param modulePositions The current distance and rotation
65 * measurements of the swerve modules.
66 * @param initialPose The starting pose estimate.
67 * @param kinematics A correctly-configured kinematics object
68 * for your drivetrain.
69 * @param stateStdDevs Standard deviations of model states.
70 * Increase these numbers to trust your
71 * model's state estimates less. This matrix
72 * is in the form [x, y, theta, s_0, ...
73 * s_n]ᵀ, with units in meters and radians, then meters.
74 * @param localMeasurementStdDevs Standard deviation of the gyro measurement.
75 * Increase this number to trust sensor
76 * readings from the gyro less. This matrix is
77 * in the form [theta, s_0, ... s_n], with
78 * units in radians followed by meters.
79 * @param visionMeasurementStdDevs Standard deviations of the vision
80 * measurements. Increase these numbers to
81 * trust global measurements from vision
82 * less. This matrix is in the form
83 * [x, y, theta]ᵀ, with units in meters and
84 * radians.
85 * @param nominalDt The time in seconds between each robot
86 * loop.
87 */
88 SwerveDrivePoseEstimator(
89 const Rotation2d& gyroAngle,
90 const wpi::array<SwerveModulePosition, NumModules>& modulePositions,
91 const Pose2d& initialPose, SwerveDriveKinematics<NumModules>& kinematics,
92 const wpi::array<double, States>& stateStdDevs,
93 const wpi::array<double, Outputs>& localMeasurementStdDevs,
94 const wpi::array<double, 3>& visionMeasurementStdDevs,
95 units::second_t nominalDt = 20_ms)
96 : m_observer([](const Vectord<States>& x,
97 const Vectord<Inputs>& u) { return u; },
98 [](const Vectord<States>& x, const Vectord<Inputs>& u) {
99 return x.template block<States - 2, 1>(2, 0);
100 },
101 stateStdDevs, localMeasurementStdDevs,
102 frc::AngleMean<States, States>(2),
103 frc::AngleMean<Outputs, States>(0),
104 frc::AngleResidual<States>(2),
105 frc::AngleResidual<Outputs>(0), frc::AngleAdd<States>(2),
106 nominalDt),
107 m_kinematics(kinematics),
108 m_nominalDt(nominalDt) {
109 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
110
111 // Create correction mechanism for vision measurements.
112 m_visionCorrect = [&](const Vectord<Inputs>& u, const Vectord<3>& y) {
113 m_observer.template Correct<3>(
114 u, y,
115 [](const Vectord<States>& x, const Vectord<Inputs>& u) {
116 return x.template block<3, 1>(0, 0);
117 },
118 m_visionContR, frc::AngleMean<3, States>(2), frc::AngleResidual<3>(2),
119 frc::AngleResidual<States>(2), frc::AngleAdd<States>(2));
120 };
121
122 // Set initial state.
123 Vectord<States> xhat;
124 auto poseVec = PoseTo3dVector(initialPose);
125 xhat(0) = poseVec(0);
126 xhat(1) = poseVec(1);
127 xhat(2) = poseVec(2);
128 for (size_t i = 0; i < NumModules; i++) {
129 xhat(3 + i) = modulePositions[i].distance.value();
130 }
131 m_observer.SetXhat(xhat);
132
133 // Calculate offsets.
134 m_gyroOffset = initialPose.Rotation() - gyroAngle;
135 m_previousAngle = initialPose.Rotation();
136 }
137
138 /**
139 * Resets the robot's position on the field.
140 *
141 * IF leftDistance and rightDistance are unspecified,
142 * You NEED to reset your encoders (to zero).
143 *
144 * The gyroscope angle does not need to be reset in the user's robot code.
145 * The library automatically takes care of offsetting the gyro angle.
146 *
147 * @param gyroAngle The angle reported by the gyroscope.
148 * @param modulePositions The current distance and rotation measurements of
149 * the swerve modules.
150 * @param pose The position on the field that your robot is at.
151 */
152 void ResetPosition(
153 const Rotation2d& gyroAngle,
154 const wpi::array<SwerveModulePosition, NumModules>& modulePositions,
155 const Pose2d& pose) {
156 // Reset state estimate and error covariance
157 m_observer.Reset();
158 m_poseBuffer.Clear();
159
160 Vectord<States> xhat;
161 auto poseVec = PoseTo3dVector(pose);
162 xhat(0) = poseVec(0);
163 xhat(1) = poseVec(1);
164 xhat(2) = poseVec(2);
165 for (size_t i = 0; i < NumModules; i++) {
166 xhat(3 + i) = modulePositions[i].distance.value();
167 }
168 m_observer.SetXhat(xhat);
169
170 m_prevTime = -1_s;
171
172 m_gyroOffset = pose.Rotation() - gyroAngle;
173 m_previousAngle = pose.Rotation();
174 }
175
176 /**
177 * Gets the pose of the robot at the current time as estimated by the Extended
178 * Kalman Filter.
179 *
180 * @return The estimated robot pose in meters.
181 */
182 Pose2d GetEstimatedPosition() const {
183 return Pose2d{m_observer.Xhat(0) * 1_m, m_observer.Xhat(1) * 1_m,
184 Rotation2d{units::radian_t{m_observer.Xhat(2)}}};
185 }
186
187 /**
188 * Sets the pose estimator's trust of global measurements. This might be used
189 * to change trust in vision measurements after the autonomous period, or to
190 * change trust as distance to a vision target increases.
191 *
192 * @param visionMeasurementStdDevs Standard deviations of the vision
193 * measurements. Increase these numbers to
194 * trust global measurements from vision
195 * less. This matrix is in the form
196 * [x, y, theta]ᵀ, with units in meters and
197 * radians.
198 */
199 void SetVisionMeasurementStdDevs(
200 const wpi::array<double, 3>& visionMeasurementStdDevs) {
201 // Create R (covariances) for vision measurements.
202 m_visionContR = frc::MakeCovMatrix(visionMeasurementStdDevs);
203 }
204
205 /**
206 * Add a vision measurement to the Unscented Kalman Filter. This will correct
207 * the odometry pose estimate while still accounting for measurement noise.
208 *
209 * This method can be called as infrequently as you want, as long as you are
210 * calling Update() every loop.
211 *
212 * To promote stability of the pose estimate and make it robust to bad vision
213 * data, we recommend only adding vision measurements that are already within
214 * one meter or so of the current pose estimate.
215 *
216 * @param visionRobotPose The pose of the robot as measured by the vision
217 * camera.
218 * @param timestamp The timestamp of the vision measurement in seconds.
219 * Note that if you don't use your own time source by
220 * calling UpdateWithTime() then you must use a
221 * timestamp with an epoch since FPGA startup
222 * (i.e. the epoch of this timestamp is the same
223 * epoch as Timer#GetFPGATimestamp.) This means
224 * that you should use Timer#GetFPGATimestamp as your
225 * time source or sync the epochs.
226 */
227 void AddVisionMeasurement(const Pose2d& visionRobotPose,
228 units::second_t timestamp) {
229 if (auto sample = m_poseBuffer.Sample(timestamp)) {
230 m_visionCorrect(Vectord<States>::Zero(),
231 PoseTo3dVector(GetEstimatedPosition().TransformBy(
232 visionRobotPose - sample.value())));
233 }
234 }
235
236 /**
237 * Adds a vision measurement to the Unscented Kalman Filter. This will correct
238 * the odometry pose estimate while still accounting for measurement noise.
239 *
240 * This method can be called as infrequently as you want, as long as you are
241 * calling Update() every loop.
242 *
243 * To promote stability of the pose estimate and make it robust to bad vision
244 * data, we recommend only adding vision measurements that are already within
245 * one meter or so of the current pose estimate.
246 *
247 * Note that the vision measurement standard deviations passed into this
248 * method will continue to apply to future measurements until a subsequent
249 * call to SetVisionMeasurementStdDevs() or this method.
250 *
251 * @param visionRobotPose The pose of the robot as measured by the
252 * vision camera.
253 * @param timestamp The timestamp of the vision measurement in
254 * seconds. Note that if you don't use your
255 * own time source by calling
256 * UpdateWithTime(), then you must use a
257 * timestamp with an epoch since FPGA startup
258 * (i.e. the epoch of this timestamp is the
259 * same epoch as
260 * frc::Timer::GetFPGATimestamp(). This means
261 * that you should use
262 * frc::Timer::GetFPGATimestamp() as your
263 * time source in this case.
264 * @param visionMeasurementStdDevs Standard deviations of the vision
265 * measurements. Increase these numbers to
266 * trust global measurements from vision
267 * less. This matrix is in the form
268 * [x, y, theta]ᵀ, with units in meters and
269 * radians.
270 */
271 void AddVisionMeasurement(
272 const Pose2d& visionRobotPose, units::second_t timestamp,
273 const wpi::array<double, 3>& visionMeasurementStdDevs) {
274 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
275 AddVisionMeasurement(visionRobotPose, timestamp);
276 }

278 /**
279 * Updates the the Unscented Kalman Filter using only wheel encoder
280 * information. This should be called every loop, and the correct loop period
281 * must be passed into the constructor of this class.
282 *
283 * @param gyroAngle The current gyro angle.
284 * @param moduleStates The current velocities and rotations of the swerve
285 * modules.
286 * @param modulePositions The current distance and rotation measurements of
287 * the swerve modules.
288 * @return The estimated pose of the robot in meters.
289 */
290 Pose2d Update(
291 const Rotation2d& gyroAngle,
292 const wpi::array<SwerveModuleState, NumModules>& moduleStates,
293 const wpi::array<SwerveModulePosition, NumModules>& modulePositions) {
294 return UpdateWithTime(units::microsecond_t(wpi::Now()), gyroAngle,
295 moduleStates, modulePositions);
296 }
297
298 /**
299 * Updates the the Unscented Kalman Filter using only wheel encoder
300 * information. This should be called every loop, and the correct loop period
301 * must be passed into the constructor of this class.
302 *
303 * @param currentTime Time at which this method was called, in seconds.
304 * @param gyroAngle The current gyro angle.
305 * @param moduleStates The current velocities and rotations of the swerve
306 * modules.
307 * @param modulePositions The current distance travelled and rotations of
308 * the swerve modules.
309 * @return The estimated pose of the robot in meters.
310 */
311 Pose2d UpdateWithTime(
312 units::second_t currentTime, const Rotation2d& gyroAngle,
313 const wpi::array<SwerveModuleState, NumModules>& moduleStates,
314 const wpi::array<SwerveModulePosition, NumModules>& modulePositions) {
315 auto dt = m_prevTime >= 0_s ? currentTime - m_prevTime : m_nominalDt;
316 m_prevTime = currentTime;
317
318 auto angle = gyroAngle + m_gyroOffset;
319 auto omega = (angle - m_previousAngle).Radians() / dt;
320
321 auto chassisSpeeds = m_kinematics.ToChassisSpeeds(moduleStates);
322 auto fieldRelativeSpeeds =
323 Translation2d{chassisSpeeds.vx * 1_s, chassisSpeeds.vy * 1_s}.RotateBy(
324 angle);
325
326 Vectord<Inputs> u;
327 u(0) = fieldRelativeSpeeds.X().value();
328 u(1) = fieldRelativeSpeeds.Y().value();
329 u(2) = omega.value();
330 for (size_t i = 0; i < NumModules; i++) {
331 u(3 + i) = moduleStates[i].speed.value();
332 }
333
334 Vectord<Outputs> localY;
335 localY(0) = angle.Radians().value();
336 for (size_t i = 0; i < NumModules; i++) {
337 localY(1 + i) = modulePositions[i].distance.value();
338 }
339
340 m_previousAngle = angle;
341
342 m_poseBuffer.AddSample(currentTime, GetEstimatedPosition());
343
344 m_observer.Predict(u, dt);
345 m_observer.Correct(u, localY);
346
347 return GetEstimatedPosition();
348 }
349
350 private:
351 UnscentedKalmanFilter<States, Inputs, Outputs> m_observer;
352 SwerveDriveKinematics<NumModules>& m_kinematics;
353 TimeInterpolatableBuffer<Pose2d> m_poseBuffer{1.5_s};
354 std::function<void(const Vectord<Inputs>& u, const Vectord<3>& y)>
355 m_visionCorrect;
356
357 Eigen::Matrix3d m_visionContR;
358
359 units::second_t m_nominalDt;
360 units::second_t m_prevTime = -1_s;
361
362 Rotation2d m_gyroOffset;
363 Rotation2d m_previousAngle;
364};
