10// This file implements the newly proposed standard C++ interfaces for hashing
11// arbitrary data and building hash functions for user-defined types. This
12// interface was originally proposed in N3333[1] and is currently under review
13// for inclusion in a future TR and/or standard.
14//
15// The primary interfaces provide are comprised of one type and three functions:
16//
17// -- 'hash_code' class is an opaque type representing the hash code for some
18// data. It is the intended product of hashing, and can be used to implement
19// hash tables, checksumming, and other common uses of hashes. It is not an
20// integer type (although it can be converted to one) because it is risky
21// to assume much about the internals of a hash_code. In particular, each
22// execution of the program has a high probability of producing a different
23// hash_code for a given input. Thus their values are not stable to save or
24// persist, and should only be used during the execution for the
25// construction of hashing datastructures.
26//
27// -- 'hash_value' is a function designed to be overloaded for each
28// user-defined type which wishes to be used within a hashing context. It
29// should be overloaded within the user-defined type's namespace and found
30// via ADL. Overloads for primitive types are provided by this library.
31//
32// -- 'hash_combine' and 'hash_combine_range' are functions designed to aid
33// programmers in easily and intuitively combining a set of data into
34// a single hash_code for their object. They should only logically be used
35// within the implementation of a 'hash_value' routine or similar context.
36//
37// Note that 'hash_combine_range' contains very special logic for hashing
38// a contiguous array of integers or pointers. This logic is *extremely* fast,
39// on a modern Intel "Gainestown" Xeon (Nehalem uarch) @2.2 GHz, these were
40// benchmarked at over 6.5 GiB/s for large keys, and <20 cycles/hash for keys
41// under 32-bytes.
42//
43//===----------------------------------------------------------------------===//

151inline uint64_t fetch64(const char *p) {
152 uint64_t result;
153 memcpy(&result, p, sizeof(result));
154 if (support::endian::system_endianness() == support::big)
155 sys::swapByteOrder(result);
156 return result;
157}

168static const uint64_t k0 = 0xc3a5c85c97cb3127ULL;
169static const uint64_t k1 = 0xb492b66fbe98f273ULL;
170static const uint64_t k2 = 0x9ae16a3b2f90404fULL;
171static const uint64_t k3 = 0xc949d7c7509e6557ULL;
172
176inline uint64_t rotate(uint64_t val, size_t shift) {
177 // Avoid shifting by 64: doing so yields an undefined result.
178 return shift == 0 ? val : ((val >> shift) | (val << (64 - shift)));
179}
180
181inline uint64_t shift_mix(uint64_t val) {
182 return val ^ (val >> 47);
183}
184
185inline uint64_t hash_16_bytes(uint64_t low, uint64_t high) {
186 // Murmur-inspired hashing.
187 const uint64_t kMul = 0x9ddfea08eb382d69ULL;
188 uint64_t a = (low ^ high) * kMul;
189 a ^= (a >> 47);
190 uint64_t b = (high ^ a) * kMul;
191 b ^= (b >> 47);
192 b *= kMul;
193 return b;
194}
195
196inline uint64_t hash_1to3_bytes(const char *s, size_t len, uint64_t seed) {
197 uint8_t a = s[0];
198 uint8_t b = s[len >> 1];
199 uint8_t c = s[len - 1];
200 uint32_t y = static_cast<uint32_t>(a) + (static_cast<uint32_t>(b) << 8);
201 uint32_t z = static_cast<uint32_t>(len + (static_cast<uint64_t>(c) << 2));
202 return shift_mix(y * k2 ^ z * k3 ^ seed) * k2;
203}
204
205inline uint64_t hash_4to8_bytes(const char *s, size_t len, uint64_t seed) {
206 uint64_t a = fetch32(s);
207 return hash_16_bytes(len + (a << 3), seed ^ fetch32(s + len - 4));
208}
209
210inline uint64_t hash_9to16_bytes(const char *s, size_t len, uint64_t seed) {
211 uint64_t a = fetch64(s);
212 uint64_t b = fetch64(s + len - 8);
213 return hash_16_bytes(seed ^ a, rotate(b + len, len)) ^ b;
214}
215
216inline uint64_t hash_17to32_bytes(const char *s, size_t len, uint64_t seed) {
217 uint64_t a = fetch64(s) * k1;
218 uint64_t b = fetch64(s + 8);
219 uint64_t c = fetch64(s + len - 8) * k2;
220 uint64_t d = fetch64(s + len - 16) * k0;
221 return hash_16_bytes(rotate(a - b, 43) + rotate(c ^ seed, 30) + d,
222 a + rotate(b ^ k3, 20) - c + len + seed);
223}
224
225inline uint64_t hash_33to64_bytes(const char *s, size_t len, uint64_t seed) {
226 uint64_t z = fetch64(s + 24);
227 uint64_t a = fetch64(s) + (len + fetch64(s + len - 16)) * k0;
228 uint64_t b = rotate(a + z, 52);
229 uint64_t c = rotate(a, 37);
230 a += fetch64(s + 8);
231 c += rotate(a, 7);
232 a += fetch64(s + 16);
233 uint64_t vf = a + z;
234 uint64_t vs = b + rotate(a, 31) + c;
235 a = fetch64(s + 16) + fetch64(s + len - 32);
236 z = fetch64(s + len - 8);
237 b = rotate(a + z, 52);
238 c = rotate(a, 37);
239 a += fetch64(s + len - 24);
240 c += rotate(a, 7);
241 a += fetch64(s + len - 16);
242 uint64_t wf = a + z;
243 uint64_t ws = b + rotate(a, 31) + c;
244 uint64_t r = shift_mix((vf + ws) * k2 + (wf + vs) * k0);
245 return shift_mix((seed ^ (r * k0)) + vs) * k2;
246}
247
248inline uint64_t hash_short(const char *s, size_t length, uint64_t seed) {
249 if (length >= 4 && length <= 8)
250 return hash_4to8_bytes(s, length, seed);
251 if (length > 8 && length <= 16)
252 return hash_9to16_bytes(s, length, seed);
253 if (length > 16 && length <= 32)
254 return hash_17to32_bytes(s, length, seed);
255 if (length > 32)
256 return hash_33to64_bytes(s, length, seed);
257 if (length != 0)
258 return hash_1to3_bytes(s, length, seed);
259
260 return k2 ^ seed;
261}

266struct hash_state {
267 uint64_t h0, h1, h2, h3, h4, h5, h6;
268
272 static hash_state create(const char *s, uint64_t seed) {
273 hash_state state = {
274 0, seed, hash_16_bytes(seed, k1), rotate(seed ^ k1, 49),
275 seed * k1, shift_mix(seed), 0 };
276 state.h6 = hash_16_bytes(state.h4, state.h5);
277 state.mix(s);
278 return state;
279 }
280
283 static void mix_32_bytes(const char *s, uint64_t &a, uint64_t &b) {
284 a += fetch64(s);
285 uint64_t c = fetch64(s + 24);
286 b = rotate(b + a + c, 21);
287 uint64_t d = a;
288 a += fetch64(s + 8) + fetch64(s + 16);
289 b += rotate(a, 44) + d;
290 a += c;
291 }
292
296 void mix(const char *s) {
297 h0 = rotate(h0 + h1 + h3 + fetch64(s + 8), 37) * k1;
298 h1 = rotate(h1 + h4 + fetch64(s + 48), 42) * k1;
299 h0 ^= h6;
300 h1 += h3 + fetch64(s + 40);
301 h2 = rotate(h2 + h5, 33) * k1;
302 h3 = h4 * k1;
303 h4 = h0 + h5;
304 mix_32_bytes(s, h3, h4);
305 h5 = h2 + h6;
306 h6 = h1 + fetch64(s + 16);
307 mix_32_bytes(s + 32, h5, h6);
308 std::swap(h2, h0);
309 }
310
313 uint64_t finalize(size_t length) {
314 return hash_16_bytes(hash_16_bytes(h3, h5) + shift_mix(h1) * k1 + h2,
315 hash_16_bytes(h4, h6) + shift_mix(length) * k1 + h0);
316 }
317};

327inline uint64_t get_execution_seed() {
328 // FIXME: This needs to be a per-execution seed. This is just a placeholder
329 // implementation. Switching to a per-execution seed is likely to flush out
330 // instability bugs and so will happen as its own commit.
331 //
332 // However, if there is a fixed seed override set the first time this is
333 // called, return that instead of the per-execution seed.
334 const uint64_t seed_prime = 0xff51afd7ed558ccdULL;
335 static uint64_t seed = fixed_seed_override ? fixed_seed_override : seed_prime;
336 return seed;
337}

345//
346// FIXME: We want to replace is_integral_or_enum and is_pointer here with
347// a predicate which asserts that comparing the underlying storage of two
348// values of the type for equality is equivalent to comparing the two values
349// for equality. For all the platforms we care about, this holds for integers
350// and pointers, but there are platforms where it doesn't and we would like to
351// support user-defined types which happen to satisfy this property.
352template <typename T> struct is_hashable_data
353 : std::integral_constant<bool, ((is_integral_or_enum<T>::value ||
354 std::is_pointer<T>::value) &&
355 64 % sizeof(T) == 0)> {};
356
357// Special case std::pair to detect when both types are viable and when there
358// is no alignment-derived padding in the pair. This is a bit of a lie because
359// std::pair isn't truly POD, but it's close enough in all reasonable
360// implementations for our use case of hashing the underlying data.
361template <typename T, typename U> struct is_hashable_data<std::pair<T, U> >
362 : std::integral_constant<bool, (is_hashable_data<T>::value &&
363 is_hashable_data<U>::value &&
364 (sizeof(T) + sizeof(U)) ==
365 sizeof(std::pair<T, U>))> {};
366
369template <typename T>
370typename std::enable_if<is_hashable_data<T>::value, T>::type
371get_hashable_data(const T &value) {
372 return value;
373}
377template <typename T>
378typename std::enable_if<!is_hashable_data<T>::value, size_t>::type
379get_hashable_data(const T &value) {
380 using ::wpi::hash_value;
381 return hash_value(value);
382}

391template <typename T>
392bool store_and_advance(char *&buffer_ptr, char *buffer_end, const T& value,
393 size_t offset = 0) {
394 size_t store_size = sizeof(value) - offset;
395 if (buffer_ptr + store_size > buffer_end)
396 return false;
397 const char *value_data = reinterpret_cast<const char *>(&value);
398 memcpy(buffer_ptr, value_data + offset, store_size);
399 buffer_ptr += store_size;
400 return true;
401}

408template <typename InputIteratorT>
409hash_code hash_combine_range_impl(InputIteratorT first, InputIteratorT last) {
410 const uint64_t seed = get_execution_seed();
411 char buffer[64], *buffer_ptr = buffer;
412 char *const buffer_end = std::end(buffer);
413 while (first != last && store_and_advance(buffer_ptr, buffer_end,
414 get_hashable_data(*first)))
415 ++first;
416 if (first == last)
417 return hash_short(buffer, buffer_ptr - buffer, seed);
418 assert(buffer_ptr == buffer_end);
419
420 hash_state state = state.create(buffer, seed);
421 size_t length = 64;
422 while (first != last) {
423 // Fill up the buffer. We don't clear it, which re-mixes the last round
424 // when only a partial 64-byte chunk is left.
425 buffer_ptr = buffer;
426 while (first != last && store_and_advance(buffer_ptr, buffer_end,
427 get_hashable_data(*first)))
428 ++first;
429
430 // Rotate the buffer if we did a partial fill in order to simulate doing
431 // a mix of the last 64-bytes. That is how the algorithm works when we
432 // have a contiguous byte sequence, and we want to emulate that here.
433 std::rotate(buffer, buffer_ptr, buffer_end);
434
435 // Mix this chunk into the current state.
436 state.mix(buffer);
437 length += buffer_ptr - buffer;
438 };
439
440 return state.finalize(length);
441}

451template <typename ValueT>
452typename std::enable_if<is_hashable_data<ValueT>::value, hash_code>::type
453hash_combine_range_impl(ValueT *first, ValueT *last) {
454 const uint64_t seed = get_execution_seed();
455 const char *s_begin = reinterpret_cast<const char *>(first);
456 const char *s_end = reinterpret_cast<const char *>(last);
457 const size_t length = std::distance(s_begin, s_end);
458 if (length <= 64)
459 return hash_short(s_begin, length, seed);
460
461 const char *s_aligned_end = s_begin + (length & ~63);
462 hash_state state = state.create(s_begin, seed);
463 s_begin += 64;
464 while (s_begin != s_aligned_end) {
465 state.mix(s_begin);
466 s_begin += 64;
467 }
468 if (length & 63)
469 state.mix(s_end - 64);
470
471 return state.finalize(length);
472}

501struct hash_combine_recursive_helper {
502 char buffer[64];
503 hash_state state;
504 const uint64_t seed;
505
506public:
511 hash_combine_recursive_helper()
512 : seed(get_execution_seed()) {}
513
520 template <typename T>
521 char *combine_data(size_t &length, char *buffer_ptr, char *buffer_end, T data) {
522 if (!store_and_advance(buffer_ptr, buffer_end, data)) {
523 // Check for skew which prevents the buffer from being packed, and do
524 // a partial store into the buffer to fill it. This is only a concern
525 // with the variadic combine because that formation can have varying
526 // argument types.
527 size_t partial_store_size = buffer_end - buffer_ptr;
528 memcpy(buffer_ptr, &data, partial_store_size);
529
530 // If the store fails, our buffer is full and ready to hash. We have to
531 // either initialize the hash state (on the first full buffer) or mix
532 // this buffer into the existing hash state. Length tracks the *hashed*
533 // length, not the buffered length.
534 if (length == 0) {
535 state = state.create(buffer, seed);
536 length = 64;
537 } else {
538 // Mix this chunk into the current state and bump length up by 64.
539 state.mix(buffer);
540 length += 64;
541 }
542 // Reset the buffer_ptr to the head of the buffer for the next chunk of
543 // data.
544 buffer_ptr = buffer;
545
546 // Try again to store into the buffer -- this cannot fail as we only
547 // store types smaller than the buffer.
548 if (!store_and_advance(buffer_ptr, buffer_end, data,
549 partial_store_size))
550 abort();
551 }
552 return buffer_ptr;
553 }

559 template <typename T, typename ...Ts>
560 hash_code combine(size_t length, char *buffer_ptr, char *buffer_end,
561 const T &arg, const Ts &...args) {
562 buffer_ptr = combine_data(length, buffer_ptr, buffer_end, get_hashable_data(arg));
563
564 // Recurse to the next argument.
565 return combine(length, buffer_ptr, buffer_end, args...);
566 }
567
573 hash_code combine(size_t length, char *buffer_ptr, char *buffer_end) {
574 // Check whether the entire set of values fit in the buffer. If so, we'll
575 // use the optimized short hashing routine and skip state entirely.
576 if (length == 0)
577 return static_cast<size_t>(hash_short(buffer, buffer_ptr - buffer, seed));
578
579 // Mix the final buffer, rotating it if we did a partial fill in order to
580 // simulate doing a mix of the last 64-bytes. That is how the algorithm
581 // works when we have a contiguous byte sequence, and we want to emulate
582 // that here.
583 std::rotate(buffer, buffer_ptr, buffer_end);
584
585 // Mix this chunk into the current state.
586 state.mix(buffer);
587 length += buffer_ptr - buffer;
588
589 return static_cast<size_t>(state.finalize(length));
590 }

607template <typename ...Ts> hash_code hash_combine(const Ts &...args) {
608 // Recursively hash each argument using a helper class.
609 ::wpi::hashing::detail::hash_combine_recursive_helper helper;
610 return helper.combine(0, helper.buffer, helper.buffer + 64, args...);
611}

623inline hash_code hash_integer_value(uint64_t value) {
624 // Similar to hash_4to8_bytes but using a seed instead of length.
625 const uint64_t seed = get_execution_seed();
626 const char *s = reinterpret_cast<const char *>(&value);
627 const uint64_t a = fetch32(s);
628 return static_cast<size_t>(hash_16_bytes(seed + (a << 3), fetch32(s + 4)));
629}
