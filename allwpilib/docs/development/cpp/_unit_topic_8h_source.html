24/**
25 * Timestamped unit.
26 *
27 * @tparam T unit type, e.g. units::meter_t
28 */
29template <typename T>
30struct TimestampedUnit {
31 TimestampedUnit() = default;
32 TimestampedUnit(int64_t time, int64_t serverTime, T value)
33 : time{time}, serverTime{serverTime}, value{value} {}
34
35 /**
36 * Time in local time base.
37 */
38 int64_t time = 0;
39
40 /**
41 * Time in server time base. May be 0 or 1 for locally set values.
42 */
43 int64_t serverTime = 0;
44
45 /**
46 * Value.
47 */
48 T value = {};
49};

239/**
240 * NetworkTables unit-typed topic. Publishers publish the type name (e.g.
241 * "meter") as the "unit" property. Type conversions are not performed--for
242 * correct behavior the publisher and subscriber must use the same unit type,
243 * but this can be checked at runtime using IsMatchingUnit().
244 *
245 * @tparam T unit type, e.g. units::meter_t
246 */
247template <typename T>
248class UnitTopic final : public Topic {
249 public:
250 using SubscriberType = UnitSubscriber<T>;
251 using PublisherType = UnitPublisher<T>;
252 using EntryType = UnitEntry<T>;
253 using ValueType = T;
254 using ParamType = T;
255 using TimestampedValueType = TimestampedUnit<T>;
256 /** The default type string for this topic type. */
257 static constexpr std::string_view kTypeString = "double";
258
259 UnitTopic() = default;
260

262 * Construct from a topic handle.
263 *
264 * @param handle Native handle

266 explicit UnitTopic(NT_Topic handle) : Topic{handle} {}

273 explicit UnitTopic(Topic topic) : Topic{topic} {}
