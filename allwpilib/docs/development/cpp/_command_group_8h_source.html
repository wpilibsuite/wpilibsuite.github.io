17/**
18 * A CommandGroup is a list of commands which are executed in sequence.
19 *
20 * Commands in a CommandGroup are added using the AddSequential() method and are
21 * called sequentially. CommandGroups are themselves Commands and can be given
22 * to other CommandGroups.
23 *
24 * CommandGroups will carry all of the requirements of their Command
25 * subcommands. Additional requirements can be specified by calling Requires()
26 * normally in the constructor.
27 *
28 * CommandGroups can also execute commands in parallel, simply by adding them
29 * using AddParallel().
30 *
31 * @see Command
32 * @see Subsystem
33 */
34class CommandGroup : public Command {
35 public:
36 CommandGroup() = default;
37
38 /**
39 * Creates a new CommandGroup with the given name.
40 *
41 * @param name The name for this command group
42 */
43 explicit CommandGroup(std::string_view name);
44
45 ~CommandGroup() override = default;
46
47 CommandGroup(CommandGroup&&) = default;
48 CommandGroup& operator=(CommandGroup&&) = default;
49
50 /**
51 * Adds a new Command to the group. The Command will be started after all the
52 * previously added Commands.
53 *
54 * Note that any requirements the given Command has will be added to the
55 * group. For this reason, a Command's requirements can not be changed after
56 * being added to a group.
57 *
58 * It is recommended that this method be called in the constructor.
59 *
60 * @param command The Command to be added
61 */
62 void AddSequential(Command* command);
63
64 /**
65 * Adds a new Command to the group with a given timeout. The Command will be
66 * started after all the previously added commands.
67 *
68 * Once the Command is started, it will be run until it finishes or the time
69 * expires, whichever is sooner. Note that the given Command will have no
70 * knowledge that it is on a timer.
71 *
72 * Note that any requirements the given Command has will be added to the
73 * group. For this reason, a Command's requirements can not be changed after
74 * being added to a group.
75 *
76 * It is recommended that this method be called in the constructor.
77 *
78 * @param command The Command to be added
79 * @param timeout The timeout
80 */
81 void AddSequential(Command* command, units::second_t timeout);
82
83 /**
84 * Adds a new child Command to the group. The Command will be started after
85 * all the previously added Commands.
86 *
87 * Instead of waiting for the child to finish, a CommandGroup will have it run
88 * at the same time as the subsequent Commands. The child will run until
89 * either it finishes, a new child with conflicting requirements is started,
90 * or the main sequence runs a Command with conflicting requirements. In the
91 * latter two cases, the child will be canceled even if it says it can't be
92 * interrupted.
93 *
94 * Note that any requirements the given Command has will be added to the
95 * group. For this reason, a Command's requirements can not be changed after
96 * being added to a group.
97 *
98 * It is recommended that this method be called in the constructor.
99 *
100 * @param command The command to be added
101 */
102 void AddParallel(Command* command);
103
104 /**
105 * Adds a new child Command to the group with the given timeout. The Command
106 * will be started after all the previously added Commands.
107 *
108 * Once the Command is started, it will run until it finishes, is interrupted,
109 * or the time expires, whichever is sooner. Note that the given Command will
110 * have no knowledge that it is on a timer.
111 *
112 * Instead of waiting for the child to finish, a CommandGroup will have it run
113 * at the same time as the subsequent Commands. The child will run until
114 * either it finishes, the timeout expires, a new child with conflicting
115 * requirements is started, or the main sequence runs a Command with
116 * conflicting requirements. In the latter two cases, the child will be
117 * canceled even if it says it can't be interrupted.
118 *
119 * Note that any requirements the given Command has will be added to the
120 * group. For this reason, a Command's requirements can not be changed after
121 * being added to a group.
122 *
123 * It is recommended that this method be called in the constructor.
124 *
125 * @param command The command to be added
126 * @param timeout The timeout
127 */
128 void AddParallel(Command* command, units::second_t timeout);
