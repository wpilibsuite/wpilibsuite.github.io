17/**
18 * A plant defined using state-space notation.
19 *
20 * A plant is a mathematical model of a system's dynamics.
21 *
22 * For more on the underlying math, read
23 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf.
24 */
25template <int States, int Inputs, int Outputs>
26class LinearSystem {
27 public:
28 /**
29 * Constructs a discrete plant with the given continuous system coefficients.
30 *
31 * @param A System matrix.
32 * @param B Input matrix.
33 * @param C Output matrix.
34 * @param D Feedthrough matrix.
35 */
36 LinearSystem(const Eigen::Matrix<double, States, States>& A,
37 const Eigen::Matrix<double, States, Inputs>& B,
38 const Eigen::Matrix<double, Outputs, States>& C,
39 const Eigen::Matrix<double, Outputs, Inputs>& D) {
40 m_A = A;
41 m_B = B;
42 m_C = C;
43 m_D = D;
44 }
45
46 LinearSystem(const LinearSystem&) = default;
47 LinearSystem& operator=(const LinearSystem&) = default;
48 LinearSystem(LinearSystem&&) = default;
49 LinearSystem& operator=(LinearSystem&&) = default;
50
51 /**
52 * Returns the system matrix A.
53 */
54 const Eigen::Matrix<double, States, States>& A() const { return m_A; }
55
56 /**
57 * Returns an element of the system matrix A.
58 *
59 * @param i Row of A.
60 * @param j Column of A.
61 */
62 double A(int i, int j) const { return m_A(i, j); }
63
64 /**
65 * Returns the input matrix B.
66 */
67 const Eigen::Matrix<double, States, Inputs>& B() const { return m_B; }
68
69 /**
70 * Returns an element of the input matrix B.
71 *
72 * @param i Row of B.
73 * @param j Column of B.
74 */
75 double B(int i, int j) const { return m_B(i, j); }
76
77 /**
78 * Returns the output matrix C.
79 */
80 const Eigen::Matrix<double, Outputs, States>& C() const { return m_C; }
81
82 /**
83 * Returns an element of the output matrix C.
84 *
85 * @param i Row of C.
86 * @param j Column of C.
87 */
88 double C(int i, int j) const { return m_C(i, j); }
89
90 /**
91 * Returns the feedthrough matrix D.
92 */
93 const Eigen::Matrix<double, Outputs, Inputs>& D() const { return m_D; }
94
95 /**
96 * Returns an element of the feedthrough matrix D.
97 *
98 * @param i Row of D.
99 * @param j Column of D.
100 */
101 double D(int i, int j) const { return m_D(i, j); }
102
103 /**
104 * Computes the new x given the old x and the control input.
105 *
106 * This is used by state observers directly to run updates based on state
107 * estimate.
108 *
109 * @param x The current state.
110 * @param clampedU The control input.
111 * @param dt Timestep for model update.
112 */
113 Eigen::Vector<double, States> CalculateX(
114 const Eigen::Vector<double, States>& x,
115 const Eigen::Vector<double, Inputs>& clampedU, units::second_t dt) const {
116 Eigen::Matrix<double, States, States> discA;
117 Eigen::Matrix<double, States, Inputs> discB;
118 DiscretizeAB<States, Inputs>(m_A, m_B, dt, &discA, &discB);
119
120 return discA * x + discB * clampedU;
121 }
122
123 /**
124 * Computes the new y given the control input.
125 *
126 * This is used by state observers directly to run updates based on state
127 * estimate.
128 *
129 * @param x The current state.
130 * @param clampedU The control input.
131 */
132 Eigen::Vector<double, Outputs> CalculateY(
133 const Eigen::Vector<double, States>& x,
134 const Eigen::Vector<double, Inputs>& clampedU) const {
135 return m_C * x + m_D * clampedU;
136 }
137
138 private:
139 /**
140 * Continuous system matrix.
141 */
142 Eigen::Matrix<double, States, States> m_A;
143
144 /**
145 * Continuous input matrix.
146 */
147 Eigen::Matrix<double, States, Inputs> m_B;
148
149 /**
150 * Output matrix.
151 */
152 Eigen::Matrix<double, Outputs, States> m_C;
153
154 /**
155 * Feedthrough matrix.
156 */
157 Eigen::Matrix<double, Outputs, Inputs> m_D;
158};
