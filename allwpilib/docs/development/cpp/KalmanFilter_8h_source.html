40template <int States, int Inputs, int Outputs>
41class KalmanFilterImpl {
42 public:
51 KalmanFilterImpl(LinearSystem<States, Inputs, Outputs>& plant,
52 const wpi::array<double, States>& stateStdDevs,
53 const wpi::array<double, Outputs>& measurementStdDevs,
54 units::second_t dt) {
55 m_plant = &plant;
56
57 auto contQ = MakeCovMatrix(stateStdDevs);
58 auto contR = MakeCovMatrix(measurementStdDevs);
59
60 Eigen::Matrix<double, States, States> discA;
61 Eigen::Matrix<double, States, States> discQ;
62 DiscretizeAQTaylor<States>(plant.A(), contQ, dt, &discA, &discQ);
63
64 auto discR = DiscretizeR<Outputs>(contR, dt);
65
66 const auto& C = plant.C();
67
68 // IsStabilizable(Aᵀ, Cᵀ) will tell us if the system is observable.
69 bool isObservable =
70 IsStabilizable<States, Outputs>(discA.transpose(), C.transpose());
71 if (!isObservable) {
72 wpi::math::MathSharedStore::ReportError(
73 "The system passed to the Kalman filter is not observable!");
74 throw std::invalid_argument(
75 "The system passed to the Kalman filter is not observable!");
76 }
77
78 Eigen::Matrix<double, States, States> P =
79 drake::math::DiscreteAlgebraicRiccatiEquation(
80 discA.transpose(), C.transpose(), discQ, discR);
81
82 // S = CPCᵀ + R
83 Eigen::Matrix<double, Outputs, Outputs> S = C * P * C.transpose() + discR;
84
85 // We want to put K = PCᵀS⁻¹ into Ax = b form so we can solve it more
86 // efficiently.
87 //
88 // K = PCᵀS⁻¹
89 // KS = PCᵀ
90 // (KS)ᵀ = (PCᵀ)ᵀ
91 // SᵀKᵀ = CPᵀ
92 //
93 // The solution of Ax = b can be found via x = A.solve(b).
94 //
95 // Kᵀ = Sᵀ.solve(CPᵀ)
96 // K = (Sᵀ.solve(CPᵀ))ᵀ
97 m_K = S.transpose().ldlt().solve(C * P.transpose()).transpose();
98
99 Reset();
100 }

103 KalmanFilterImpl& operator=(KalmanFilterImpl&&) = default;

108 const Eigen::Matrix<double, States, Outputs>& K() const { return m_K; }
109
116 double K(int i, int j) const { return m_K(i, j); }
117
121 const Eigen::Vector<double, States>& Xhat() const { return m_xHat; }
122
128 double Xhat(int i) const { return m_xHat(i); }
129
135 void SetXhat(const Eigen::Vector<double, States>& xHat) { m_xHat = xHat; }
136
143 void SetXhat(int i, double value) { m_xHat(i) = value; }
144
148 void Reset() { m_xHat.setZero(); }
149
156 void Predict(const Eigen::Vector<double, Inputs>& u, units::second_t dt) {
157 m_xHat = m_plant->CalculateX(m_xHat, u, dt);
158 }
159
166 void Correct(const Eigen::Vector<double, Inputs>& u,
167 const Eigen::Vector<double, Outputs>& y) {
168 // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − (Cx̂ₖ₊₁⁻ + Duₖ₊₁))
169 m_xHat += m_K * (y - (m_plant->C() * m_xHat + m_plant->D() * u));
170 }
171
172 private:
173 LinearSystem<States, Inputs, Outputs>* m_plant;
174
178 Eigen::Matrix<double, States, Outputs> m_K;
179
183 Eigen::Vector<double, States> m_xHat;
184};
