39 template <int States, int Inputs, int Outputs>
40 class KalmanFilterImpl {
41  public:
50  KalmanFilterImpl(LinearSystem<States, Inputs, Outputs>& plant,
51  const wpi::array<double, States>& stateStdDevs,
52  const wpi::array<double, Outputs>& measurementStdDevs,
53  units::second_t dt) {
54  m_plant = &plant;
55 
56  auto contQ = MakeCovMatrix(stateStdDevs);
57  auto contR = MakeCovMatrix(measurementStdDevs);
58 
59  Eigen::Matrix<double, States, States> discA;
60  Eigen::Matrix<double, States, States> discQ;
61  DiscretizeAQTaylor<States>(plant.A(), contQ, dt, &discA, &discQ);
62 
63  auto discR = DiscretizeR<Outputs>(contR, dt);
64 
65  const auto& C = plant.C();
66 
67  // IsStabilizable(A^T, C^T) will tell us if the system is observable.
68  bool isObservable =
69  IsStabilizable<States, Outputs>(discA.transpose(), C.transpose());
70  if (!isObservable) {
71  wpi::math::MathSharedStore::ReportError(
72  "The system passed to the Kalman filter is not observable!");
73  throw std::invalid_argument(
74  "The system passed to the Kalman filter is not observable!");
75  }
76 
77  Eigen::Matrix<double, States, States> P =
78  drake::math::DiscreteAlgebraicRiccatiEquation(
79  discA.transpose(), C.transpose(), discQ, discR);
80 
81  Eigen::Matrix<double, Outputs, Outputs> S = C * P * C.transpose() + discR;
82 
83  // We want to put K = PC^T S^-1 into Ax = b form so we can solve it more
84  // efficiently.
85  //
86  // K = PC^T S^-1
87  // KS = PC^T
88  // (KS)^T = (PC^T)^T
89  // S^T K^T = CP^T
90  //
91  // The solution of Ax = b can be found via x = A.solve(b).
92  //
93  // K^T = S^T.solve(CP^T)
94  // K = (S^T.solve(CP^T))^T
95  m_K = S.transpose().ldlt().solve(C * P.transpose()).transpose();
96 
97  Reset();
98  }

101  KalmanFilterImpl& operator=(KalmanFilterImpl&&) = default;

164  void Correct(const Eigen::Matrix<double, Inputs, 1>& u,
165  const Eigen::Matrix<double, Outputs, 1>& y) {
166  m_xHat += m_K * (y - (m_plant->C() * m_xHat + m_plant->D() * u));
167  }
