30/**
31 * A class for driving Mecanum drive platforms.
32 *
33 * Mecanum drives are rectangular with one wheel on each corner. Each wheel has
34 * rollers toed in 45 degrees toward the front or back. When looking at the
35 * wheels from the top, the roller axles should form an X across the robot.
36 *
37 * Drive base diagram:
38 * <pre>
39 * \\_______/
40 * \\ | | /
41 * | |
42 * /_|___|_\\
43 * / \\
44 * </pre>
45 *
46 * Each Drive() function provides different inverse kinematic relations for a
47 * Mecanum drive robot.
48 *
49 * The positive Y axis points ahead, the positive X axis points to the right,
50 * and the positive Z axis points down. Rotations follow the right-hand rule, so
51 * clockwise rotation around the Z axis is positive.
52 *
53 * Note: the axis conventions used in this class differ from DifferentialDrive.
54 * This may change in a future year's WPILib release.
55 *
56 * Inputs smaller then 0.02 will be set to 0, and larger values will be scaled
57 * so that the full range is still used. This deadband value can be changed
58 * with SetDeadband().
59 *
60 * MotorSafety is enabled by default. The DriveCartesian or DrivePolar
61 * methods should be called periodically to avoid Motor Safety timeouts.
62 */

64 public wpi::Sendable,
65 public wpi::SendableHelper<MecanumDrive> {

93 /**
94 * Drive method for Mecanum platform.
95 *
96 * Angles are measured clockwise from the positive X axis. The robot's speed
97 * is independent from its angle or rotation rate.
98 *
99 * @param ySpeed The robot's speed along the Y axis [-1.0..1.0]. Forward is
100 * positive.
101 * @param xSpeed The robot's speed along the X axis [-1.0..1.0]. Right is
102 * positive.
103 * @param zRotation The robot's rotation rate around the Z axis [-1.0..1.0].
104 * Clockwise is positive.
105 * @param gyroAngle The current angle reading from the gyro in degrees around
106 * the Z axis. Use this to implement field-oriented controls.
107 */
108 void DriveCartesian(double ySpeed, double xSpeed, double zRotation,
109 double gyroAngle = 0.0);
110
111 /**
112 * Drive method for Mecanum platform.
113 *
114 * Angles are measured clockwise from the positive X axis. The robot's speed
115 * is independent from its angle or rotation rate.
116 *
117 * @param magnitude The robot's speed at a given angle [-1.0..1.0]. Forward is
118 * positive.
119 * @param angle The angle around the Z axis at which the robot drives in
120 * degrees [-180..180].
121 * @param zRotation The robot's rotation rate around the Z axis [-1.0..1.0].
122 * Clockwise is positive.
123 */
124 void DrivePolar(double magnitude, double angle, double zRotation);
125
126 /**
127 * Cartesian inverse kinematics for Mecanum platform.
128 *
129 * Angles are measured clockwise from the positive X axis. The robot's speed
130 * is independent from its angle or rotation rate.
131 *
132 * @param ySpeed The robot's speed along the Y axis [-1.0..1.0]. Forward is
133 * positive.
134 * @param xSpeed The robot's speed along the X axis [-1.0..1.0]. Right is
135 * positive.
136 * @param zRotation The robot's rotation rate around the Z axis [-1.0..1.0].
137 * Clockwise is positive.
138 * @param gyroAngle The current angle reading from the gyro in degrees around
139 * the Z axis. Use this to implement field-oriented controls.
140 * @return Wheel speeds [-1.0..1.0].
141 */
142 static WheelSpeeds DriveCartesianIK(double ySpeed, double xSpeed,
143 double zRotation, double gyroAngle = 0.0);
144
145 void StopMotor() override;
146 std::string GetDescription() const override;
147

149

153 SpeedController* m_frontRightMotor;
154 SpeedController* m_rearRightMotor;
155
156 bool reported = false;
