80 /// A stateless \concept{concept_rawallocator,RawAllocator} that allocates memory using the virtual memory allocation functions.
81 /// It does not prereserve any memory and will always reserve and commit combined.
82 /// \ingroup allocator
83 class virtual_memory_allocator
84 : WPI_EBO(detail::global_leak_checker<detail::virtual_memory_allocator_leak_handler>)
85 {
86 public:
87 using is_stateful = std::false_type;
88
89 virtual_memory_allocator() noexcept = default;
90 virtual_memory_allocator(virtual_memory_allocator&&) noexcept {}
91 ~virtual_memory_allocator() noexcept = default;
92
93 virtual_memory_allocator& operator=(virtual_memory_allocator&&) noexcept
94 {
95 return *this;
96 }
97
98 /// \effects A \concept{concept_rawallocator,RawAllocator} allocation function.
99 /// It uses \ref virtual_memory_reserve followed by \ref virtual_memory_commit for the allocation.
100 /// The number of pages allocated will be the minimum to hold \c size continuous bytes,
101 /// i.e. \c size will be rounded up to the next multiple.
102 /// If debug fences are activated, one additional page before and after the memory will be allocated.
103 /// \returns A pointer to a \concept{concept_node,node}, it will never be \c nullptr.
104 /// It will always be aligned on a fence boundary, regardless of the alignment parameter.
105 /// \throws An exception of type \ref out_of_memory or whatever is thrown by its handler if the allocation fails.
106 void* allocate_node(std::size_t size, std::size_t alignment);
107
108 /// \effects A \concept{concept_rawallocator,RawAllocator} deallocation function.
109 /// It calls \ref virtual_memory_decommit followed by \ref virtual_memory_release for the deallocation.
110 void deallocate_node(void* node, std::size_t size, std::size_t alignment) noexcept;
111
112 /// \returns The maximum node size by returning the maximum value.
113 std::size_t max_node_size() const noexcept;
114
115 /// \returns The maximum alignment which is the same as the \ref virtual_memory_page_size.
116 std::size_t max_alignment() const noexcept;
117 };

126 /// A \concept{concept_blockallocator,BlockAllocator} that reserves virtual memory and commits it part by part.
127 /// It is similar to \ref memory_stack but does not support growing and uses virtual memory,
128 /// also meant for big blocks not small allocations.
129 /// \ingroup allocator
130 class virtual_block_allocator
131 {
132 public:
133 /// \effects Creates it giving it the block size and the total number of blocks it can allocate.
134 /// It reserves enough virtual memory for <tt>block_size * no_blocks</tt>.
135 /// \requires \c block_size must be non-zero and a multiple of the \ref virtual_memory_page_size.
136 /// \c no_blocks must be bigger than \c 1.
137 /// \throws \ref out_of_memory if it cannot reserve the virtual memory.
138 explicit virtual_block_allocator(std::size_t block_size, std::size_t no_blocks);
139
140 /// \effects Releases the reserved virtual memory.
141 ~virtual_block_allocator() noexcept;
142
143 /// @{
144 /// \effects Moves the block allocator, it transfers ownership over the reserved area.
145 /// This does not invalidate any memory blocks.
146 virtual_block_allocator(virtual_block_allocator&& other) noexcept
147 : cur_(other.cur_), end_(other.end_), block_size_(other.block_size_)
148 {
149 other.cur_ = other.end_ = nullptr;
150 other.block_size_ = 0;
151 }
152
153 virtual_block_allocator& operator=(virtual_block_allocator&& other) noexcept
154 {
155 virtual_block_allocator tmp(detail::move(other));
156 swap(*this, tmp);
157 return *this;
158 }
159 /// @}
160
161 /// \effects Swaps the ownership over the reserved memory.
162 /// This does not invalidate any memory blocks.
163 friend void swap(virtual_block_allocator& a, virtual_block_allocator& b) noexcept
164 {
165 detail::adl_swap(a.cur_, b.cur_);
166 detail::adl_swap(a.end_, b.end_);
167 detail::adl_swap(a.block_size_, b.block_size_);
168 }
169
170 /// \effects Allocates a new memory block by committing the next \ref next_block_size() number of bytes.
171 /// \returns The \ref memory_block committed.
172 /// \throws \ref out_of_memory if it cannot commit the memory or the \ref capacity_left() is exhausted.
173 memory_block allocate_block();
174
175 /// \effects Deallocates the last allocated memory block by decommitting it.
176 /// This block will be returned again on the next call to \ref allocate_block().
177 /// \requires \c block must be the current top block of the memory,
178 /// this is guaranteed by \ref memory_arena.
179 void deallocate_block(memory_block block) noexcept;
180
181 /// \returns The next block size, this is the block size of the constructor.
182 std::size_t next_block_size() const noexcept
183 {
184 return block_size_;
185 }
186
187 /// \returns The number of blocks that can be committed until it runs out of memory.
188 std::size_t capacity_left() const noexcept
189 {
190 return static_cast<std::size_t>(end_ - cur_) / block_size_;
191 }
192
193 private:
194 allocator_info info() noexcept;
195
196 char * cur_, *end_;
197 std::size_t block_size_;
198 };
