23template <int Rows, int Cols, typename Matrix, typename T, typename... Ts>
24void MatrixImpl(Matrix& result, T elem, Ts... elems) {
25 constexpr int count = Rows * Cols - (sizeof...(Ts) + 1);
26
27 result(count / Cols, count % Cols) = elem;
28 if constexpr (sizeof...(Ts) > 0) {
29 MatrixImpl<Rows, Cols>(result, elems...);
30 }
31}

61template <int States, int Inputs>
62bool IsStabilizableImpl(const Eigen::Matrix<double, States, States>& A,
63 const Eigen::Matrix<double, States, Inputs>& B) {
64 Eigen::EigenSolver<Eigen::Matrix<double, States, States>> es{A};
65
66 for (int i = 0; i < States; ++i) {
67 if (es.eigenvalues()[i].real() * es.eigenvalues()[i].real() +
68 es.eigenvalues()[i].imag() * es.eigenvalues()[i].imag() <
69 1) {
70 continue;
71 }
72
73 Eigen::Matrix<std::complex<double>, States, States + Inputs> E;
74 E << es.eigenvalues()[i] * Eigen::Matrix<std::complex<double>, States,
75 States>::Identity() -
76 A,
77 B;
78
79 Eigen::ColPivHouseholderQR<
80 Eigen::Matrix<std::complex<double>, States, States + Inputs>>
81 qr{E};
82 if (qr.rank() < States) {
83 return false;
84 }
85 }
86 return true;
87}

115/**
116 * Creates a cost matrix from the given vector for use with LQR.
117 *
118 * The cost matrix is constructed using Bryson's rule. The inverse square of
119 * each element in the input is taken and placed on the cost matrix diagonal.
120 *
121 * @param costs An array. For a Q matrix, its elements are the maximum allowed
122 * excursions of the states from the reference. For an R matrix,
123 * its elements are the maximum allowed excursions of the control
124 * inputs from no actuation.
125 * @return State excursion or control effort cost matrix.
126 */
127template <typename... Ts, typename = std::enable_if_t<
128 std::conjunction_v<std::is_same<double, Ts>...>>>
129Eigen::Matrix<double, sizeof...(Ts), sizeof...(Ts)> MakeCostMatrix(
130 Ts... costs) {
131 Eigen::DiagonalMatrix<double, sizeof...(Ts)> result;
132 detail::CostMatrixImpl(result.diagonal(), costs...);
133 return result;
134}

157/**
158 * Creates a cost matrix from the given vector for use with LQR.
159 *
160 * The cost matrix is constructed using Bryson's rule. The inverse square of
161 * each element in the input is taken and placed on the cost matrix diagonal.
162 *
163 * @param costs An array. For a Q matrix, its elements are the maximum allowed
164 * excursions of the states from the reference. For an R matrix,
165 * its elements are the maximum allowed excursions of the control
166 * inputs from no actuation.
167 * @return State excursion or control effort cost matrix.
168 */
169template <size_t N>
170Eigen::Matrix<double, N, N> MakeCostMatrix(const std::array<double, N>& costs) {
171 Eigen::DiagonalMatrix<double, N> result;
172 auto& diag = result.diagonal();
173 for (size_t i = 0; i < N; ++i) {
174 diag(i) = 1.0 / std::pow(costs[i], 2);
175 }
176 return result;
177}

210/**
211 * Creates a vector of normally distributed white noise with the given noise
212 * intensities for each element.
213 *
214 * @param stdDevs An array whose elements are the standard deviations of each
215 * element of the noise vector.
216 * @return White noise vector.
217 */
218template <int N>
219Eigen::Vector<double, N> MakeWhiteNoiseVector(
220 const std::array<double, N>& stdDevs) {
221 std::random_device rd;
222 std::mt19937 gen{rd()};
223
224 Eigen::Vector<double, N> result;
225 for (int i = 0; i < N; ++i) {
226 // Passing a standard deviation of 0.0 to std::normal_distribution is
227 // undefined behavior
228 if (stdDevs[i] == 0.0) {
229 result(i) = 0.0;
230 } else {
231 std::normal_distribution distr{0.0, stdDevs[i]};
232 result(i) = distr(gen);
233 }
234 }
235 return result;
236}

258/**
259 * Returns true if (A, B) is a stabilizable pair.
260 *
261 * (A, B) is stabilizable if and only if the uncontrollable eigenvalues of A, if
262 * any, have absolute values less than one, where an eigenvalue is
263 * uncontrollable if rank(λI - A, B) < n where n is the number of states.
264 *
265 * @tparam States The number of states.
266 * @tparam Inputs The number of inputs.
267 * @param A System matrix.
268 * @param B Input matrix.
269 */
270template <int States, int Inputs>
271bool IsStabilizable(const Eigen::Matrix<double, States, States>& A,
272 const Eigen::Matrix<double, States, Inputs>& B) {
273 return detail::IsStabilizableImpl<States, Inputs>(A, B);
274}

276/**
277 * Returns true if (A, C) is a detectable pair.
278 *
279 * (A, C) is detectable if and only if the unobservable eigenvalues of A, if
280 * any, have absolute values less than one, where an eigenvalue is unobservable
281 * if rank(λI - A; C) < n where n is the number of states.
282 *
283 * @tparam States The number of states.
284 * @tparam Outputs The number of outputs.
285 * @param A System matrix.
286 * @param C Output matrix.
287 */
288template <int States, int Outputs>
289bool IsDetectable(const Eigen::Matrix<double, States, States>& A,
290 const Eigen::Matrix<double, Outputs, States>& C) {
291 return detail::IsStabilizableImpl<States, Outputs>(A.transpose(),
292 C.transpose());
293}

338/**
339 * Normalize all inputs if any excedes the maximum magnitude. Useful for systems
340 * such as differential drivetrains.
341 *
342 * @tparam Inputs The number of inputs.
343 * @param u The input vector.
344 * @param maxMagnitude The maximum magnitude any input can have.
345 * @return The normalizedInput
346 */
347template <int Inputs>
348Eigen::Vector<double, Inputs> NormalizeInputVector(
349 const Eigen::Vector<double, Inputs>& u, double maxMagnitude) {
350 double maxValue = u.template lpNorm<Eigen::Infinity>();
351
352 if (maxValue > maxMagnitude) {
353 return u * maxMagnitude / maxValue;
354 }
355 return u;
356}
