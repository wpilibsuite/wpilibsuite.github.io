23template <int Rows, int Cols, typename Matrix, typename T, typename... Ts>
24void MatrixImpl(Matrix& result, T elem, Ts... elems) {
25 constexpr int count = Rows * Cols - (sizeof...(Ts) + 1);
26
27 result(count / Cols, count % Cols) = elem;
28 if constexpr (sizeof...(Ts) > 0) {
29 MatrixImpl<Rows, Cols>(result, elems...);
30 }
31}

50void WhiteNoiseVectorImpl(Matrix& result, T elem, Ts... elems) {
51 std::random_device rd;

53 std::normal_distribution<> distr{0.0, elem};

61template <int States, int Inputs>
62bool IsStabilizableImpl(const Eigen::Matrix<double, States, States>& A,
63 const Eigen::Matrix<double, States, Inputs>& B) {
64 Eigen::EigenSolver<Eigen::Matrix<double, States, States>> es{A};
65
66 for (int i = 0; i < States; ++i) {
67 if (es.eigenvalues()[i].real() * es.eigenvalues()[i].real() +
68 es.eigenvalues()[i].imag() * es.eigenvalues()[i].imag() <
69 1) {
70 continue;
71 }
72
73 Eigen::Matrix<std::complex<double>, States, States + Inputs> E;
74 E << es.eigenvalues()[i] * Eigen::Matrix<std::complex<double>, States,
75 States>::Identity() -
76 A,
77 B;
78
79 Eigen::ColPivHouseholderQR<
80 Eigen::Matrix<std::complex<double>, States, States + Inputs>>
81 qr(E);
82 if (qr.rank() < States) {
83 return false;
84 }
85 }
86 return true;
87}

169template <size_t N>
170Eigen::Matrix<double, N, N> MakeCostMatrix(const std::array<double, N>& costs) {
171 Eigen::DiagonalMatrix<double, N> result;
172 auto& diag = result.diagonal();
173 for (size_t i = 0; i < N; ++i) {
174 diag(i) = 1.0 / std::pow(costs[i], 2);

218template <int N>
219Eigen::Vector<double, N> MakeWhiteNoiseVector(
220 const std::array<double, N>& stdDevs) {
221 std::random_device rd;
222 std::mt19937 gen{rd()};
223
224 Eigen::Vector<double, N> result;
225 for (int i = 0; i < N; ++i) {
226 // Passing a standard deviation of 0.0 to std::normal_distribution is
227 // undefined behavior
228 if (stdDevs[i] == 0.0) {
229 result(i) = 0.0;
230 } else {
231 std::normal_distribution distr{0.0, stdDevs[i]};
232 result(i) = distr(gen);
233 }
234 }
235 return result;
236}

302template <int Inputs>

304 const Eigen::Vector<double, Inputs>& u,
305 const Eigen::Vector<double, Inputs>& umin,
306 const Eigen::Vector<double, Inputs>& umax) {
307 Eigen::Vector<double, Inputs> result;

323template <int Inputs>
324Eigen::Vector<double, Inputs> NormalizeInputVector(
325 const Eigen::Vector<double, Inputs>& u, double maxMagnitude) {
326 double maxValue = u.template lpNorm<Eigen::Infinity>();
327
328 if (maxValue > maxMagnitude) {
329 return u * maxMagnitude / maxValue;
330 }
331 return u;
332}
