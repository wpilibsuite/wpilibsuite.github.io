21/**
22 * A class used to bind command scheduling to events. The
23 * Trigger class is a base for all command-event-binding classes, and so the
24 * methods are named fairly abstractly; for purpose-specific wrappers, see

27 * This class is provided by the NewCommands VendorDep

29 * @see Button

51 /**
52 * Binds a command to start when the trigger becomes active. Takes a
53 * raw pointer, and so is non-owning; users are responsible for the lifespan
54 * of the command.
55 *
56 * @param command The command to bind.
57 * @param interruptible Whether the command should be interruptible.

62 /**
63 * Binds a command to start when the trigger becomes active. Transfers
64 * command ownership to the button scheduler, so the user does not have to
65 * worry about lifespan - rvalue refs will be *moved*, lvalue refs will be
66 * *copied.*
67 *
68 * @param command The command to bind.
69 * @param interruptible Whether the command should be interruptible.
70 * @return The trigger, for chained calls.
71 */
72 template <class T, typename = std::enable_if_t<std::is_base_of_v<
73 Command, std::remove_reference_t<T>>>>
74 Trigger WhenActive(T&& command, bool interruptible = true) {
75 CommandScheduler::GetInstance().AddButton(
76 [pressedLast = m_isActive(), *this,
77 command = std::make_unique<std::remove_reference_t<T>>(
78 std::forward<T>(command)),
79 interruptible]() mutable {
80 bool pressed = m_isActive();
81
82 if (!pressedLast && pressed) {
83 command->Schedule(interruptible);
84 }
85
86 pressedLast = pressed;
87 });
88
89 return *this;
90 }

121 /**
122 * Binds a command to be started repeatedly while the trigger is active, and
123 * canceled when it becomes inactive. Transfers command ownership to the
124 * button scheduler, so the user does not have to worry about lifespan -
125 * rvalue refs will be *moved*, lvalue refs will be *copied.*

281 /**
282 * Binds a command to start when the trigger becomes active, and be canceled
283 * when it again becomes active. Transfers command ownership to the button
284 * scheduler, so the user does not have to worry about lifespan - rvalue refs
285 * will be *moved*, lvalue refs will be *copied.*
286 *
287 * @param command The command to bind.
288 * @param interruptible Whether the command should be interruptible.
289 * @return The trigger, for chained calls.
290 */
291 template <class T, typename = std::enable_if_t<std::is_base_of_v<
292 Command, std::remove_reference_t<T>>>>
293 Trigger ToggleWhenActive(T&& command, bool interruptible = true) {
294 CommandScheduler::GetInstance().AddButton(
295 [pressedLast = m_isActive(), *this,
296 command = std::make_unique<std::remove_reference_t<T>>(
297 std::forward<T>(command)),
298 interruptible]() mutable {
299 bool pressed = m_isActive();
300
301 if (!pressedLast && pressed) {
302 if (command->IsScheduled()) {
303 command->Cancel();
304 } else {
305 command->Schedule(interruptible);
306 }
307 }
308
309 pressedLast = pressed;
310 });
311 return *this;
312 }

324 /**
325 * Composes two triggers with logical AND.
326 *
327 * @return A trigger which is active when both component triggers are active.
328 */
329 Trigger operator&&(Trigger rhs) {
330 return Trigger([*this, rhs] { return m_isActive() && rhs.m_isActive(); });
331 }
332
333 /**
334 * Composes two triggers with logical OR.
335 *
336 * @return A trigger which is active when either component trigger is active.
337 */
338 Trigger operator||(Trigger rhs) {
339 return Trigger([*this, rhs] { return m_isActive() || rhs.m_isActive(); });
340 }
341
342 /**
343 * Composes a trigger with logical NOT.
344 *
345 * @return A trigger which is active when the component trigger is inactive,
346 * and vice-versa.
347 */
348 Trigger operator!() {
349 return Trigger([*this] { return !m_isActive(); });
350 }
351
352 /**
353 * Creates a new debounced trigger from this trigger - it will become active
354 * when this trigger has been active for longer than the specified period.
355 *
356 * @param debounceTime The debounce period.
357 * @param type The debounce type.
358 * @return The debounced trigger.
359 */
360 Trigger Debounce(units::second_t debounceTime,
361 frc::Debouncer::DebounceType type =
362 frc::Debouncer::DebounceType::kRising);
