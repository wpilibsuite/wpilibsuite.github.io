27 /// Storage for a \ref static_allocator.
28 /// Its constructor will take a reference to it and use it for its allocation.
29 /// The storage type is simply a \c char array aligned for maximum alignment.
30 /// \note It is not allowed to access the memory of the storage.
31 /// \ingroup allocator
32 template <std::size_t Size>
33 struct static_allocator_storage
34 {
35 alignas(detail::max_alignment) char storage[Size];
36 };
37
38 static_assert(sizeof(static_allocator_storage<1024>) == 1024, "");
39 static_assert(alignof(static_allocator_storage<1024>) == detail::max_alignment, "");

43 /// A stateful \concept{concept_rawallocator,RawAllocator} that uses a fixed sized storage for the allocations.
44 /// It works on a \ref static_allocator_storage and uses its memory for all allocations.
45 /// Deallocations are not supported, memory cannot be marked as freed.<br>
46 /// \note It is not allowed to share an \ref static_allocator_storage between multiple \ref static_allocator objects.
47 /// \ingroup allocator
48 class static_allocator
49 {
50 public:
51 using is_stateful = std::true_type;
52
53 /// \effects Creates it by passing it a \ref static_allocator_storage by reference.
54 /// It will take the address of the storage and use its memory for the allocation.
55 /// \requires The storage object must live as long as the allocator object.
56 /// It must not be shared between multiple allocators,
57 /// i.e. the object must not have been passed to a constructor before.
58 template <std::size_t Size>
59 static_allocator(static_allocator_storage<Size>& storage) noexcept
60 : stack_(&storage), end_(stack_.top() + Size)
61 {
62 }
63
64 /// \effects A \concept{concept_rawallocator,RawAllocator} allocation function.
65 /// It uses the specified \ref static_allocator_storage.
66 /// \returns A pointer to a \concept{concept_node,node}, it will never be \c nullptr.
67 /// \throws An exception of type \ref out_of_memory or whatever is thrown by its handler if the storage is exhausted.
68 void* allocate_node(std::size_t size, std::size_t alignment);
69
70 /// \effects A \concept{concept_rawallocator,RawAllocator} deallocation function.
71 /// It does nothing, deallocation is not supported by this allocator.
72 void deallocate_node(void*, std::size_t, std::size_t) noexcept {}
73
74 /// \returns The maximum node size which is the capacity remaining inside the \ref static_allocator_storage.
75 std::size_t max_node_size() const noexcept
76 {
77 return static_cast<std::size_t>(end_ - stack_.top());
78 }
79
80 /// \returns The maximum possible value since there is no alignment restriction
81 /// (except indirectly through the size of the \ref static_allocator_storage).
82 std::size_t max_alignment() const noexcept
83 {
84 return std::size_t(-1);
85 }
86
87 private:
88 allocator_info info() const noexcept;
89
90 detail::fixed_memory_stack stack_;
91 const char* end_;
92 };

100 /// A \concept{concept_blockallocator,BlockAllocator} that allocates the blocks from a fixed size storage.
101 /// It works on a \ref static_allocator_storage and uses it for all allocations,
102 /// deallocations are only allowed in reversed order which is guaranteed by \ref memory_arena.
103 /// \note It is not allowed to share an \ref static_allocator_storage between multiple \ref static_allocator objects.
104 /// \ingroup allocator
105 class static_block_allocator
106 {
107 public:
108 /// \effects Creates it by passing it the block size and a \ref static_allocator_storage by reference.
109 /// It will take the address of the storage and use it to allocate \c block_size'd blocks.
110 /// \requires The storage object must live as long as the allocator object.
111 /// It must not be shared between multiple allocators,
112 /// i.e. the object must not have been passed to a constructor before.
113 /// The size of the \ref static_allocator_storage must be a multiple of the (non-null) block size.
114 template <std::size_t Size>
115 static_block_allocator(std::size_t block_size,
116 static_allocator_storage<Size>& storage) noexcept
117 : cur_(static_cast<char*>(static_cast<void*>(&storage))),
118 end_(cur_ + Size),
119 block_size_(block_size)
120 {
121 WPI_MEMORY_ASSERT(block_size <= Size);
122 WPI_MEMORY_ASSERT(Size % block_size == 0u);
123 }
124
125 ~static_block_allocator() noexcept = default;
126
127 /// @{
128 /// \effects Moves the block allocator, it transfers ownership over the \ref static_allocator_storage.
129 /// This does not invalidate any memory blocks.
130 static_block_allocator(static_block_allocator&& other) noexcept
131 : cur_(other.cur_), end_(other.end_), block_size_(other.block_size_)
132 {
133 other.cur_ = other.end_ = nullptr;
134 other.block_size_ = 0;
135 }
136
137 static_block_allocator& operator=(static_block_allocator&& other) noexcept
138 {
139 static_block_allocator tmp(detail::move(other));
140 swap(*this, tmp);
141 return *this;
142 }
143 /// @}
144
145 /// \effects Swaps the ownership over the \ref static_allocator_storage.
146 /// This does not invalidate any memory blocks.
147 friend void swap(static_block_allocator& a, static_block_allocator& b) noexcept
148 {
149 detail::adl_swap(a.cur_, b.cur_);
150 detail::adl_swap(a.end_, b.end_);
151 detail::adl_swap(a.block_size_, b.block_size_);
152 }
153
154 /// \effects Allocates a new block by returning the \ref next_block_size() bytes.
155 /// \returns The new memory block.
156 memory_block allocate_block();
157
158 /// \effects Deallocates the last memory block by marking the block as free again.
159 /// This block will be returned again by the next call to \ref allocate_block().
160 /// \requires \c block must be the current top block of the memory,
161 /// this is guaranteed by \ref memory_arena.
162 void deallocate_block(memory_block block) noexcept;
163
164 /// \returns The next block size, this is the size passed to the constructor.
165 std::size_t next_block_size() const noexcept
166 {
167 return block_size_;
168 }
169
170 private:
171 allocator_info info() const noexcept;
172
173 char * cur_, *end_;
174 std::size_t block_size_;
175 };
