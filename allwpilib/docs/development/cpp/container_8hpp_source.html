214 /// @{
215 /// Convenience function to create a container adapter using a certain
216 /// \concept{concept_rawallocator,RawAllocator}. \returns An empty adapter with an
217 /// implementation container using a reference to a given allocator. \ingroup adapter
218 template <typename T, class RawAllocator, class Container = deque<T, RawAllocator>>
219 std::stack<T, Container> make_stack(RawAllocator& allocator)
220 {
221 return std::stack<T, Container>{Container(allocator)};
222 }
223
224 /// \copydoc make_stack
225 template <typename T, class RawAllocator, class Container = deque<T, RawAllocator>>
226 std::queue<T, Container> make_queue(RawAllocator& allocator)
227 {
228 return std::queue<T, Container>{Container(allocator)};
229 }
230
231 /// \copydoc make_stack
232 template <typename T, class RawAllocator, class Container = deque<T, RawAllocator>,
233 class Compare = std::less<T>>
234 std::priority_queue<T, Container, Compare> make_priority_queue(RawAllocator& allocator,
235 Compare comp = {})
236 {
237 return std::priority_queue<T, Container, Compare>{detail::move(comp),
238 Container(allocator)};
239 }
