25/**
26 * Process handle.
27 * Process handles will spawn a new process and allow the user to control it
28 * and establish communication channels with it using streams.
29 */
30class Process final : public HandleImpl<Process, uv_process_t> {
31 struct private_init {};
32
33 public:
34 explicit Process(const private_init&) {}
35 ~Process() noexcept override = default;
36
37 /**
38 * Structure for Spawn() option temporaries. This is a reference type, so if
39 * this value is stored outside of a temporary, be careful about overwriting
40 * what it points to.
41 */
42 struct Option {
43 enum Type {
44 kNone,
45 kArg,
46 kEnv,
47 kCwd,
48 kUid,
49 kGid,
50 kSetFlags,
51 kClearFlags,
52 kStdioIgnore,
53 kStdioInheritFd,
54 kStdioInheritPipe,
55 kStdioCreatePipe
56 };
57
58 Option() : m_type(kNone) {}
59
60 /*implicit*/ Option(const char* arg) { // NOLINT
61 m_data.str = arg;
62 }
63
64 /*implicit*/ Option(const std::string& arg) { // NOLINT
65 m_data.str = arg.data();
66 }
67
68 /*implicit*/ Option(std::string_view arg) // NOLINT
69 : m_strData(arg) {
70 m_data.str = m_strData.c_str();
71 }
72
73 /*implicit*/ Option(const SmallVectorImpl<char>& arg) // NOLINT
74 : m_strData(arg.data(), arg.size()) {
75 m_data.str = m_strData.c_str();
76 }
77
78 explicit Option(Type type) : m_type(type) {}
79
80 Type m_type = kArg;
81 std::string m_strData;
82 union {
83 const char* str;
84 uv_uid_t uid;
85 uv_gid_t gid;
86 unsigned int flags;
87 struct {
88 size_t index;
89 union {
90 int fd;
91 Pipe* pipe;
92 };
93 unsigned int flags;
94 } stdio;
95 } m_data;
96 };
97
98 /**
99 * Set environment variable for the subprocess. If not set, the parent's
100 * environment is used.
101 * @param env environment variable
102 */
103 static Option Env(std::string_view env) {
104 Option o(Option::kEnv);
105 o.m_strData = env;
106 o.m_data.str = o.m_strData.c_str();
107 return o;
108 }
109
110 /**
111 * Set the current working directory for the subprocess.
112 * @param cwd current working directory
113 */
114 static Option Cwd(std::string_view cwd) {
115 Option o(Option::kCwd);
116 o.m_strData = cwd;
117 o.m_data.str = o.m_strData.c_str();
118 return o;
119 }
120
121 /**
122 * Set the child process' user id.
123 * @param uid user id
124 */
125 static Option Uid(uv_uid_t uid) {
126 Option o(Option::kUid);
127 o.m_data.uid = uid;
128 return o;
129 }
130
131 /**
132 * Set the child process' group id.
133 * @param gid group id
134 */
135 static Option Gid(uv_gid_t gid) {
136 Option o(Option::kGid);
137 o.m_data.gid = gid;
138 return o;
139 }
140
141 /**
142 * Set spawn flags.
143 * @param flags Bitmask values from uv_process_flags.
144 */
145 static Option SetFlags(unsigned int flags) {
146 Option o(Option::kSetFlags);
147 o.m_data.flags = flags;
148 return o;
149 }
150
151 /**
152 * Clear spawn flags.
153 * @param flags Bitmask values from uv_process_flags.
154 */
155 static Option ClearFlags(unsigned int flags) {
156 Option o(Option::kClearFlags);
157 o.m_data.flags = flags;
158 return o;
159 }
160
161 /**
162 * Explicitly ignore a stdio.
163 * @param index stdio index
164 */
165 static Option StdioIgnore(size_t index) {
166 Option o(Option::kStdioIgnore);
167 o.m_data.stdio.index = index;
168 return o;
169 }
170
171 /**
172 * Inherit a file descriptor from the parent process.
173 * @param index stdio index
174 * @param fd parent file descriptor
175 */
176 static Option StdioInherit(size_t index, int fd) {
177 Option o(Option::kStdioInheritFd);
178 o.m_data.stdio.index = index;
179 o.m_data.stdio.fd = fd;
180 return o;
181 }
182
183 /**
184 * Inherit a pipe from the parent process.
185 * @param index stdio index
186 * @param pipe pipe
187 */
188 static Option StdioInherit(size_t index, Pipe& pipe) {
189 Option o(Option::kStdioInheritPipe);
190 o.m_data.stdio.index = index;
191 o.m_data.stdio.pipe = &pipe;
192 return o;
193 }
194
195 /**
196 * Create a pipe between the child and the parent.
197 * @param index stdio index
198 * @param pipe pipe
199 * @param flags Some combination of UV_READABLE_PIPE, UV_WRITABLE_PIPE, and
200 * UV_OVERLAPPED_PIPE (Windows only, ignored on Unix).
201 */
202 static Option StdioCreatePipe(size_t index, Pipe& pipe, unsigned int flags) {
203 Option o(Option::kStdioCreatePipe);
204 o.m_data.stdio.index = index;
205 o.m_data.stdio.pipe = &pipe;
206 o.m_data.stdio.flags = flags;
207 return o;
208 }
209
210 /**
211 * Disables inheritance for file descriptors / handles that this process
212 * inherited from its parent. The effect is that child processes spawned
213 * by this process don't accidentally inherit these handles.
214 *
215 * It is recommended to call this function as early in your program as
216 * possible, before the inherited file descriptors can be closed or
217 * duplicated.
218 */
219 static void DisableStdioInheritance() { uv_disable_stdio_inheritance(); }
220
221 /**
222 * Starts a process. If the process is not successfully spawned, an error
223 * is generated on the loop and this function returns nullptr.
224 *
225 * Possible reasons for failing to spawn would include (but not be limited to)
226 * the file to execute not existing, not having permissions to use the setuid
227 * or setgid specified, or not having enough memory to allocate for the new
228 * process.
229 *
230 * @param loop Loop object where this handle runs.
231 * @param file Path pointing to the program to be executed
232 * @param options Process options
233 */
234 static std::shared_ptr<Process> SpawnArray(Loop& loop, std::string_view file,
235 span<const Option> options);
236
237 static std::shared_ptr<Process> SpawnArray(
238 Loop& loop, std::string_view file,
239 std::initializer_list<Option> options) {
240 return SpawnArray(loop, file, {options.begin(), options.end()});
241 }
242
243 template <typename... Args>
244 static std::shared_ptr<Process> Spawn(Loop& loop, std::string_view file,
245 const Args&... options) {
246 return SpawnArray(loop, file, {options...});
247 }
248
249 /**
250 * Starts a process. If the process is not successfully spawned, an error
251 * is generated on the loop and this function returns nullptr.
252 *
253 * Possible reasons for failing to spawn would include (but not be limited to)
254 * the file to execute not existing, not having permissions to use the setuid
255 * or setgid specified, or not having enough memory to allocate for the new
256 * process.
257 *
258 * @param loop Loop object where this handle runs.
259 * @param file Path pointing to the program to be executed
260 * @param options Process options
261 */
262 static std::shared_ptr<Process> SpawnArray(const std::shared_ptr<Loop>& loop,
263 std::string_view file,
264 span<const Option> options) {
265 return SpawnArray(*loop, file, options);
266 }
267
268 static std::shared_ptr<Process> SpawnArray(
269 const std::shared_ptr<Loop>& loop, std::string_view file,
270 std::initializer_list<Option> options) {
271 return SpawnArray(*loop, file, options);
272 }
273
274 template <typename... Args>
275 static std::shared_ptr<Process> Spawn(const std::shared_ptr<Loop>& loop,
276 std::string_view file,
277 const Args&... options) {
278 return SpawnArray(*loop, file, {options...});
279 }
280
281 /**
282 * Sends the specified signal to the process.
283 * @param signum signal number
284 */
285 void Kill(int signum) { Invoke(&uv_process_kill, GetRaw(), signum); }
286
287 /**
288 * Sends the specified signal to the given PID.
289 * @param pid process ID
290 * @param signum signal number
291 * @return 0 on success, otherwise error code.
292 */
293 static int Kill(int pid, int signum) noexcept { return uv_kill(pid, signum); }
294
295 /**
296 * Get the process ID.
297 * @return Process ID.
298 */
299 uv_pid_t GetPid() const noexcept { return GetRaw()->pid; }
300
301 /**
302 * Signal generated when the process exits. The parameters are the exit
303 * status and the signal that caused the process to terminate, if any.
304 */
305 sig::Signal<int64_t, int> exited;
306};
