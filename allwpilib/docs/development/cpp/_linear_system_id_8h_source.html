33 /**
34 * Constructs the state-space model for an elevator.
35 *
36 * States: [[position], [velocity]]
37 * Inputs: [[voltage]]
38 * Outputs: [[position]]
39 *
40 * @param motor Instance of DCMotor.
41 * @param m Carriage mass.
42 * @param r Pulley radius.
43 * @param G Gear ratio from motor to carriage.
44 * @throws std::domain_error if m <= 0, r <= 0, or G <= 0.
45 */
46 static LinearSystem<2, 1, 1> ElevatorSystem(DCMotor motor,
47 units::kilogram_t m,
48 units::meter_t r, double G) {
49 if (m <= 0_kg) {
50 throw std::domain_error("m must be greater than zero.");
51 }
52 if (r <= 0_m) {
53 throw std::domain_error("r must be greater than zero.");
54 }
55 if (G <= 0.0) {
56 throw std::domain_error("G must be greater than zero.");
57 }
58
59 Matrixd<2, 2> A{{0.0, 1.0},
60 {0.0, (-std::pow(G, 2) * motor.Kt /
61 (motor.R * units::math::pow<2>(r) * m * motor.Kv))
62 .value()}};
63 Matrixd<2, 1> B{0.0, (G * motor.Kt / (motor.R * r * m)).value()};
64 Matrixd<1, 2> C{1.0, 0.0};
65 Matrixd<1, 1> D{0.0};
66
67 return LinearSystem<2, 1, 1>(A, B, C, D);
68 }

101 /**
102 * Constructs the state-space model for a 1 DOF velocity-only system from
103 * system identification data.
104 *
105 * You MUST use an SI unit (i.e. meters or radians) for the Distance template
106 * argument. You may still use non-SI units (such as feet or inches) for the
107 * actual method arguments; they will automatically be converted to SI
108 * internally.
109 *
110 * States: [[velocity]]
111 * Inputs: [[voltage]]
112 * Outputs: [[velocity]]
113 *
114 * The parameters provided by the user are from this feedforward model:
115 *
116 * u = K_v v + K_a a
117 *
118 * @param kV The velocity gain, in volt seconds per distance.
119 * @param kA The acceleration gain, in volt seconds² per distance.
120 * @throws std::domain_error if kV <= 0 or kA <= 0.
121 */
122 template <typename Distance, typename = std::enable_if_t<
123 std::is_same_v<units::meter, Distance> ||
124 std::is_same_v<units::radian, Distance>>>
125 static LinearSystem<1, 1, 1> IdentifyVelocitySystem(
126 decltype(1_V / Velocity_t<Distance>(1)) kV,
127 decltype(1_V / Acceleration_t<Distance>(1)) kA) {
128 if (kV <= decltype(kV){0}) {
129 throw std::domain_error("Kv must be greater than zero.");
130 }
131 if (kA <= decltype(kA){0}) {
132 throw std::domain_error("Ka must be greater than zero.");
133 }
134
135 Matrixd<1, 1> A{-kV.value() / kA.value()};
136 Matrixd<1, 1> B{1.0 / kA.value()};
137 Matrixd<1, 1> C{1.0};
138 Matrixd<1, 1> D{0.0};
139
140 return LinearSystem<1, 1, 1>(A, B, C, D);
141 }

158 * u = K_v v + K_a a
159 *

161 * @param kA The acceleration gain, in volt seconds² per distance.
162 * @throws std::domain_error if kV <= 0 or kA <= 0.

174 throw std::domain_error("Ka must be greater than zero.");

176
177 Matrixd<2, 2> A{{0.0, 1.0}, {0.0, -kV.value() / kA.value()}};
178 Matrixd<2, 1> B{0.0, 1.0 / kA.value()};
179 Matrixd<1, 2> C{1.0, 0.0};
180 Matrixd<1, 1> D{0.0};
181
182 return LinearSystem<2, 1, 1>(A, B, C, D);

185 /**
186 * Constructs the state-space model for a 2 DOF drivetrain velocity system
187 * from system identification data.
188 *
189 * States: [[left velocity], [right velocity]]
190 * Inputs: [[left voltage], [right voltage]]
191 * Outputs: [[left velocity], [right velocity]]
192 *
193 * @param kVlinear The linear velocity gain in volts per (meter per second).
194 * @param kAlinear The linear acceleration gain in volts per (meter per
195 * second squared).
196 * @param kVangular The angular velocity gain in volts per (meter per second).
197 * @param kAangular The angular acceleration gain in volts per (meter per
198 * second squared).
199 * @throws domain_error if kVlinear <= 0, kAlinear <= 0, kVangular <= 0,
200 * or kAangular <= 0.
201 */
202 static LinearSystem<2, 2, 2> IdentifyDrivetrainSystem(
203 decltype(1_V / 1_mps) kVlinear, decltype(1_V / 1_mps_sq) kAlinear,
204 decltype(1_V / 1_mps) kVangular, decltype(1_V / 1_mps_sq) kAangular) {
205 if (kVlinear <= decltype(kVlinear){0}) {
206 throw std::domain_error("Kv,linear must be greater than zero.");
207 }
208 if (kAlinear <= decltype(kAlinear){0}) {
209 throw std::domain_error("Ka,linear must be greater than zero.");
210 }
211 if (kVangular <= decltype(kVangular){0}) {
212 throw std::domain_error("Kv,angular must be greater than zero.");
213 }
214 if (kAangular <= decltype(kAangular){0}) {
215 throw std::domain_error("Ka,angular must be greater than zero.");
216 }
217
218 double A1 = -(kVlinear.value() / kAlinear.value() +
219 kVangular.value() / kAangular.value());
220 double A2 = -(kVlinear.value() / kAlinear.value() -
221 kVangular.value() / kAangular.value());
222 double B1 = 1.0 / kAlinear.value() + 1.0 / kAangular.value();
223 double B2 = 1.0 / kAlinear.value() - 1.0 / kAangular.value();
224
225 Matrixd<2, 2> A = 0.5 * Matrixd<2, 2>{{A1, A2}, {A2, A1}};
226 Matrixd<2, 2> B = 0.5 * Matrixd<2, 2>{{B1, B2}, {B2, B1}};
227 Matrixd<2, 2> C{{1.0, 0.0}, {0.0, 1.0}};
228 Matrixd<2, 2> D{{0.0, 0.0}, {0.0, 0.0}};
229
230 return LinearSystem<2, 2, 2>(A, B, C, D);
231 }

233 /**
234 * Constructs the state-space model for a 2 DOF drivetrain velocity system
235 * from system identification data.
236 *
237 * States: [[left velocity], [right velocity]]
238 * Inputs: [[left voltage], [right voltage]]
239 * Outputs: [[left velocity], [right velocity]]
240 *
241 * @param kVlinear The linear velocity gain in volts per (meter per second).
242 * @param kAlinear The linear acceleration gain in volts per (meter per
243 * second squared).
244 * @param kVangular The angular velocity gain in volts per (radian per
245 * second).
246 * @param kAangular The angular acceleration gain in volts per (radian per
247 * second squared).
248 * @param trackwidth The width of the drivetrain.
249 * @throws domain_error if kVlinear <= 0, kAlinear <= 0, kVangular <= 0,
250 * kAangular <= 0, or trackwidth <= 0.
251 */
252 static LinearSystem<2, 2, 2> IdentifyDrivetrainSystem(
253 decltype(1_V / 1_mps) kVlinear, decltype(1_V / 1_mps_sq) kAlinear,
254 decltype(1_V / 1_rad_per_s) kVangular,
255 decltype(1_V / 1_rad_per_s_sq) kAangular, units::meter_t trackwidth) {
256 if (kVlinear <= decltype(kVlinear){0}) {
257 throw std::domain_error("Kv,linear must be greater than zero.");
258 }
259 if (kAlinear <= decltype(kAlinear){0}) {
260 throw std::domain_error("Ka,linear must be greater than zero.");
261 }
262 if (kVangular <= decltype(kVangular){0}) {
263 throw std::domain_error("Kv,angular must be greater than zero.");
264 }
265 if (kAangular <= decltype(kAangular){0}) {
266 throw std::domain_error("Ka,angular must be greater than zero.");
267 }
268 if (trackwidth <= 0_m) {
269 throw std::domain_error("r must be greater than zero.");
270 }
271
272 // We want to find a factor to include in Kv,angular that will convert
273 // `u = Kv,angular omega` to `u = Kv,angular v`.
274 //
275 // v = omega r
276 // omega = v/r
277 // omega = 1/r v
278 // omega = 1/(trackwidth/2) v
279 // omega = 2/trackwidth v
280 //
281 // So multiplying by 2/trackwidth converts the angular gains from V/(rad/s)
282 // to V/(m/s).
283 return IdentifyDrivetrainSystem(kVlinear, kAlinear,
284 kVangular * 2.0 / trackwidth * 1_rad,
285 kAangular * 2.0 / trackwidth * 1_rad);
286 }

351 /**
352 * Constructs the state-space model for a drivetrain.
353 *
354 * States: [[left velocity], [right velocity]]
355 * Inputs: [[left voltage], [right voltage]]
356 * Outputs: [[left velocity], [right velocity]]
357 *
358 * @param motor Instance of DCMotor.
359 * @param m Drivetrain mass.
360 * @param r Wheel radius.
361 * @param rb Robot radius.
362 * @param J Moment of inertia.
363 * @param G Gear ratio from motor to wheel.
364 * @throws std::domain_error if m <= 0, r <= 0, rb <= 0, J <= 0, or
365 * G <= 0.
366 */
367 static LinearSystem<2, 2, 2> DrivetrainVelocitySystem(
368 const DCMotor& motor, units::kilogram_t m, units::meter_t r,
369 units::meter_t rb, units::kilogram_square_meter_t J, double G) {
370 if (m <= 0_kg) {
371 throw std::domain_error("m must be greater than zero.");
372 }
373 if (r <= 0_m) {
374 throw std::domain_error("r must be greater than zero.");
375 }
376 if (rb <= 0_m) {
377 throw std::domain_error("rb must be greater than zero.");
378 }
379 if (J <= 0_kg_sq_m) {
380 throw std::domain_error("J must be greater than zero.");
381 }
382 if (G <= 0.0) {
383 throw std::domain_error("G must be greater than zero.");
384 }
385
386 auto C1 = -std::pow(G, 2) * motor.Kt /
387 (motor.Kv * motor.R * units::math::pow<2>(r));
388 auto C2 = G * motor.Kt / (motor.R * r);
389
390 Matrixd<2, 2> A{{((1 / m + units::math::pow<2>(rb) / J) * C1).value(),
391 ((1 / m - units::math::pow<2>(rb) / J) * C1).value()},
392 {((1 / m - units::math::pow<2>(rb) / J) * C1).value(),
393 ((1 / m + units::math::pow<2>(rb) / J) * C1).value()}};
394 Matrixd<2, 2> B{{((1 / m + units::math::pow<2>(rb) / J) * C2).value(),
395 ((1 / m - units::math::pow<2>(rb) / J) * C2).value()},
396 {((1 / m - units::math::pow<2>(rb) / J) * C2).value(),
397 ((1 / m + units::math::pow<2>(rb) / J) * C2).value()}};
398 Matrixd<2, 2> C{{1.0, 0.0}, {0.0, 1.0}};
399 Matrixd<2, 2> D{{0.0, 0.0}, {0.0, 0.0}};
400
401 return LinearSystem<2, 2, 2>(A, B, C, D);
402 }
