18/**
19 * Constructs a plant inversion model-based feedforward from a LinearSystem.
20 *
21 * The feedforward is calculated as <strong> u_ff = B<sup>+</sup> (r_k+1 - A
22 * r_k) </strong>, where <strong> B<sup>+</sup> </strong> is the pseudoinverse
23 * of B.
24 *
25 * For more on the underlying math, read
26 * https://file.tavsys.net/control/controls-engineering-in-frc.pdf.
27 *
28 * @tparam States The number of states.
29 * @tparam Inputs The number of inputs.
30 */
31template <int States, int Inputs>
32class LinearPlantInversionFeedforward {
33 public:
34 /**
35 * Constructs a feedforward with the given plant.
36 *
37 * @tparam Outputs The number of outputs.
38 * @param plant The plant being controlled.
39 * @param dt Discretization timestep.
40 */
41 template <int Outputs>
42 LinearPlantInversionFeedforward(
43 const LinearSystem<States, Inputs, Outputs>& plant, units::second_t dt)
44 : LinearPlantInversionFeedforward(plant.A(), plant.B(), dt) {}
45
46 /**
47 * Constructs a feedforward with the given coefficients.
48 *
49 * @param A Continuous system matrix of the plant being controlled.
50 * @param B Continuous input matrix of the plant being controlled.
51 * @param dt Discretization timestep.
52 */
53 LinearPlantInversionFeedforward(
54 const Eigen::Matrix<double, States, States>& A,
55 const Eigen::Matrix<double, States, Inputs>& B, units::second_t dt)
56 : m_dt(dt) {
57 DiscretizeAB<States, Inputs>(A, B, dt, &m_A, &m_B);
58 Reset();
59 }
60
61 /**
62 * Returns the previously calculated feedforward as an input vector.
63 *
64 * @return The calculated feedforward.
65 */
66 const Eigen::Vector<double, Inputs>& Uff() const { return m_uff; }
67
68 /**
69 * Returns an element of the previously calculated feedforward.
70 *
71 * @param i Row of uff.
72 *
73 * @return The row of the calculated feedforward.
74 */
75 double Uff(int i) const { return m_uff(i, 0); }
76
77 /**
78 * Returns the current reference vector r.
79 *
80 * @return The current reference vector.
81 */
82 const Eigen::Vector<double, States>& R() const { return m_r; }
83
84 /**
85 * Returns an element of the reference vector r.
86 *
87 * @param i Row of r.
88 *
89 * @return The row of the current reference vector.
90 */
91 double R(int i) const { return m_r(i, 0); }
92
93 /**
94 * Resets the feedforward with a specified initial state vector.
95 *
96 * @param initialState The initial state vector.
97 */
98 void Reset(const Eigen::Vector<double, States>& initialState) {
99 m_r = initialState;
100 m_uff.setZero();
101 }
102
103 /**
104 * Resets the feedforward with a zero initial state vector.
105 */
106 void Reset() {
107 m_r.setZero();
108 m_uff.setZero();
109 }
110
111 /**
112 * Calculate the feedforward with only the desired
113 * future reference. This uses the internally stored "current"
114 * reference.
115 *
116 * If this method is used the initial state of the system is the one set using
117 * Reset(const Eigen::Vector<double, States>&). If the initial state is not
118 * set it defaults to a zero vector.
119 *
120 * @param nextR The reference state of the future timestep (k + dt).
121 *
122 * @return The calculated feedforward.
123 */
124 Eigen::Vector<double, Inputs> Calculate(
125 const Eigen::Vector<double, States>& nextR) {
126 return Calculate(m_r, nextR); // NOLINT
127 }
128
129 /**
130 * Calculate the feedforward with current and future reference vectors.
131 *
132 * @param r The reference state of the current timestep (k).
133 * @param nextR The reference state of the future timestep (k + dt).
134 *
135 * @return The calculated feedforward.
136 */
137 Eigen::Vector<double, Inputs> Calculate(
138 const Eigen::Vector<double, States>& r,
139 const Eigen::Vector<double, States>& nextR) {
140 m_uff = m_B.householderQr().solve(nextR - (m_A * r));
141 m_r = nextR;
142 return m_uff;
143 }
144
145 private:
146 Eigen::Matrix<double, States, States> m_A;
147 Eigen::Matrix<double, States, Inputs> m_B;
148
149 units::second_t m_dt;
150
151 // Current reference
152 Eigen::Vector<double, States> m_r;
153
154 // Computed feedforward
155 Eigen::Vector<double, Inputs> m_uff;
156};
