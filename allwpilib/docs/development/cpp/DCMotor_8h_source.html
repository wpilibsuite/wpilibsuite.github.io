52  constexpr DCMotor(units::volt_t nominalVoltage,
53  units::newton_meter_t stallTorque,
54  units::ampere_t stallCurrent, units::ampere_t freeCurrent,
55  units::radians_per_second_t freeSpeed, int numMotors = 1)
56  : nominalVoltage(nominalVoltage),
57  stallTorque(stallTorque * numMotors),
58  stallCurrent(stallCurrent * numMotors),
59  freeCurrent(freeCurrent * numMotors),
60  freeSpeed(freeSpeed),
61  R(nominalVoltage / this->stallCurrent),
62  Kv(freeSpeed / (nominalVoltage - R * this->freeCurrent)),
63  Kt(this->stallTorque / this->stallCurrent) {}

71  constexpr units::ampere_t Current(units::radians_per_second_t speed,
72  units::volt_t inputVoltage) const {
73  return -1.0 / Kv / R * speed + 1.0 / R * inputVoltage;
74  }
