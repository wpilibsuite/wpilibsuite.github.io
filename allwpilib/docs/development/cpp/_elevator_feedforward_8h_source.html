12namespace frc {
13/**
14 * A helper class that computes feedforward outputs for a simple elevator
15 * (modeled as a motor acting against the force of gravity).
16 */
17template <class Distance>
18class ElevatorFeedforward {
19 public:
20 using Velocity =
21 units::compound_unit<Distance, units::inverse<units::seconds>>;
22 using Acceleration =
23 units::compound_unit<Velocity, units::inverse<units::seconds>>;
24 using kv_unit = units::compound_unit<units::volts, units::inverse<Velocity>>;
25 using ka_unit =
26 units::compound_unit<units::volts, units::inverse<Acceleration>>;
27
28 ElevatorFeedforward() = default;
29
30 /**
31 * Creates a new ElevatorFeedforward with the specified gains.
32 *
33 * @param kS The static gain, in volts.
34 * @param kG The gravity gain, in volts.
35 * @param kV The velocity gain, in volt seconds per distance.
36 * @param kA The acceleration gain, in volt seconds² per distance.
37 */
38 constexpr ElevatorFeedforward(
39 units::volt_t kS, units::volt_t kG, units::unit_t<kv_unit> kV,
40 units::unit_t<ka_unit> kA = units::unit_t<ka_unit>(0))
41 : kS(kS), kG(kG), kV(kV), kA(kA) {}
42
43 /**
44 * Calculates the feedforward from the gains and setpoints.
45 *
46 * @param velocity The velocity setpoint, in distance per second.
47 * @param acceleration The acceleration setpoint, in distance per second².
48 * @return The computed feedforward, in volts.
49 */
50 constexpr units::volt_t Calculate(units::unit_t<Velocity> velocity,
51 units::unit_t<Acceleration> acceleration =
52 units::unit_t<Acceleration>(0)) {
53 return kS * wpi::sgn(velocity) + kG + kV * velocity + kA * acceleration;
54 }
55
56 // Rearranging the main equation from the calculate() method yields the
57 // formulas for the methods below:
58
59 /**
60 * Calculates the maximum achievable velocity given a maximum voltage supply
61 * and an acceleration. Useful for ensuring that velocity and
62 * acceleration constraints for a trapezoidal profile are simultaneously
63 * achievable - enter the acceleration constraint, and this will give you
64 * a simultaneously-achievable velocity constraint.
65 *
66 * @param maxVoltage The maximum voltage that can be supplied to the elevator.
67 * @param acceleration The acceleration of the elevator.
68 * @return The maximum possible velocity at the given acceleration.
69 */
70 constexpr units::unit_t<Velocity> MaxAchievableVelocity(
71 units::volt_t maxVoltage, units::unit_t<Acceleration> acceleration) {
72 // Assume max velocity is positive
73 return (maxVoltage - kS - kG - kA * acceleration) / kV;
74 }
75
76 /**
77 * Calculates the minimum achievable velocity given a maximum voltage supply
78 * and an acceleration. Useful for ensuring that velocity and
79 * acceleration constraints for a trapezoidal profile are simultaneously
80 * achievable - enter the acceleration constraint, and this will give you
81 * a simultaneously-achievable velocity constraint.
82 *
83 * @param maxVoltage The maximum voltage that can be supplied to the elevator.
84 * @param acceleration The acceleration of the elevator.
85 * @return The minimum possible velocity at the given acceleration.
86 */
87 constexpr units::unit_t<Velocity> MinAchievableVelocity(
88 units::volt_t maxVoltage, units::unit_t<Acceleration> acceleration) {
89 // Assume min velocity is negative, ks flips sign
90 return (-maxVoltage + kS - kG - kA * acceleration) / kV;
91 }
92
93 /**
94 * Calculates the maximum achievable acceleration given a maximum voltage
95 * supply and a velocity. Useful for ensuring that velocity and
96 * acceleration constraints for a trapezoidal profile are simultaneously
97 * achievable - enter the velocity constraint, and this will give you
98 * a simultaneously-achievable acceleration constraint.
99 *
100 * @param maxVoltage The maximum voltage that can be supplied to the elevator.
101 * @param velocity The velocity of the elevator.
102 * @return The maximum possible acceleration at the given velocity.
103 */
104 constexpr units::unit_t<Acceleration> MaxAchievableAcceleration(
105 units::volt_t maxVoltage, units::unit_t<Velocity> velocity) {
106 return (maxVoltage - kS * wpi::sgn(velocity) - kG - kV * velocity) / kA;
107 }
108
109 /**
110 * Calculates the minimum achievable acceleration given a maximum voltage
111 * supply and a velocity. Useful for ensuring that velocity and
112 * acceleration constraints for a trapezoidal profile are simultaneously
113 * achievable - enter the velocity constraint, and this will give you
114 * a simultaneously-achievable acceleration constraint.
115 *
116 * @param maxVoltage The maximum voltage that can be supplied to the elevator.
117 * @param velocity The velocity of the elevator.
118 * @return The minimum possible acceleration at the given velocity.
119 */
120 constexpr units::unit_t<Acceleration> MinAchievableAcceleration(
121 units::volt_t maxVoltage, units::unit_t<Velocity> velocity) {
122 return MaxAchievableAcceleration(-maxVoltage, velocity);
123 }
124
125 units::volt_t kS{0};
126 units::volt_t kG{0};
127 units::unit_t<kv_unit> kV{0};
128 units::unit_t<ka_unit> kA{0};
129};
130} // namespace frc
