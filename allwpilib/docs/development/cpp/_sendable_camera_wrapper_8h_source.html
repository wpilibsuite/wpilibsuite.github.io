40/**
41 * A wrapper to make video sources sendable and usable from Shuffleboard.
42 */
43class SendableCameraWrapper
44 : public wpi::Sendable,
45 public wpi::SendableHelper<SendableCameraWrapper> {
46 private:
47 struct private_init {};
48
49 public:
50 /**
51 * Creates a new sendable wrapper. Private constructor to avoid direct
52 * instantiation with multiple wrappers floating around for the same camera.
53 *
54 * @param cameraName the name of the camera to wrap
55 */
56 SendableCameraWrapper(std::string_view cameraName, const private_init&);
57
58 /**
59 * Gets a sendable wrapper object for the given video source, creating the
60 * wrapper if one does not already exist for the source.
61 *
62 * @param source the video source to wrap
63 * @return a sendable wrapper object for the video source, usable in
64 * Shuffleboard via ShuffleboardTab::Add() and ShuffleboardLayout::Add()
65 */
66 static SendableCameraWrapper& Wrap(const cs::VideoSource& source);
67 static SendableCameraWrapper& Wrap(CS_Source source);
68
69 static SendableCameraWrapper& Wrap(std::string_view cameraName,
70 wpi::span<const std::string> cameraUrls);
71
72 void InitSendable(wpi::SendableBuilder& builder) override;
73
74 private:
75 std::string m_uri;
76};

79inline SendableCameraWrapper::SendableCameraWrapper(std::string_view name,
80 const private_init&)
81 : m_uri(detail::kProtocol) {
82 detail::AddToSendableRegistry(this, name);
83 m_uri += name;
84}

86inline SendableCameraWrapper& SendableCameraWrapper::Wrap(
87 const cs::VideoSource& source) {
88 return Wrap(source.GetHandle());
89}

91inline SendableCameraWrapper& SendableCameraWrapper::Wrap(CS_Source source) {
92 CS_Status status = 0;
93 auto name = cs::GetSourceName(source, &status);
94 auto& wrapper = detail::GetSendableCameraWrapper(name);
95 if (!wrapper) {
96 wrapper = std::make_shared<SendableCameraWrapper>(name, private_init{});
97 }
98 return *wrapper;
99}
100
101inline SendableCameraWrapper& SendableCameraWrapper::Wrap(
102 std::string_view cameraName, wpi::span<const std::string> cameraUrls) {
103 auto& wrapper = detail::GetSendableCameraWrapper(cameraName);
104 if (!wrapper) {
105 wrapper =
106 std::make_shared<SendableCameraWrapper>(cameraName, private_init{});
107 }
108 auto streams = fmt::format("/CameraPublisher/{}/streams", cameraName);
109 nt::NetworkTableInstance::GetDefault().GetEntry(streams).SetStringArray(
110 cameraUrls);
111 return *wrapper;
112}
