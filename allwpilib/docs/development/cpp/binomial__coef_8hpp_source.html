27template<typename T>
28constexpr
29T
30binomial_coef_recur(const T n, const T k)
31noexcept
32{
33 return( // edge cases
34 (k == T(0) || n == k) ? T(1) : // deals with 0 choose 0 case
35 n == T(0) ? T(0) :
36 // else
37 binomial_coef_recur(n-1,k-1) + binomial_coef_recur(n-1,k) );
38}
