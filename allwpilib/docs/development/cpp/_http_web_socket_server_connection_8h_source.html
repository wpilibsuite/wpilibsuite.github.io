22/**
23 * A server-side HTTP connection that also accepts WebSocket upgrades.
24 *
25 * @tparam Derived derived class for std::enable_shared_from_this.
26 */
27template <typename Derived>
28class HttpWebSocketServerConnection
29 : public HttpServerConnection,
30 public std::enable_shared_from_this<Derived> {
31 public:
32 /**
33 * Constructor.
34 *
35 * @param stream network stream
36 * @param protocols Acceptable subprotocols
37 */
38 HttpWebSocketServerConnection(std::shared_ptr<uv::Stream> stream,
39 span<const std::string_view> protocols);
40
41 /**
42 * Constructor.
43 *
44 * @param stream network stream
45 * @param protocols Acceptable subprotocols
46 */
47 HttpWebSocketServerConnection(
48 std::shared_ptr<uv::Stream> stream,
49 std::initializer_list<std::string_view> protocols)
50 : HttpWebSocketServerConnection(stream,
51 {protocols.begin(), protocols.end()}) {}
52
53 protected:
54 /**
55 * Check that an incoming WebSocket upgrade is okay. This is called prior
56 * to accepting the upgrade (so prior to ProcessWsUpgrade()).
57 *
58 * The implementation should check other headers and return true if the
59 * WebSocket connection should be accepted.
60 *
61 * @param protocol negotiated subprotocol
62 */
63 virtual bool IsValidWsUpgrade(std::string_view protocol) { return true; }
64
65 /**
66 * Process an incoming WebSocket upgrade. This is called after the header
67 * reader has been disconnected and the websocket has been accepted.
68 *
69 * The implementation should set up appropriate callbacks on the websocket
70 * object to continue communication.
71 *
72 * @note When a WebSocket upgrade occurs, the stream user data is replaced
73 * with the websocket, and the websocket user data points to "this".
74 * Replace the websocket user data with caution!
75 */
76 virtual void ProcessWsUpgrade() = 0;
77
78 /**
79 * WebSocket connection; not valid until ProcessWsUpgrade is called.
80 */
81 WebSocket* m_websocket = nullptr;
82
83 private:
84 WebSocketServerHelper m_helper;
85 SmallVector<std::string, 2> m_protocols;
86};
