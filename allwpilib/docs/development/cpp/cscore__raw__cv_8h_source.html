157 inline void RawCvSource::PutFrame(cv::Mat& image) {
158  m_status = 0;
159  rawFrame.data = reinterpret_cast<char*>(image.data);
160  rawFrame.width = image.cols;
161  rawFrame.height = image.rows;
162  rawFrame.totalData = image.total() * image.channels();
163  rawFrame.pixelFormat = image.channels() == 3 ? CS_PIXFMT_BGR : CS_PIXFMT_GRAY;
164  PutSourceFrame(m_handle, rawFrame, &m_status);
165 }

193 inline uint64_t RawCvSink::GrabFrameDirect(cv::Mat& image, double timeout) {
194  rawFrame.height = 0;
195  rawFrame.width = 0;
196  rawFrame.pixelFormat = CS_PixelFormat::CS_PIXFMT_BGR;
197  m_status = RawSink::GrabFrame(rawFrame, timeout);
198  if (m_status <= 0) {
199  return m_status;
200  }
201  image = cv::Mat{rawFrame.height, rawFrame.width, CV_8UC3, rawFrame.data};
202  return m_status;
203 }
