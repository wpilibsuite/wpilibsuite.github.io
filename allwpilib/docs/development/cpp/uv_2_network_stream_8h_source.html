38/**
39 * Network stream handle.
40 * This is an abstract type; there are two network stream implementations (Tcp
41 * and Pipe).
42 */
43class NetworkStream : public Stream {
44 public:
45 static constexpr int kDefaultBacklog = 128;
46
47 std::shared_ptr<NetworkStream> shared_from_this() {
48 return std::static_pointer_cast<NetworkStream>(Handle::shared_from_this());
49 }
50
51 std::shared_ptr<const NetworkStream> shared_from_this() const {
52 return std::static_pointer_cast<const NetworkStream>(
53 Handle::shared_from_this());
54 }
55
56 /**
57 * Start listening for incoming connections. When a new incoming connection
58 * is received the connection signal is generated.
59 * @param backlog the number of connections the kernel might queue, same as
60 * listen(2).
61 */
62 void Listen(int backlog = kDefaultBacklog);
63
64 /**
65 * Start listening for incoming connections. This is a convenience wrapper
66 * around `Listen(int)` that also connects a callback to the connection
67 * signal. When a new incoming connection is received the connection signal
68 * is generated (and the callback is called).
69 * @param callback the callback to call when a connection is received.
70 * `Accept()` should be called from this callback.
71 * @param backlog the number of connections the kernel might queue, same as
72 * listen(2).
73 */
74 void Listen(std::function<void()> callback, int backlog = kDefaultBacklog);
75
76 /**
77 * Accept incoming connection.
78 *
79 * This call is used in conjunction with `Listen()` to accept incoming
80 * connections. Call this function after receiving a ListenEvent event to
81 * accept the connection.
82 * An error signal will be emitted in case of errors.
83 *
84 * When the connection signal is emitted it is guaranteed that this
85 * function will complete successfully the first time. If you attempt to use
86 * it more than once, it may fail.
87 * It is suggested to only call this function once per connection signal.
88 *
89 * @return The stream handle for the accepted connection, or nullptr on error.
90 */
91 std::shared_ptr<NetworkStream> Accept() {
92 return DoAccept()->shared_from_this();
93 }
94
95 /**
96 * Accept incoming connection.
97 *
98 * This call is used in conjunction with `Listen()` to accept incoming
99 * connections. Call this function after receiving a connection signal to
100 * accept the connection.
101 * An error signal will be emitted in case of errors.
102 *
103 * When the connection signal is emitted it is guaranteed that this
104 * function will complete successfully the first time. If you attempt to use
105 * it more than once, it may fail.
106 * It is suggested to only call this function once per connection signal.
107 *
108 * @param client Client stream object.
109 * @return False on error.
110 */
111 bool Accept(const std::shared_ptr<NetworkStream>& client) {
112 return Invoke(&uv_accept, GetRawStream(), client->GetRawStream());
113 }
114
115 /**
116 * Signal generated when an incoming connection is received.
117 */
118 sig::Signal<> connection;
119
120 protected:
121 explicit NetworkStream(uv_stream_t* uv_stream) : Stream{uv_stream} {}
122
123 virtual NetworkStream* DoAccept() = 0;
124};
