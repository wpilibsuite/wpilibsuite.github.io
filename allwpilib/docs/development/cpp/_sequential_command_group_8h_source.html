43 * @param commands the commands to include in this group.
44 */
45 explicit SequentialCommandGroup(

48 /**
49 * Creates a new SequentialCommandGroup. The given commands will be run

51 * finishes.
52 *
53 * @param commands the commands to include in this group.
54 */
55 template <class... Types,
56 typename = std::enable_if_t<std::conjunction_v<
57 std::is_base_of<Command, std::remove_reference_t<Types>>...>>>
58 explicit SequentialCommandGroup(Types&&... commands) {
59 AddCommands(std::forward<Types>(commands)...);
60 }
