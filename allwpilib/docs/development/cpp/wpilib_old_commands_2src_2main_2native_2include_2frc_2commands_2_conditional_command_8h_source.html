13/**
14 * A ConditionalCommand is a Command that starts one of two commands.
15 *
16 * A ConditionalCommand uses the Condition method to determine whether it should
17 * run onTrue or onFalse.
18 *
19 * A ConditionalCommand adds the proper Command to the Scheduler during
20 * Initialize() and then IsFinished() will return true once that Command has
21 * finished executing.
22 *
23 * If no Command is specified for onFalse, the occurrence of that condition
24 * will be a no-op.
25 *
26 * A ConditionalCommand will require the superset of subsystems of the onTrue
27 * and onFalse commands.
28 *
29 * This class is provided by the OldCommands VendorDep
30 *
31 * @see Command
32 * @see Scheduler
33 */
34class ConditionalCommand : public Command {
35 public:
36 /**
37 * Creates a new ConditionalCommand with given onTrue and onFalse Commands.
38 *
39 * @param onTrue The Command to execute if Condition() returns true
40 * @param onFalse The Command to execute if Condition() returns false
41 */
42 explicit ConditionalCommand(Command* onTrue, Command* onFalse = nullptr);
43
44 /**
45 * Creates a new ConditionalCommand with given onTrue and onFalse Commands.
46 *
47 * @param name The name for this command group
48 * @param onTrue The Command to execute if Condition() returns true
49 * @param onFalse The Command to execute if Condition() returns false
50 */
51 ConditionalCommand(std::string_view name, Command* onTrue,
52 Command* onFalse = nullptr);
53
54 ~ConditionalCommand() override = default;
55
56 ConditionalCommand(ConditionalCommand&&) = default;
57 ConditionalCommand& operator=(ConditionalCommand&&) = default;
58
59 protected:
60 /**
61 * The Condition to test to determine which Command to run.
62 *
63 * @return true if m_onTrue should be run or false if m_onFalse should be run.
64 */
65 virtual bool Condition() = 0;
66
67 void _Initialize() override;
68 void _Cancel() override;
69 bool IsFinished() override;
70 void _Interrupted() override;
71
72 private:
73 // The Command to execute if Condition() returns true
74 Command* m_onTrue;
75
76 // The Command to execute if Condition() returns false
77 Command* m_onFalse;
78
79 // Stores command chosen by condition
80 Command* m_chosenCommand = nullptr;
81};
