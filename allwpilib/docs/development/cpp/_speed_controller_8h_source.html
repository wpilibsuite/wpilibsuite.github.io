12/**
13 * Interface for speed controlling devices.
14 *
15 * @deprecated Use MotorController.
16 */
17class WPI_DEPRECATED("use MotorController") SpeedController {
18 public:
19 virtual ~SpeedController() = default;
20
21 /**
22 * Common interface for setting the speed of a speed controller.
23 *
24 * @param speed The speed to set. Value should be between -1.0 and 1.0.
25 */
26 virtual void Set(double speed) = 0;
27
28 /**
29 * Sets the voltage output of the SpeedController. Compensates for
30 * the current bus voltage to ensure that the desired voltage is output even
31 * if the battery voltage is below 12V - highly useful when the voltage
32 * outputs are "meaningful" (e.g. they come from a feedforward calculation).
33 *
34 * <p>NOTE: This function *must* be called regularly in order for voltage
35 * compensation to work properly - unlike the ordinary set function, it is not
36 * "set it and forget it."
37 *
38 * @param output The voltage to output.
39 */
40 virtual void SetVoltage(units::volt_t output);
41
42 /**
43 * Common interface for getting the current set speed of a speed controller.
44 *
45 * @return The current set speed. Value is between -1.0 and 1.0.
46 */
47 virtual double Get() const = 0;
48
49 /**
50 * Common interface for inverting direction of a speed controller.
51 *
52 * @param isInverted The state of inversion, true is inverted.
53 */
54 virtual void SetInverted(bool isInverted) = 0;
55
56 /**
57 * Common interface for returning the inversion state of a speed controller.
58 *
59 * @return isInverted The state of inversion, true is inverted.
60 */
61 virtual bool GetInverted() const = 0;
62
63 /**
64 * Common interface for disabling a motor.
65 */
66 virtual void Disable() = 0;
67
68 /**
69 * Common interface to stop the motor until Set is called again.
70 */
71 virtual void StopMotor() = 0;
72};
73
74} // namespace frc
