10namespace frc {
11
12/**
13 * Interface for speed controlling devices.
14 */
15class WPI_DEPRECATED("use MotorController") SpeedController {
16 public:
17 virtual ~SpeedController() = default;
18
19 /**
20 * Common interface for setting the speed of a speed controller.
21 *
22 * @param speed The speed to set. Value should be between -1.0 and 1.0.
23 */
24 virtual void Set(double speed) = 0;
25
26 /**
27 * Sets the voltage output of the SpeedController. Compensates for
28 * the current bus voltage to ensure that the desired voltage is output even
29 * if the battery voltage is below 12V - highly useful when the voltage
30 * outputs are "meaningful" (e.g. they come from a feedforward calculation).
31 *
32 * <p>NOTE: This function *must* be called regularly in order for voltage
33 * compensation to work properly - unlike the ordinary set function, it is not
34 * "set it and forget it."
35 *
36 * @param output The voltage to output.
37 */
38 virtual void SetVoltage(units::volt_t output);
39
40 /**
41 * Common interface for getting the current set speed of a speed controller.
42 *
43 * @return The current set speed. Value is between -1.0 and 1.0.
44 */
45 virtual double Get() const = 0;
46
47 /**
48 * Common interface for inverting direction of a speed controller.
49 *
50 * @param isInverted The state of inversion, true is inverted.
51 */
52 virtual void SetInverted(bool isInverted) = 0;
53
54 /**
55 * Common interface for returning the inversion state of a speed controller.
56 *
57 * @return isInverted The state of inversion, true is inverted.
58 */
59 virtual bool GetInverted() const = 0;
60
61 /**
62 * Common interface for disabling a motor.
63 */
64 virtual void Disable() = 0;
65
66 /**
67 * Common interface to stop the motor until Set is called again.
68 */
69 virtual void StopMotor() = 0;
70};
71
72} // namespace frc
