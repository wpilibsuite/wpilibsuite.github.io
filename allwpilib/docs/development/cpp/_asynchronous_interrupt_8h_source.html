17namespace frc {
18/**
19 * Class for handling asynchronous interrupts using a callback thread.
20 *
21 * <p> By default, interrupts will occur on rising edge. Callbacks are disabled
22 * by default, and Enable() must be called before they will occur.
23 *
24 * <p> Both rising and falling edges can be indicated in one callback if both a
25 * rising and falling edge occurred since the previous callback.
26 *
27 * <p>Synchronous (blocking) interrupts are handled by the SynchronousInterrupt
28 * class.
29 */
30class AsynchronousInterrupt {
31 public:
32 /**
33 * Construct an Asynchronous Interrupt from a Digital Source.
34 *
35 * <p> At construction, the interrupt will trigger on the rising edge.
36 *
37 * <p> The first bool in the callback indicates the rising edge triggered the
38 * interrupt, the second bool is falling edge.
39 *
40 * @param source the DigitalSource the interrupts are triggered from
41 * @param callback the callback function to call when the interrupt is
42 * triggered
43 */
44 AsynchronousInterrupt(DigitalSource& source,
45 std::function<void(bool, bool)> callback);
46
47 /**
48 * Construct an Asynchronous Interrupt from a Digital Source.
49 *
50 * <p> At construction, the interrupt will trigger on the rising edge.
51 *
52 * <p> The first bool in the callback indicates the rising edge triggered the
53 * interrupt, the second bool is falling edge.
54 *
55 * @param source the DigitalSource the interrupts are triggered from
56 * @param callback the callback function to call when the interrupt is
57 * triggered
58 */
59 AsynchronousInterrupt(DigitalSource* source,
60 std::function<void(bool, bool)> callback);
61
62 /**
63 * Construct an Asynchronous Interrupt from a Digital Source.
64 *
65 * <p> At construction, the interrupt will trigger on the rising edge.
66 *
67 * <p> The first bool in the callback indicates the rising edge triggered the
68 * interrupt, the second bool is falling edge.
69 *
70 * @param source the DigitalSource the interrupts are triggered from
71 * @param callback the callback function to call when the interrupt is
72 * triggered
73 */
74 AsynchronousInterrupt(std::shared_ptr<DigitalSource> source,
75 std::function<void(bool, bool)> callback);
76
77 /**
78 * Construct an Asynchronous Interrupt from a Digital Source.
79 *
80 * <p> At construction, the interrupt will trigger on the rising edge.
81 *
82 * @param source the DigitalSource the interrupts are triggered from
83 * @param f the callback function to call when the interrupt is triggered
84 * @param arg the first argument, interrupt was triggered on rising edge
85 * @param args the remaing arguments, interrupt was triggered on falling edge
86 */
87 template <typename Callable, typename Arg, typename... Args>
88 AsynchronousInterrupt(DigitalSource& source, Callable&& f, Arg&& arg,
89 Args&&... args)
90 : AsynchronousInterrupt(
91 source, std::bind(std::forward<Callable>(f), std::forward<Arg>(arg),
92 std::forward<Args>(args)...)) {}
93
94 /**
95 * Construct an Asynchronous Interrupt from a Digital Source.
96 *
97 * <p> At construction, the interrupt will trigger on the rising edge.
98 *
99 * @param source the DigitalSource the interrupts are triggered from
100 * @param f the callback function to call when the interrupt is triggered
101 * @param arg the first argument, interrupt was triggered on rising edge
102 * @param args the remaing arguments, interrupt was triggered on falling edge
103 */
104 template <typename Callable, typename Arg, typename... Args>
105 AsynchronousInterrupt(DigitalSource* source, Callable&& f, Arg&& arg,
106 Args&&... args)
107 : AsynchronousInterrupt(
108 source, std::bind(std::forward<Callable>(f), std::forward<Arg>(arg),
109 std::forward<Args>(args)...)) {}
110
111 /**
112 * Construct an Asynchronous Interrupt from a Digital Source.
113 *
114 * <p> At construction, the interrupt will trigger on the rising edge.
115 *
116 * @param source the DigitalSource the interrupts are triggered from
117 * @param f the callback function to call when the interrupt is triggered
118 * @param arg the first argument, interrupt was triggered on rising edge
119 * @param args the remaing arguments, interrupt was triggered on falling edge
120 */
121 template <typename Callable, typename Arg, typename... Args>
122 AsynchronousInterrupt(std::shared_ptr<DigitalSource> source, Callable&& f,
123 Arg&& arg, Args&&... args)
124 : AsynchronousInterrupt(
125 source, std::bind(std::forward<Callable>(f), std::forward<Arg>(arg),
126 std::forward<Args>(args)...)) {}
127
128 ~AsynchronousInterrupt();
129
130 /**
131 * Enables interrupt callbacks. Before this, callbacks will not occur. Does
132 * nothing if already enabled.
133 */
134 void Enable();
135
136 /**
137 * Disables interrupt callbacks. Does nothing if already disabled.
138 */
139 void Disable();
140
141 /**
142 * Set which edges to trigger the interrupt on.
143 *
144 * @param risingEdge %Trigger on rising edge
145 * @param fallingEdge %Trigger on falling edge
146 */
147 void SetInterruptEdges(bool risingEdge, bool fallingEdge);
148
149 /**
150 * Get the timestamp of the last rising edge.
151 *
152 * <p>This function does not require the interrupt to be enabled to work.
153 *
154 * <p>This only works if rising edge was configured using SetInterruptEdges.
155 * @return the timestamp in seconds relative to GetFPGATime
156 */
157 units::second_t GetRisingTimestamp();
158
159 /**
160 * Get the timestamp of the last falling edge.
161 *
162 * <p>This function does not require the interrupt to be enabled to work.
163 *
164 * <p>This only works if falling edge was configured using SetInterruptEdges.
165 * @return the timestamp in seconds relative to GetFPGATime
166 */
167 units::second_t GetFallingTimestamp();
168
169 private:
170 void ThreadMain();
171
172 std::atomic_bool m_keepRunning{false};
173 std::thread m_thread;
174 SynchronousInterrupt m_interrupt;
175 std::function<void(bool, bool)> m_callback;
176};
