23/**
24 * Represents a time-parameterized trajectory. The trajectory contains of
25 * various States that represent the pose, curvature, time elapsed, velocity,
26 * and acceleration at that point.
27 */
28class WPILIB_DLLEXPORT Trajectory {
29 public:
30 /**
31 * Represents one point on the trajectory.
32 */
33 struct WPILIB_DLLEXPORT State {
34 // The time elapsed since the beginning of the trajectory.
35 units::second_t t = 0_s;
36
37 // The speed at that point of the trajectory.
38 units::meters_per_second_t velocity = 0_mps;
39
40 // The acceleration at that point of the trajectory.
41 units::meters_per_second_squared_t acceleration = 0_mps_sq;
42
43 // The pose at that point of the trajectory.
44 Pose2d pose;
45
46 // The curvature at that point of the trajectory.
47 units::curvature_t curvature{0.0};
48
49 /**
50 * Checks equality between this State and another object.
51 *
52 * @param other The other object.
53 * @return Whether the two objects are equal.
54 */
55 bool operator==(const State& other) const;
56
57 /**
58 * Checks inequality between this State and another object.
59 *
60 * @param other The other object.
61 * @return Whether the two objects are not equal.
62 */
63 bool operator!=(const State& other) const;
64
65 /**
66 * Interpolates between two States.
67 *
68 * @param endValue The end value for the interpolation.
69 * @param i The interpolant (fraction).
70 *
71 * @return The interpolated state.
72 */
73 State Interpolate(State endValue, double i) const;
74 };
75
76 Trajectory() = default;
77
78 /**
79 * Constructs a trajectory from a vector of states.
80 */
81 explicit Trajectory(const std::vector<State>& states);
82
83 /**
84 * Returns the overall duration of the trajectory.
85 * @return The duration of the trajectory.
86 */
87 units::second_t TotalTime() const { return m_totalTime; }
88
89 /**
90 * Return the states of the trajectory.
91 * @return The states of the trajectory.
92 */
93 const std::vector<State>& States() const { return m_states; }
94
95 /**
96 * Sample the trajectory at a point in time.
97 *
98 * @param t The point in time since the beginning of the trajectory to sample.
99 * @return The state at that point in time.
100 */
101 State Sample(units::second_t t) const;
102
103 /**
104 * Transforms all poses in the trajectory by the given transform. This is
105 * useful for converting a robot-relative trajectory into a field-relative
106 * trajectory. This works with respect to the first pose in the trajectory.
107 *
108 * @param transform The transform to transform the trajectory by.
109 * @return The transformed trajectory.
110 */
111 Trajectory TransformBy(const Transform2d& transform);
112
113 /**
114 * Transforms all poses in the trajectory so that they are relative to the
115 * given pose. This is useful for converting a field-relative trajectory
116 * into a robot-relative trajectory.
117 *
118 * @param pose The pose that is the origin of the coordinate frame that
119 * the current trajectory will be transformed into.
120 * @return The transformed trajectory.
121 */
122 Trajectory RelativeTo(const Pose2d& pose);
123
124 /**
125 * Concatenates another trajectory to the current trajectory. The user is
126 * responsible for making sure that the end pose of this trajectory and the
127 * start pose of the other trajectory match (if that is the desired behavior).
128 *
129 * @param other The trajectory to concatenate.
130 * @return The concatenated trajectory.
131 */
132 Trajectory operator+(const Trajectory& other) const;
133
134 /**
135 * Returns the initial pose of the trajectory.
136 *
137 * @return The initial pose of the trajectory.
138 */
139 Pose2d InitialPose() const { return Sample(0_s).pose; }
140
141 /**
142 * Checks equality between this Trajectory and another object.
143 *
144 * @param other The other object.
145 * @return Whether the two objects are equal.
146 */
147 bool operator==(const Trajectory& other) const;
148
149 /**
150 * Checks inequality between this Trajectory and another object.
151 *
152 * @param other The other object.
153 * @return Whether the two objects are inequal.
154 */
155 bool operator!=(const Trajectory& other) const;
156
157 private:
158 std::vector<State> m_states;
159 units::second_t m_totalTime = 0_s;
160};
