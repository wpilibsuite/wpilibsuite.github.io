186 template <int Derivative, int Samples>
187 static auto BackwardFiniteDifference(units::second_t period) {
188 // See
189 // https://en.wikipedia.org/wiki/Finite_difference_coefficient#Arbitrary_stencil_points
190 //
191 // For a given list of stencil points s of length n and the order of
192 // derivative d < n, the finite difference coefficients can be obtained by
193 // solving the following linear system for the vector a.
194 //
195 // @verbatim
196 // [s₁⁰ ⋯ sₙ⁰ ][a₁] [ δ₀,d ]
197 // [ ⋮ ⋱ ⋮ ][⋮ ] = d! [ ⋮ ]
198 // [s₁ⁿ⁻¹ ⋯ sₙⁿ⁻¹][aₙ] [δₙ₋₁,d]
199 // @endverbatim
200 //
201 // where δᵢ,ⱼ are the Kronecker delta. For backward finite difference, the
202 // stencil points are the range [-n + 1, 0]. The FIR gains are the elements
203 // of the vector a in reverse order divided by hᵈ.
204 //
205 // The order of accuracy of the approximation is of the form O(hⁿ⁻ᵈ).
206
207 static_assert(Derivative >= 1,
208 "Order of derivative must be greater than or equal to one.");
209 static_assert(Samples > 0, "Number of samples must be greater than zero.");
210 static_assert(Derivative < Samples,
211 "Order of derivative must be less than number of samples.");
212
213 Eigen::Matrix<double, Samples, Samples> S;
214 for (int row = 0; row < Samples; ++row) {
215 for (int col = 0; col < Samples; ++col) {
216 double s = 1 - Samples + col;
217 S(row, col) = std::pow(s, row);
218 }
219 }
220
221 // Fill in Kronecker deltas: https://en.wikipedia.org/wiki/Kronecker_delta
222 Eigen::Vector<double, Samples> d;
223 for (int i = 0; i < Samples; ++i) {
224 d(i) = (i == Derivative) ? Factorial(Derivative) : 0.0;
225 }
226
227 Eigen::Vector<double, Samples> a =
228 S.householderQr().solve(d) / std::pow(period.to<double>(), Derivative);
229
230 // Reverse gains list
231 std::vector<double> gains;
232 for (int i = Samples - 1; i >= 0; --i) {
233 gains.push_back(a(i));
234 }
235
236 return LinearFilter(gains, {});
237 }

242 void Reset() {
243 std::fill(m_inputs.begin(), m_inputs.end(), T{0.0});
244 std::fill(m_outputs.begin(), m_outputs.end(), T{0.0});
245 }

254 T Calculate(T input) {
255 T retVal{0.0};
256
257 // Rotate the inputs
258 if (m_inputGains.size() > 0) {
259 m_inputs.push_front(input);
260 }
261
262 // Calculate the new value
263 for (size_t i = 0; i < m_inputGains.size(); ++i) {
264 retVal += m_inputs[i] * m_inputGains[i];
265 }
266 for (size_t i = 0; i < m_outputGains.size(); ++i) {
267 retVal -= m_outputs[i] * m_outputGains[i];
268 }
269
270 // Rotate the outputs
271 if (m_outputGains.size() > 0) {
272 m_outputs.push_front(retVal);
273 }
274
275 return retVal;
276 }

282 std::vector<double> m_outputGains;

289 static constexpr int Factorial(int n) {
290 if (n < 2) {
291 return 1;
292 } else {
293 return n * Factorial(n - 1);
294 }
