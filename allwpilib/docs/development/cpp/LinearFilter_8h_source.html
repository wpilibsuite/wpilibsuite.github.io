166  T Calculate(T input) {
167  T retVal = T(0.0);
168 
169  // Rotate the inputs
170  m_inputs.push_front(input);
171 
172  // Calculate the new value
173  for (size_t i = 0; i < m_inputGains.size(); i++) {
174  retVal += m_inputs[i] * m_inputGains[i];
175  }
176  for (size_t i = 0; i < m_outputGains.size(); i++) {
177  retVal -= m_outputs[i] * m_outputGains[i];
178  }
179 
180  // Rotate the outputs
181  m_outputs.push_front(retVal);
182 
183  return retVal;
184  }
