182  T Calculate(T input) {
183  T retVal{0.0};
184 
185  // Rotate the inputs
186  if (m_inputGains.size() > 0) {
187  m_inputs.push_front(input);
188  }
189 
190  // Calculate the new value
191  for (size_t i = 0; i < m_inputGains.size(); ++i) {
192  retVal += m_inputs[i] * m_inputGains[i];
193  }
194  for (size_t i = 0; i < m_outputGains.size(); ++i) {
195  retVal -= m_outputs[i] * m_outputGains[i];
196  }
197 
198  // Rotate the outputs
199  if (m_outputGains.size() > 0) {
200  m_outputs.push_front(retVal);
201  }
202 
203  return retVal;
204  }
