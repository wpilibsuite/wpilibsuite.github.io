28 /// A stateful \concept{concept_rawallocator,RawAllocator} that is designed for allocations in a loop.
29 /// It uses `N` stacks for the allocation, one of them is always active.
30 /// Allocation uses the currently active stack.
31 /// Calling \ref iteration_allocator::next_iteration() at the end of the loop,
32 /// will make the next stack active for allocation,
33 /// effectively releasing all of its memory.
34 /// Any memory allocated will thus be usable for `N` iterations of the loop.
35 /// This type of allocator is a generalization of the double frame allocator.
36 /// \ingroup allocator
37 template <std::size_t N, class BlockOrRawAllocator = default_allocator>
38 class iteration_allocator
39 : WPI_EBO(detail::iteration_block_allocator<BlockOrRawAllocator>)
40 {
41 public:
42 using allocator_type = detail::iteration_block_allocator<BlockOrRawAllocator>;
43
44 /// \effects Creates it with a given initial block size and and other constructor arguments for the \concept{concept_blockallocator,BlockAllocator}.
45 /// It will allocate the first (and only) block and evenly divide it on all the stacks it uses.
46 template <typename... Args>
47 explicit iteration_allocator(std::size_t block_size, Args&&... args)
48 : allocator_type(block_size, detail::forward<Args>(args)...), cur_(0u)
49 {
50 block_ = get_allocator().allocate_block();
51 auto cur = static_cast<char*>(block_.memory);
52 auto size_each = block_.size / N;
53 for (auto i = 0u; i != N; ++i)
54 {
55 stacks_[i] = detail::fixed_memory_stack(cur);
56 cur += size_each;
57 }
58 }
59
60 iteration_allocator(iteration_allocator&& other) noexcept
61 : allocator_type(detail::move(other)),
62 block_(other.block_),
63 cur_(detail::move(other.cur_))
64 {
65 for (auto i = 0u; i != N; ++i)
66 stacks_[i] = detail::move(other.stacks_[i]);
67
68 other.cur_ = N;
69 }
70
71 ~iteration_allocator() noexcept
72 {
73 if (cur_ < N)
74 get_allocator().deallocate_block(block_);
75 }
76
77 iteration_allocator& operator=(iteration_allocator&& other) noexcept
78 {
79 allocator_type::operator=(detail::move(other));
80 block_ = other.block_;
81 cur_ = other.cur_;
82
83 for (auto i = 0u; i != N; ++i)
84 stacks_[i] = detail::move(other.stacks_[i]);
85
86 other.cur_ = N;
87
88 return *this;
89 }
90
91 /// \effects Allocates a memory block of given size and alignment.
92 /// It simply moves the top marker of the currently active stack.
93 /// \returns A \concept{concept_node,node} with given size and alignment.
94 /// \throws \ref out_of_fixed_memory if the current stack does not have any memory left.
95 /// \requires \c size and \c alignment must be valid.
96 void* allocate(std::size_t size, std::size_t alignment)
97 {
98 auto& stack = stacks_[cur_];
99
100 auto fence = detail::debug_fence_size;
101 auto offset = detail::align_offset(stack.top() + fence, alignment);
102 if (!stack.top()
103 || (fence + offset + size + fence > std::size_t(block_end(cur_) - stack.top())))
104 WPI_THROW(out_of_fixed_memory(info(), size));
105 return stack.allocate_unchecked(size, offset);
106 }
107
108 /// \effects Allocates a memory block of given size and alignment
109 /// similar to \ref allocate().
110 /// \returns A \concept{concept_node,node} with given size and alignment
111 /// or `nullptr` if the current stack does not have any memory left.
112 void* try_allocate(std::size_t size, std::size_t alignment) noexcept
113 {
114 auto& stack = stacks_[cur_];
115 return stack.allocate(block_end(cur_), size, alignment);
116 }
117
118 /// \effects Goes to the next internal stack.
119 /// This will clear the stack whose \ref max_iterations() lifetime has reached,
120 /// and use it for all allocations in this iteration.
121 /// \note This function should be called at the end of the loop.
122 void next_iteration() noexcept
123 {
124 WPI_MEMORY_ASSERT_MSG(cur_ != N, "moved-from allocator");
125 cur_ = (cur_ + 1) % N;
126 stacks_[cur_].unwind(block_start(cur_));
127 }
128
129 /// \returns The number of iteration each allocation will live.
130 /// This is the template parameter `N`.
131 static std::size_t max_iterations() noexcept
132 {
133 return N;
134 }
135
136 /// \returns The index of the current iteration.
137 /// This is modulo \ref max_iterations().
138 std::size_t cur_iteration() const noexcept
139 {
140 return cur_;
141 }
142
143 /// \returns A reference to the \concept{concept_blockallocator,BlockAllocator} used for managing the memory.
144 /// \requires It is undefined behavior to move this allocator out into another object.
145 allocator_type& get_allocator() noexcept
146 {
147 return *this;
148 }
149
150 /// \returns The amount of memory remaining in the stack with the given index.
151 /// This is the number of bytes that are available for allocation.
152 std::size_t capacity_left(std::size_t i) const noexcept
153 {
154 return std::size_t(block_end(i) - stacks_[i].top());
155 }
156
157 /// \returns The amount of memory remaining in the currently active stack.
158 std::size_t capacity_left() const noexcept
159 {
160 return capacity_left(cur_iteration());
161 }
162
163 private:
164 allocator_info info() const noexcept
165 {
166 return {WPI_MEMORY_LOG_PREFIX "::iteration_allocator", this};
167 }
168
169 char* block_start(std::size_t i) const noexcept
170 {
171 WPI_MEMORY_ASSERT_MSG(i <= N, "moved from state");
172 auto ptr = static_cast<char*>(block_.memory);
173 return ptr + (i * block_.size / N);
174 }
175
176 char* block_end(std::size_t i) const noexcept
177 {
178 WPI_MEMORY_ASSERT_MSG(i < N, "moved from state");
179 return block_start(i + 1);
180 }
181
182 detail::fixed_memory_stack stacks_[N];
183 memory_block block_;
184 std::size_t cur_;
185
186 friend allocator_traits<iteration_allocator<N, BlockOrRawAllocator>>;
187 friend composable_allocator_traits<iteration_allocator<N, BlockOrRawAllocator>>;
188 };
