103 // non changeable pointer to an Allocator that keeps a lock
104 // I don't think EBO is necessary here...
105 template <class Alloc, class Mutex>
106 class locked_allocator
107 {
108 public:
109 locked_allocator(Alloc& alloc, Mutex& m) noexcept : mutex_(&m), alloc_(&alloc)
110 {
111 mutex_->lock();
112 }
113
114 locked_allocator(locked_allocator&& other) noexcept
115 : mutex_(other.mutex_), alloc_(other.alloc_)
116 {
117 other.mutex_ = nullptr;
118 other.alloc_ = nullptr;
119 }
120
121 ~locked_allocator() noexcept
122 {
123 if (mutex_)
124 mutex_->unlock();
125 }
126
127 locked_allocator& operator=(locked_allocator&& other) noexcept = delete;
128
129 Alloc& operator*() const noexcept
130 {
131 WPI_MEMORY_ASSERT(alloc_);
132 return *alloc_;
133 }
134
135 Alloc* operator->() const noexcept
136 {
137 WPI_MEMORY_ASSERT(alloc_);
138 return alloc_;
139 }
140
141 private:
142 Mutex* mutex_; // don't use unqiue_lock to avoid dependency
143 Alloc* alloc_;
144 };

147 locked_allocator<Alloc, Mutex> lock_allocator(Alloc& a, Mutex& m)
