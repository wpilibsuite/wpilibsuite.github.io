85  const Eigen::Matrix<double, 3, 1>& u) { return u; },

89  },
90  stateStdDevs, localMeasurementStdDevs,
91  frc::AngleMean<3, 3>(2), frc::AngleMean<1, 3>(0),
92  frc::AngleResidual<3>(2), frc::AngleResidual<1>(0),
93  frc::AngleAdd<3>(2), nominalDt),

98  // Create correction mechanism for vision measurements.
99  m_visionCorrect = [&](const Eigen::Matrix<double, 3, 1>& u,
100  const Eigen::Matrix<double, 3, 1>& y) {
101  m_observer.Correct<3>(
102  u, y,
103  [](const Eigen::Matrix<double, 3, 1>& x,
104  const Eigen::Matrix<double, 3, 1>& u) { return x; },
105  m_visionDiscR, frc::AngleMean<3, 3>(2), frc::AngleResidual<3>(2),
106  frc::AngleResidual<3>(2), frc::AngleAdd<3>(2));
107  };

109  // Set initial state.

111 
112  // Calculate offsets.

203  template <typename... ModuleState>
204  Pose2d Update(const Rotation2d& gyroAngle, ModuleState&&... moduleStates) {
205  return UpdateWithTime(units::microsecond_t(wpi::Now()), gyroAngle,
206  moduleStates...);

220  template <typename... ModuleState>
221  Pose2d UpdateWithTime(units::second_t currentTime,
222  const Rotation2d& gyroAngle,
223  ModuleState&&... moduleStates) {
224  auto dt = m_prevTime >= 0_s ? currentTime - m_prevTime : m_nominalDt;
225  m_prevTime = currentTime;
226 
227  auto angle = gyroAngle + m_gyroOffset;
228  auto omega = (angle - m_previousAngle).Radians() / dt;
229 
230  auto chassisSpeeds = m_kinematics.ToChassisSpeeds(moduleStates...);
231  auto fieldRelativeSpeeds =
232  Translation2d(chassisSpeeds.vx * 1_s, chassisSpeeds.vy * 1_s)
233  .RotateBy(angle);
234 
235  auto u =
236  frc::MakeMatrix<3, 1>(fieldRelativeSpeeds.X().template to<double>(),
237  fieldRelativeSpeeds.Y().template to<double>(),
238  omega.template to<double>());
239 
240  auto localY = frc::MakeMatrix<1, 1>(angle.Radians().template to<double>());
241  m_previousAngle = angle;
242 
243  m_latencyCompensator.AddObserverState(m_observer, u, localY, currentTime);
244 
245  m_observer.Predict(u, dt);
246  m_observer.Correct(u, localY);
247 
248  return GetEstimatedPosition();
249  }
