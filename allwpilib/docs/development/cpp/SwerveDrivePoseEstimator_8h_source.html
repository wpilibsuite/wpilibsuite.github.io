76 SwerveDrivePoseEstimator(
77 const Rotation2d& gyroAngle, const Pose2d& initialPose,
78 SwerveDriveKinematics<NumModules>& kinematics,
79 const wpi::array<double, 3>& stateStdDevs,
80 const wpi::array<double, 1>& localMeasurementStdDevs,
81 const wpi::array<double, 3>& visionMeasurementStdDevs,
82 units::second_t nominalDt = 0.02_s)
83 : m_observer([](const Eigen::Matrix<double, 3, 1>& x,
84 const Eigen::Matrix<double, 3, 1>& u) { return u; },
85 [](const Eigen::Matrix<double, 3, 1>& x,
86 const Eigen::Matrix<double, 3, 1>& u) {
87 return x.block<1, 1>(2, 0);
88 },
89 stateStdDevs, localMeasurementStdDevs,
90 frc::AngleMean<3, 3>(2), frc::AngleMean<1, 3>(0),
91 frc::AngleResidual<3>(2), frc::AngleResidual<1>(0),
92 frc::AngleAdd<3>(2), nominalDt),
93 m_kinematics(kinematics),
94 m_nominalDt(nominalDt) {
95 SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
96
97 // Create correction mechanism for vision measurements.
98 m_visionCorrect = [&](const Eigen::Matrix<double, 3, 1>& u,
99 const Eigen::Matrix<double, 3, 1>& y) {
100 m_observer.Correct<3>(
101 u, y,
102 [](const Eigen::Matrix<double, 3, 1>& x,
103 const Eigen::Matrix<double, 3, 1>& u) { return x; },
104 m_visionContR, frc::AngleMean<3, 3>(2), frc::AngleResidual<3>(2),
105 frc::AngleResidual<3>(2), frc::AngleAdd<3>(2));
106 };
107
108 // Set initial state.
109 m_observer.SetXhat(PoseTo3dVector(initialPose));
110
111 // Calculate offsets.
112 m_gyroOffset = initialPose.Rotation() - gyroAngle;
113 m_previousAngle = initialPose.Rotation();
114 }

240 template <typename... ModuleState>
241 Pose2d Update(const Rotation2d& gyroAngle, ModuleState&&... moduleStates) {
242 return UpdateWithTime(units::microsecond_t(wpi::Now()), gyroAngle,
243 moduleStates...);
244 }

257 template <typename... ModuleState>
258 Pose2d UpdateWithTime(units::second_t currentTime,
259 const Rotation2d& gyroAngle,
260 ModuleState&&... moduleStates) {
261 auto dt = m_prevTime >= 0_s ? currentTime - m_prevTime : m_nominalDt;
262 m_prevTime = currentTime;
263
264 auto angle = gyroAngle + m_gyroOffset;
265 auto omega = (angle - m_previousAngle).Radians() / dt;
266
267 auto chassisSpeeds = m_kinematics.ToChassisSpeeds(moduleStates...);
268 auto fieldRelativeSpeeds =
269 Translation2d(chassisSpeeds.vx * 1_s, chassisSpeeds.vy * 1_s)
270 .RotateBy(angle);
271
272 auto u =
273 frc::MakeMatrix<3, 1>(fieldRelativeSpeeds.X().template to<double>(),
274 fieldRelativeSpeeds.Y().template to<double>(),
275 omega.template to<double>());
276
277 auto localY = frc::MakeMatrix<1, 1>(angle.Radians().template to<double>());
278 m_previousAngle = angle;
279
280 m_latencyCompensator.AddObserverState(m_observer, u, localY, currentTime);
281
282 m_observer.Predict(u, dt);
283 m_observer.Correct(u, localY);
284
285 return GetEstimatedPosition();
286 }
287
288 private:
289 UnscentedKalmanFilter<3, 3, 1> m_observer;
290 SwerveDriveKinematics<NumModules>& m_kinematics;
291 KalmanFilterLatencyCompensator<3, 3, 1, UnscentedKalmanFilter<3, 3, 1>>
292 m_latencyCompensator;
293 std::function<void(const Eigen::Matrix<double, 3, 1>& u,
294 const Eigen::Matrix<double, 3, 1>& y)>
295 m_visionCorrect;
296
297 Eigen::Matrix3d m_visionContR;
298
299 units::second_t m_nominalDt;
300 units::second_t m_prevTime = -1_s;
301
302 Rotation2d m_gyroOffset;
303 Rotation2d m_previousAngle;
304
305 template <int Dim>
306 static wpi::array<double, Dim> StdDevMatrixToArray(

308 wpi::array<double, Dim> array;
309 for (size_t i = 0; i < Dim; ++i) {
310 array[i] = vector(i);
311 }
312 return array;
313 }
