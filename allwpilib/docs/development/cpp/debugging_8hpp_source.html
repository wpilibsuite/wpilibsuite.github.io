19 /// The magic values that are used for debug filling.
20 /// If \ref WPI_MEMORY_DEBUG_FILL is \c true, memory will be filled to help detect use-after-free or missing initialization errors.
21 /// These are the constants for the different types.
22 /// \ingroup core
23 enum class debug_magic : unsigned char
24 {
25 /// Marks internal memory used by the allocator - "allocated block".
26 internal_memory = 0xAB,
27 /// Marks internal memory currently not used by the allocator - "freed block".
28 internal_freed_memory = 0xFB,
29 /// Marks allocated, but not yet used memory - "clean memory".
30 new_memory = 0xCD,
31 /// Marks freed memory - "dead memory".
32 freed_memory = 0xDD,
33 /// Marks buffer memory used to ensure proper alignment.
34 /// This memory can also serve as \ref debug_magic::fence_memory.
35 alignment_memory = 0xED,
36 /// Marks buffer memory used to protect against overflow - "fence memory".
37 /// The option \ref WPI_MEMORY_DEBUG_FENCE controls the size of a memory fence that will be placed before or after a memory block.
38 /// It helps catching buffer overflows.
39 fence_memory = 0xFD
40 };

42 /// The type of the handler called when a memory leak is detected.
43 /// Leak checking can be controlled via the option \ref WPI_MEMORY_DEBUG_LEAK_CHECK
44 /// and only affects calls through the \ref allocator_traits, not direct calls.
45 /// The handler gets the \ref allocator_info and the amount of memory leaked.
46 /// This can also be negative, meaning that more memory has been freed than allocated.
47 /// \requiredbe A leak handler shall log the leak, abort the program, do nothing or anything else that seems appropriate.
48 /// It must not throw any exceptions since it is called in the cleanup process.
49 /// \defaultbe On a hosted implementation it logs the leak to \c stderr and returns, continuing execution.
50 /// On a freestanding implementation it does nothing.
51 /// \ingroup core
52 using leak_handler = void (*)(const allocator_info& info, std::ptrdiff_t amount);
