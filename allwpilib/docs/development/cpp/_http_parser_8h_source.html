18/**
19 * HTTP protocol parser. Performs incremental parsing with callbacks for each
20 * part of the HTTP protocol. As this is incremental, it's suitable for use
21 * with event based frameworks that provide arbitrary chunks of data.
22 */
23class HttpParser {
24 public:
25 enum Type {
26 kRequest = HTTP_REQUEST,
27 kResponse = HTTP_RESPONSE,
28 kBoth = HTTP_BOTH
29 };
30
31 /**
32 * Returns the library version. Bits 16-23 contain the major version number,
33 * bits 8-15 the minor version number and bits 0-7 the patch level.
34 */
35 static uint32_t GetParserVersion();
36
37 /**
38 * Constructor.
39 * @param type Type of parser (request or response or both)
40 */
41 explicit HttpParser(Type type);
42
43 /**
44 * Reset the parser to initial state.
45 * This allows reusing the same parser object from request to request.
46 * @param type Type of parser (request or response or both)
47 */
48 void Reset(Type type);
49
50 /**
51 * Set the maximum accepted length for URLs, field names, and field values.
52 * The default is 1024.
53 * @param len maximum length
54 */
55 void SetMaxLength(size_t len) { m_maxLength = len; }
56
57 /**
58 * Executes the parser. An empty input is treated as EOF.
59 * @param in input data
60 * @return Trailing input data after the parse.
61 */
62 std::string_view Execute(std::string_view in) {
63 in.remove_prefix(
64 http_parser_execute(&m_parser, &m_settings, in.data(), in.size()));
65 return in;
66 }
67
68 /**
69 * Get HTTP major version.
70 */
71 unsigned int GetMajor() const { return m_parser.http_major; }
72
73 /**
74 * Get HTTP minor version.
75 */
76 unsigned int GetMinor() const { return m_parser.http_minor; }
77
78 /**
79 * Get HTTP status code. Valid only on responses. Valid in and after
80 * the OnStatus() callback has been called.
81 */
82 unsigned int GetStatusCode() const { return m_parser.status_code; }
83
84 /**
85 * Get HTTP method. Valid only on requests.
86 */
87 http_method GetMethod() const {
88 return static_cast<http_method>(m_parser.method);
89 }
90
91 /**
92 * Determine if an error occurred.
93 * @return False if no error.
94 */
95 bool HasError() const { return m_parser.http_errno != HPE_OK; }
96
97 /**
98 * Get error number.
99 */
100 http_errno GetError() const {
101 return static_cast<http_errno>(m_parser.http_errno);
102 }
103
104 /**
105 * Abort the parse. Call this from a callback handler to indicate an error.
106 * This will result in GetError() returning one of the callback-related
107 * errors (e.g. HPE_CB_message_begin).
108 */
109 void Abort() { m_aborted = true; }
110
111 /**
112 * Determine if an upgrade header was present and the parser has exited
113 * because of that. Should be checked when Execute() returns in addition to
114 * checking GetError().
115 * @return True if upgrade header, false otherwise.
116 */
117 bool IsUpgrade() const { return m_parser.upgrade; }
118
119 /**
120 * If this returns false in the headersComplete or messageComplete
121 * callback, then this should be the last message on the connection.
122 * If you are the server, respond with the "Connection: close" header.
123 * If you are the client, close the connection.
124 */
125 bool ShouldKeepAlive() const { return http_should_keep_alive(&m_parser); }
126
127 /**
128 * Pause the parser.
129 * @param paused True to pause, false to unpause.
130 */
131 void Pause(bool paused) { http_parser_pause(&m_parser, paused); }
132
133 /**
134 * Checks if this is the final chunk of the body.
135 */
136 bool IsBodyFinal() const { return http_body_is_final(&m_parser); }
137
138 /**
139 * Get URL. Valid in and after the url callback has been called.
140 */
141 std::string_view GetUrl() const { return m_urlBuf.str(); }
142
143 /**
144 * Message begin callback.
145 */
146 sig::Signal<> messageBegin;
147
148 /**
149 * URL callback.
150 *
151 * The parameter to the callback is the complete URL string.
152 */
153 sig::Signal<std::string_view> url;
154
155 /**
156 * Status callback.
157 *
158 * The parameter to the callback is the complete status string.
159 * GetStatusCode() can be used to get the numeric status code.
160 */
161 sig::Signal<std::string_view> status;
162
163 /**
164 * Header field callback.
165 *
166 * The parameters to the callback are the field name and field value.
167 */
168 sig::Signal<std::string_view, std::string_view> header;
169
170 /**
171 * Headers complete callback.
172 *
173 * The parameter to the callback is whether the connection should be kept
174 * alive. If this is false, then this should be the last message on the
175 * connection. If you are the server, respond with the "Connection: close"
176 * header. If you are the client, close the connection.
177 */
178 sig::Signal<bool> headersComplete;
179
180 /**
181 * Body data callback.
182 *
183 * The parameters to the callback is the data chunk and whether this is the
184 * final chunk of data in the message. Note this callback will be called
185 * multiple times arbitrarily (e.g. it's possible that it may be called with
186 * just a few characters at a time).
187 */
188 sig::Signal<std::string_view, bool> body;
189
190 /**
191 * Headers complete callback.
192 *
193 * The parameter to the callback is whether the connection should be kept
194 * alive. If this is false, then this should be the last message on the
195 * connection. If you are the server, respond with the "Connection: close"
196 * header. If you are the client, close the connection.
197 */
198 sig::Signal<bool> messageComplete;
199
200 /**
201 * Chunk header callback.
202 *
203 * The parameter to the callback is the chunk size.
204 */
205 sig::Signal<uint64_t> chunkHeader;
206
207 /**
208 * Chunk complete callback.
209 */
210 sig::Signal<> chunkComplete;
211
212 private:
213 http_parser m_parser;
214 http_parser_settings m_settings;
215
216 size_t m_maxLength = 1024;
217 enum { kStart, kUrl, kStatus, kField, kValue } m_state = kStart;
218 SmallString<128> m_urlBuf;
219 SmallString<32> m_fieldBuf;
220 SmallString<128> m_valueBuf;
221
222 bool m_aborted = false;
223};
