12namespace frc2 {
13/**
14 * A subsystem that uses a ProfiledPIDController to control an output. The
15 * controller is run synchronously from the subsystem's periodic() method.
16 *
17 * @see ProfiledPIDController
18 */
19template <class Distance>
20class ProfiledPIDSubsystem : public SubsystemBase {
21 using Distance_t = units::unit_t<Distance>;
22 using Velocity =
23 units::compound_unit<Distance, units::inverse<units::seconds>>;
24 using Velocity_t = units::unit_t<Velocity>;
25 using State = typename frc::TrapezoidProfile<Distance>::State;
26
27 public:
28 /**
29 * Creates a new ProfiledPIDSubsystem.
30 *
31 * @param controller the ProfiledPIDController to use
32 * @param initialPosition the initial goal position of the subsystem
33 */
34 explicit ProfiledPIDSubsystem(frc::ProfiledPIDController<Distance> controller,
35 Distance_t initialPosition = Distance_t{0})
36 : m_controller{controller} {
37 SetGoal(initialPosition);
38 }
39
40 void Periodic() override {
41 if (m_enabled) {
42 UseOutput(m_controller.Calculate(GetMeasurement(), m_goal),
43 m_controller.GetSetpoint());
44 }
45 }
46
47 /**
48 * Sets the goal state for the subsystem.
49 *
50 * @param goal The goal state for the subsystem's motion profile.
51 */
52 void SetGoal(State goal) { m_goal = goal; }
53
54 /**
55 * Sets the goal state for the subsystem. Goal velocity assumed to be zero.
56 *
57 * @param goal The goal position for the subsystem's motion profile.
58 */
59 void SetGoal(Distance_t goal) { m_goal = State{goal, Velocity_t(0)}; }
60
61 /**
62 * Enables the PID control. Resets the controller.
63 */
64 virtual void Enable() {
65 m_controller.Reset(GetMeasurement());
66 m_enabled = true;
67 }
68
69 /**
70 * Disables the PID control. Sets output to zero.
71 */
72 virtual void Disable() {
73 UseOutput(0, State{Distance_t(0), Velocity_t(0)});
74 m_enabled = false;
75 }
76
77 /**
78 * Returns whether the controller is enabled.
79 *
80 * @return Whether the controller is enabled.
81 */
82 bool IsEnabled() { return m_enabled; }
83
84 /**
85 * Returns the ProfiledPIDController.
86 *
87 * @return The controller.
88 */
89 frc::ProfiledPIDController<Distance>& GetController() { return m_controller; }
90
91 protected:
92 frc::ProfiledPIDController<Distance> m_controller;
93 bool m_enabled{false};
94
95 /**
96 * Returns the measurement of the process variable used by the
97 * ProfiledPIDController.
98 *
99 * @return the measurement of the process variable
100 */
101 virtual Distance_t GetMeasurement() = 0;
102
103 /**
104 * Uses the output from the ProfiledPIDController.
105 *
106 * @param output the output of the ProfiledPIDController
107 * @param setpoint the setpoint state of the ProfiledPIDController, for
108 * feedforward
109 */
110 virtual void UseOutput(double output, State setpoint) = 0;
111
112 private:
113 State m_goal;
114};
115} // namespace frc2
