11 /* General Handle Data Layout
12  * Bits 0-15: Handle Index
13  * Bits 16-23: 8 bit rolling reset detection
14  * Bits 24-30: Handle Type
15  * Bit 31: 1 if handle error, 0 if no error
16  *
17  * Other specialized handles will use different formats, however Bits 24-31 are
18  * always reserved for type and error handling.
19  */

128 inline int16_t getHandleTypedIndex(HAL_Handle handle, HAL_HandleEnum enumType,
129  int16_t version) {
130  if (!isHandleType(handle, enumType)) {
131  return InvalidHandleIndex;
132  }
133 #if !defined(__FRC_ROBORIO__)
134  if (!isHandleCorrectVersion(handle, version)) {
135  return InvalidHandleIndex;
136  }
137 #endif
