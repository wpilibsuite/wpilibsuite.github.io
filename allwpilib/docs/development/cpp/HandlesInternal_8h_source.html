11/* General Handle Data Layout
12 * Bits 0-15: Handle Index
13 * Bits 16-23: 8 bit rolling reset detection
14 * Bits 24-30: Handle Type
15 * Bit 31: 1 if handle error, 0 if no error
16 *
17 * Other specialized handles will use different formats, however Bits 24-31 are
18 * always reserved for type and error handling.
19 */

132inline int16_t getHandleTypedIndex(HAL_Handle handle, HAL_HandleEnum enumType,
133 int16_t version) {

135 return InvalidHandleIndex;

137#if !defined(__FRC_ROBORIO__)
138 if (!isHandleCorrectVersion(handle, version)) {
139 return InvalidHandleIndex;

161inline int16_t getPortHandleChannel(HAL_PortHandle handle) {
162 if (!isHandleType(handle, HAL_HandleEnum::Port)) {
163 return InvalidHandleIndex;
164 }
165 return static_cast<uint8_t>(handle & 0xff);
