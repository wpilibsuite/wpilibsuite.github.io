203  void Predict(const Eigen::Matrix<double, Inputs, 1>& u, units::second_t dt) {
204  m_dt = dt;
205 
206  // Discretize Q before projecting mean and covariance forward
207  Eigen::Matrix<double, States, States> contA =
208  NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
209  Eigen::Matrix<double, States, States> discA;
210  Eigen::Matrix<double, States, States> discQ;
211  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
212 
213  Eigen::Matrix<double, States, 2 * States + 1> sigmas =
214  m_pts.SigmaPoints(m_xHat, m_P);
215 
216  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
217  Eigen::Matrix<double, States, 1> x =
218  sigmas.template block<States, 1>(0, i);
219  m_sigmasF.template block<States, 1>(0, i) = RK4(m_f, x, u, dt);
220  }
221 
222  auto ret = UnscentedTransform<States, States>(
223  m_sigmasF, m_pts.Wm(), m_pts.Wc(), m_meanFuncX, m_residualFuncX);
224  m_xHat = std::get<0>(ret);
225  m_P = std::get<1>(ret);
226 
227  m_P += discQ;
228  }

301  template <int Rows>
302  void Correct(const Eigen::Matrix<double, Inputs, 1>& u,
303  const Eigen::Matrix<double, Rows, 1>& y,
304  std::function<Eigen::Matrix<double, Rows, 1>(
305  const Eigen::Matrix<double, States, 1>&,
306  const Eigen::Matrix<double, Inputs, 1>&)>
307  h,
308  const Eigen::Matrix<double, Rows, Rows>& R,
309  std::function<Eigen::Matrix<double, Rows, 1>(
310  const Eigen::Matrix<double, Rows, 2 * States + 1>&,
311  const Eigen::Matrix<double, 2 * States + 1, 1>&)>
312  meanFuncY,
313  std::function<Eigen::Matrix<double, Rows, 1>(
314  const Eigen::Matrix<double, Rows, 1>&,
315  const Eigen::Matrix<double, Rows, 1>&)>
316  residualFuncY,
317  std::function<Eigen::Matrix<double, States, 1>(
318  const Eigen::Matrix<double, States, 1>&,
319  const Eigen::Matrix<double, States, 1>&)>
320  residualFuncX,
321  std::function<Eigen::Matrix<double, States, 1>(
322  const Eigen::Matrix<double, States, 1>&,
323  const Eigen::Matrix<double, States, 1>)>
324  addFuncX) {
325  const Eigen::Matrix<double, Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
326 
327  // Transform sigma points into measurement space
328  Eigen::Matrix<double, Rows, 2 * States + 1> sigmasH;
329  Eigen::Matrix<double, States, 2 * States + 1> sigmas =
330  m_pts.SigmaPoints(m_xHat, m_P);
331  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
332  sigmasH.template block<Rows, 1>(0, i) =
333  h(sigmas.template block<States, 1>(0, i), u);
334  }
335 
336  // Mean and covariance of prediction passed through UT
337  auto [yHat, Py] = UnscentedTransform<States, Rows>(
338  sigmasH, m_pts.Wm(), m_pts.Wc(), meanFuncY, residualFuncY);
339  Py += discR;
340 
341  // Compute cross covariance of the state and the measurements
342  Eigen::Matrix<double, States, Rows> Pxy;
343  Pxy.setZero();
344  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
345  Pxy +=
346  m_pts.Wc(i) *
347  (residualFuncX(m_sigmasF.template block<States, 1>(0, i), m_xHat)) *
348  (residualFuncY(sigmasH.template block<Rows, 1>(0, i), yHat))
349  .transpose();
350  }
351 
352  // K = P_{xy} Py^-1
353  // K^T = P_y^T^-1 P_{xy}^T
354  // P_y^T K^T = P_{xy}^T
355  // K^T = P_y^T.solve(P_{xy}^T)
356  // K = (P_y^T.solve(P_{xy}^T)^T
357  Eigen::Matrix<double, States, Rows> K =
358  Py.transpose().ldlt().solve(Pxy.transpose()).transpose();
359 
360  m_xHat = addFuncX(m_xHat, K * residualFuncY(y, yHat));
361  m_P -= K * Py * K.transpose();
362  }
