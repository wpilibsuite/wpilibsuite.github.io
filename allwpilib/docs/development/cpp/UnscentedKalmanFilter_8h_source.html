204  void Predict(const Eigen::Matrix<double, Inputs, 1>& u, units::second_t dt) {
205  m_dt = dt;
206 
207  // Discretize Q before projecting mean and covariance forward
208  Eigen::Matrix<double, States, States> contA =
209  NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
210  Eigen::Matrix<double, States, States> discA;
211  Eigen::Matrix<double, States, States> discQ;
212  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
213 
214  Eigen::Matrix<double, States, 2 * States + 1> sigmas =
215  m_pts.SigmaPoints(m_xHat, m_P);
216 
217  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
218  Eigen::Matrix<double, States, 1> x =
219  sigmas.template block<States, 1>(0, i);
220  m_sigmasF.template block<States, 1>(0, i) = RK4(m_f, x, u, dt);
221  }
222 
223  auto ret = UnscentedTransform<States, States>(
224  m_sigmasF, m_pts.Wm(), m_pts.Wc(), m_meanFuncX, m_residualFuncX);
225  m_xHat = std::get<0>(ret);
226  m_P = std::get<1>(ret);
227 
228  m_P += discQ;
229  }

302  template <int Rows>
303  void Correct(const Eigen::Matrix<double, Inputs, 1>& u,
304  const Eigen::Matrix<double, Rows, 1>& y,
305  std::function<Eigen::Matrix<double, Rows, 1>(
306  const Eigen::Matrix<double, States, 1>&,
307  const Eigen::Matrix<double, Inputs, 1>&)>
308  h,
309  const Eigen::Matrix<double, Rows, Rows>& R,
310  std::function<Eigen::Matrix<double, Rows, 1>(
311  const Eigen::Matrix<double, Rows, 2 * States + 1>&,
312  const Eigen::Matrix<double, 2 * States + 1, 1>&)>
313  meanFuncY,
314  std::function<Eigen::Matrix<double, Rows, 1>(
315  const Eigen::Matrix<double, Rows, 1>&,
316  const Eigen::Matrix<double, Rows, 1>&)>
317  residualFuncY,
318  std::function<Eigen::Matrix<double, States, 1>(
319  const Eigen::Matrix<double, States, 1>&,
320  const Eigen::Matrix<double, States, 1>&)>
321  residualFuncX,
322  std::function<Eigen::Matrix<double, States, 1>(
323  const Eigen::Matrix<double, States, 1>&,
324  const Eigen::Matrix<double, States, 1>)>
325  addFuncX) {
326  const Eigen::Matrix<double, Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
327 
328  // Transform sigma points into measurement space
329  Eigen::Matrix<double, Rows, 2 * States + 1> sigmasH;
330  Eigen::Matrix<double, States, 2 * States + 1> sigmas =
331  m_pts.SigmaPoints(m_xHat, m_P);
332  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
333  sigmasH.template block<Rows, 1>(0, i) =
334  h(sigmas.template block<States, 1>(0, i), u);
335  }
336 
337  // Mean and covariance of prediction passed through UT
338  auto [yHat, Py] = UnscentedTransform<States, Rows>(
339  sigmasH, m_pts.Wm(), m_pts.Wc(), meanFuncY, residualFuncY);
340  Py += discR;
341 
342  // Compute cross covariance of the state and the measurements
343  Eigen::Matrix<double, States, Rows> Pxy;
344  Pxy.setZero();
345  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
346  Pxy +=
347  m_pts.Wc(i) *
348  (residualFuncX(m_sigmasF.template block<States, 1>(0, i), m_xHat)) *
349  (residualFuncY(sigmasH.template block<Rows, 1>(0, i), yHat))
350  .transpose();
351  }
352 
353  // K = P_{xy} Py^-1
354  // K^T = P_y^T^-1 P_{xy}^T
355  // P_y^T K^T = P_{xy}^T
356  // K^T = P_y^T.solve(P_{xy}^T)
357  // K = (P_y^T.solve(P_{xy}^T)^T
358  Eigen::Matrix<double, States, Rows> K =
359  Py.transpose().ldlt().solve(Pxy.transpose()).transpose();
360 
361  m_xHat = addFuncX(m_xHat, K * residualFuncY(y, yHat));
362  m_P -= K * Py * K.transpose();
363  }
