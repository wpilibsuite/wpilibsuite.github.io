40  constexpr SimpleMotorFeedforward(
41  units::volt_t kS, units::unit_t<kv_unit> kV,
42  units::unit_t<ka_unit> kA = units::unit_t<ka_unit>(0))
43  : kS(kS), kV(kV), kA(kA) {}

52  constexpr units::volt_t Calculate(units::unit_t<Velocity> velocity,
53  units::unit_t<Acceleration> acceleration =
54  units::unit_t<Acceleration>(0)) const {
55  return kS * wpi::sgn(velocity) + kV * velocity + kA * acceleration;
56  }
57 
67  units::volt_t Calculate(units::unit_t<Velocity> currentVelocity,
68  units::unit_t<Velocity> nextVelocity,
69  units::second_t dt) const {
70  auto plant = LinearSystemId::IdentifyVelocitySystem<Distance>(kV, kA);
71  LinearPlantInversionFeedforward<1, 1> feedforward{plant, dt};
72 
73  Eigen::Matrix<double, 1, 1> r;
74  r << currentVelocity.template to<double>();
75 
76  Eigen::Matrix<double, 1, 1> nextR;
77  nextR << nextVelocity.template to<double>();
78 
79  return kS * wpi::sgn(currentVelocity.template to<double>()) +
80  units::volt_t{feedforward.Calculate(r, nextR)(0)};
81  }

97  constexpr units::unit_t<Velocity> MaxAchievableVelocity(
98  units::volt_t maxVoltage,
99  units::unit_t<Acceleration> acceleration) const {
100  // Assume max velocity is positive
101  return (maxVoltage - kS - kA * acceleration) / kV;
102  }
103 
115  constexpr units::unit_t<Velocity> MinAchievableVelocity(
116  units::volt_t maxVoltage,
117  units::unit_t<Acceleration> acceleration) const {
118  // Assume min velocity is positive, ks flips sign
119  return (-maxVoltage + kS - kA * acceleration) / kV;
120  }
121 
133  constexpr units::unit_t<Acceleration> MaxAchievableAcceleration(
134  units::volt_t maxVoltage, units::unit_t<Velocity> velocity) const {
135  return (maxVoltage - kS * wpi::sgn(velocity) - kV * velocity) / kA;
136  }
137 
149  constexpr units::unit_t<Acceleration> MinAchievableAcceleration(
150  units::volt_t maxVoltage, units::unit_t<Velocity> velocity) const {
151  return MaxAchievableAcceleration(-maxVoltage, velocity);
152  }
153 
154  units::volt_t kS{0};
155  units::unit_t<kv_unit> kV{0};
156  units::unit_t<ka_unit> kA{0};
157 };
