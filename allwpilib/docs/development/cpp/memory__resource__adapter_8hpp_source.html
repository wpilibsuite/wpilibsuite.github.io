94 /// Wraps a \concept{concept_rawallocator,RawAllocator} and makes it a \ref memory_resource.
95 /// \ingroup adapter
96 template <class RawAllocator>
97 class memory_resource_adapter
98 : public memory_resource,
99 WPI_EBO(allocator_traits<RawAllocator>::allocator_type)
100 {
101 public:
102 using allocator_type = typename allocator_traits<RawAllocator>::allocator_type;
103
104 /// \effects Creates the resource by moving in the allocator.
105 memory_resource_adapter(allocator_type&& other) noexcept
106 : allocator_type(detail::move(other))
107 {
108 }
109
110 /// @{
111 /// \returns A reference to the wrapped allocator.
112 allocator_type& get_allocator() noexcept
113 {
114 return *this;
115 }
116
117 const allocator_type& get_allocator() const noexcept
118 {
119 return *this;
120 }
121 /// @}
122
123 protected:
124 using traits_type = allocator_traits<RawAllocator>;
125
126 /// \effects Allocates raw memory with given size and alignment.
127 /// It forwards to \c allocate_node() or \c allocate_array() depending on the size.
128 /// \returns The new memory as returned by the \concept{concept_rawallocator,RawAllocator}.
129 /// \throws Anything thrown by the allocation function.
130 void* do_allocate(std::size_t bytes, std::size_t alignment) override
131 {
132 auto max = traits_type::max_node_size(*this);
133 if (bytes <= max)
134 return traits_type::allocate_node(*this, bytes, alignment);
135 auto div = bytes / max;
136 auto mod = bytes % max;
137 auto n = div + (mod != 0);
138 return traits_type::allocate_array(*this, n, max, alignment);
139 }
140
141 /// \effects Deallocates memory previously allocated by \ref do_allocate.
142 /// It forwards to \c deallocate_node() or \c deallocate_array() depending on the size.
143 /// \throws Nothing.
144 void do_deallocate(void* p, std::size_t bytes, std::size_t alignment) override
145 {
146 auto max = traits_type::max_node_size(*this);
147 if (bytes <= max)
148 traits_type::deallocate_node(*this, p, bytes, alignment);
149 else
150 {
151 auto div = bytes / max;
152 auto mod = bytes % max;
153 auto n = div + (mod != 0);
154 traits_type::deallocate_array(*this, p, n, max, alignment);
155 }
156 }
157
158 /// \returns Whether or not \c *this is equal to \c other
159 /// by comparing the addresses.
160 bool do_is_equal(const memory_resource& other) const noexcept override
161 {
162 return this == &other;
163 }
164 };

166 /// Wraps a \ref memory_resource and makes it a \concept{concept_rawallocator,RawAllocator}.
167 /// \ingroup adapter
168 class memory_resource_allocator
169 {
170 public:
171 /// \effects Creates it by giving it a pointer to the \ref memory_resource.
172 /// \requires \c ptr must not be \c nullptr.
173 memory_resource_allocator(memory_resource* ptr) noexcept : ptr_(ptr)
174 {
175 WPI_MEMORY_ASSERT(ptr);
176 }
177
178 /// \effects Allocates a node by forwarding to the \c allocate() function.
179 /// \returns The node as returned by the \ref memory_resource.
180 /// \throws Anything thrown by the \c allocate() function.
181 void* allocate_node(std::size_t size, std::size_t alignment)
182 {
183 return ptr_->allocate(size, alignment);
184 }
185
186 /// \effects Deallocates a node by forwarding to the \c deallocate() function.
187 void deallocate_node(void* ptr, std::size_t size, std::size_t alignment) noexcept
188 {
189 ptr_->deallocate(ptr, size, alignment);
190 }
191
192 /// \returns The maximum alignment which is the maximum value of type \c std::size_t.
193 std::size_t max_alignment() const noexcept
194 {
195 return std::size_t(-1);
196 }
197
198 /// \returns A pointer to the used \ref memory_resource, this is never \c nullptr.
199 memory_resource* resource() const noexcept
200 {
201 return ptr_;
202 }
203
204 private:
205 memory_resource* ptr_;
206 };
